pub fn new(store_base_lba: u64) -> Self

store_base_lba = api::disk::PART_CELLSTORE_BASE_LBA. The cell-store FAT is mounted with prefix /bin so /bin/<cell> strips to <cell> at the FAT root — matching the layout gen_disk writes into the cell-store.
