1/// Temperature + humidity reading from an SHT3x sensor.
2pub struct Reading {
3    /// Temperature in tenths of °C (e.g., 253 = 25.3 °C).
4    pub temp_cx10: i32,
5    /// Relative humidity in tenths of percent (e.g., 610 = 61.0 %).
6    pub hum_px10: u32,
7    /// True when no sensor responded and synthetic data was substituted.
8    pub simulated: bool,
9}
10
11/// Decode a 6-byte SHT3x high-precision single-shot response.
12///
13/// Byte layout: [T_MSB, T_LSB, T_CRC, H_MSB, H_LSB, H_CRC].
14/// CRC bytes are ignored — the I2C bus itself is the error boundary.
15///
16/// Returns `None` when the response looks like a bus error (MSBs == 0xFF,
17/// which signals "no slave pulled SDA low during the data phase").
18pub fn parse(buf: &[u8; 6]) -> Option<Reading> {
19    if buf[0] == 0xFF && buf[1] == 0xFF {
20        return None;
21    }
22    let raw_t = ((buf[0] as u32) << 8) | buf[1] as u32;
23    let raw_h = ((buf[3] as u32) << 8) | buf[4] as u32;
24    // T[°C]  = -45 + 175 × raw_T / 65535
25    // RH[%]  = 100 × raw_H / 65535
26    let temp_cx10 = -450i32 + (1750i64 * raw_t as i64 / 65535) as i32;
27    let hum_px10 = (1000u64 * raw_h as u64 / 65535) as u32;
28    Some(Reading {
29        temp_cx10,
30        hum_px10,
31        simulated: false,
32    })
33}
34
35/// Synthetic fallback when no I2C slave is present.
36///
37/// Walks a plausible range so the demo shows animated output rather
38/// than a static value.  `tick` increments once per poll cycle.
39pub fn synthetic(tick: u32) -> Reading {
40    Reading {
41        temp_cx10: 240 + (tick % 50) as i32, // 24.0 → 28.9 °C
42        hum_px10: 590 + (tick % 50),         // 59.0 → 63.9 %
43        simulated: true,
44    }
45}
