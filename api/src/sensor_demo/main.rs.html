24#[no_mangle]
25pub fn main() {
26    println("[sensor-demo] SHT3x via bit-bang I2C (pin 0=SCL, pin 1=SDA, addr 0x44)");
27
28    match Pl061Gpio::open() {
29        Ok(gpio) => run_with_gpio(gpio),
30        Err(_) => {
31            println("[sensor-demo] GPIO unavailable — synthetic-only mode");
32            run_synthetic();
33        }
34    }
35}
36
37// Bounded so GPIO is released for other Driver Cells (e.g. pwm-demo, spi-demo).
38const DEMO_CYCLES: u32 = 3;
39
40fn run_with_gpio(gpio: Pl061Gpio) {
41    let mut i2c = BitBangI2c::new(gpio);
42    for tick in 0..DEMO_CYCLES {
43        let r = poll_sensor(&mut i2c, tick);
44        print_reading(&r);
45        sleep_1s();
46    }
47}
48
49fn poll_sensor(i2c: &mut impl ViI2c<Error = hal_i2c::I2cError>, tick: u32) -> sht3x::Reading {
50    // SHT3x high-precision single-shot: write [0x2C, 0x06], read 6 bytes.
51    let mut buf = [0u8; 6];
52    match i2c.write_read(0x44, &[0x2C, 0x06], &mut buf) {
53        Ok(()) => sht3x::parse(&buf).unwrap_or_else(|| sht3x::synthetic(tick)),
54        // NackAddress: no slave on bus — expected in QEMU without a real sensor.
55        Err(_) => sht3x::synthetic(tick),
56    }
57}
58
59fn print_reading(r: &sht3x::Reading) {
60    let label = if r.simulated { " [sim]" } else { "" };
61    let t_int = r.temp_cx10 / 10;
62    let t_frac = (r.temp_cx10 % 10).abs();
63    // When temp is between -0.9 and -0.1°C, t_int == 0 but the value is still negative.
64    let t_sign = if r.temp_cx10 < 0 && t_int == 0 {
65        "-"
66    } else {
67        ""
68    };
69    let h_int = r.hum_px10 / 10;
70    let h_frac = r.hum_px10 % 10;
71    println(
72        alloc::format!(
73            "T={}{}.{}C H={}.{}%{}",
74            t_sign,
75            t_int,
76            t_frac,
77            h_int,
78            h_frac,
79            label
80        )
81        .as_str(),
82    );
83}
84
85fn run_synthetic() {
86    for tick in 0..DEMO_CYCLES {
87        print_reading(&sht3x::synthetic(tick));
88        sleep_1s();
89    }
90}

92/// Block for approximately 1 s.
93///
94/// Uses `RecvTimeout` as a sleep primitive: 100 scheduler ticks × 10 ms/tick.
95/// A stray message wakes us early — fine for a polling demo, we just loop again.
96fn sleep_1s() {
97    let mut buf = [0u8; 64];
98    let _ = sys_recv_timeout(0, &mut buf, 100);
99}
