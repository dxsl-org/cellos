15/// Slide direction used internally to know which prev/current offset to apply
16/// once canvas transform support is added.
17#[derive(Clone, Copy, PartialEq)]
18pub enum SlideDir {
19    Forward,
20    Backward,
21}
22
23/// Stack navigator — displays one page at a time.
24///
25/// Call `push()` / `pop()` to navigate; the navigator swaps the active page.
26/// Slide animation is scaffolded (`slide_t`, `tick_animation`) but painting
27/// falls back to the current page only until canvas transform support lands.
28///
29/// Wrap `StackNavigator` as the root widget of your `ViApp`. Call `push()` and
30/// `pop()` directly on the navigator (the inner `Router` is also pub for key
31/// inspection via `router.current_key()`).
32pub struct StackNavigator<K: Clone + PartialEq + 'static> {
33    pub router: Router<K>,
34    current: Box<dyn ViNode>,
35    prev: Option<Box<dyn ViNode>>,
36    /// Animation progress: 0.0 = transition start, 1.0 = complete.
37    slide_t: f32,
38    slide_dir: SlideDir,
39    animating: bool,
40    bounds: Cell<Rect>,
41}
42
43impl<K: Clone + PartialEq + 'static> StackNavigator<K> {
44    /// Create a navigator with `initial_widget` displayed first.
45    pub fn new(router: Router<K>, initial_widget: Box<dyn ViNode>) -> Self {
46        Self {
47            router,
48            current: initial_widget,
49            prev: None,
50            slide_t: 1.0,
51            slide_dir: SlideDir::Forward,
52            animating: false,
53            bounds: Cell::new(Rect::ZERO),
54        }
55    }
56
57    /// Navigate forward (push). Returns `false` if the key is not registered.
58    pub fn push(&mut self, key: K) -> bool {
59        if let Some(new_widget) = self.router.push(key) {
60            let old = core::mem::replace(&mut self.current, new_widget);
61            self.prev = Some(old);
62            self.slide_t = 0.0;
63            self.slide_dir = SlideDir::Forward;
64            self.animating = true;
65            true
66        } else {
67            false
68        }
69    }
70
71    /// Navigate back (pop). Returns `false` if already at root.
72    pub fn pop(&mut self) -> bool {
73        if let Some((_, new_widget)) = self.router.pop() {
74            let old = core::mem::replace(&mut self.current, new_widget);
75            self.prev = Some(old);
76            self.slide_t = 0.0;
77            self.slide_dir = SlideDir::Backward;
78            self.animating = true;
79            true
80        } else {
81            false
82        }
83    }
84
85    pub fn can_pop(&self) -> bool {
86        self.router.can_pop()
87    }
88
89    /// Advance slide animation by `dt_ms` milliseconds (200 ms total transition).
90    ///
91    /// Returns `true` while animating; returns `false` once complete and the
92    /// previous page has been dropped. Call from the app's animation loop.
93    pub fn tick_animation(&mut self, dt_ms: u32) -> bool {
94        if !self.animating {
95            return false;
96        }
97        self.slide_t += dt_ms as f32 / 200.0;
98        if self.slide_t >= 1.0 {
99            self.slide_t = 1.0;
100            self.animating = false;
101            self.prev = None;
102        }
103        self.animating
104    }
105}
106
107impl<K: Clone + PartialEq + 'static> ViNode for StackNavigator<K> {
108    fn layout(&mut self, constraints: Constraints) -> Size {
109        let size = self.current.layout(constraints);
110        if let Some(prev) = &mut self.prev {
111            prev.layout(constraints);
112        }
113        self.bounds
114            .set(Rect::from_origin_size(constraints.origin, size));
115        size
116    }
117
118    fn bounds(&self) -> Rect {
119        self.bounds.get()
120    }
121
122    fn paint(&self, cx: &mut RenderCtx<'_>) {
123        // Canvas has no transform support — paint current page only.
124        // Slide animation visuals are deferred until push_transform lands on ViCanvas.
125        self.current.paint(cx);
126    }
127
128    fn event(&mut self, event: &Event) -> bool {
129        // Block input during transitions to avoid double-firing on both pages.
130        if self.animating {
131            return true;
132        }
133        self.current.event(event)
134    }
135
136    fn collect_dirty_handles(&mut self, region: DirtyRegion) -> Vec<SubscriptionHandle> {
137        let mut handles = self.current.collect_dirty_handles(Rc::clone(&region));
138        if let Some(prev) = &mut self.prev {
139            handles.extend(prev.collect_dirty_handles(Rc::clone(&region)));
140        }
141        handles
142    }
143
144    fn is_focusable(&self) -> bool {
145        false
146    }
147
148    fn collect_focusable_bounds(&mut self) -> Vec<Rect> {
149        if self.animating {
150            return Vec::new();
151        }
152        self.current.collect_focusable_bounds()
153    }
154
155    fn activate_at(&mut self, target: Rect) -> bool {
156        self.current.activate_at(target)
157    }
158}
