2extern crate alloc;
3use alloc::{boxed::Box, vec::Vec};
4
5use crate::node::ViNode;
6
7/// Generic router mapping page keys to builder functions.
8///
9/// `K` must be `Clone + PartialEq + 'static` (e.g., `&'static str` or a custom enum).
10/// Pages are built lazily — each `push` / `pop` / `replace` invokes the builder fresh.
11/// Lazy page constructor registered per route key.
12type PageBuilder = Box<dyn Fn() -> Box<dyn ViNode>>;
13
14pub struct Router<K: Clone + PartialEq + 'static> {
15    routes: Vec<(K, PageBuilder)>,
16    history: Vec<K>,
17}
18
19impl<K: Clone + PartialEq + 'static> Router<K> {
20    pub fn new(initial: K) -> Self {
21        Self {
22            routes: Vec::new(),
23            history: alloc::vec![initial],
24        }
25    }
26
27    /// Register a page builder for a key.
28    pub fn register(&mut self, key: K, builder: impl Fn() -> Box<dyn ViNode> + 'static) {
29        self.routes.push((key, Box::new(builder)));
30    }
31
32    /// Push a new page key onto the history stack.
33    ///
34    /// Returns the built widget, or `None` if the key is not registered.
35    pub fn push(&mut self, key: K) -> Option<Box<dyn ViNode>> {
36        let widget = self.build(&key)?;
37        self.history.push(key);
38        Some(widget)
39    }
40
41    /// Pop the current page.
42    ///
43    /// Returns `(new_current_key, widget)`, or `None` if already at root or the
44    /// target key has no registered builder.
45    ///
46    /// # Invariant
47    /// History is only mutated when the build succeeds, so a missing-builder
48    /// failure leaves history unchanged (no silent corruption).
49    pub fn pop(&mut self) -> Option<(K, Box<dyn ViNode>)> {
50        if self.history.len() <= 1 {
51            return None;
52        }
53        // Peek at the destination key before touching history.
54        let key = self.history[self.history.len() - 2].clone();
55        let widget = self.build(&key)?;
56        // Build succeeded — now commit the history change.
57        self.history.pop();
58        Some((key, widget))
59    }
60
61    /// Replace the current page without touching history.
62    ///
63    /// Returns the built widget, or `None` if the key is not registered.
64    pub fn replace(&mut self, key: K) -> Option<Box<dyn ViNode>> {
65        let widget = self.build(&key)?;
66        if let Some(last) = self.history.last_mut() {
67            *last = key;
68        }
69        Some(widget)
70    }
71
72    pub fn can_pop(&self) -> bool {
73        self.history.len() > 1
74    }
75
76    pub fn current_key(&self) -> &K {
77        self.history.last().unwrap()
78    }
79
80    fn build(&self, key: &K) -> Option<Box<dyn ViNode>> {
81        self.routes
82            .iter()
83            .find(|(k, _)| k == key)
84            .map(|(_, builder)| builder())
85    }
86}
