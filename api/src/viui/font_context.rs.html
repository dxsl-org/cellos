2//! `FontContext` — optional scalable font state for ViUI v2.
3//!
4//! Wraps an optional `GlyphAtlas` (fontdue-backed) with a default text size.
5//! When no font is loaded, all text falls back to the 8×8 bitmap font in
6//! `FramebufferCanvas::draw_text()`.
7//!
8//! # Usage
9//!
10//! ```rust,ignore
11//! // No font — bitmap fallback everywhere
12//! let ctx = FontContext::no_font();
13//!
14//! // Load a TTF font at default size 16px
15//! let ctx = FontContext::with_font(include_bytes!("../assets/font.ttf"), 16.0)
16//!               .expect("invalid font bytes");
17//! ```
18
19use ostd::font_atlas::GlyphAtlas;
20
21/// Scalable font state shared across a full frame's paint pass.
22///
23/// Passed through `RenderCtx` to every `ViNode::paint()` call.
24/// Widgets that do not need scalable text ignore the `atlas` field entirely.
25pub struct FontContext {
26    /// Cached glyph rasterizer. `None` = bitmap 8×8 fallback.
27    pub atlas: Option<GlyphAtlas>,
28    /// Default font size in pixels used by text widgets.
29    pub size_px: f32,
30}
31
32impl FontContext {
33    /// No scalable font — all text widgets use the 8×8 bitmap fallback.
34    pub fn no_font() -> Self {
35        Self {
36            atlas: None,
37            size_px: 16.0,
38        }
39    }
40
41    /// Load a TrueType / OpenType font from raw bytes.
42    ///
43    /// Returns `None` when the font data is invalid.
44    pub fn with_font(font_bytes: &[u8], size_px: f32) -> Option<Self> {
45        GlyphAtlas::new(font_bytes).map(|atlas| Self {
46            atlas: Some(atlas),
47            size_px,
48        })
49    }
50
51    /// Convenience: `with_font()` with a fallback to `no_font()`.
52    ///
53    /// Use when the font is optional — degraded but non-panicking.
54    pub fn try_font(font_bytes: &[u8], size_px: f32) -> Self {
55        Self::with_font(font_bytes, size_px).unwrap_or_else(FontContext::no_font)
56    }
57
58    /// True when a scalable atlas is loaded.
59    pub fn has_font(&self) -> bool {
60        self.atlas.is_some()
61    }
62}
