2//! ScrollArea — vertically scrollable container wrapping any `ViNode` child.
3//!
4//! # Scroll implementation notes
5//!
6//! `ViCanvas` has no `translate()` method (G1 constraint). Scroll is implemented
7//! by shifting the child's layout origin by `-scroll_y` during the layout pass so
8//! the child's cached bounds are already in the correct (scrolled) screen position.
9//! `clip_push` hides out-of-viewport content.
10//!
11//! This means `layout()` must be re-run whenever `scroll_y` changes. `ViApp`
12//! already re-lays out on every frame when dirty — so a scroll event marks the
13//! region dirty and the subsequent layout + paint shows the shifted content.
14//!
15//! # Scrollbar
16//!
17//! A 6 px-wide thumb scrollbar is drawn on the right edge when content overflows.
18
19extern crate alloc;
20use alloc::{boxed::Box, vec::Vec};
21use core::cell::Cell;
22
23use crate::canvas::Color;
24use crate::dirty::DirtyRegion;
25use crate::event::Event;
26use crate::layout::{Constraints, Point, Rect, Size};
27use crate::node::ViNode;
28use crate::render_ctx::RenderCtx;
29use crate::signal::SubscriptionHandle;
30
31const SCROLL_SPEED: f32 = 30.0;
32const SCROLLBAR_W: f32 = 6.0;
33const THUMB_MIN_H: f32 = 24.0;
34
35pub struct ScrollArea {
36    child: Box<dyn ViNode>,
37    /// Current vertical scroll offset in pixels (top of visible window).
38    scroll_y: Cell<f32>,
39    /// Full unclipped height of the child content.
40    content_height: Cell<f32>,
41    bounds_cache: Cell<Rect>,
42}
43
44impl ScrollArea {
45    pub fn new(child: Box<dyn ViNode>) -> Self {
46        Self {
47            child,
48            scroll_y: Cell::new(0.0),
49            content_height: Cell::new(0.0),
50            bounds_cache: Cell::new(Rect::ZERO),
51        }
52    }
53
54    /// Maximum scroll offset — zero when content fits in the viewport.
55    fn max_scroll(&self) -> f32 {
56        (self.content_height.get() - self.bounds_cache.get().h).max(0.0)
57    }
58}
59
60impl ViNode for ScrollArea {
61    fn layout(&mut self, constraints: Constraints) -> Size {
62        let size = constraints.constrain(Size {
63            w: constraints.max.w,
64            h: constraints.max.h,
65        });
66        self.bounds_cache
67            .set(Rect::from_origin_size(constraints.origin, size));
68
69        let scroll = self.scroll_y.get();
70
71        // Child gets full width minus scrollbar, unbounded height.
72        // Origin is shifted up by scroll_y so child bounds are already scrolled.
73        let child_constraints = Constraints {
74            origin: Point::new(constraints.origin.x, constraints.origin.y - scroll),
75            min: Size { w: 0.0, h: 0.0 },
76            max: Size {
77                w: (size.w - SCROLLBAR_W).max(0.0),
78                // Unlimited height — measure the full content.
79                h: f32::MAX,
80            },
81        };
82        let child_size = self.child.layout(child_constraints);
83        self.content_height.set(child_size.h);
84
85        size
86    }
87
88    fn bounds(&self) -> Rect {
89        self.bounds_cache.get()
90    }
91
92    fn paint(&self, cx: &mut RenderCtx<'_>) {
93        let b = self.bounds_cache.get();
94        let scroll = self.scroll_y.get();
95
96        // Clip to viewport — hides child content that scrolled out of view
97        cx.canvas.clip_push(b);
98
99        // Child's bounds are already shifted by -scroll_y (set during layout).
100        // Paint proceeds normally; clipping hides anything outside the viewport.
101        self.child.paint(cx);
102
103        cx.canvas.clip_pop();
104
105        // Scrollbar — only rendered when content overflows
106        let content_h = self.content_height.get();
107        if content_h > b.h {
108            let track = Rect {
109                x: b.x + b.w - SCROLLBAR_W,
110                y: b.y,
111                w: SCROLLBAR_W,
112                h: b.h,
113            };
114            cx.canvas.fill_rect(track, Color::rgb(25, 25, 38));
115
116            let max_scroll = (content_h - b.h).max(1.0);
117            let thumb_ratio = b.h / content_h;
118            let thumb_h = (thumb_ratio * b.h).max(THUMB_MIN_H).min(b.h);
119            let thumb_top = b.y + (scroll / max_scroll) * (b.h - thumb_h);
120            let thumb = Rect {
121                x: track.x + 1.0,
122                y: thumb_top.clamp(b.y, b.y + b.h - thumb_h),
123                w: SCROLLBAR_W - 2.0,
124                h: thumb_h,
125            };
126            cx.canvas.fill_rect(thumb, Color::rgb(80, 80, 110));
127        }
128    }
129
130    fn event(&mut self, event: &Event) -> bool {
131        let b = self.bounds_cache.get();
132
133        match event {
134            Event::Scroll { pos, delta_y } if b.contains(*pos) => {
135                // delta_y: negative = scroll down (conventional mouse wheel direction)
136                let new_y =
137                    (self.scroll_y.get() - delta_y * SCROLL_SPEED).clamp(0.0, self.max_scroll());
138                self.scroll_y.set(new_y);
139                true
140            }
141            _ => self.child.event(event),
142        }
143    }
144
145    fn collect_dirty_handles(&mut self, region: DirtyRegion) -> Vec<SubscriptionHandle> {
146        self.child.collect_dirty_handles(region)
147    }
148}
