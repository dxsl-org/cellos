2//! Card — panel container with a background fill, border, and inner padding.
3
4extern crate alloc;
5use alloc::{boxed::Box, vec::Vec};
6use core::cell::Cell;
7
8use crate::canvas::Color;
9use crate::dirty::DirtyRegion;
10use crate::event::Event;
11use crate::layout::{Constraints, Point, Rect, Size};
12use crate::node::ViNode;
13use crate::render_ctx::RenderCtx;
14use crate::signal::SubscriptionHandle;
15
16/// Container that draws a tinted panel behind its child widget.
17///
18/// `padding` shrinks the child's allocated area on all sides.
19/// The child's bounds are set relative to the card's padded interior.
20pub struct Card {
21    child: Box<dyn ViNode>,
22    padding: f32,
23    bg_color: Color,
24    border_color: Color,
25    bounds_cache: Cell<Rect>,
26}
27
28impl Card {
29    /// Wrap `child` with default padding (8 px), dark background, and subtle border.
30    pub fn new(child: Box<dyn ViNode>) -> Self {
31        Self {
32            child,
33            padding: 8.0,
34            bg_color: Color::rgb(30, 30, 45),
35            border_color: Color::rgb(60, 60, 80),
36            bounds_cache: Cell::new(Rect::ZERO),
37        }
38    }
39
40    /// Uniform padding on all sides.
41    pub fn padding(mut self, p: f32) -> Self {
42        self.padding = p;
43        self
44    }
45
46    /// Background fill color.
47    pub fn bg(mut self, c: Color) -> Self {
48        self.bg_color = c;
49        self
50    }
51
52    /// Border color.
53    pub fn border(mut self, c: Color) -> Self {
54        self.border_color = c;
55        self
56    }
57}
58
59impl ViNode for Card {
60    fn layout(&mut self, constraints: Constraints) -> Size {
61        let pad = self.padding;
62
63        // Give child the interior area — origin shifted in by padding.
64        let inner = Constraints {
65            origin: Point::new(constraints.origin.x + pad, constraints.origin.y + pad),
66            min: Size { w: 0.0, h: 0.0 },
67            max: Size {
68                w: (constraints.max.w - pad * 2.0).max(0.0),
69                h: (constraints.max.h - pad * 2.0).max(0.0),
70            },
71        };
72        let inner_size = self.child.layout(inner);
73
74        let size = constraints.constrain(Size {
75            w: inner_size.w + pad * 2.0,
76            h: inner_size.h + pad * 2.0,
77        });
78        self.bounds_cache
79            .set(Rect::from_origin_size(constraints.origin, size));
80        size
81    }
82
83    fn bounds(&self) -> Rect {
84        self.bounds_cache.get()
85    }
86
87    fn paint(&self, cx: &mut RenderCtx<'_>) {
88        let b = self.bounds_cache.get();
89
90        // Background fill
91        cx.canvas.fill_rect(b, self.bg_color);
92
93        // 1 px border — 4 edge lines
94        let x0 = b.x;
95        let y0 = b.y;
96        let x1 = b.x + b.w;
97        let y1 = b.y + b.h;
98        cx.canvas
99            .draw_line(Point::new(x0, y0), Point::new(x1, y0), self.border_color); // top
100        cx.canvas
101            .draw_line(Point::new(x1, y0), Point::new(x1, y1), self.border_color); // right
102        cx.canvas
103            .draw_line(Point::new(x1, y1), Point::new(x0, y1), self.border_color); // bottom
104        cx.canvas
105            .draw_line(Point::new(x0, y1), Point::new(x0, y0), self.border_color); // left
106
107        // Child paints inside the padded interior
108        self.child.paint(cx);
109    }
110
111    fn event(&mut self, event: &Event) -> bool {
112        self.child.event(event)
113    }
114
115    fn collect_dirty_handles(&mut self, region: DirtyRegion) -> Vec<SubscriptionHandle> {
116        self.child.collect_dirty_handles(region)
117    }
118}
