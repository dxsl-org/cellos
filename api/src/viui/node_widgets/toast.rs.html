2//! Toast notification widget — transient, auto-dismiss, non-blocking overlay.
3//!
4//! Toasts are positioned at the bottom-center of the screen. Auto-dismiss is
5//! driven by `ToastEntry::elapsed_ms`, which is incremented by `ViApp::tick_with_dt`.
6//! Set `duration_ms = 0` for a manual-dismiss-only toast.
7
8extern crate alloc;
9use alloc::string::String;
10use core::cell::Cell;
11
12use crate::canvas::Color;
13use crate::dirty::DirtyRegion;
14use crate::event::Event;
15use crate::layout::{Constraints, Point, Rect, Size};
16use crate::node::ViNode;
17use crate::render_ctx::RenderCtx;
18use crate::signal::SubscriptionHandle;

20// ─── ToastKind ───────────────────────────────────────────────────────────────
21
22/// Semantic category of a toast — controls background colour.
23#[derive(Clone)]
24pub enum ToastKind {
25    Info,
26    Success,
27    Warning,
28    Error,
29}
30
31// ─── ToastConfig ─────────────────────────────────────────────────────────────
32
33/// Parameters for a transient toast notification.
34#[derive(Clone)]
35pub struct ToastConfig {
36    pub message: String,
37    pub kind: ToastKind,
38    /// Auto-dismiss after this many milliseconds. `0` = manual dismiss only.
39    pub duration_ms: u32,
40}
41
42// ─── ToastEntry ──────────────────────────────────────────────────────────────
43
44/// Live toast tracked by `ViApp` — wraps the config with timer state and a
45/// pre-built widget so `paint()` does not need to allocate each frame.
46pub struct ToastEntry {
47    pub config: ToastConfig,
48    pub elapsed_ms: u32,
49    /// Pre-built widget. Laid out every frame by `ViApp` in Path A.
50    pub widget: Toast,
51}
52
53// ─── Toast ───────────────────────────────────────────────────────────────────
54
55/// Rendered toast overlay widget.
56///
57/// Positioned at the bottom-center of the screen constraints. The widget is
58/// non-interactive: all events pass through.
59pub struct Toast {
60    message: String,
61    kind: ToastKind,
62    bounds: Cell<Rect>,
63}
64
65impl Toast {
66    /// Construct from a `ToastConfig`.
67    pub fn new(config: &ToastConfig) -> Self {
68        Self {
69            message: config.message.clone(),
70            kind: config.kind.clone(),
71            bounds: Cell::new(Rect::ZERO),
72        }
73    }
74
75    fn bg_color(&self) -> Color {
76        match self.kind {
77            ToastKind::Info => Color::rgb(40, 80, 160),
78            ToastKind::Success => Color::rgb(30, 120, 60),
79            ToastKind::Warning => Color::rgb(160, 120, 20),
80            ToastKind::Error => Color::rgb(160, 40, 40),
81        }
82    }
83}
84
85impl ViNode for Toast {
86    fn layout(&mut self, constraints: Constraints) -> Size {
87        let w = constraints.max.w.min(400.0);
88        let h = 44.0f32;
89        // Centre horizontally; sit 16 px above the bottom edge.
90        let x = constraints.origin.x + (constraints.max.w - w) / 2.0;
91        let y = constraints.origin.y + constraints.max.h - h - 16.0;
92        self.bounds.set(Rect { x, y, w, h });
93        Size::new(w, h)
94    }
95
96    fn bounds(&self) -> Rect {
97        self.bounds.get()
98    }
99
100    fn paint(&self, cx: &mut RenderCtx<'_>) {
101        let b = self.bounds.get();
102        cx.canvas.fill_rect(b, self.bg_color());
103        // 12 px left pad, vertical centre of 44 px = ~14 px from top (8 px glyph).
104        cx.draw_text(
105            Point::new(b.x + 12.0, b.y + 14.0),
106            &self.message,
107            Color::WHITE,
108        );
109    }
110
111    /// Toasts are non-interactive — never consume events.
112    fn event(&mut self, _event: &Event) -> bool {
113        false
114    }
115
116    fn collect_dirty_handles(
117        &mut self,
118        _region: DirtyRegion,
119    ) -> alloc::vec::Vec<SubscriptionHandle> {
120        alloc::vec::Vec::new()
121    }
122}
