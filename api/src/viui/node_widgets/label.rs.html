2//! Label v2 — text widget driven by a `Signal<String>`.
3
4use crate::canvas::Color;
5use crate::dirty::DirtyRegion;
6use crate::event::Event;
7use crate::layout::{Constraints, Rect, Size};
8use crate::node::ViNode;
9use crate::render_ctx::RenderCtx;
10use crate::signal::{Signal, SubscriptionHandle};
11
12extern crate alloc;
13use alloc::{string::String, vec::Vec};
14
15/// Text display widget.
16///
17/// `text` is a `Signal<String>` — call `signal.set()` from anywhere to update
18/// the displayed text. The app runner will repaint on the next tick.
19pub struct Label {
20    pub text: Signal<String>,
21    pub color: Color,
22    bounds: Rect,
23    cached_byte_len: usize,
24    cached_char_count: usize,
25}
26
27impl Label {
28    pub fn new(text: Signal<String>) -> Self {
29        Self {
30            text,
31            color: Color::WHITE,
32            bounds: Rect::ZERO,
33            cached_byte_len: 0,
34            cached_char_count: 0,
35        }
36    }
37
38    pub fn with_color(mut self, color: Color) -> Self {
39        self.color = color;
40        self
41    }
42}
43
44impl ViNode for Label {
45    fn layout(&mut self, constraints: Constraints) -> Size {
46        let byte_len = self.text.get().len();
47        if byte_len != self.cached_byte_len {
48            self.cached_char_count = self.text.get().chars().count();
49            self.cached_byte_len = byte_len;
50        }
51        // Use 8.0px char width as conservative fallback; RenderCtx provides
52        // better metrics at paint time but layout doesn't have cx access.
53        let desired = Size {
54            w: self.cached_char_count as f32 * 8.0,
55            h: 16.0, // accommodate scalable font (no longer hard-coded 8px)
56        };
57        let size = constraints.constrain(desired);
58        self.bounds = Rect::from_origin_size(constraints.origin, size);
59        size
60    }
61
62    fn bounds(&self) -> Rect {
63        self.bounds
64    }
65
66    fn paint(&self, cx: &mut RenderCtx<'_>) {
67        let pos = crate::layout::Point::new(self.bounds.x, self.bounds.y);
68        cx.draw_text(pos, &self.text.get(), self.color);
69    }
70
71    fn event(&mut self, _event: &Event) -> bool {
72        false
73    }
74
75    fn collect_dirty_handles(&mut self, region: DirtyRegion) -> Vec<SubscriptionHandle> {
76        let rect = self.bounds;
77        let h = self.text.subscribe(move || {
78            region.borrow_mut().mark(rect);
79        });
80        alloc::vec![h]
81    }
82}
