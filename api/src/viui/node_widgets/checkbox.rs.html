2//! CheckBox — toggleable boolean widget with optional text label.
3
4extern crate alloc;
5use alloc::{boxed::Box, string::String, vec::Vec};
6use core::cell::Cell;
7
8use crate::canvas::Color;
9use crate::dirty::DirtyRegion;
10use crate::event::{Event, MouseButton};
11use crate::layout::{Constraints, Point, Rect, Size};
12use crate::node::ViNode;
13use crate::render_ctx::RenderCtx;
14use crate::signal::{Signal, SubscriptionHandle};
15
16/// Total widget height in pixels.
17const HEIGHT: f32 = 24.0;
18/// Side length of the toggle box.
19const BOX_SIZE: f32 = 16.0;
20/// Horizontal gap between box right edge and label text.
21const LABEL_GAP: f32 = 6.0;
22
23/// Boolean toggle with a checkmark glyph and an optional text label.
24///
25/// Clicking anywhere in the widget bounds inverts `checked`.
26/// `on_toggle` fires after the state change with the new value.
27pub struct CheckBox {
28    /// Current checked state.
29    pub checked: Signal<bool>,
30    /// Label shown to the right of the box. Empty string hides the label.
31    pub label: Signal<String>,
32    on_toggle: Option<Box<dyn Fn(bool)>>,
33    bounds_cache: Cell<Rect>,
34}
35
36impl CheckBox {
37    pub fn new(checked: Signal<bool>) -> Self {
38        Self {
39            checked,
40            label: Signal::new(String::new()),
41            on_toggle: None,
42            bounds_cache: Cell::new(Rect::ZERO),
43        }
44    }
45
46    /// Attach a reactive label signal.
47    pub fn label(mut self, s: Signal<String>) -> Self {
48        self.label = s;
49        self
50    }
51
52    /// Attach a static label string.
53    pub fn label_str(self, s: impl Into<String>) -> Self {
54        self.label(Signal::new(s.into()))
55    }
56
57    /// Callback fired with the new `bool` value after every toggle.
58    pub fn on_toggle(mut self, f: impl Fn(bool) + 'static) -> Self {
59        self.on_toggle = Some(Box::new(f));
60        self
61    }
62}
63
64impl ViNode for CheckBox {
65    fn layout(&mut self, constraints: Constraints) -> Size {
66        let size = constraints.constrain(Size {
67            w: constraints.max.w,
68            h: HEIGHT,
69        });
70        self.bounds_cache
71            .set(Rect::from_origin_size(constraints.origin, size));
72        size
73    }
74
75    fn bounds(&self) -> Rect {
76        self.bounds_cache.get()
77    }
78
79    fn paint(&self, cx: &mut RenderCtx<'_>) {
80        let b = self.bounds_cache.get();
81
82        // Vertically center the box within the widget height.
83        let box_y = b.y + (b.h - BOX_SIZE) * 0.5;
84        let box_rect = Rect {
85            x: b.x + 2.0,
86            y: box_y,
87            w: BOX_SIZE,
88            h: BOX_SIZE,
89        };
90
91        // Box fill
92        cx.canvas.fill_rect(box_rect, Color::rgb(30, 30, 45));
93
94        // Box border — 4 edge lines
95        let bc = Color::rgb(100, 100, 130);
96        let x0 = box_rect.x;
97        let y0 = box_rect.y;
98        let x1 = x0 + box_rect.w;
99        let y1 = y0 + box_rect.h;
100        cx.canvas
101            .draw_line(Point::new(x0, y0), Point::new(x1, y0), bc); // top
102        cx.canvas
103            .draw_line(Point::new(x1, y0), Point::new(x1, y1), bc); // right
104        cx.canvas
105            .draw_line(Point::new(x1, y1), Point::new(x0, y1), bc); // bottom
106        cx.canvas
107            .draw_line(Point::new(x0, y1), Point::new(x0, y0), bc); // left
108
109        // Checkmark: two-segment ✓ path when checked
110        if *self.checked.get() {
111            let accent = Color::rgb(80, 140, 220);
112            // Left segment: bottom-left corner towards center-bottom
113            let start = Point::new(x0 + 3.0, y0 + BOX_SIZE * 0.55);
114            let mid = Point::new(x0 + BOX_SIZE * 0.35, y1 - 3.0);
115            // Right segment: center-bottom up to top-right
116            let end = Point::new(x1 - 3.0, y0 + 4.0);
117            cx.canvas.draw_line(start, mid, accent);
118            cx.canvas.draw_line(mid, end, accent);
119        }
120
121        // Label text (if non-empty)
122        let label = self.label.get();
123        if !label.is_empty() {
124            let lx = b.x + 2.0 + BOX_SIZE + LABEL_GAP;
125            let ly = b.y + (b.h - cx.line_height()) * 0.5;
126            cx.draw_text(
127                Point::new(lx, ly.max(b.y)),
128                &label,
129                Color::rgb(220, 220, 230),
130            );
131        }
132    }
133
134    fn is_focusable(&self) -> bool {
135        true
136    }
137
138    fn activate(&mut self) -> bool {
139        let new_val = !*self.checked.get();
140        self.checked.set(new_val);
141        if let Some(cb) = &self.on_toggle {
142            cb(new_val);
143        }
144        true
145    }
146
147    fn event(&mut self, event: &Event) -> bool {
148        let b = self.bounds_cache.get();
149
150        let toggled = match event {
151            Event::MousePress {
152                pos,
153                button: MouseButton::Left,
154            } if b.contains(*pos) => true,
155            Event::TouchBegin { pos, .. } if b.contains(*pos) => true,
156            _ => false,
157        };
158
159        if toggled {
160            let new_val = !*self.checked.get();
161            self.checked.set(new_val);
162            if let Some(cb) = &self.on_toggle {
163                cb(new_val);
164            }
165        }
166
167        toggled
168    }
169
170    fn collect_dirty_handles(&mut self, region: DirtyRegion) -> Vec<SubscriptionHandle> {
171        let bounds = self.bounds_cache.get();
172        let region2 = region.clone();
173        let h1 = self.checked.subscribe(move || {
174            region.borrow_mut().mark(bounds);
175        });
176        let h2 = self.label.subscribe(move || {
177            region2.borrow_mut().mark(bounds);
178        });
179        alloc::vec![h1, h2]
180    }
181}
