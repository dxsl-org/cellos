2//! Minimal tick-based app runner for ViUI v2.
3//!
4//! # Repaint strategy
5//!
6//! Two independent dirty flags:
7//! - `layout_dirty`: set by consumed events or `mark_dirty()`. Triggers full
8//!   layout + Signal re-subscribe (updates captured bounds) + full repaint.
9//! - `dirty_region`: set by Signal subscriptions. Triggers partial repaint with
10//!   the cached layout — no layout pass, O(m) paint where m = dirty widgets.
11//!
12//! Path A (structural): event consumed → layout → re-subscribe → mark_all
13//! Path B (signal-only): signal fires → mark(widget_bounds) → partial repaint
14//!
15//! # Animation
16//!
17//! Add `Box<dyn Animatable>` via `add_animation()`. On every `tick_with_dt()`
18//! call all active animations are advanced by `dt_ms` before event processing.
19//! Animations that change a `Signal<f32>` trigger Path B automatically.
20//!
21//! # Usage
22//!
23//! ```rust,ignore
24//! let count = Signal::new(0i32);
25//! let label = Label::new(count.map(|n| format!("Count: {n}")));
26//! let cnt2  = count.clone();
27//! let btn   = Button::new("Increment", move || cnt2.update(|n| *n += 1));
28//! let root  = vstack!(label, btn);
29//!
30//! let mut app = ViApp::new(Box::new(root), Box::new(renderer));
31//! loop {
32//!     app.tick(&collect_input_events());
33//!     // or: app.tick_with_dt(&events, elapsed_ms);
34//! }
35//! ```

55/// Tracks Tab-order focus across all focusable widgets in the tree.
56///
57/// `list` is rebuilt after every layout pass (bounds are stale until then).
58/// `idx` is clamped or cleared whenever the list shrinks.
59struct FocusState {
60    /// Screen rects of all focusable widgets, in tree (tab) order.
61    list: alloc::vec::Vec<crate::layout::Rect>,
62    /// Currently focused index into `list`, or `None` when no widget is focused.
63    idx: Option<usize>,
64}
65
66impl FocusState {
67    fn new() -> Self {
68        Self {
69            list: alloc::vec::Vec::new(),
70            idx: None,
71        }
72    }
73
74    /// Advance or reverse focus by one step, wrapping around.
75    fn advance(&mut self, reverse: bool) {
76        if self.list.is_empty() {
77            self.idx = None;
78            return;
79        }
80        self.idx = Some(match self.idx {
81            None => {
82                if reverse {
83                    self.list.len() - 1
84                } else {
85                    0
86                }
87            }
88            Some(i) => {
89                if reverse {
90                    if i == 0 {
91                        self.list.len() - 1
92                    } else {
93                        i - 1
94                    }
95                } else {
96                    (i + 1) % self.list.len()
97                }
98            }
99        });
100    }
101
102    /// Returns the screen rect of the currently focused widget, if any.
103    fn current_bounds(&self) -> Option<crate::layout::Rect> {
104        self.idx.and_then(|i| self.list.get(i).copied())
105    }
106}

110/// Delay before the first software repeat fires (milliseconds).
111const KEY_REPEAT_DELAY_MS: u64 = 500;
112
113/// Interval between subsequent software repeat events (milliseconds).
114const KEY_REPEAT_INTERVAL_MS: u64 = 50;
115
116/// Tracks which key is currently held for software key-repeat injection.
117///
118/// # Hardware vs software repeat
119/// The input service already sends `state=Repeated` events for hardware
120/// auto-repeat (see `api::input::KeyState::Repeated`). `input_bridge::parse_input_message`
121/// treats those identically to `Pressed`, so hardware repeat already works without
122/// this struct. `KeyRepeatState` provides *software* repeat for platforms that do NOT
123/// forward hardware repeat (serial consoles, touch keyboards, synthetic input in tests).
124/// When hardware repeat arrives it naturally resets `held_ms` via the Pressed arm, so
125/// the two paths do not fight.
126struct KeyRepeatState {
127    /// The key currently held, or `None` when no key is pressed.
128    held_key: Option<KeyCode>,
129    /// Modifier state captured at the moment the key was pressed.
130    held_mods: Modifiers,
131    /// Accumulated milliseconds since the key was first pressed (reset on new press).
132    held_ms: u64,
133    /// Accumulated milliseconds since the last repeat event fired (reset on each fire).
134    since_last_ms: u64,
135}
136
137impl KeyRepeatState {
138    const fn new() -> Self {
139        Self {
140            held_key: None,
141            held_mods: Modifiers {
142                shift: false,
143                ctrl: false,
144                alt: false,
145            },
146            held_ms: 0,
147            since_last_ms: 0,
148        }
149    }
150}

152/// Tick-based app runner.
153///
154/// Call `tick()` or `tick_with_dt()` from the cell's main loop, passing any
155/// buffered input events and — for animation — the milliseconds elapsed since
156/// the last call.
157pub struct ViApp {
158    root: Box<dyn ViNode>,
159    renderer: Box<dyn ViRenderer>,
160    font_ctx: FontContext,
161    animations: Vec<Box<dyn Animatable>>,
162    dirty_region: DirtyRegion,
163    dirty_handles: Vec<SubscriptionHandle>,
164    layout_dirty: bool,
165    /// Active design-token set. Widgets read colors/spacing from `cx.theme`.
166    theme: Box<dyn ViTheme>,
167    /// Software key-repeat state. See `KeyRepeatState` for rationale.
168    key_repeat: KeyRepeatState,
169    /// Tab-order focus state. Rebuilt after every layout pass.
170    focus: FocusState,
171    /// Last known mouse position, updated on every MouseMove.
172    /// Injected into MousePress/MouseRelease events whose pos is (0,0) because
173    /// the input service encodes button events without position (caller must track).
174    mouse_pos: crate::layout::Point,
175    /// Stack of overlay widgets (dialogs, drop-down popups). Rendered and
176    /// dispatched on top of `root`; topmost entry wins events first.
177    overlays: Vec<OverlayEntry>,
178    /// Deferred overlay mutations queued by widgets during `event()`.
179    /// Drained once per frame after all input events are processed.
180    action_queue: OverlayActionQueue,
181    /// Active toast notifications, in chronological order.
182    toast_entries: VecDeque<crate::node_widgets::toast::ToastEntry>,
183}

185impl ViApp {
186    /// Create a new app with no scalable font (8×8 bitmap fallback).
187    ///
188    /// The first tick always renders a full frame.
189    pub fn new(root: Box<dyn ViNode>, renderer: Box<dyn ViRenderer>) -> Self {
190        let dirty_region: DirtyRegion = Rc::new(RefCell::new(DirtyRect::new()));
191        Self {
192            root,
193            renderer,
194            font_ctx: FontContext::no_font(),
195            animations: Vec::new(),
196            dirty_region,
197            dirty_handles: Vec::new(),
198            layout_dirty: true,
199            theme: Box::new(DarkTheme),
200            key_repeat: KeyRepeatState::new(),
201            focus: FocusState::new(),
202            mouse_pos: crate::layout::Point::new(0.0, 0.0),
203            overlays: Vec::new(),
204            action_queue: new_action_queue(),
205            toast_entries: VecDeque::new(),
206        }
207    }

211    /// Return a clone of the action queue.
212    ///
213    /// Pass this clone to `Dialog` / `DropDown` constructors so they can push
214    /// `OverlayAction` items without holding a borrow on `ViApp`.
215    pub fn action_queue(&self) -> OverlayActionQueue {
216        Rc::clone(&self.action_queue)
217    }

219    /// Push a widget as a blocking or non-blocking overlay.
220    ///
221    /// `blocking = true` — modal: all input is consumed by the overlay, the
222    /// root does not see events while it is on screen (e.g. Dialog).
223    /// `blocking = false` — non-modal: input passes through when not consumed
224    /// by the overlay (e.g. DropDown popup).
225    pub fn push_overlay(&mut self, widget: Box<dyn ViNode>, blocking: bool) {
226        self.overlays.push(OverlayEntry {
227            widget,
228            blocking,
229            dismiss_outside: !blocking,
230            anchor_bounds: None,
231        });
232        self.layout_dirty = true;
233    }
234
235    /// Pop the topmost overlay. No-op when the overlay stack is empty.
236    pub fn pop_overlay(&mut self) {
237        self.overlays.pop();
238        self.layout_dirty = true;
239    }

241    /// Show a transient toast notification.
242    ///
243    /// Toasts are rendered above all content and auto-dismiss after
244    /// `config.duration_ms` milliseconds (`0` = never auto-dismiss).
245    pub fn show_toast(&mut self, config: crate::node_widgets::toast::ToastConfig) {
246        let widget = crate::node_widgets::toast::Toast::new(&config);
247        self.toast_entries
248            .push_back(crate::node_widgets::toast::ToastEntry {
249                config,
250                elapsed_ms: 0,
251                widget,
252            });
253        self.layout_dirty = true;
254    }

256    /// Override the default 8×8 bitmap font with a scalable TTF font.
257    ///
258    /// `font_bytes`: raw TTF/OTF data (use `include_bytes!` for embedded fonts).
259    /// `size_px`: default text height in pixels.
260    /// Returns `self` unchanged when the font bytes are invalid.
261    pub fn with_font(mut self, font_bytes: &[u8], size_px: f32) -> Self {
262        if let Some(ctx) = FontContext::with_font(font_bytes, size_px) {
263            self.font_ctx = ctx;
264        }
265        self
266    }

273    /// Register an animation that is advanced every `tick_with_dt()` call.

275    /// Animations typically wrap an `AnimatedSignal<f32>` and drive a widget

277    pub fn add_animation(&mut self, anim: Box<dyn Animatable>) {

281    /// Process events and render if dirty. Returns `true` if a frame was rendered.
282    ///
283    /// Equivalent to `tick_with_dt(events, 0)` — animations are not advanced.
284    pub fn tick(&mut self, events: &[Event]) -> bool {
285        self.tick_with_dt(events, 0)
286    }
287
288    /// Process events, advance animations by `dt_ms` milliseconds, and render
289    /// if dirty. Returns `true` if a frame was rendered.
290    ///
291    /// `dt_ms`: milliseconds elapsed since the last call. Pass `0` to skip
292    /// animation advancement (useful for initial render or event-only ticks).
293    pub fn tick_with_dt(&mut self, events: &[Event], dt_ms: u32) -> bool {
294        // ── Advance animations ────────────────────────────────────────────────
295        // Animations update Signal<f32> values which trigger dirty-region marks
296        // via their subscriptions — no explicit dirty flag needed here.
297        if dt_ms > 0 {
298            for anim in &mut self.animations {
299                anim.tick(dt_ms);
300            }
301        }
302
303        // ── Drain action queue from previous frame ────────────────────────────
304        // Widgets may have pushed actions during the last tick's event pass.
305        // Drain before the new event pass so actions are visible immediately.
306        self.drain_action_queue();
307
308        // ── Process input events ──────────────────────────────────────────────
309        for e in events {
310            // Tab focus cycling — consumed by the focus system, not dispatched to root.
311            if let Event::KeyPress {
312                key: KeyCode::Tab,
313                modifiers,
314            } = e
315            {
316                self.focus.advance(modifiers.shift);
317                self.layout_dirty = true;
318                continue;
319            }
320
321            // Enter activation — route directly to the focused widget via activate_at().
322            // When a blocking overlay is active, activate_at on the overlay is more
323            // appropriate; skip root activation in that case.
324            if let Event::KeyPress {
325                key: KeyCode::Enter,
326                ..
327            } = e
328            {
329                let top_blocking = self.overlays.last().map(|o| o.blocking).unwrap_or(false);
330                if !top_blocking {
331                    if let Some(b) = self.focus.current_bounds() {
332                        if self.root.activate_at(b) {
333                            self.layout_dirty = true;
334                        }
335                    }
336                }
337                continue;
338            }
339
340            // Track mouse position for button-event injection.
341            // parse_mouse_button in input_bridge sets pos=(0,0) because the input
342            // service encodes button events without coordinates — we inject the last
343            // known position here so hit-testing in Button/Slider works correctly.
344            if let Event::MouseMove { pos } = e {
345                self.mouse_pos = *pos;
346            }
347
348            // Patch mouse events that arrive with pos=(0,0) — substitute the last
349            // tracked position so widget hit-testing is correct.
350            // parse_mouse_button and parse_mouse_scroll in input_bridge set pos=(0,0)
351            // because the input service encodes those events without coordinates.
352            let patched;
353            let ev: &Event = match e {
354                Event::MousePress { pos, button } if pos.x == 0.0 && pos.y == 0.0 => {
355                    patched = Event::MousePress {
356                        pos: self.mouse_pos,
357                        button: *button,
358                    };
359                    &patched
360                }
361                Event::MouseRelease { pos, button } if pos.x == 0.0 && pos.y == 0.0 => {
362                    patched = Event::MouseRelease {
363                        pos: self.mouse_pos,
364                        button: *button,
365                    };
366                    &patched
367                }
368                Event::Scroll { pos, delta_y } if pos.x == 0.0 && pos.y == 0.0 => {
369                    patched = Event::Scroll {
370                        pos: self.mouse_pos,
371                        delta_y: *delta_y,
372                    };
373                    &patched
374                }
375                other => other,
376            };
377
378            // Track held key for software repeat injection (see KeyRepeatState).
379            match ev {
380                Event::KeyPress { key, modifiers } => {
381                    self.key_repeat.held_key = Some(*key);
382                    self.key_repeat.held_mods = *modifiers;
383                    self.key_repeat.held_ms = 0;
384                    self.key_repeat.since_last_ms = 0;
385                }
386                Event::KeyRelease { .. } => {
387                    self.key_repeat.held_key = None;
388                }
389                _ => {}
390            }
391
392            // Route event: topmost overlay wins, then falls through if non-blocking.
393            // Snapshot overlay metadata before the mutable borrow for event().
394            let overlay_meta = self
395                .overlays
396                .last()
397                .map(|t| (t.blocking, t.dismiss_outside, t.widget.bounds()));
398
399            let consumed = if let Some(top) = self.overlays.last_mut() {
400                let top_blocking = top.blocking;
401                let overlay_consumed = top.widget.event(ev);
402                if top_blocking {
403                    // Blocking overlay: never dispatch to root.
404                    overlay_consumed
405                } else if !overlay_consumed {
406                    // Non-blocking overlay did not consume — dispatch to root.
407                    self.root.event(ev)
408                } else {
409                    true
410                }
411            } else {
412                self.root.event(ev)
413            };
414
415            // Enforce dismiss_outside: if the topmost non-blocking overlay has
416            // dismiss_outside=true and a press/touch landed outside its bounds,
417            // pop it now. The popup's own event() intentionally returns false for
418            // outside clicks (it cannot pop itself during the borrow), so this is
419            // the only place the dismiss is enforced.
420            if let Some((blocking, dismiss_outside, overlay_bounds)) = overlay_meta {
421                if !blocking && dismiss_outside {
422                    let outside_press = match ev {
423                        Event::MousePress { pos, .. } => !overlay_bounds.contains(*pos),
424                        Event::TouchBegin { pos, .. } => !overlay_bounds.contains(*pos),
425                        _ => false,
426                    };
427                    if outside_press {
428                        self.action_queue.borrow_mut().push(OverlayAction::Pop);
429                    }
430                }
431            }
432
433            if consumed {
434                self.layout_dirty = true;
435            }
436        }
437
438        // ── Software key-repeat injection ─────────────────────────────────────
439        // Only runs when time is advancing (dt_ms > 0) and a key is held.
440        // Platforms that send hardware repeat events (api::input::KeyState::Repeated)
441        // already inject repeats via input_bridge, so this path fires only when
442        // held_ms crosses the thresholds without hardware-repeat interruption.
443        if dt_ms > 0 {
444            if let Some(key) = self.key_repeat.held_key {
445                self.key_repeat.held_ms += dt_ms as u64;
446                self.key_repeat.since_last_ms += dt_ms as u64;
447
448                if self.key_repeat.held_ms >= KEY_REPEAT_DELAY_MS
449                    && self.key_repeat.since_last_ms >= KEY_REPEAT_INTERVAL_MS
450                {
451                    self.key_repeat.since_last_ms = 0;
452                    let repeat_event = Event::KeyPress {
453                        key,
454                        modifiers: self.key_repeat.held_mods,
455                    };
456                    let consumed = if let Some(top) = self.overlays.last_mut() {
457                        let top_blocking = top.blocking;
458                        let overlay_consumed = top.widget.event(&repeat_event);
459                        if top_blocking {
460                            overlay_consumed
461                        } else if !overlay_consumed {
462                            self.root.event(&repeat_event)
463                        } else {
464                            true
465                        }
466                    } else {
467                        self.root.event(&repeat_event)
468                    };
469                    if consumed {
470                        self.layout_dirty = true;
471                    }
472                }
473            }
474        }
475
476        // ── Drain action queue (widgets may have queued actions this frame) ────
477        self.drain_action_queue();
478
479        // ── Advance toast timers ──────────────────────────────────────────────
480        if dt_ms > 0 && !self.toast_entries.is_empty() {
481            let prev_len = self.toast_entries.len();
482            for entry in &mut self.toast_entries {
483                entry.elapsed_ms = entry.elapsed_ms.saturating_add(dt_ms);
484            }
485            self.toast_entries.retain(|entry| {
486                entry.config.duration_ms == 0 || entry.elapsed_ms < entry.config.duration_ms
487            });
488            if self.toast_entries.len() != prev_len {
489                self.layout_dirty = true;
490            }
491        }
492
493        // ── Path A: structural change → full layout + re-subscribe ────────────
494        if self.layout_dirty {
495            self.layout_dirty = false;
496            let (w, h) = self.renderer.size();
497            let screen = Constraints::root(Size::new(w as f32, h as f32));
498
499            // Layout root widget.
500            self.root.layout(screen);
501
502            // Layout all overlay widgets (they need fresh screen constraints).
503            for entry in &mut self.overlays {
504                entry.widget.layout(screen);
505            }
506
507            // Layout toast widgets.
508            for entry in &mut self.toast_entries {
509                entry.widget.layout(screen);
510            }
511
512            // Rebuild focus list.  When a blocking overlay is active, focus is
513            // confined to that overlay's focusable bounds.
514            if self.overlays.last().map(|e| e.blocking).unwrap_or(false) {
515                self.focus.list = self
516                    .overlays
517                    .last_mut()
518                    .map(|e| e.widget.collect_focusable_bounds())
519                    .unwrap_or_default();
520            } else {
521                self.focus.list = self.root.collect_focusable_bounds();
522            }
523            // Clamp idx if the list shrank (e.g. widget removed).
524            if let Some(i) = self.focus.idx {
525                if i >= self.focus.list.len() {
526                    self.focus.idx = None;
527                }
528            }
529
530            // Rebuild signal subscriptions for root + overlays.
531            self.dirty_handles = self
532                .root
533                .collect_dirty_handles(Rc::clone(&self.dirty_region));
534            for entry in &mut self.overlays {
535                let handles = entry
536                    .widget
537                    .collect_dirty_handles(Rc::clone(&self.dirty_region));
538                self.dirty_handles.extend(handles);
539            }
540
541            self.dirty_region.borrow_mut().mark_all(w as f32, h as f32);
542        }
543
544        // ── Path B: render accumulated dirty region ───────────────────────────
545        let damage = self.dirty_region.borrow_mut().take();
546        if damage.is_none() {
547            return false;
548        }
549
550        // Snapshot focus bounds before the borrow split — Option<Rect> is Copy.
551        let focus_bounds = self.focus.current_bounds();
552
553        // Split borrows: renderer, font_ctx, theme, root, overlays, and
554        // toast_entries are all disjoint struct fields.
555        let renderer = &mut self.renderer;
556        let font_ctx = &mut self.font_ctx;
557        let theme = &*self.theme;
558        let root = &self.root;
559        let overlays = &self.overlays;
560        let toast_entries = &self.toast_entries;
561
562        renderer.render(damage, &mut |canvas| {
563            let mut cx = RenderCtx {
564                canvas,
565                font: font_ctx,
566                theme,
567            };
568            root.paint(&mut cx);
569
570            // Paint overlays bottom-to-top.
571            for entry in overlays.iter() {
572                if entry.blocking {
573                    // Dim the background with a semi-transparent scrim.
574                    let (sw, sh) = (cx.canvas.width() as f32, cx.canvas.height() as f32);
575                    cx.canvas.fill_rect(
576                        crate::layout::Rect::new(0.0, 0.0, sw, sh),
577                        crate::canvas::Color::bgra(0, 0, 0, 160),
578                    );
579                }
580                entry.widget.paint(&mut cx);
581            }
582
583            // Paint toasts above all overlays.
584            for entry in toast_entries.iter() {
585                entry.widget.paint(&mut cx);
586            }
587
588            // Post-paint focus ring — drawn on top of all widget content.
589            if let Some(b) = focus_bounds {
590                cx.canvas.draw_rect_border(b, cx.theme.accent(), 2.0);
591            }
592        });
593        true
594    }

598    /// Drain the action queue and apply each `OverlayAction` to `self`.
599    ///
600    /// Called twice per frame: once before event processing (to apply actions
601    /// from the previous frame) and once after (to apply actions queued this
602    /// frame). The double drain ensures single-frame round-trip latency.
603    fn drain_action_queue(&mut self) {
604        // Collect into a local vec to release the borrow on action_queue before
605        // mutating `self.overlays` / `self.toast_entries`.
606        let actions: Vec<OverlayAction> = self.action_queue.borrow_mut().drain(..).collect();
607        for action in actions {
608            match action {
609                OverlayAction::Push(entry) => {
610                    self.overlays.push(entry);
611                    self.layout_dirty = true;
612                }
613                OverlayAction::Pop => {
614                    self.overlays.pop();
615                    self.layout_dirty = true;
616                }
617                OverlayAction::PopAll => {
618                    self.overlays.clear();
619                    self.layout_dirty = true;
620                }
621                OverlayAction::ShowToast(config) => {
622                    let widget = crate::node_widgets::toast::Toast::new(&config);
623                    self.toast_entries
624                        .push_back(crate::node_widgets::toast::ToastEntry {
625                            config,
626                            elapsed_ms: 0,
627                            widget,
628                        });
629                    self.layout_dirty = true;
630                }
631            }
632        }
633    }

635    /// Force a full repaint on the next `tick()` call.
636    pub fn mark_dirty(&mut self) {
637        self.layout_dirty = true;
638    }
