2//! GPU draw command types and command buffer recorder.
3//!
4//! `GpuCmd` encodes one draw operation without executing it. Commands are
5//! accumulated into `GpuCommandBuffer` during a paint pass and replayed by
6//! a `CommandExecutor` — allowing damage-rect filtering and future hardware
7//! GPU execution without changing widget code.
8
9use crate::canvas::{Color, TextStyle};
10use crate::layout::{Point, Rect};
11use alloc::{string::String, vec::Vec};
12
13// ─── GpuCmd ──────────────────────────────────────────────────────────────────
14
15/// A single recorded draw operation.
16#[derive(Debug)]
17pub enum GpuCmd {
18    /// Fill a pre-clipped solid rectangle.
19    FillRect { rect: Rect, color: Color },
20    /// Draw a line from `a` to `b` (Bresenham, handled by executor).
21    DrawLine { a: Point, b: Point, color: Color },
22    /// Draw text at `pos`. `style.size_px == 0` = bitmap 8×8 fallback.
23    DrawText {
24        pos: Point,
25        text: String,
26        style: TextStyle,
27    },
28    /// Blit raw BGRA pixels. Destination may extend beyond current clip —
29    /// executor applies clipping during playback.
30    DrawImage {
31        dest: Rect,
32        pixels: Vec<u8>,
33        src_stride: u32,
34    },
35    /// Zero-alloc path for text ≤ 127 bytes (covers all typical single-line UI strings).
36    /// Bytes are always valid UTF-8 — written from `&str` in `GpuCanvas::draw_text`.
37    DrawTextShort {
38        pos: Point,
39        bytes: [u8; 128],
40        len: u8,
41        style: TextStyle,
42    },
43}
44
45impl GpuCmd {
46    /// Conservative bounding rect used for damage-rect filtering.
47    ///
48    /// Returns `None` only if the command is inherently unclippable; callers
49    /// that see `None` must always execute the command.
50    pub fn bounding_rect(&self) -> Option<Rect> {
51        match self {
52            GpuCmd::FillRect { rect, .. } => Some(*rect),
53            GpuCmd::DrawLine { a, b, .. } => {
54                let x = a.x.min(b.x);
55                let y = a.y.min(b.y);
56                let x2 = a.x.max(b.x);
57                let y2 = a.y.max(b.y);
58                // Min 1px so a horizontal/vertical line still has area.
59                Some(Rect {
60                    x,
61                    y,
62                    w: (x2 - x).max(1.0),
63                    h: (y2 - y).max(1.0),
64                })
65            }
66            GpuCmd::DrawText { pos, text, .. } => {
67                // 8×8 bitmap font: estimate width = chars × 8px
68                Some(Rect {
69                    x: pos.x,
70                    y: pos.y,
71                    w: text.len() as f32 * 8.0,
72                    h: 8.0,
73                })
74            }
75            GpuCmd::DrawImage { dest, .. } => Some(*dest),
76            GpuCmd::DrawTextShort { pos, len, .. } => Some(Rect {
77                x: pos.x,
78                y: pos.y,
79                w: *len as f32 * 8.0,
80                h: 8.0,
81            }),
82        }
83    }
84}

86// ─── RecordedCmd ─────────────────────────────────────────────────────────────
87
88/// A `GpuCmd` with its bounding rect pre-computed at record time.
89///
90/// `bounds` mirrors `GpuCmd::bounding_rect()` but is evaluated once in
91/// `GpuCommandBuffer::push()` so the executor reads a stored field instead
92/// of recomputing it every frame.
93pub struct RecordedCmd {
94    pub cmd: GpuCmd,
95    pub bounds: Option<Rect>,
96}
97
98// ─── GpuCommandBuffer ────────────────────────────────────────────────────────
99
100/// Ordered list of draw commands recorded during one paint pass.
101pub struct GpuCommandBuffer {
102    cmds: Vec<RecordedCmd>,
103}
104
105impl GpuCommandBuffer {
106    pub fn new() -> Self {
107        Self { cmds: Vec::new() }
108    }
109
110    /// Append a command, pre-computing its bounding rect once at record time.
111    pub fn push(&mut self, cmd: GpuCmd) {
112        let bounds = cmd.bounding_rect();
113        self.cmds.push(RecordedCmd { cmd, bounds });
114    }
115
116    /// Iterate recorded commands with pre-computed bounds.
117    pub fn recorded_slice(&self) -> &[RecordedCmd] {
118        &self.cmds
119    }
120
121    pub fn len(&self) -> usize {
122        self.cmds.len()
123    }
124
125    pub fn is_empty(&self) -> bool {
126        self.cmds.is_empty()
127    }
128
129    /// Clear all commands while retaining the Vec's heap allocation for the next frame.
130    pub fn clear(&mut self) {
131        self.cmds.clear();
132    }
133}
134
135impl Default for GpuCommandBuffer {
136    fn default() -> Self {
137        Self::new()
138    }
