2//! `GpuRenderer<E>` — `ViRenderer` implementation backed by a `CommandExecutor`.
3//!
4//! The paint closure runs against a `GpuCanvas` (recording mode), then the
5//! accumulated `GpuCommandBuffer` is handed to the executor for playback with
6//! damage-rect filtering. Widget code is unchanged — it calls the same
7//! `ViCanvas` trait methods regardless of the active backend.
8
9use crate::canvas::ViCanvas;
10use crate::executor::CommandExecutor;
11use crate::gpu_canvas::GpuCanvas;
12use crate::gpu_cmd::GpuCommandBuffer;
13use crate::layout::Rect;
14use crate::renderer::ViRenderer;
15
16// ─── GpuRenderer ─────────────────────────────────────────────────────────────
17
18/// Generic renderer that records draw commands and replays via `E`.
19///
20/// Use `GpuRenderer<CpuExecutor>` as a drop-in replacement for
21/// `FramebufferRenderer` with damage-rect optimization enabled.
22/// Swap `E` for a hardware GPU executor in G2+ without touching widget code.
23pub struct GpuRenderer<E: CommandExecutor> {
24    executor: E,
25    width: u32,
26    height: u32,
27    buf: GpuCommandBuffer,
28}
29
30impl<E: CommandExecutor> GpuRenderer<E> {
31    pub fn new(executor: E, width: u32, height: u32) -> Self {
32        Self {
33            executor,
34            width,
35            height,
36            buf: GpuCommandBuffer::new(),
37        }
38    }
39
40    /// Unwrap the executor (e.g. to reclaim the inner `ViSurface`).
41    pub fn into_executor(self) -> E {
42        self.executor
43    }
44}
45
46impl<E: CommandExecutor> ViRenderer for GpuRenderer<E> {
47    fn render(&mut self, damage: Option<Rect>, draw: &mut dyn FnMut(&mut dyn ViCanvas)) {
48        self.buf.clear();
49        {
50            let mut canvas = GpuCanvas::new(&mut self.buf, self.width, self.height);
51            draw(&mut canvas);
52        }
53        // NLL field split: &self.buf (immutable) + &mut self.executor separate fields
54        self.executor.execute(&self.buf, damage);
55    }
56
57    fn size(&self) -> (u32, u32) {
58        (self.width, self.height)
59    }
60}
