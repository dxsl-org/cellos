2//! `RenderCtx` — combined canvas + font context + theme for `ViNode::paint()`.
3//!
4//! Bundles a mutable canvas reference with the frame's `FontContext` and
5//! active `ViTheme` so every widget has access to drawing primitives, scalable
6//! glyphs, and design tokens through a single argument.
7//!
8//! # Why a struct instead of two arguments?
9//!
10//! `ViNode::paint()` is an object-safe trait method. Rust requires object-safe
11//! methods to have no more than one unsized parameter (the receiver). Passing
12//! two `&mut dyn …` arguments is fine syntactically but introducing `RenderCtx`
13//! is cleaner, extensible, and avoids ambiguity at call sites.
14
15use crate::canvas::{Color, TextStyle, ViCanvas};
16use crate::font_context::FontContext;
17use crate::layout::Point;
18use crate::theme::ViTheme;
19
20/// Combined draw surface + font state + active theme for one paint pass.
21///
22/// Passed by mutable reference through the entire widget tree during paint.
23/// Containers forward `cx` directly to children — no intermediate clone or reborrow.
24pub struct RenderCtx<'a> {
25    /// The pixel drawing surface for this frame.
26    pub canvas: &'a mut dyn ViCanvas,
27    /// Scalable glyph state. `atlas` may be `None` (bitmap fallback).
28    pub font: &'a mut FontContext,
29    /// Active design token set. Widgets read colors and spacing from here
30    /// rather than hardcoding values.
31    pub theme: &'a dyn ViTheme,
32}
33
34impl<'a> RenderCtx<'a> {
35    /// Draw text, using the scalable atlas when available, otherwise 8×8 bitmap.
36    ///
37    /// Prefer this helper over calling `canvas.draw_text` directly so the
38    /// fallback logic is in one place.
39    #[inline]
40    pub fn draw_text(&mut self, pos: Point, text: &str, color: Color) {
41        match self.font.atlas.as_mut() {
42            Some(atlas) => {
43                let px = self.font.size_px;
44                self.canvas.draw_text_scaled(pos, text, px, color, atlas);
45            }
46            None => {
47                self.canvas
48                    .draw_text(pos, text, TextStyle { color, size_px: 0 });
49            }
50        }
51    }
52
53    /// Draw text at an explicit pixel size, with atlas or bitmap fallback.
54    #[inline]
55    pub fn draw_text_at_size(&mut self, pos: Point, text: &str, color: Color, size_px: f32) {
56        match self.font.atlas.as_mut() {
57            Some(atlas) => self
58                .canvas
59                .draw_text_scaled(pos, text, size_px, color, atlas),
60            None => self
61                .canvas
62                .draw_text(pos, text, TextStyle { color, size_px: 0 }),
63        }
64    }
65
66    /// Estimated character width at the current font size. Used for layout.
67    ///
68    /// Returns an approximation — exact metrics require a full layout pass with
69    /// the atlas. Widgets that need tight text layout should call
70    /// `self.font.atlas.as_mut().map(|a| a.advance(ch, px))` directly.
71    #[inline]
72    pub fn char_width(&self) -> f32 {
73        if self.font.has_font() {
74            self.font.size_px * 0.6
75        } else {
76            8.0
77        }
78    }
79
80    /// Line height at the current font size.
81    #[inline]
82    pub fn line_height(&self) -> f32 {
83        if self.font.has_font() {
84            self.font.size_px * 1.2
85        } else {
86            8.0
87        }
88    }
89
90    /// Re-borrow as a shorter-lived `RenderCtx`. Allows passing `cx` to child
91    /// widgets when the parent still needs access afterward (split borrow).
92    #[inline]
93    pub fn reborrow(&mut self) -> RenderCtx<'_> {
94        RenderCtx {
95            canvas: self.canvas,
96            font: self.font,
97            theme: self.theme,
98        }
99    }
100}
