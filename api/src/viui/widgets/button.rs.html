1//! Button widget — OrbTK WidgetFlags + decoupled event→message pattern.
2
3use alloc::string::String;
4
5use crate::canvas::TextStyle;
6use crate::event::{Event, EventCx, EventStatus, MouseButton};
7use crate::layout::{Constraints, LayoutNode, Padding, Point, Rect, Size};
8use crate::widget::{PaintCx, ViWidget, WidgetId};
9use crate::widgets::label::Label;
10
11pub struct Button {
12    pub id: WidgetId,
13    label: Label,
14    pub padding: Padding,
15    // Direct-field state (OrbTK pattern: event() writes, paint() reads)
16    hovered: bool,
17    pressed: bool,
18    /// True for exactly one frame after a completed click. Reset each dispatch cycle.
19    pub just_clicked: bool,
20}
21
22impl Button {
23    pub fn new(id: WidgetId, text: impl Into<String>) -> Self {
24        Self {
25            id,
26            label: Label::new(text),
27            padding: Padding::all(6.0),
28            hovered: false,
29            pressed: false,
30            just_clicked: false,
31        }
32    }
33
34    pub fn with_padding(mut self, padding: Padding) -> Self {
35        self.padding = padding;
36        self
37    }
38
39    /// Call once per frame after dispatch to consume and reset the click flag.
40    pub fn clicked(&mut self) -> bool {
41        let v = self.just_clicked;
42        self.just_clicked = false;
43        v
44    }
45}
46
47impl ViWidget for Button {
48    fn layout(&self, constraints: Constraints) -> LayoutNode {
49        let text_size = self.label.measure();
50        let desired = Size {
51            w: text_size.w + self.padding.h_total(),
52            h: text_size.h + self.padding.v_total(),
53        };
54        let size = constraints.constrain(desired);
55        let bounds = Rect::from_origin_size(constraints.origin, size);
56        LayoutNode::leaf(bounds)
57    }
58
59    fn paint(&self, cx: &mut PaintCx) {
60        let ts = self.label.measure();
61        let bounds = Rect::from_origin_size(
62            cx.origin,
63            Size {
64                w: ts.w + self.padding.h_total(),
65                h: ts.h + self.padding.v_total(),
66            },
67        );
68
69        let bg = if self.pressed {
70            cx.theme.button_pressed()
71        } else if self.hovered {
72            cx.theme.button_hovered()
73        } else {
74            cx.theme.button_normal()
75        };
76        cx.canvas.fill_rect(bounds, bg);
77
78        let border = cx.theme.border();
79        let b = bounds;
80        for (a, bb) in [
81            (Point::new(b.x, b.y), Point::new(b.x + b.w, b.y)),
82            (Point::new(b.x + b.w, b.y), Point::new(b.x + b.w, b.y + b.h)),
83            (Point::new(b.x + b.w, b.y + b.h), Point::new(b.x, b.y + b.h)),
84            (Point::new(b.x, b.y + b.h), Point::new(b.x, b.y)),
85        ] {
86            cx.canvas.draw_line(a, bb, border);
87        }
88
89        let text_pos = Point::new(
90            cx.origin.x + self.padding.left,
91            cx.origin.y + self.padding.top,
92        );
93        cx.canvas.draw_text(
94            text_pos,
95            &self.label.text,
96            TextStyle {
97                color: cx.theme.text_primary(),
98                size_px: cx.theme.font_size_body(),
99            },
100        );
101    }
102
103    fn event(&mut self, cx: &mut EventCx, e: &Event) -> EventStatus {
104        let bounds = cx.bounds();
105        match e {
106            Event::MouseMove { pos } => {
107                let now_hovered = bounds.contains(*pos);
108                if now_hovered != self.hovered {
109                    self.hovered = now_hovered;
110                    cx.mark_dirty();
111                }
112                EventStatus::Ignored
113            }
114            Event::MousePress {
115                pos,
116                button: MouseButton::Left,
117            } => {
118                if bounds.contains(*pos) {
119                    self.pressed = true;
120                    cx.mark_dirty();
121                    EventStatus::Consumed
122                } else {
123                    EventStatus::Ignored
124                }
125            }
126            Event::MouseRelease {
127                pos,
128                button: MouseButton::Left,
129            } => {
130                let was_pressed = self.pressed;
131                self.pressed = false;
132                if was_pressed {
133                    cx.mark_dirty();
134                    if bounds.contains(*pos) {
135                        self.just_clicked = true;
136                    }
137                    EventStatus::Consumed
138                } else {
139                    EventStatus::Ignored
140                }
141            }
142            _ => EventStatus::Ignored,
143        }
144    }
145}
