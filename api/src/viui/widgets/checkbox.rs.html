1//! Checkbox widget — toggle with label.
2
3use alloc::string::String;
4
5use crate::canvas::{Color, TextStyle};
6use crate::event::{Event, EventCx, EventStatus, MouseButton};
7use crate::layout::{Constraints, LayoutNode, Point, Rect, Size};
8use crate::widget::{PaintCx, ViWidget, WidgetId};
9use crate::widgets::label::Label;
10
11const BOX_SIZE: f32 = 14.0;
12const GAP: f32 = 6.0;
13const CHECK: Color = Color::WHITE;
14
15pub struct Checkbox {
16    pub id: WidgetId,
17    pub checked: bool,
18    label: Label,
19    hovered: bool,
20}
21
22impl Checkbox {
23    pub fn new(id: WidgetId, checked: bool, label: impl Into<String>) -> Self {
24        Self {
25            id,
26            checked,
27            label: Label::new(label),
28            hovered: false,
29        }
30    }
31
32    /// Consume and return the latest checked state.
33    pub fn is_checked(&self) -> bool {
34        self.checked
35    }
36}
37
38impl ViWidget for Checkbox {
39    fn layout(&self, constraints: Constraints) -> LayoutNode {
40        let label_w = self.label.measure().w;
41        let desired = Size {
42            w: BOX_SIZE + GAP + label_w,
43            h: BOX_SIZE,
44        };
45        let size = constraints.constrain(desired);
46        LayoutNode::leaf(Rect::from_origin_size(constraints.origin, size))
47    }
48
49    fn paint(&self, cx: &mut PaintCx) {
50        let orig = cx.origin;
51        let box_rect = Rect::new(orig.x, orig.y, BOX_SIZE, BOX_SIZE);
52
53        let bg = if self.hovered {
54            cx.theme.button_hovered()
55        } else {
56            cx.theme.surface()
57        };
58        cx.canvas.fill_rect(box_rect, bg);
59
60        let border = cx.theme.border();
61        for (a, b) in [
62            (
63                Point::new(orig.x, orig.y),
64                Point::new(orig.x + BOX_SIZE, orig.y),
65            ),
66            (
67                Point::new(orig.x + BOX_SIZE, orig.y),
68                Point::new(orig.x + BOX_SIZE, orig.y + BOX_SIZE),
69            ),
70            (
71                Point::new(orig.x + BOX_SIZE, orig.y + BOX_SIZE),
72                Point::new(orig.x, orig.y + BOX_SIZE),
73            ),
74            (
75                Point::new(orig.x, orig.y + BOX_SIZE),
76                Point::new(orig.x, orig.y),
77            ),
78        ] {
79            cx.canvas.draw_line(a, b, border);
80        }
81
82        if self.checked {
83            // Checkmark: two diagonal lines forming a tick
84            let m = 3.0; // margin
85            cx.canvas.draw_line(
86                Point::new(orig.x + m, orig.y + BOX_SIZE / 2.0),
87                Point::new(orig.x + BOX_SIZE / 2.0 - 1.0, orig.y + BOX_SIZE - m),
88                CHECK,
89            );
90            cx.canvas.draw_line(
91                Point::new(orig.x + BOX_SIZE / 2.0 - 1.0, orig.y + BOX_SIZE - m),
92                Point::new(orig.x + BOX_SIZE - m, orig.y + m),
93                CHECK,
94            );
95        }
96
97        let text_pos = Point::new(orig.x + BOX_SIZE + GAP, orig.y + (BOX_SIZE - 8.0) / 2.0);
98        cx.canvas.draw_text(
99            text_pos,
100            &self.label.text,
101            TextStyle {
102                color: cx.theme.text_primary(),
103                size_px: cx.theme.font_size_body(),
104            },
105        );
106    }
107
108    fn event(&mut self, cx: &mut EventCx, e: &Event) -> EventStatus {
109        let bounds = cx.bounds();
110        match e {
111            Event::MouseMove { pos } => {
112                let h = bounds.contains(*pos);
113                if h != self.hovered {
114                    self.hovered = h;
115                    cx.mark_dirty();
116                }
117                EventStatus::Ignored
118            }
119            Event::MousePress {
120                pos,
121                button: MouseButton::Left,
122            } => {
123                if bounds.contains(*pos) {
124                    self.checked = !self.checked;
125                    cx.mark_dirty();
126                    EventStatus::Consumed
127                } else {
128                    EventStatus::Ignored
129                }
130            }
131            _ => EventStatus::Ignored,
132        }
133    }
134}
