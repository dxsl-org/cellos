2//! `ViNode` trait — v2 widget interface for the Reactive Signal Tree.
3//!
4//! Compared to v1 `ViWidget`:
5//! - No `Msg` generic — callbacks live in widget structs as `Box<dyn Fn()>`
6//! - No `WidgetStateStore` — state lives in `Signal<T>` fields
7//! - `layout()` returns `Size` (containers recurse into children directly)
8//! - `paint()` takes `&mut dyn ViCanvas` directly (no `PaintCx` wrapper)
9//! - `event()` returns `bool` instead of `EventStatus`
10
11extern crate alloc;
12use alloc::vec::Vec;
13
14use crate::dirty::DirtyRegion;
15use crate::event::Event;
16use crate::layout::{Constraints, Rect, Size};
17use crate::render_ctx::RenderCtx;
18use crate::signal::SubscriptionHandle;
19
20/// v2 widget trait — Reactive Signal Tree node.
21///
22/// All implementations must cache their final `Rect` from `layout()` and
23/// return it from `bounds()` — `paint()` and `event()` use it for positioning
24/// and hit-testing respectively.
25pub trait ViNode: 'static {
26    /// Compute layout given available space; store the result as cached bounds.
27    ///
28    /// `constraints.origin` is the top-left of the assigned slot.
29    /// Returns the actual size consumed (≤ `constraints.max`).
30    fn layout(&mut self, constraints: Constraints) -> Size;
31
32    /// Cached bounds from the last `layout()` call. Returns `Rect::ZERO` before
33    /// first layout.
34    fn bounds(&self) -> Rect;
35
36    /// Paint into `cx` using the cached bounds.
37    ///
38    /// `cx` carries both the draw surface (`canvas`) and the scalable font
39    /// state (`font`). Containers must forward `cx` to children via
40    /// `cx.reborrow()` or by passing each child a disjoint sub-borrow.
41    fn paint(&self, cx: &mut RenderCtx<'_>);
42
43    /// Handle an input event. Returns `true` if consumed (stops bubbling to parent).
44    fn event(&mut self, event: &Event) -> bool;
45
46    /// Subscribe all internal `Signal`s to mark `region` dirty when they change.
47    ///
48    /// Called by `ViApp` after each layout pass. The returned handles keep
49    /// subscriptions alive; `ViApp` stores them and drops them before the next
50    /// layout so closures capture fresh bounds. Containers must recurse into
51    /// children. Default: returns empty vec (widgets with no Signals).
52    fn collect_dirty_handles(&mut self, _region: DirtyRegion) -> Vec<SubscriptionHandle> {
53        Vec::new()
54    }
55
56    /// Returns true if this widget can receive keyboard focus.
57    ///
58    /// Leaf interactive widgets (Button, CheckBox, Slider, TextEdit) override
59    /// this to return `true`. Containers and decorative widgets leave the default.
60    fn is_focusable(&self) -> bool {
61        false
62    }
63
64    /// Collect screen rects of all focusable descendants (and self, if focusable).
65    ///
66    /// Called by `ViApp` after each layout pass to build the Tab-order list.
67    /// Containers must override to recurse into children. Default: returns `self.bounds()`
68    /// when `is_focusable()` is true, otherwise empty.
69    fn collect_focusable_bounds(&mut self) -> alloc::vec::Vec<crate::layout::Rect> {
70        if self.is_focusable() {
71            alloc::vec![self.bounds()]
72        } else {
73            alloc::vec::Vec::new()
74        }
75    }
76
77    /// Activate the widget (called when Enter is pressed while this widget is focused).
78    ///
79    /// Returns `true` if the activation was consumed (triggers layout re-check).
80    /// Button: fires `on_click`. CheckBox: toggles state. Default: no-op → false.
81    fn activate(&mut self) -> bool {
82        false
83    }
84
85    /// Walk the tree and activate the widget whose bounds exactly match `target`.
86    ///
87    /// Default impl handles leaf widgets: if focusable and bounds match, calls
88    /// `activate()`. Containers must override to recurse into children.
89    fn activate_at(&mut self, target: crate::layout::Rect) -> bool {
90        if self.is_focusable() && self.bounds() == target {
91            self.activate()
92        } else {
93            false
94        }
95    }
96}
