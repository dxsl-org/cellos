1//! Layer-2 hardware security self-tests.
2//!
3//! Runs at boot when the `test-hooks` feature is active. Each test prints
4//! `{FEATURE}-SELFTEST: PASS` or `{FEATURE}-SELFTEST: SKIP` (never FAIL for
5//! expected "feature absent" — only FAIL for feature present but broken).
6//!
7//! ⚠️ Never ship `test-hooks` in a production image. These tests exercise
8//! raw hardware state at boot and emit extra serial output that changes the
9//! boot log structure CI greps against.
10//!
11//! # Preconditions
12//! Call after all hardware init but before the scheduler starts. The tests
13//! must run in Ring-0 / EL1 (kernel mode) with interrupts still masked.
14
15// ─── AArch64 MTE self-test ─────────────────────────────────────────────────
16
17/// AArch64 MTE tag round-trip self-test.
18///
19/// Tags a 32-byte static buffer with colour 0x5, verifies both granules read
20/// back as 0x5, re-tags with colour 0xA, and verifies the new tag.
21///
22/// # SKIP conditions
23/// - MTE2 not present in ID_AA64PFR1_EL1[11:8].
24///
25/// # FAIL conditions
26/// - Tag round-trip returns wrong value (MTE asm broken or SCTLR not set).
27#[cfg(all(target_arch = "aarch64", feature = "test-hooks"))]
28pub fn run_mte_selftest() {
29    // Read ID_AA64PFR1_EL1 — the MTE field is [11:8].
30    // ≥ 2 means MTE2 (full load+store tag checking) is available.
31    let pfr1: u64;
32    // SAFETY: mrs from a read-only ID register; no memory side effects.
33    unsafe {
34        core::arch::asm!(
35            "mrs {}, id_aa64pfr1_el1",
36            out(reg) pfr1,
37            options(nomem, nostack)
38        );
39    }
40    if ((pfr1 >> 8) & 0xF) < 2 {
41        puts_aa64("[SELFTEST] MTE-SELFTEST: SKIP: MTE2 unavailable\n");
42        return;
43    }
44
45    // 32-byte test buffer — two MTE granules (16 bytes each).
46    // `static mut` is safe here: single-threaded boot code, no other Rust
47    // reference exists, and the buffer is only written via raw asm (STG).
48    static mut MTE_BUF: [u8; 32] = [0u8; 32];
49    // addr_of_mut! is safe — no reference to the static is created.
50    let ptr = core::ptr::addr_of_mut!(MTE_BUF) as *mut u8;
51
52    // Tag both granules with colour 0x5.
53    // SAFETY: ptr is 16-byte aligned (static alignment), len is a multiple of
54    // 16, and the region is valid Normal-Tagged kernel memory.
55    unsafe {
56        tag_region(ptr, 32, 0x5);
57    }
58
59    // SAFETY: ptr points into the freshly tagged region.
60    let tag0 = unsafe { get_tag(ptr) };
61    // SAFETY: ptr+16 is within the 32-byte MTE_BUF (granule 2).
62    let tag1 = unsafe { get_tag(ptr.add(16)) };
63    if tag0 != 0x5 || tag1 != 0x5 {
64        puts_aa64("[SELFTEST] MTE-SELFTEST: FAIL: tag round-trip (0x5) failed\n");
65        return;
66    }
67
68    // Re-tag with colour 0xA (simulates allocation-tag change on "free").
69    // SAFETY: same constraints as above.
70    unsafe {
71        tag_region(ptr, 32, 0xA);
72    }
73    let new_tag = unsafe { get_tag(ptr) };
74    if new_tag != 0xA {
75        puts_aa64("[SELFTEST] MTE-SELFTEST: FAIL: re-tag (0xA) failed\n");
76        return;
77    }
78
79    puts_aa64("[SELFTEST] MTE-SELFTEST: PASS\n");
80}

82/// Tag `len` bytes starting at `ptr` with MTE colour `color`.
83///
84/// Uses STG (Store Tag) to write the allocation tag for each 16-byte granule.
85/// Requires MTE2+ and ATA/ATA0 set in SCTLR_EL1 (done by `hal::mte::init()`).
86///
87/// # Safety
88/// - `ptr` must be 16-byte aligned.
89/// - `len` must be a non-zero multiple of 16.
90/// - The range `[ptr, ptr + len)` must be valid kernel memory (Normal-Tagged).
91/// - Must be called from EL1 with interrupts masked (boot path).
92#[cfg(all(target_arch = "aarch64", feature = "test-hooks"))]
93unsafe fn tag_region(ptr: *mut u8, len: usize, color: u8) {
94    let tag = (color & 0xF) as u64;
95    let mut cur = ptr as u64;
96    let end = cur + len as u64;
97    while cur < end {
98        // Bits [59:56] carry the allocation tag; TBI strips the top byte at
99        // load/store so normal memory accesses are unaffected.
100        let tagged_ptr = (cur & !(0xFu64 << 56)) | (tag << 56);
101        // SAFETY: STG touches only the tag memory for the 16-byte granule at
102        // `tagged_ptr`. The caller guarantees the range is valid.
103        unsafe {
104            core::arch::asm!(
105                "stg {p}, [{p}]",
106                p = in(reg) tagged_ptr,
107                options(nostack)
108            );
109        }
110        cur += 16;
111    }
112}

114/// Return the 4-bit MTE allocation tag for the granule at `ptr`.
115///
116/// Uses LDG (Load Tag) which places the tag in bits [59:56] of the output.
117///
118/// # Safety
119/// - `ptr` must point inside a valid Normal-Tagged region that was previously
120///   tagged with [`tag_region`].
121#[cfg(all(target_arch = "aarch64", feature = "test-hooks"))]
122unsafe fn get_tag(ptr: *const u8) -> u8 {
123    let mut result: u64 = ptr as u64;
124    // SAFETY: LDG reads only the tag memory for the granule at `ptr`.
125    unsafe {
126        core::arch::asm!(
127            "ldg {r}, [{r}]",
128            r = inout(reg) result,
129            options(nostack)
130        );
131    }
132    ((result >> 56) & 0xF) as u8
133}

145/// x86_64 PKU PKRU value and register self-test.
146///
147/// Verifies:
148/// 1. `pkru_for_key(0)` returns 0x00000000 (trusted-core all-access).
149/// 2. `pkru_for_key(1)` returns 0x55555550 (allow keys 0+1, deny rest).
150/// 3. RDPKRU in kernel mode returns 0 (kernel PKRU = all-access).
151///
152/// # SKIP conditions
153/// - `PKU_ACTIVE == 0` (PKU not enabled — either unavailable or IBT absent).
154///
155/// # FAIL conditions
156/// - Any computed PKRU value deviates from the expected constant.
157/// - RDPKRU returns a non-zero value in kernel context.
158#[cfg(all(target_arch = "x86_64", feature = "test-hooks"))]
159pub fn run_pku_selftest() {
160    use crate::hal::pku;
161
162    // SAFETY: PKU_ACTIVE is written once at boot (single-threaded init), then
163    // treated as read-only.  No concurrent Rust reference exists after boot.
164    if unsafe { pku::PKU_ACTIVE } == 0 {
165        puts_x86("[SELFTEST] PKU-SELFTEST: SKIP: PKU not active\n");
166        return;
167    }
168
169    // Key 0 = trusted-core → PKRU = 0 (all-access).
170    let k0 = pku::pkru_for_key(0);
171    if k0 != 0x0000_0000 {
172        puts_x86("[SELFTEST] PKU-SELFTEST: FAIL: key 0 PKRU wrong\n");
173        return;
174    }
175
176    // Key 1 = standard Tier-1 Rust cell.
177    // All-deny = 0x5555_5555; clear bits [1:0] (key 0 AD), clear bits [3:2] (key 1 AD).
178    // Expected: 0x5555_5550.
179    let k1 = pku::pkru_for_key(1);
180    if k1 != 0x5555_5550 {
181        puts_x86("[SELFTEST] PKU-SELFTEST: FAIL: key 1 PKRU wrong\n");
182        return;
183    }
184
185    // RDPKRU in Ring-0: the kernel always runs with PKRU = 0 (all-access).
186    // ECX must be 0; EAX receives PKRU, EDX receives 0.
187    let pkru_val: u32;
188    // SAFETY: RDPKRU from Ring-0 with ECX=0; PKU_ACTIVE=1 means CR4.PKE is set
189    // so this instruction is supported. RDPKRU reads ECX (implicit, must be 0)
190    // and writes PKRU to EAX; EDX is cleared.
191    unsafe {
192        core::arch::asm!(
193            "xor ecx, ecx",  // ECX must be 0 for RDPKRU (privileged guard)
194            "rdpkru",
195            out("eax") pkru_val,
196            out("edx") _,
197            options(nomem, nostack)
198        );
199    }
200    if pkru_val != 0 {
201        puts_x86("[SELFTEST] PKU-SELFTEST: FAIL: kernel PKRU != 0\n");
202        return;
203    }
204
205    puts_x86("[SELFTEST] PKU-SELFTEST: PASS\n");
206}
