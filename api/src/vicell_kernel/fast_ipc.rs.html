1//! Kernel-owned fast-IPC dispatch table — the single canonical instance.
2//!
3//! In a Single Address Space there is no privilege wall between Cells: a trusted
4//! Cell calling a service handler is just an indirect call (~3 cycles) versus
5//! ~100+ for an `ecall` round-trip. For the fast path to work, ONE handler
6//! pointer must be shared by the VFS cell (which registers it), client cells
7//! (which call it), and the kernel (which nulls it if VFS faults).
8//!
9//! Because Cells are separately-loaded ELFs (each with its own copy of any
10//! `static`), the shared instance cannot live in a per-cell library — it lives
11//! HERE, in the kernel. Cells reach `register_vfs`/`call_vfs` by name through the
12//! loader's global-symbol-table resolution (see `loader::dynsym`); the kernel
13//! uses `set_vfs_handler_cell`/`clear_vfs_if_cell` directly.
14//!
15//! ## Safety invariant
16//! The handler pointer is published once at VFS startup (before any client call)
17//! with `Release` ordering, read with `Acquire`, and only ever nulled on VFS
18//! fault. Single-hart QEMU: no concurrent modification.
19
20use api::fast_ipc::{TrustedHandle, VfsCell};
21use api::ipc::{VfsRequest, IPC_BUF_SIZE};
22use core::sync::atomic::{AtomicPtr, AtomicUsize, Ordering};
23
24/// Signature of a registered VFS fast-IPC handler: read `req`, write the
25/// response into `out`, return the number of bytes written.
26pub type VfsFastHandler = unsafe fn(req: &VfsRequest<'_>, out: &mut [u8; IPC_BUF_SIZE]) -> usize;
27
28static VFS_HANDLER_PTR: AtomicPtr<()> = AtomicPtr::new(core::ptr::null_mut());
29/// Raw CellId that registered the handler; 0 = unregistered. Lets the kernel
30/// null the pointer when (and only when) that specific cell faults.
31static VFS_HANDLER_CELL: AtomicUsize = AtomicUsize::new(0);
32
33/// Register the VFS fast-IPC handler. Called once by the VFS cell at startup
34/// (resolved to this kernel symbol via the loader global symbol table).
35///
36/// `#[no_mangle]` so a Cell's undefined `register_vfs` import resolves here.
37#[no_mangle]
38pub extern "Rust" fn register_vfs(handler: VfsFastHandler) {
39    // SAFETY: fn-ptr → *mut () for atomic storage; recovered with the same type
40    // in `call_vfs`. Published Release so the handler body is visible to Acquire readers.
41    VFS_HANDLER_PTR.store(handler as *mut (), Ordering::Release);
42}
43
44/// Record which cell owns the registered handler (kernel-internal; called from
45/// the VFS spawn path so a later fault of that cell can null the pointer).
46pub fn set_vfs_handler_cell(cell_id_raw: usize) {
47    VFS_HANDLER_CELL.store(cell_id_raw, Ordering::Relaxed);
48}
49
50/// Null the handler pointer iff `cell_id_raw` is the registered owner. Called by
51/// the kernel fault path so a future `call_vfs` does not jump into dead VFS code.
52pub fn clear_vfs_if_cell(cell_id_raw: usize) {
53    if VFS_HANDLER_CELL.load(Ordering::Relaxed) == cell_id_raw && cell_id_raw != 0 {
54        VFS_HANDLER_PTR.store(core::ptr::null_mut(), Ordering::Release);
55        VFS_HANDLER_CELL.store(0, Ordering::Relaxed);
56    }
57}

59/// RAII guard that restores the S-mode interrupt-enable bit (SIE) on drop.
60///
61/// Constructed by disabling SIE and recording its prior state.  `Drop` restores
62/// it, so SIE is always restored even if the handler panics (Rust drop glue runs
63/// before the panic handler, giving the guard a chance to clean up).
64struct SieGuard(
65    /// `true` if SIE was set before we disabled it; `false` = was already clear.
66    bool,
67);
68
69impl SieGuard {
70    /// Disable SIE and return a guard that will restore it.
71    ///
72    /// # Safety
73    /// Must be called from S-mode.
74    #[inline]
75    unsafe fn disable() -> Self {
76        #[cfg(target_arch = "riscv64")]
77        {
78            let v: usize;
79            // SAFETY: csrrci reads-and-clears sstatus.SIE (bit 1) atomically.
80            core::arch::asm!("csrrci {}, sstatus, 0x2", out(reg) v);
81            Self(v & 0x2 != 0)
82        }
83        #[cfg(not(target_arch = "riscv64"))]
84        Self(false)
85    }
86}
87
88impl Drop for SieGuard {
89    fn drop(&mut self) {
90        if self.0 {
91            // SAFETY: restoring SIE to the value saved in disable(); S-mode only.
92            #[cfg(target_arch = "riscv64")]
93            unsafe {
94                core::arch::asm!("csrsi sstatus, 0x2");
95            }
96        }
97    }
98}

100/// Call the registered VFS handler directly, bypassing the `ecall` trap. Returns
101/// bytes written into `out`, or 0 if no handler is registered (caller falls back
102/// to the `sys_send`/`sys_recv` path).
103///
104/// `#[no_mangle]` so a client Cell's undefined `call_vfs` import resolves here.
105///
106/// # Note (PIE limitation)
107/// For non-PIE cells (current default), each cell ELF links `libs/ostd` statically
108/// and gets its own copy of `VFS_HANDLER_PTR` — so `call_vfs` in the shell reads
109/// null and always takes the ecall fallback.  The fast path becomes effective once
110/// cells are compiled as PIE and the loader patches JUMP_SLOT relocations to this
111/// kernel function.  The fallback is always safe.
112///
113/// # Safety
114/// The caller must own `out` exclusively for the call. `_handle` documents that
115/// the caller was granted fast-path access; it is not enforced at runtime.
116#[no_mangle]
117pub unsafe extern "Rust" fn call_vfs(
118    _handle: TrustedHandle<VfsCell>,
119    req: &VfsRequest<'_>,
120    out: &mut [u8; IPC_BUF_SIZE],
121) -> usize {
122    let ptr = VFS_HANDLER_PTR.load(Ordering::Acquire);
123    if ptr.is_null() {
124        return 0; // VFS not yet registered — caller falls back to ecall path.
125    }
126    // SAFETY: ptr was stored by register_vfs from a valid VfsFastHandler.
127    let handler: VfsFastHandler = core::mem::transmute(ptr);
128
129    // Disable S-mode interrupts for the handler's duration. The VFS FAT16 driver
130    // holds a spinlock; timer preemption mid-handler to another VFS caller would
131    // deadlock on it. SieGuard restores SIE on drop — safe even on handler panic.
132    // SAFETY: called from S-mode trap handler context.
133    let _sie = SieGuard::disable();
134
135    handler(req, out)
136}

138/// Resolve a kernel-exported symbol name to its runtime address — the loader's
139/// "Global Symbol Table" lookup for a Cell's undefined dynamic symbols. Returns
140/// `None` for names the kernel does not intentionally export.
141///
142/// Hand-maintained: add an arm here when a Cell is permitted to call a kernel
143/// function by name. (A `static` table can't hold these — `fn as usize` is not
144/// permitted in const eval — so resolution is a runtime match.)
145pub fn resolve_export(name: &str) -> Option<usize> {
146    match name {
147        "register_vfs" => Some(register_vfs as *const () as usize),
148        "call_vfs" => Some(call_vfs as *const () as usize),
149        _ => None,
150    }
151}
