1//! Resource Registry — exclusive MMIO region grants for Driver Cells.
2//!
3//! A Driver Cell calls `sys_request_mmio(base, len)` (added in Phase 03).
4//! The kernel checks here before handing an `MmioRegion` to the Cell:
5//!
6//! 1. **Allowlist**: the requested range must fall within a known-safe
7//!    device window for the current QEMU target.  Unknown ranges are rejected
8//!    so a misbehaving Cell cannot map arbitrary kernel memory as MMIO.
9//!
10//! 2. **Exclusive ownership**: at most one Cell may hold a given MMIO range.
11//!    A second `request_mmio` for an overlapping range returns `AlreadyExists`.
12//!
13//! 3. **Release-on-exit**: `release_for(cell_id)` frees all ranges owned by
14//!    a Cell.  Call this from every Cell-exit path alongside
15//!    `cell_quota::deregister`.
16//!
17//! # v1 scope
18//! Allowlist is hardcoded per QEMU target (DTB discovery deferred to v2).
19//!
20//! | Target | Device | Base | Size |
21//! |--------|--------|------|------|
22//! | QEMU ARM virt (aarch64) | PL011 UART0 | 0x0900_0000 | 0x1000 |
23//! | QEMU ARM virt (aarch64) | PL061 GPIO  | 0x0903_0000 | 0x1000 |
24//! | QEMU RISC-V virt (riscv64) | (none yet — kernel serial owns UART) | — | — |
25
26use crate::sync::Spinlock;
27use alloc::collections::BTreeMap;
28use types::{CellId, ViError, ViResult};

30// ---------------------------------------------------------------------------
31// Device-class tags (parameterized MMIO capability)
32// ---------------------------------------------------------------------------
33
34/// UART serial device window. Set in a cell's `mmio_devices` when its manifest
35/// declares `uart = true`.
36pub const DEV_UART: u8 = 1 << 0;
37/// GPIO controller window. Set when the manifest declares `gpio = true`.
38pub const DEV_GPIO: u8 = 1 << 1;
39/// PCIe device BAR window. Set on tasks with `PcieDriverCap` — not a manifest flag
40/// (all 8 manifest bits are occupied); gated by the ZST cap instead.
41pub const DEV_PCIE: u8 = 1 << 2;
42/// CAN bus controller window (v2 manifest — freed by the u16 flags widening).
43/// Set when the manifest declares `can = true`.
44pub const DEV_CAN: u8 = 1 << 3;
45/// ADC controller window (v2 manifest). Set when the manifest declares `adc = true`.
46pub const DEV_ADC: u8 = 1 << 4;
47
48// ---------------------------------------------------------------------------
49// Allowlist (per QEMU machine, v1 hardcoded)
50// ---------------------------------------------------------------------------
51
52/// `(base, len, device_class)` triples a Driver Cell may request. The device
53/// class scopes the capability: a cell may claim a range only if it declared
54/// the matching device (manifest gpio/uart flag), so a GPIO-only cell cannot
55/// grab the UART window and vice-versa.
56#[cfg(all(target_arch = "aarch64", feature = "board-rpi3"))]
57const ALLOWED: &[(usize, usize, u8)] = &[
58    (0x3F20_0000, 0x1_0000, DEV_GPIO), // BCM2837 GPIO — Raspberry Pi 3 (54 pins)
59    (0x3F21_5000, 0x0_1000, DEV_UART), // BCM mini UART (AUX block) — RPi 3
60                                       // BCM I2C (0x3F804000), SPI (0x3F204000) added when respective drivers land.
61];
62
63#[cfg(all(target_arch = "aarch64", not(feature = "board-rpi3")))]
64const ALLOWED: &[(usize, usize, u8)] = &[
65    (0x0900_0000, 0x1000, DEV_UART), // PL011 UART0  — QEMU ARM virt
66    (0x0903_0000, 0x1000, DEV_GPIO), // PL061 GPIO   — QEMU ARM virt
67];
68
69/// SiFive GPIO0 for QEMU `sifive_u` machine (FU540/FU740).
70/// The kernel serial driver owns NS16550 at 0x1000_0000 — excluded from allowlist.
71#[cfg(target_arch = "riscv64")]
72const ALLOWED: &[(usize, usize, u8)] = &[
73    (0x1001_2000, 0x1000, DEV_GPIO), // SiFive GPIO0 — QEMU sifive_u machine
74];
75
76#[cfg(target_arch = "x86_64")]
77const ALLOWED: &[(usize, usize, u8)] = &[];
78
79#[cfg(not(any(
80    target_arch = "aarch64",
81    target_arch = "riscv64",
82    target_arch = "x86_64"
83)))]
84const ALLOWED: &[(usize, usize, u8)] = &[];

86// ---------------------------------------------------------------------------
87// Registry state
88// ---------------------------------------------------------------------------
89
90/// Maps MMIO base address → (len, owner CellId).
91static REGISTRY: Spinlock<BTreeMap<usize, (usize, CellId)>> = Spinlock::new(BTreeMap::new());
92
93/// Dynamically discovered PCIe BAR windows (base → len).
94/// Populated by `pcie_ecam::init()` after the ECAM scan; consumed by
95/// `request_mmio` when the caller holds `PcieDriverCap` (DEV_PCIE).
96static PCIE_BARS: Spinlock<BTreeMap<usize, usize>> = Spinlock::new(BTreeMap::new());
97
98// ---------------------------------------------------------------------------
99// Public API
100// ---------------------------------------------------------------------------
101
102/// Force-release this module's lock during fault teardown.
103///
104/// # Safety
105/// Single-hart kernel; called only from the fault/panic path with interrupts
106/// disabled.  Force-unlocking an already-free Spinlock is a no-op.
107pub unsafe fn force_unlock_locks() {
108    REGISTRY.force_unlock();
109    // SAFETY: same contract as REGISTRY above.
110    unsafe {
111        PCIE_BARS.force_unlock();
112    }
113}

115/// Register a PCIe BAR window discovered during ECAM scan.
116///
117/// Called by `pcie_ecam::init()` for every non-zero BAR on every device.
118/// Driver Cells may subsequently call `sys_request_mmio` on these ranges
119/// if they hold `PcieDriverCap`.
120pub fn register_pcie_bar(base: usize, len: usize) {
121    if base != 0 && len != 0 {
122        PCIE_BARS.lock().insert(base, len);
123    }
124}
125
126/// Return `true` if `[base, base+len)` is a known PCIe BAR window.
127///
128/// Used by the `RequestMmio` handler to decide whether to take the PCIe path.
129pub fn is_pcie_bar(base: usize, len: usize) -> bool {
130    let guard = PCIE_BARS.lock();
131    guard.get(&base).is_some_and(|&bar_len| len <= bar_len)
132}
133
134/// Request exclusive MMIO ownership without allowlist validation (Platform Cell only).
135///
136/// Bypasses the per-arch ALLOWED list and the PCIE_BARS table. The overlap check
137/// still runs — two cells cannot share a byte. Used by the PlatformCap bypass path
138/// in `sys_request_mmio` so the Platform Cell can claim the ECAM config-space window
139/// (which is not a device BAR and therefore not in either allowlist).
140pub fn request_mmio_unchecked(cell_id: CellId, base: usize, len: usize) -> ViResult<()> {
141    let end = base.checked_add(len).ok_or(ViError::InvalidInput)?;
142    let mut reg = REGISTRY.lock();
143    for (&eb, &(el, _)) in reg.iter() {
144        let ee = eb + el;
145        if !(end <= eb || base >= ee) {
146            return Err(ViError::AlreadyExists);
147        }
148    }
149    reg.insert(base, (len, cell_id));
150    Ok(())
151}

153/// Request exclusive ownership of `[base, base+len)` for `cell_id`.
154///
155/// Returns:
156/// - `Ok(())` — range is now owned by the caller; construct `MmioRegion` and
157///   hand it to the Cell.
158/// - `Err(PermissionDenied)` — range not in allowlist, or its device class is
159///   not among `allowed_devices` (the cell's declared `mmio_devices`).
160/// - `Err(AlreadyExists)` — range overlaps an already-granted region.
161/// - `Err(InvalidInput)` — arithmetic overflow in `base + len`.
162pub fn request_mmio(cell_id: CellId, base: usize, len: usize, allowed_devices: u8) -> ViResult<()> {
163    // 1. Allowlist check — the range must fall inside a known device window
164    //    AND that window's device class must be one the cell declared.
165    let end = base.checked_add(len).ok_or(ViError::InvalidInput)?;
166
167    // PCIe path: validate against the dynamic BAR table populated by pcie_ecam.
168    let in_allowlist = if allowed_devices & DEV_PCIE != 0 {
169        let bars = PCIE_BARS.lock();
170        bars.get(&base).is_some_and(|&bar_len| len <= bar_len)
171    } else {
172        // GPIO/UART path: static per-arch allowlist.
173        ALLOWED.iter().any(|&(ab, al, class)| {
174            let ae = ab + al;
175            base >= ab && end <= ae && (class & allowed_devices != 0)
176        })
177    };
178    if !in_allowlist {
179        return Err(ViError::PermissionDenied);
180    }
181
182    // 2. Overlap check — no two cells may share a byte
183    let mut reg = REGISTRY.lock();
184    for (&eb, &(el, _owner)) in reg.iter() {
185        let ee = eb + el;
186        // Ranges overlap when NOT (end ≤ eb OR base ≥ ee)
187        if !(end <= eb || base >= ee) {
188            return Err(ViError::AlreadyExists);
189        }
190    }
191
192    reg.insert(base, (len, cell_id));
193    Ok(())
194}

196/// Release all MMIO regions owned by `cell_id`.
197///
198/// Call this from every Cell-exit path (Exit syscall, ForceExit, fault, watchdog).
199pub fn release_for(cell_id: CellId) {
200    REGISTRY
201        .lock()
202        .retain(|_base, &mut (_len, owner)| owner != cell_id);
203}
204
205/// Return the task ID (TID) of the cell that currently owns the MMIO region
206/// whose base address exactly matches `base`.
207///
208/// Returns `None` if no cell has requested that exact base address.
209/// Used by the GPIO IRQ handler to route interrupts to the current MMIO owner.
210pub fn lookup_mmio_owner(base: usize) -> Option<usize> {
211    REGISTRY
212        .lock()
213        .get(&base)
214        .map(|&(_len, cell_id)| cell_id.0 as usize)
215}
216
217/// Current number of registered regions (diagnostics).
218pub fn region_count() -> usize {
219    REGISTRY.lock().len()
220}

222// ---------------------------------------------------------------------------
223// PCIe BDF ownership (for sys_grant_dma authorization)
224// ---------------------------------------------------------------------------

226/// Maps PCIe BDF → owning task ID.
227///
228/// Kernel drivers (NIC, NVMe) are not registered here — they bypass `sys_grant_dma`
229/// and call `iommu::map_dma_for_cell(0, bdf, ...)` directly during init.
230/// Only userspace Driver Cells that receive PCIe device ownership via capability
231/// delegation need to register here.
232static BDF_OWNERS: Spinlock<alloc::collections::BTreeMap<u32, usize>> =
233    Spinlock::new(alloc::collections::BTreeMap::new());
234
235/// Register a PCIe BDF as owned by task `tid`.
236///
237/// Called when a Driver Cell is granted ownership of a PCIe device.
238pub fn register_bdf_owner(bdf: u32, tid: usize) {
239    BDF_OWNERS.lock().insert(bdf, tid);
240}
241
242/// Return the task ID that currently owns `bdf`, or `None` if unowned.
243pub fn owner_of_bdf(bdf: u32) -> Option<usize> {
244    BDF_OWNERS.lock().get(&bdf).copied()
245}
246
247/// Release all BDF ownerships held by task `tid` (called on Cell exit).
248pub fn release_bdfs_for(tid: usize) {
249    BDF_OWNERS.lock().retain(|_bdf, &mut owner| owner != tid);
250}

252/// Force-unlock BDF_OWNERS during fault teardown.
253///
254/// # Safety
255/// See `force_unlock_locks` above — same contract.
256pub unsafe fn force_unlock_bdf_locks() {
257    BDF_OWNERS.force_unlock();
