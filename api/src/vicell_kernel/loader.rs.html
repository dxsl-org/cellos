6/// Tracks whether a block-I/O cell has registered the VFS fast-IPC handler pointer.
7/// Set to `true` on first registration; subsequent registrations (hot-swap path) log
8/// a warning and re-point the handler.  Never reset — warm boot / snapshot restore
9/// skips `spawn_from_path`, so re-registration never fires spuriously.

37/// Spawn a cell by reading its ELF from a filesystem path.
38///
39/// Resolution order:
40/// 1. If the early-boot cell table has been probed (via `early::EarlyLoader::probe`),
41///    reads the ELF directly from the block device at the known LBA.
42/// 2. Otherwise returns `ViError::NotFound` — the caller must ensure the early
43///    table is probed before calling `spawn_from_path` during bootstrapping.
44///
45/// After the ELF is loaded into memory, relocations are applied and the cell is
46/// enqueued via `crate::task::spawn_from_mem`.
47///
48/// # Errors
49/// - `ViError::NotFound` — path absent from the bootstrap table.
50/// - `ViError::InvalidInput` — malformed ELF or unsupported relocation.
51/// - `ViError::OutOfMemory` — cannot allocate frames for segments.
52///   Legacy hardcoded path grants for cells lacking a `__ViCell_manifest`.
53///   Mirrors the pre-manifest behavior; only `/bin/` paths gain privilege. The
54///   returned set is still subject to spawner intersection in `spawn_from_path`.
55fn legacy_path_caps(path: &str) -> crate::task::cap::CapSet {
56    let mut c = crate::task::cap::CapSet::EMPTY;
57    if path.starts_with("/bin/") {
58        if path.ends_with("/bin/vfs") {
59            c.block_io = true;
60            c.block_regions = 0b11; // legacy: P1 + P4 (pre-P03, no SRV bit)
61        }
62        if path.ends_with("/bin/net") {
63            c.network = true;
64        }
65        if path.ends_with("/bin/shell") || path.ends_with("/bin/init") {
66            c.spawn = true;
67        }
68    }
69    c
70}

72/// Spawn a cell from a filesystem path. `spawner` sets the capability ceiling:
73/// `Root` (boot/init) grants the full manifest; `User(tid)`/`Ceiling(caps)`
74/// intersect the manifest with the spawner's caps (P2 monotonic downgrade).
75pub fn spawn_from_path(path: &str, spawner: crate::task::cap::Spawner) -> ViResult<usize> {
76    // Validate path: non-empty, leading slash, bounded length, no traversal sequences.
77    // Reject '..' and '//' to prevent a future VFS-backed spawn from escaping /bin/
78    // via a /bin/-prefixed traversal path (defense-in-depth; currently harmless since
79    // the early loader uses exact-match, but cheap to enforce here unconditionally).
80    if path.is_empty()
81        || !path.starts_with('/')
82        || path.len() > disk_layout::MAX_CELL_PATH
83        || path.contains("..")
84        || path.contains("//")
85    {
86        log::error!("[loader] invalid path {:?}", path);
87        return Err(ViError::InvalidInput);
88    }
89
90    log::info!("[loader] SpawnFromPath: {}", path);
91
92    // Read ELF bytes from the early bootstrap table.
93    let elf_bytes = early::EarlyLoader::read_file(path)?;
94    spawn_gated(&elf_bytes, path, spawner)
95}

97/// Verify and spawn a cell from ELF bytes already resident in memory.
98///
99/// Shared by `spawn_from_path` (bytes from the boot ramdisk / P2 table) and the
100/// `SpawnFromElf` syscall (bytes a userspace spawner read from VFS). The trust
101/// gate — Ed25519 signature, manifest-privilege check, capability intersection
102/// (P2 downgrade), operator policy, syscall allowlist, cluster membership,
103/// memory quota, and integrity measurement — is applied to the BYTES, so the
104/// source of the bytes is irrelevant to trust. `path` is advisory: it drives the
105/// `/bin/` privilege check, policy lookup, measurement label, and the path-based
106/// capability grants below. A caller lying about `path` can only LOSE privilege
107/// (a non-`/bin/` hint yields the user ceiling), never gain it.
108pub fn spawn_gated(
109    elf_bytes: &[u8],
110    path: &str,
111    spawner: crate::task::cap::Spawner,
112) -> ViResult<usize> {
113    // ── Binary signature gate ─────────────────────────────────────────────────
114    // Verify the Ed25519 signature in __ViCell_sig before any ELF parsing or
115    // task creation. With `signing-required`, an absent signature is treated
116    // the same as an invalid one (fail-closed). In dev mode (default), an
117    // absent signature is permitted so unsigned dev cells keep working.
118    match crate::signing::extract_sig(elf_bytes) {
119        Some(sig) => {
120            if !crate::signing::verify_cell(elf_bytes, &sig) {
121                log::warn!("[loader] DENY {:?}: cell signature INVALID", path);
122                crate::audit::log_event(
123                    crate::audit::AuditEvent::CellSignatureFailed,
124                    &crate::audit::encode_u32x2(0, 0),
125                );
126                return Err(ViError::PermissionDenied);
127            }
128            crate::audit::log_event(
129                crate::audit::AuditEvent::CellSignatureVerified,
130                &crate::audit::encode_u32x2(0, 0),
131            );
132        }
133        None if crate::signing::signing_required() => {
134            log::warn!(
135                "[loader] DENY {:?}: no __ViCell_sig (signing-required)",
136                path
137            );
138            crate::audit::log_event(
139                crate::audit::AuditEvent::CellSignatureFailed,
140                &crate::audit::encode_u32x2(0, 0),
141            );
142            return Err(ViError::PermissionDenied);
143        }
144        None => {
145            // Dev mode: unsigned cell permitted.
146        }
147    }
148
149    let elf_loader = ElfLoader;
150
151    // Read capability manifest from `__ViCell_manifest` ELF section.
152    // Absent or malformed → None (falls back to legacy hardcoded path grants).
153    let manifest_opt: Option<api::manifest::CellManifest> =
154        match elf_loader.get_section(elf_bytes, "__ViCell_manifest") {
155            Ok(bytes) => api::manifest::CellManifest::from_bytes(bytes),
156            Err(_) => None,
157        };
158
159    // Privilege gate: a user Cell (path NOT under /bin/) may NOT declare any
160    // privileged capability.  Runs BEFORE spawn_from_mem — no task is created
161    // for a rejected Cell.
162    if let Some(ref m) = manifest_opt {
163        if !path.starts_with("/bin/") && m.declares_any_privilege() {
164            log::error!(
165                "[loader] DENY spawn {:?}: user cell over-declares caps (flags={:#04x})",
166                path,
167                m.flags
168            );
169            crate::audit::log_event(
170                crate::audit::AuditEvent::CellSpawnDenied,
171                &crate::audit::encode_u32x2(m.flags as u32, 0u32),
172            );
173            return Err(ViError::PermissionDenied);
174        }
175    }
176
177    // Extract cell name from the last path component (e.g. "/bin/shell" → "shell").
178    let name = path.rsplit('/').next().unwrap_or(path);
179
180    // Spawn via the in-memory path.  spawn_from_mem now applies .rela.dyn
181    // relocations internally, so no separate apply_relocations call is needed.
182    let (tid, _load_base) =
183        crate::task::spawn_from_mem(elf_bytes, name, CellId(0), alloc::vec::Vec::new())
184            .map_err(|_| ViError::OutOfMemory)?;
185
186    // Assign a unique CellId based on the task ID so per-cell quota and
187    // capability checks are correctly scoped.  `spawn_from_mem` defaults to
188    // CellId(0) (kernel), which would make every path-spawned cell share the
189    // kernel's quota slot (charge() short-circuits for cell_id == 0).
190    let cell_id = CellId(tid as u64);
191    if let Some(sched) = crate::task::SCHEDULER.lock().as_mut() {
192        if let Some(task) = sched.tasks.get_mut(&tid) {
193            task.cell_id = cell_id;
194        }
195    }
196
197    crate::audit::log_event(
198        crate::audit::AuditEvent::CellSpawn,
199        &crate::audit::encode_u32x2(tid as u32, 0u32),
200    );
201
202    // Integrity measurement (IMA-style): hash the ELF image and record it in the
203    // append-only measurement log BEFORE the cell is scheduled. Evidence for
204    // future DICE/EAT attestation — orthogonal to (and complements) Cell signing.
205    crate::measurement_log::measure(tid, path, elf_bytes);
206
207    // Read per-Cell syscall allowlist from ELF section __ViCell_syscalls.
208    // The section (if present) contains a u64 LE bitset; absent = permit-all.
209    {
210        let allowlist = match ElfLoader.get_section(elf_bytes, "__ViCell_syscalls") {
211            Ok(bytes) if bytes.len() >= 8 => {
212                // SAFETY: bytes slice is valid data from the loaded ELF.
213                u64::from_le_bytes(
214                    bytes[..8]
215                        .try_into()
216                        .expect("8-byte __ViCell_syscalls section"),
217                )
218            }
219            _ => u64::MAX, // no section → permit-all (backwards compatible)
220        };
221        if let Some(sched) = crate::task::SCHEDULER.lock().as_mut() {
222            if let Some(task) = sched.tasks.get_mut(&tid) {
223                task.syscall_allowlist = allowlist;
224            }
225        }
226    }
227
228    // Read cluster membership from ELF section __ViCell_cluster.
229    // Layout: u8 mode, u8 pad[7], u64 cluster_id (LE) = 16 bytes.
230    // Absent section → Isolated mode (mode=0, cluster_id=0); backwards compatible.
231    {
232        let (mode, cid) = match ElfLoader.get_section(elf_bytes, "__ViCell_cluster") {
233            Ok(bytes) if bytes.len() >= 16 => {
234                // SAFETY: bytes slice is valid data from the loaded ELF.
235                let mode = bytes[0];
236                let cid = u64::from_le_bytes(bytes[8..16].try_into().expect("8-byte cluster_id"));
237                (mode, cid)
238            }
239            _ => (0u8, 0u64), // no section → Isolated, no cluster
240        };
241        if let Some(sched) = crate::task::SCHEDULER.lock().as_mut() {
242            if let Some(task) = sched.tasks.get_mut(&tid) {
243                task.cluster_mode = mode;
244                task.cluster_id = cid;
245            }
246        }
247    }
248
249    // Register per-cell memory quota (4 MiB default) using the real CellId.
250    crate::memory::cell_quota::register(cell_id, crate::memory::cell_quota::DEFAULT_QUOTA_BYTES);
251
252    // ── Capability grant (P2 — spawn-time monotonic downgrade) ───────────────
253    // 1. `requested` = caps the manifest declares (absent → legacy path grants).
254    // 2. `granted`   = requested ∩ spawner-ceiling — a cell cannot hand a child a
255    //    cap it does not itself hold. `init` (Spawner::Root) is the root authority
256    //    and is exempt; HotSwap passes the replaced cell's caps as the ceiling.
257    use crate::task::cap::{CapSet, Spawner};
258    // `requested` = manifest (or legacy) caps, PLUS the privileged path-triggered
259    // caps (pcie_driver/platform/supervisor) layered on by `with_path_caps`. Folding
260    // them into `requested` means the single ceiling intersection below governs them
261    // too — closing the pre-P-TRUST hole where they were minted by a raw path match
262    // after (and blind to) the intersection.
263    let requested: CapSet = match manifest_opt {
264        Some(ref m) => CapSet::from_manifest(m),
265        None => legacy_path_caps(path),
266    }
267    .with_path_caps(path);
268    // Snapshot the spawner's caps in its OWN lock scope; the guard is DROPPED
269    // before the child-mutation lock below (the Spinlock is non-reentrant).
270    let after_spawner: CapSet = match spawner {
271        Spawner::Root => requested,
272        Spawner::Ceiling(ceil) => requested.intersect(ceil),
273        Spawner::User(stid) => {
274            let ceil = crate::task::SCHEDULER
275                .lock()
276                .as_ref()
277                .and_then(|s| s.tasks.get(&stid))
278                .map(|t| CapSet::of_task(t))
279                .unwrap_or(CapSet::EMPTY); // unknown spawner → no caps (fail-safe)
280            requested.intersect(ceil)
281        }
282    };
283    // 3. Operator policy (P5/Phase 04): `granted = after_spawner ∩ policy(path)`,
284    //    with trusted-core recovery + fail-safe. `policy::apply` takes the POLICY
285    //    lock internally — called OUTSIDE the SCHEDULER guard above to avoid lock
286    //    nesting. `init` (Root) is exempt (it is the loader OF the policy).
287    let granted: CapSet = match spawner {
288        Spawner::Root => after_spawner,
289        _ => crate::policy::apply(path, tid, after_spawner),
290    };
291
292    if let Some(sched) = crate::task::SCHEDULER.lock().as_mut() {
293        if let Some(task) = sched.tasks.get_mut(&tid) {
294            granted.apply_to(task);
295
296            // x86_64 PKU: derive the protection-key domain from the granted caps
297            // AND the manifest's declared tier (Manifest v2). See
298            // `cap::granted_tier` for the floor-not-ceiling invariant this encodes.
299            // Tiers map 1:1 onto PKU keys 0-3 (TRUSTED_CORE=0 .. UNTRUSTED=3);
300            // `pkru_for_key` handles any key in range generically.
301            // On non-x86_64 targets these fields default to 0 and are never consulted.
302            #[cfg(target_arch = "x86_64")]
303            {
304                let requested_tier = manifest_opt
305                    .as_ref()
306                    .map(|m| m.tier())
307                    .unwrap_or(api::manifest::TIER_LEGACY);
308                let key: u8 = crate::task::cap::granted_tier(&granted, requested_tier);
309                task.pku_key = key;
310                task.pku_value = crate::hal::pku::pkru_for_key(key);
311            }
312
313            // Privileged path-caps (PcieDriverCap / SupervisorCap) are now written by
314            // `granted.apply_to(task)` above — they ride in the CapSet and are bounded
315            // by the ceiling intersection (P-TRUST). Only the two exceptions remain here:
316            //
317            // 1. PlatformCap — ceiling-gated via `granted.platform`, THEN the one-holder
318            //    singleton latch (`try_grant_platform`). It is not written by `apply_to`
319            //    because a plain bool cannot enforce the singleton.
320            if granted.platform {
321                match crate::task::cap::try_grant_platform() {
322                    Some(cap) => task.platform_cap = Some(cap),
323                    None => {
324                        log::error!("[loader] PlatformCap already granted — refusing 2nd /bin/platform spawn");
325                        return Err(types::ViError::PermissionDenied);
326                    }
327                }
328            }
329            // 2. VFS cell-store block region (0b1000) — DEFERRED post-policy raw grant.
330            //    Folding it into the ceiling (like the caps above) would be zeroed by
331            //    the `/bin/vfs` operator-policy entry (`block_regions = 0b111`), breaking
332            //    VFS. Closing this channel needs a POLICY.BIN re-bake that grants vfs
333            //    0b1111; tracked as a P-TRUST follow-up. Lower severity than the caps
334            //    above (partition access, not DMA-anywhere).
335            if path == "/bin/vfs" {
336                task.block_regions |= 0b1000;
337            }
338        }
339    }
340
341    // Side effect keyed off the GRANTED (not requested) block-io bit: the VFS
342    // fast-IPC handler must point at whoever actually received block_io.
343    if granted.block_io {
344        // Re-registration is valid on VFS hot-swap; just re-point the handler.
345        let already = BLOCK_IO_REGISTERED.swap(true, Ordering::SeqCst);
346        if already {
347            log::warn!("[loader] block_io re-registration — VFS hot-swap or second block_io cell");
348        }
349        crate::fast_ipc::set_vfs_handler_cell(cell_id.0 as usize);
350    }
351    // Register the input service endpoint so console_drv can route UART bytes to it.
352    // (Service-registry registration is done by init via sys_register_service.)
353    if path.ends_with("/bin/input") {
354        crate::task::drivers::driver_cell::set_input_cell(tid);
355    }
356    Ok(tid)
357}
