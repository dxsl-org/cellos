1//! Kernel audit ring buffer — 256 KB in-memory event log.
2//!
3//! Records key kernel events (IPC, file, net, spawn, fault, exit) with a
4//! monotonic timestamp for post-mortem analysis.  A background Cell (or shell
5//! builtin) drains the ring and appends to `/data/kernel.log`.
6//!
7//! # Concurrency
8//! The timer ISR may preempt a syscall-context `log_event()` call, creating
9//! two apparent producers.  We guard each write by briefly disabling S-mode
10//! interrupts (`csrci sstatus, 0x2`) — safe on single-hart.
11//!
12//! # Overflow
13//! When the ring is full, new writes are dropped and `DROPPED` is incremented.
14//! No blocking, no corruption.
15//!
16//! # Record format
17//! ```
18//! [u64 mtime_ticks LE][u8 event_type][u8 payload_len][payload…]
19//! ```
20//! Total minimum: 10 bytes per event.
21
22use core::cell::UnsafeCell;
23use core::sync::atomic::{AtomicUsize, Ordering};
24
25const BUF_SIZE: usize = 256 * 1024; // must be a power of two
26const MASK: usize = BUF_SIZE - 1;
27
28/// Kernel audit event type byte.
29#[repr(u8)]
30#[allow(dead_code)] // reason: variants logged by different kernel subsystems
31pub enum AuditEvent {
32    IpcSend = 1,
33    IpcRecv = 2,
34    FileOpen = 3,
35    FileWrite = 4,
36    NetTx = 5,
37    NetRx = 6,
38    CellSpawn = 7,
39    CellFault = 8,
40    CellExit = 9,
41    CellSpawnDenied = 10,
42    /// An RT-priority cell's `RecvTimeout` deadline elapsed before its awaited
43    /// message arrived — a missed control-loop cycle. Payload: `encode_u32x2(cell_id,
44    /// cumulative_miss_count)`. Observability only (no enforcement): makes RT misses
45    /// visible for post-mortem / tuning once real-hardware bench data is available.
46    RtDeadlineMiss = 11,
47    /// An RT-priority cell crossed the CPU-monopoly *warning* threshold (a fraction
48    /// of the watchdog budget) without yielding — an early signal BEFORE the hard
49    /// watchdog kill. Payload: `encode_u32x2(cell_id, run_ticks)`. One-shot per episode.
50    RtCpuOverrun = 12,
51    /// A cell that opted into liveness heartbeating missed its deadline — a silent hang
52    /// (deadlock / stuck loop) the CPU watchdog cannot detect. The kernel terminates it
53    /// for supervised restart. Payload: `encode_u32x2(cell_id, tid)`.
54    CellHung = 13,
55    /// A cell invoked a syscall not present in its `__ViCell_syscalls` allowlist.
56    /// Payload: `encode_u32x2(caller_tid, allowlist_bit)`.
57    SyscallDenied = 14,
58    /// A cell's ELF image was measured at spawn (integrity measurement, IMA-style).
59    /// Payload: `encode_u32x2(tid, sha256_prefix_le_u32)`. The full digest and the
60    /// rolling aggregate live in [`crate::measurement_log`].
61    CellMeasure = 15,
62    /// The signed operator policy was loaded + verified at boot (P5).
63    /// Payload: `encode_u32x2(entry_count, 0)`.
64    PolicyLoaded = 16,
65    /// The operator policy failed signature verification or parsing — fail-closed.
66    /// Payload: `encode_u32x2(reason_code, 0)`.
67    PolicyInvalid = 17,
68    /// No operator policy present in VIFS1. Payload: `encode_u32x2(0, 0)`.
69    PolicyAbsent = 18,
70    /// Operator policy narrowed a cell's spawn-time caps. Payload:
71    /// `encode_u32x2(tid, dropped_flags)` (dropped_flags: bit0 block_io, bit1
72    /// network, bit2 spawn, bit3 hypervisor).
73    CapNarrowedByPolicy = 19,
74    /// Runtime revocation: a SpawnCap holder stripped capabilities from a live cell.
75    /// Payload: `encode_u32x2(target_tid, cap_mask)` where `cap_mask` matches
76    /// `api::syscall::cap_mask` constants (bit0=block_io, bit1=network, bit2=spawn,
77    /// bit3=hypervisor, bits8-15=mmio_devices, bits16-23=block_regions).
78    CapRevoked = 20,
79    /// Cell binary signature verified at spawn (binary integrity gate).
80    /// Payload: `encode_u32x2(tid, 0)`.
81    CellSignatureVerified = 21,
82    /// Cell binary signature absent or invalid — spawn denied.
83    /// Payload: `encode_u32x2(tid, 0)`. Path is logged separately via `log::warn!`.
84    CellSignatureFailed = 22,
85}

87struct AuditRing {
88    buf: UnsafeCell<[u8; BUF_SIZE]>,
89    head: AtomicUsize,
90    tail: AtomicUsize,
91    dropped: AtomicUsize,
92}
93
94// SAFETY: single-hart kernel; interrupt-disable in log_event prevents
95// timer ISR from preempting a partial write.
96unsafe impl Sync for AuditRing {}
97
98static RING: AuditRing = AuditRing {
99    // SAFETY: UnsafeCell<[u8; N]> in a static is always zero-initialised.
100    buf: UnsafeCell::new([0u8; BUF_SIZE]),
101    head: AtomicUsize::new(0),
102    tail: AtomicUsize::new(0),
103    dropped: AtomicUsize::new(0),
104};
105
106/// Write a kernel audit event to the ring.
107///
108/// Disables S-mode interrupts for the duration of the write so the timer ISR
109/// cannot preempt a partial record.  Drops silently on ring-full.
110pub fn log_event(event: AuditEvent, payload: &[u8]) {
111    let plen = payload.len().min(255) as u8;
112    let record_len = 10 + plen as usize;
113
114    // Disable S-mode interrupts: prevent timer ISR from racing this write.
115    // SAFETY: single-hart; restoring sie after the write preserves the invariant.
116    let sie_was_set = {
117        #[cfg(target_arch = "riscv64")]
118        {
119            let v: usize;
120            // SAFETY: csrrci clears SIE (bit 1) and returns old sstatus.
121            unsafe { core::arch::asm!("csrrci {}, sstatus, 0x2", out(reg) v) };
122            v & 0x2 != 0
123        }
124        #[cfg(not(target_arch = "riscv64"))]
125        {
126            false
127        }
128    };
129
130    let head = RING.head.load(Ordering::Relaxed);
131    let tail = RING.tail.load(Ordering::Acquire);
132
133    // Drop-on-full: never overwrite consumer bytes.
134    if head.wrapping_sub(tail) + record_len > BUF_SIZE {
135        RING.dropped.fetch_add(1, Ordering::Relaxed);
136        restore_sie(sie_was_set);
137        return;
138    }
139
140    #[cfg(target_arch = "riscv64")]
141    let mtime = hal::common::timer::read_mtime().to_le_bytes();
142    #[cfg(not(target_arch = "riscv64"))]
143    let mtime = 0u64.to_le_bytes();
144    let buf = unsafe { &mut *RING.buf.get() };
145    let mut pos = head;
146    for &b in mtime
147        .iter()
148        .chain(core::slice::from_ref(&(event as u8)))
149        .chain(core::slice::from_ref(&plen))
150        .chain(&payload[..plen as usize])
151    {
152        buf[pos & MASK] = b;
153        pos = pos.wrapping_add(1);
154    }
155
156    // Publish the write — consumer sees it only after this Release store.
157    RING.head
158        .store(head.wrapping_add(record_len), Ordering::Release);
159
160    restore_sie(sie_was_set);
161}
162
163#[inline(always)]
164fn restore_sie(was_set: bool) {
165    if was_set {
166        #[cfg(target_arch = "riscv64")]
167        // SAFETY: restoring SIE to its prior state.
168        unsafe {
169            core::arch::asm!("csrsi sstatus, 0x2");
170        }
171    }
172}

174/// Drain up to `out.len()` bytes from the ring.  Returns bytes copied.
175///
176/// Called by the log-flusher Cell (consumer side).
177pub fn drain(out: &mut [u8]) -> usize {
178    let head = RING.head.load(Ordering::Acquire);
179    let tail = RING.tail.load(Ordering::Relaxed);
180    let available = head.wrapping_sub(tail);
181    if available == 0 {
182        return 0;
183    }
184
185    let to_copy = available.min(out.len());
186    let buf = unsafe { &*RING.buf.get() };
187    for (i, byte) in out[..to_copy].iter_mut().enumerate() {
188        *byte = buf[(tail.wrapping_add(i)) & MASK];
189    }
190    RING.tail
191        .store(tail.wrapping_add(to_copy), Ordering::Release);
192    to_copy
193}
194
195/// Number of records dropped due to ring-full since boot.
196pub fn dropped_count() -> usize {
197    RING.dropped.load(Ordering::Relaxed)
198}

202/// Encode two `u32` values into a fixed 8-byte payload.
203pub fn encode_u32x2(a: u32, b: u32) -> [u8; 8] {
204    let mut out = [0u8; 8];
205    out[..4].copy_from_slice(&a.to_le_bytes());
206    out[4..].copy_from_slice(&b.to_le_bytes());
207    out
208}
