1//! Per-Cell integrity measurement log (Linux IMA-style, TPM-free).
2//!
3//! On every `spawn_from_path`, the loader records `SHA-256(elf_bytes)` here
4//! BEFORE the cell is scheduled to run. Entries form an append-only log plus a
5//! rolling aggregate (`agg = SHA256(agg || entry_hash)`) — the single value a
6//! future DICE/EAT remote-attestation token signs to prove the exact software
7//! that ran on this device. See
8//! `docs/research/research-cell-security-permissions.md` §3.6.
9//!
10//! This is *measurement* (evidence), not *enforcement*: it does not block a
11//! cell. Signature-based rejection (Cell binary signing) is a separate,
12//! orthogonal gate. Pairing the two gives "measured + verified launch".
13
14use crate::sync::Spinlock;
15use alloc::string::String;
16use alloc::vec::Vec;
17
18/// Soft cap on retained entries — bounds kernel memory. Beyond this individual
19/// entries stop being appended (the aggregate still advances, so attestation
20/// integrity is preserved) and a warning is logged. Cell spawns are bounded in
21/// practice on G1.
22const MAX_ENTRIES: usize = 256;
23
24/// One measured cell: its TID, the SHA-256 of its ELF image, and its path.
25pub struct MeasureEntry {
26    pub tid: u32,
27    pub hash: [u8; 32],
28    pub path: String,
29}
30
31struct Log {
32    entries: Vec<MeasureEntry>,
33    aggregate: [u8; 32],
34}
35
36static LOG: Spinlock<Log> = Spinlock::new(Log {
37    entries: Vec::new(),
38    aggregate: [0u8; 32],
39});
40
41/// Force-release this module's lock during fault teardown.
42///
43/// # Safety
44/// Single-hart kernel; called only from the fault/panic path with interrupts
45/// disabled. Force-unlocking an already-free Spinlock is a no-op.
46pub unsafe fn force_unlock_locks() {
47    LOG.force_unlock();
48}
49
50/// Measure a cell's ELF image: hash it, append to the log, extend the aggregate,
51/// and emit a `CellMeasure` audit event. Returns the digest.
52///
53/// Call BEFORE the cell is scheduled (the loader does this right after spawn,
54/// while still single-threaded in kernel context — the cell cannot have run yet).
55pub fn measure(tid: usize, path: &str, elf: &[u8]) -> [u8; 32] {
56    let hash = crate::sha256::sha256(elf);
57
58    {
59        let mut log = LOG.lock();
60        // Extend the aggregate: agg = SHA256(agg || hash).
61        let mut buf = [0u8; 64];
62        buf[..32].copy_from_slice(&log.aggregate);
63        buf[32..].copy_from_slice(&hash);
64        log.aggregate = crate::sha256::sha256(&buf);
65
66        if log.entries.len() < MAX_ENTRIES {
67            log.entries.push(MeasureEntry {
68                tid: tid as u32,
69                hash,
70                path: String::from(path),
71            });
72        } else {
73            log::warn!(
74                "[measure] log full ({} entries) — aggregate still advancing",
75                MAX_ENTRIES
76            );
77        }
78    }
79
80    // Audit: tid + first 4 hash bytes for quick correlation (full hash lives here).
81    let hp = u32::from_le_bytes([hash[0], hash[1], hash[2], hash[3]]);
82    crate::audit::log_event(
83        crate::audit::AuditEvent::CellMeasure,
84        &crate::audit::encode_u32x2(tid as u32, hp),
85    );
86    hash
87}
88
89/// Rolling aggregate over every measured cell (for future remote attestation).
90pub fn aggregate() -> [u8; 32] {
91    LOG.lock().aggregate
92}
93
94/// Number of entries currently retained (diagnostics).
95pub fn entry_count() -> usize {
96    LOG.lock().entries.len()
97}
