1//! Heap snapshot serialization and warm-boot restore for Instant On.
2//!
3//! Saves all allocated physical frames to a reserved sector range on the VirtIO
4//! block device.  On the next boot, `try_restore()` replays those frames back
5//! to their original physical addresses — skipping ELF loading, heap init, and
6//! scheduler setup — for a dramatically faster boot.
7//!
8//! # Architecture constraint
9//! ALL allocated frames must be snapshotted, including kernel `.bss`/`.data`
10//! pages that hold global statics (`SCHEDULER`, `BLOCK_DEVICE`, etc.).  BSS is
11//! re-zeroed at every cold boot, so without these pages the scheduler would be
12//! `None` when `try_restore()` calls `yield_cpu()`.
13//!
14//! # Performance characterization (Phase 29-4 benchmark)
15//!
16//! | Platform          | Throughput  | 4 MB snapshot | 8 MB snapshot |
17//! |-------------------|-------------|---------------|---------------|
18//! | QEMU TCG          | ~30 MB/s    | ~133 ms       | ~266 ms       |
19//! | StarFive VF2 eMMC | ~100+ MB/s  | ~40 ms ✓      | ~80 ms ✓      |
20//! | QEMU + /dev/shm   | ~2000 MB/s  | ~2 ms ✓       | ~4 ms ✓       |
21//!
22//! Sub-100 ms is a **real-hardware claim** (eMMC 100+ MB/s).  QEMU TCG
23//! emulates VirtIO at ~30 MB/s — use `/dev/shm`-backed disk for QEMU sub-100ms.
24//! Timing is logged by both `serialize_snapshot()` and `try_restore()`:
25//!   `[snapshot] warm boot: N frames restored in X ms`
26
27use crate::memory::frame::FRAME_ALLOCATOR;
28use crate::task::drivers::block;

30/// Reserved LBA range for snapshot storage in `disk_v3.img` — MBR partition P3.
31/// Sector 0 = 48-byte header; sectors 1+ = allocated frame data (8 sectors/frame).
32/// The old hardcoded 200_000 sat INSIDE the FAT32 data area and would corrupt
33/// /data files past ~100 MB; P3 is a dedicated raw region (disk_layout.rs).
34pub const SNAPSHOT_BASE_LBA: u64 = crate::loader::disk_layout::PART_SNAPSHOT_BASE_LBA;
35
36/// Snapshot format version — increment on breaking header layout changes.
37pub const SNAPSHOT_FORMAT_VERSION: u16 = 1;
38
39/// Magic bytes identifying a ViCell snapshot image (little-endian `VICU`).
40pub const SNAPSHOT_MAGIC: u32 = 0x5543_4956; // 'U','C','I','V' as bytes on disk
41
42/// Git SHA short hash baked in at compile time.  Snapshot is invalid if this
43/// changes (i.e., the kernel was recompiled since the snapshot was taken).
44const KERNEL_GIT_SHA: &str = env!("VERGEN_GIT_SHA");
45
46/// Parse the first 8 hex chars of the git SHA into a u64.
47fn kernel_hash() -> u64 {
48    let s = KERNEL_GIT_SHA.trim();
49    let end = s.len().min(8);
50    u64::from_str_radix(&s[..end], 16).unwrap_or(0)
51}

53/// 48-byte snapshot header at the start of the snapshot sector.
54#[repr(C)]
55pub struct SnapshotHeader {
56    /// Magic: 0x5543_4956 ("UCIV" on disk = "VICU" LE).
57    pub magic: u32,
58    /// Format version; cold boot if this doesn't match `SNAPSHOT_FORMAT_VERSION`.
59    pub version: u16,
60    /// Reserved flags.
61    pub flags: u16,
62    /// Kernel git SHA (first 8 hex chars → u64).  Invalidates on rebuild.
63    pub kernel_hash: u64,
64    /// Physical start of the snapshotted region.
65    pub pa_base: u64,
66    /// Physical end (exclusive) of the snapshotted region.
67    pub pa_end: u64,
68    /// Number of 4096-byte frames stored (allocated frames only).
69    pub frame_count: u32,
70    /// CRC32 of (header with this field = 0) + all frame data.
71    pub crc32: u32,
72}
73
74// Compile-time size guarantee — SnapshotHeader must fit in one 512-byte sector.
75// Layout: u32+u16+u16+u64+u64+u64+u32+u32 = 4+2+2+8+8+8+4+4 = 40 bytes.
76const _: () = assert!(core::mem::size_of::<SnapshotHeader>() == 40);

78// ── Serialization ────────────────────────────────────────────────────────────
79
80/// Serialize all allocated physical frames to the reserved disk sector range.
81///
82/// Writes allocated-only frames (excludes free frames) to reduce snapshot size.
83/// The snapshot includes kernel `.bss`/`.data` pages — required so that global
84/// statics like `SCHEDULER` survive the warm-boot restore path.
85///
86/// Returns the number of frames written on success.
87///
88/// # Safety constraints
89/// Must be called with all cells quiesced (at a `yield_cpu()` point) so no
90/// task stack is mid-function-call when the memory image is frozen.
91pub fn serialize_snapshot() -> Result<u32, &'static str> {
92    #[cfg(target_arch = "riscv64")]
93    let t0 = hal::common::timer::read_mtime();
94    #[cfg(not(target_arch = "riscv64"))]
95    let _t0 = 0u64;
96
97    let guard = FRAME_ALLOCATOR.lock();
98    let allocator = guard.as_ref().ok_or("frame allocator not initialized")?;
99
100    let pa_base = allocator.memory_start();
101    let pa_end = allocator.memory_end();
102    let total = allocator.total_frames();
103
104    let mut hasher = crc32fast::Hasher::new();
105    let mut current_lba = SNAPSHOT_BASE_LBA + 1; // sector 0 reserved for header
106    let mut frame_count = 0u32;
107
108    // Walk every frame; write only allocated ones.
109    // We hold the frame-allocator lock ONLY to enumerate — we do NOT hold it
110    // across the VirtIO write (which would mask timer interrupts for seconds).
111    // Instead: collect allocated frame addresses first, then write without lock.
112    // For Phase 29 MVP: hold the lock across writes (single-threaded kernel, safe).
113    for frame_idx in 0..total {
114        if !allocator.is_frame_allocated(frame_idx) {
115            continue;
116        }
117
118        let pa = allocator.frame_addr(frame_idx);
119        // Exclude MMIO: any physical address below RAM base 0x80000000.
120        // Correct kernels should never allocate MMIO frames, but guard anyway.
121        if pa < 0x8000_0000 {
122            continue;
123        }
124
125        // Write 8 × 512-byte sectors per 4096-byte frame.
126        for sector_offset in 0..8usize {
127            let byte_offset = sector_offset * 512;
128            let mut buf = [0u8; 512];
129            // SAFETY: `pa` is a valid allocated RAM frame within [pa_base, pa_end).
130            // The frame allocator bitmap confirmed it is in use.  We are single-hart
131            // and the scheduler is quiesced — no concurrent writes to this frame.
132            unsafe {
133                let src = (pa + byte_offset) as *const u8;
134                core::ptr::copy_nonoverlapping(src, buf.as_mut_ptr(), 512);
135            }
136            hasher.update(&buf);
137            block::write_sector(current_lba, &buf)
138                .map_err(|_| "write_sector failed during snapshot")?;
139            current_lba += 1;
140        }
141        frame_count += 1;
142    }
143    drop(guard); // release frame allocator lock before header write
144
145    // Compute final CRC and write header to sector 0 of snapshot region.
146    let final_crc = hasher.finalize();
147    let header = SnapshotHeader {
148        magic: SNAPSHOT_MAGIC,
149        version: SNAPSHOT_FORMAT_VERSION,
150        flags: 0,
151        kernel_hash: kernel_hash(),
152        pa_base: pa_base as u64,
153        pa_end: pa_end as u64,
154        frame_count,
155        crc32: final_crc,
156    };
157    let mut header_sector = [0u8; 512];
158    // SAFETY: SnapshotHeader is repr(C) 48 bytes; copying to a 512-byte buffer is safe.
159    unsafe {
160        core::ptr::copy_nonoverlapping(
161            &header as *const SnapshotHeader as *const u8,
162            header_sector.as_mut_ptr(),
163            core::mem::size_of::<SnapshotHeader>(),
164        );
165    }
166    block::write_sector(SNAPSHOT_BASE_LBA, &header_sector)
167        .map_err(|_| "write header sector failed")?;
168
169    #[cfg(target_arch = "riscv64")]
170    let elapsed_ms = (hal::common::timer::read_mtime().wrapping_sub(t0)) / 10_000;
171    #[cfg(not(target_arch = "riscv64"))]
172    let elapsed_ms = 0u64;
173
174    log::info!(
175        "[snapshot] wrote {} frames ({} KiB) in {} ms to LBA {}",
176        frame_count,
177        frame_count as usize * 4,
178        elapsed_ms,
179        SNAPSHOT_BASE_LBA
180    );
181    Ok(frame_count)
182}

184// ── Warm-boot restore ────────────────────────────────────────────────────────
185
186/// Attempt to restore the kernel from a previously written snapshot.
187///
188/// Returns `true` if warm boot succeeded — the caller must skip cold-boot
189/// cell initialization.  Returns `false` on any validation failure.
190///
191/// # Calling contract
192/// Must be called AFTER VirtIO block is initialized (needed for disk reads)
193/// and BEFORE `EarlyLoader::probe()` or `task::init()` (cells are about to be
194/// replaced by the restored task set).
195pub fn try_restore() -> bool {
196    // Read snapshot header.
197    let mut header_sector = [0u8; 512];
198    if block::read_sector(SNAPSHOT_BASE_LBA, &mut header_sector).is_err() {
199        log::info!("[snapshot] no block device → cold boot");
200        return false;
201    }
202
203    // SAFETY: header_sector is 512 bytes; SnapshotHeader is repr(C) 48 bytes;
204    // the first 48 bytes are cast as a header — no pointer provenance issues.
205    let header: &SnapshotHeader = unsafe { &*(header_sector.as_ptr() as *const SnapshotHeader) };
206
207    if header.magic != SNAPSHOT_MAGIC {
208        log::info!(
209            "[snapshot] no valid snapshot (magic {:08X}) → cold boot",
210            header.magic
211        );
212        return false;
213    }
214    if header.version != SNAPSHOT_FORMAT_VERSION {
215        log::info!(
216            "[snapshot] format version mismatch ({} != {}) → cold boot",
217            header.version,
218            SNAPSHOT_FORMAT_VERSION
219        );
220        return false;
221    }
222    if header.kernel_hash != kernel_hash() {
223        log::info!("[snapshot] kernel changed → cold boot (invalidating stale snapshot)");
224        invalidate_snapshot();
225        return false;
226    }
227
228    let frame_count = header.frame_count as usize;
229    let pa_base = header.pa_base as usize;
230    let saved_crc = header.crc32;
231
232    log::info!(
233        "[snapshot] valid: {} frames at PA 0x{:X} → verifying CRC…",
234        frame_count,
235        pa_base
236    );
237
238    // Verify CRC32 over (header with crc32 field = 0) + all frame sectors.
239    {
240        let mut hasher = crc32fast::Hasher::new();
241        let mut hdr_copy = header_sector;
242        // Zero the crc32 field for consistent hashing.
243        // Layout: magic(4)+version(2)+flags(2)+kernel_hash(8)+pa_base(8)+pa_end(8)+frame_count(4)+crc32(4)
244        // crc32 offset = 4+2+2+8+8+8+4 = 36.
245        hdr_copy[36..40].copy_from_slice(&[0u8; 4]);
246        hasher.update(&hdr_copy[..core::mem::size_of::<SnapshotHeader>()]);
247
248        let mut buf = [0u8; 512];
249        let crc_lba_end = SNAPSHOT_BASE_LBA + 1 + frame_count as u64 * 8;
250        for frame_lba in (SNAPSHOT_BASE_LBA + 1)..crc_lba_end {
251            if block::read_sector(frame_lba, &mut buf).is_err() {
252                log::warn!("[snapshot] read error during CRC verify → cold boot");
253                return false;
254            }
255            hasher.update(&buf);
256        }
257        if hasher.finalize() != saved_crc {
258            log::warn!("[snapshot] CRC32 mismatch → cold boot (corrupted snapshot)");
259            invalidate_snapshot();
260            return false;
261        }
262    }
263
264    log::info!("[snapshot] CRC ok → restoring {} frames…", frame_count);
265    #[cfg(target_arch = "riscv64")]
266    let t_restore_start = hal::common::timer::read_mtime();
267    #[cfg(not(target_arch = "riscv64"))]
268    let _t_restore_start = 0u64;
269
270    // Restore frames to their original physical addresses.
271    // This overwrites ALL of physical RAM including kernel .bss/.data — global
272    // statics (SCHEDULER, BLOCK_DEVICE, etc.) are restored to their runtime values.
273    let mut frame_lba = SNAPSHOT_BASE_LBA + 1;
274    for frame_idx in 0..frame_count {
275        let pa = pa_base + frame_idx * 4096;
276        let frame_ptr = pa as *mut u8;
277        for sector_offset in 0..8usize {
278            let mut buf = [0u8; 512];
279            if block::read_sector(frame_lba, &mut buf).is_err() {
280                log::error!("[snapshot] read failed at frame {} → cold boot", frame_idx);
281                return false;
282            }
283            // SAFETY: pa is within [pa_base, pa_end) confirmed by CRC.
284            // pa_base >= 0x80000000 (RAM base); single-hart with quiesced cells.
285            unsafe {
286                core::ptr::copy_nonoverlapping(
287                    buf.as_ptr(),
288                    frame_ptr.add(sector_offset * 512),
289                    512,
290                );
291            }
292            frame_lba += 1;
293        }
294    }
295
296    log::info!("[snapshot] frames restored → reinitializing hardware");
297
298    // Reinitialize hardware — MMIO registers reset on every power cycle.
299    // This MUST happen after frame restore because init_driver() writes device
300    // registers that were cleared by hardware reset.
301    #[cfg(target_arch = "riscv64")]
302    crate::hal::common::plic::init();
303
304    // Re-run VirtIO init: device registers were reset by power cycle, so the
305    // virtqueue / descriptor-ring state inside the restored BLOCK_DEVICE struct
306    // no longer matches device-side state.  init_driver() re-registers queues.
307    crate::task::drivers::init();
308
309    // Re-arm the scheduler timer.
310    #[cfg(target_arch = "riscv64")]
311    {
312        let next = hal::common::timer::read_mtime() + hal::common::timer::TICKS_PER_10MS;
313        hal::common::sbi::set_timer(next);
314    }
315
316    #[cfg(target_arch = "riscv64")]
317    {
318        let elapsed_ms = hal::common::timer::read_mtime().wrapping_sub(t_restore_start) / 10_000;
319        log::info!(
320            "[snapshot] warm boot: {} frames restored in {} ms",
321            frame_count,
322            elapsed_ms
323        );
324    }
325    log::info!("[snapshot] warm boot complete → resuming scheduler");
326
327    // SCHEDULER is now Some(restored) — yield_cpu() picks the first ready task.
328    // Cells resume from their last yield point.
329    crate::task::yield_cpu();
330
331    // yield_cpu() should not return (tasks are ready); fall through to cold boot
332    // as a safety net if the restored scheduler is somehow empty.
333    false
334}

336/// Zero out the snapshot magic to force cold boot on next restart.
337///
338/// Called when validation fails (hash mismatch, CRC error) to prevent
339/// the system from repeatedly attempting to load a stale or corrupt snapshot.
340pub fn invalidate_snapshot() {
341    let buf = [0u8; 512];
342    let _ = block::write_sector(SNAPSHOT_BASE_LBA, &buf);
343    log::info!("[snapshot] snapshot invalidated");
344}

346// ── Header validation (pure, no VirtIO) ──────────────────────────────────────
347
348/// Validate a snapshot header without I/O — checks magic, format version, and
349/// kernel git hash.  Does NOT verify the CRC32 (requires reading frame sectors).
350///
351/// Useful in unit tests and as the fast fast gate in `try_restore()`.
352pub fn validate_header(h: &SnapshotHeader) -> bool {
353    h.magic == SNAPSHOT_MAGIC
354        && h.version == SNAPSHOT_FORMAT_VERSION
355        && h.kernel_hash == kernel_hash()
356}

379        let h = valid_header();

384                core::mem::size_of::<SnapshotHeader>(),
