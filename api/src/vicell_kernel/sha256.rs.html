3//! Self-contained, `no_std`, zero-dependency: the kernel has no crypto crate and
4//! adding one to the PIC kernel build is avoided here. Used only for integrity
5//! *measurement* (loader → [`crate::measurement_log`]), never for secrecy — so a
6//! compact, audited, single-file implementation is preferable to a dependency.

86/// Compute the SHA-256 digest of `data`.
87pub fn sha256(data: &[u8]) -> [u8; 32] {
88    let mut h = H0;
89    let bitlen = (data.len() as u64).wrapping_mul(8);
90
91    let mut chunks = data.chunks_exact(64);
92    for block in chunks.by_ref() {
93        // SAFETY of unwrap: chunks_exact yields exactly-64-byte slices.
94        process(&mut h, block.try_into().unwrap());
95    }
96    let rem = chunks.remainder();
97
98    // Padding: 0x80, zeros, then 64-bit big-endian bit length. Needs one extra
99    // block when the remainder leaves < 8 bytes for the length field.
100    let mut tail = [0u8; 128];
101    tail[..rem.len()].copy_from_slice(rem);
102    tail[rem.len()] = 0x80;
103    let taillen = if rem.len() < 56 { 64 } else { 128 };
104    tail[taillen - 8..taillen].copy_from_slice(&bitlen.to_be_bytes());
105    process(&mut h, tail[..64].try_into().unwrap());
106    if taillen == 128 {
107        process(&mut h, tail[64..128].try_into().unwrap());
108    }
109
110    let mut out = [0u8; 32];
111    for (i, word) in h.iter().enumerate() {
112        out[i * 4..i * 4 + 4].copy_from_slice(&word.to_be_bytes());
113    }
114    out
115}

117fn process(h: &mut [u32; 8], block: &[u8; 64]) {
118    let mut w = [0u32; 64];
119    for i in 0..16 {
120        w[i] = u32::from_be_bytes([
121            block[i * 4],
122            block[i * 4 + 1],
123            block[i * 4 + 2],
124            block[i * 4 + 3],
125        ]);
126    }
127    for i in 16..64 {
128        let s0 = w[i - 15].rotate_right(7) ^ w[i - 15].rotate_right(18) ^ (w[i - 15] >> 3);
129        let s1 = w[i - 2].rotate_right(17) ^ w[i - 2].rotate_right(19) ^ (w[i - 2] >> 10);
130        w[i] = w[i - 16]
131            .wrapping_add(s0)
132            .wrapping_add(w[i - 7])
133            .wrapping_add(s1);
134    }
135
136    let (mut a, mut b, mut c, mut d, mut e, mut f, mut g, mut hh) =
137        (h[0], h[1], h[2], h[3], h[4], h[5], h[6], h[7]);
138
139    for i in 0..64 {
140        let s1 = e.rotate_right(6) ^ e.rotate_right(11) ^ e.rotate_right(25);
141        let ch = (e & f) ^ ((!e) & g);
142        let t1 = hh
143            .wrapping_add(s1)
144            .wrapping_add(ch)
145            .wrapping_add(K[i])
146            .wrapping_add(w[i]);
147        let s0 = a.rotate_right(2) ^ a.rotate_right(13) ^ a.rotate_right(22);
148        let maj = (a & b) ^ (a & c) ^ (b & c);
149        let t2 = s0.wrapping_add(maj);
150        hh = g;
151        g = f;
152        f = e;
153        e = d.wrapping_add(t1);
154        d = c;
155        c = b;
156        b = a;
157        a = t1.wrapping_add(t2);
158    }
159
160    h[0] = h[0].wrapping_add(a);
161    h[1] = h[1].wrapping_add(b);
162    h[2] = h[2].wrapping_add(c);
163    h[3] = h[3].wrapping_add(d);
164    h[4] = h[4].wrapping_add(e);
165    h[5] = h[5].wrapping_add(f);
166    h[6] = h[6].wrapping_add(g);
167    h[7] = h[7].wrapping_add(hh);
168}

170#[cfg(test)]
171mod tests {
172    use super::sha256;
173
174    fn hex(d: &[u8; 32]) -> alloc::string::String {
175        use core::fmt::Write;
176        let mut s = alloc::string::String::new();
177        for b in d {
178            let _ = write!(s, "{:02x}", b);
179        }
180        s
181    }
182
183    #[test]
184    fn nist_empty() {
185        // SHA-256("") — FIPS 180-4 test vector.
186        assert_eq!(
187            hex(&sha256(b"")),
188            "e3b0c44298fc1c149afbf4c8996fb92427ae41e4649b934ca495991b7852b855"
189        );
190    }
191
192    #[test]
193    fn nist_abc() {
194        // SHA-256("abc") — FIPS 180-4 test vector.
195        assert_eq!(
196            hex(&sha256(b"abc")),
197            "ba7816bf8f01cfea414140de5dae2223b00361a396177a9cb410ff61f20015ad"
198        );
199    }
200
201    #[test]
202    fn nist_two_block() {
203        // 56 bytes — forces the two-block padding path (rem.len() >= 56).
204        let msg = b"abcdbcdecdefdefgefghfghighijhijkijkljklmklmnlmnomnopnopq";
205        assert_eq!(
206            hex(&sha256(msg)),
207            "248d6a61d20638b8e5c026930c3e6039a33ce45964ff2167f6ecedd419db06c1"
208        );
