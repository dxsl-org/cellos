1//! Kernel Service Registry — stable `service_id → current provider tid` mapping.
2//!
3//! The supervisor (init) respawns a dead service under a NEW tid. Clients that
4//! addressed the service by its old tid would break. This registry adds an
5//! indirection: the supervisor registers each service's tid under a well-known
6//! `service_id` ([`api::syscall::service`]) and re-registers the new tid on
7//! respawn; a client resolves `service_id → tid` right before sending, so it
8//! reconnects transparently. Keeping the map in the kernel (the never-die core)
9//! means it survives any service's death, and a dead provider is auto-cleared
10//! ([`clear_tid`]) so a lookup in the death→respawn window returns "none" (the
11//! client retries) instead of a stale tid.
12//!
13//! Only `SpawnCap` holders may `register` (enforced at the syscall dispatch),
14//! so a cell cannot hijack, e.g., the VFS endpoint — the trusted supervisor owns
15//! the namespace. `lookup` is open to all cells.
16
17use crate::sync::Spinlock;
18use alloc::collections::BTreeMap;
19
20/// Upper bound on distinct registered services. Bounds kernel memory and matches
21/// the small, fixed set of well-known service IDs — a runaway registrar cannot
22/// grow the map without bound.
23pub const MAX_SERVICES: usize = 32;
24
25/// `service_id` → current provider task id. `0` is never stored (it is the ABI
26/// "no provider" sentinel returned by `lookup`).
27static REGISTRY: Spinlock<BTreeMap<u16, usize>> = Spinlock::new(BTreeMap::new());
28
29/// Force-release this module's lock during fault teardown.
30///
31/// # Safety
32/// Single-hart; called only from the fault/panic path with interrupts disabled.
33pub unsafe fn force_unlock_locks() {
34    REGISTRY.force_unlock();
35}
36
37/// Register `tid` as the current provider of `service_id`, replacing any prior
38/// entry. Returns `false` (rejected) if the registry is full and `service_id` is
39/// new, or if `tid` is 0 (the reserved "none" sentinel). The SpawnCap authority
40/// check is performed by the caller (syscall dispatch), not here.
41pub fn register(service_id: u16, tid: usize) -> bool {
42    if tid == 0 {
43        return false;
44    }
45    let mut map = REGISTRY.lock();
46    if map.len() >= MAX_SERVICES && !map.contains_key(&service_id) {
47        log::warn!(
48            "[service-registry] full ({} entries); rejecting id {}",
49            MAX_SERVICES,
50            service_id
51        );
52        return false;
53    }
54    map.insert(service_id, tid);
55    log::info!("[service-registry] {} -> tid {}", service_id, tid);
56    true
57}
58
59/// Resolve `service_id` to its current provider tid, or `None` if no live
60/// provider is registered. The syscall layer maps `None` to the ABI value 0.
61pub fn lookup(service_id: u16) -> Option<usize> {
62    REGISTRY.lock().get(&service_id).copied()
63}
64
65/// Remove every registration that points at `tid`. Called from `exit_task` when a
66/// task dies so a client never resolves a service to a dead provider; the
67/// supervisor re-registers the replacement's tid on respawn.
68pub fn clear_tid(tid: usize) {
69    let mut map = REGISTRY.lock();
70    let before = map.len();
71    map.retain(|_, &mut t| t != tid);
72    if map.len() != before {
73        log::info!(
74            "[service-registry] cleared stale entries for dead tid {}",
75            tid
76        );
77    }
78}

80#[cfg(test)]
81mod tests {
82    use super::*;
83
84    #[test]
85    fn register_then_lookup() {
86        assert!(register(api::syscall::service::VFS, 4));
87        assert_eq!(lookup(api::syscall::service::VFS), Some(4));
88    }
89
90    #[test]
91    fn reject_zero_tid() {
92        assert!(!register(api::syscall::service::NET, 0));
93        assert_eq!(lookup(api::syscall::service::NET), None);
94    }
95
96    #[test]
97    fn reregister_updates_tid() {
98        register(api::syscall::service::INPUT, 7);
99        register(api::syscall::service::INPUT, 9);
100        assert_eq!(lookup(api::syscall::service::INPUT), Some(9));
101    }
102
103    #[test]
104    fn clear_tid_removes_dead_provider() {
105        register(api::syscall::service::CONFIG, 12);
106        clear_tid(12);
107        assert_eq!(lookup(api::syscall::service::CONFIG), None);
108    }
109}
