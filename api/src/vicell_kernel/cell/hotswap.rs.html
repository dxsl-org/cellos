1//! Hot-swap orchestrator — live-replaces a Cell without message loss.
2//!
3//! # Protocol (5 steps)
4//!
5//! 1. **Freeze** — kernel marks the old cell `Frozen { swap_id }`; the FROZEN set
6//!    is set so incoming IPC can be queued instead of delivered (Phase 02 drains
7//!    the queue; this phase stubs the queue as a no-op).
8//! 2. **Serialize** — kernel sends `AppEvent::Snapshot { swap_id }` to the old cell
9//!    via an IPC envelope.  The cell serializes state and calls `sys_state_stash(key, …)`;
10//!    the kernel polls the stash for the key.  On timeout the swap aborts (old cell
11//!    is unfrozen and continues running).
12//! 3. **Spawn** — kernel loads the new ELF via `loader::spawn_from_path`; the new
13//!    cell starts and blocks in its init recv loop.
14//! 4. **Deserialize** — kernel sends `AppEvent::Restore { key }` to the new cell.
15//!    The cell calls `sys_state_restore(key, …)` then `sys_hotswap_ready()`.  The
16//!    kernel polls the new task's `hotswap_ready` flag.
17//! 5. **Unfreeze** — kernel re-registers the service entry for the new cell tid,
18//!    terminates the old cell via the internal path (bypasses the Frozen kill-guard),
19//!    and (Phase 02) drains queued IPC to the new cell.
20//!
21//! # Lock ordering
22//!
23//! `FROZEN` (leaf) is always acquired **before** `SCHEDULER` is dropped, never
24//! while holding it.  SCHEDULER → FROZEN ordering is safe (one-way dependency).
25

36// ─── Swap-ID counter ─────────────────────────────────────────────────────────
37
38/// Monotonically increasing counter assigning a unique ID to each swap sequence.
39///
40/// `0` is the null sentinel and is never returned by `next_swap_id()`.
41static SWAP_ID_COUNTER: AtomicU64 = AtomicU64::new(1);
42
43fn next_swap_id() -> u64 {
44    SWAP_ID_COUNTER.fetch_add(1, Ordering::Relaxed)
45}

47// ─── Timeout constant ────────────────────────────────────────────────────────

49/// How long (scheduler ticks, ≈10 ms each) to wait for the old cell to serialize
50/// and for the new cell to signal ready.  5 seconds.
51const HOTSWAP_TIMEOUT_TICKS: u64 = 500;

53// ─── IPC envelope byte constants ─────────────────────────────────────────────
54
55/// App SDK magic byte (mirrors `ostd::app::APP_MSG_MAGIC`).
56const APP_MSG_MAGIC: u8 = 0xAC;
57
58/// Envelope discriminant for `AppEvent::Snapshot` (hot-swap Step 2).
59/// Envelope layout: `[0xAC, 0xF0, swap_id_le8 (8 bytes)]` = 10 bytes total.
60const DISC_SNAPSHOT: u8 = 0xF0;
61
62/// Envelope discriminant for `AppEvent::Restore` (hot-swap Step 4).
63/// Envelope layout: `[0xAC, 0xF1, key (64 bytes)]` = 66 bytes total.
64const DISC_RESTORE: u8 = 0xF1;

66// ─── Freeze registry ─────────────────────────────────────────────────────────
67
68/// Global freeze set — cell ids whose incoming IPC should be queued rather than
69/// delivered.  Phase 02 will use this to buffer then flush messages to the new
70/// cell.  In Phase 01 the queuing is a no-op stub: existing callers simply fail
71/// to send if the cell is frozen (same as before).
72///
73/// Lock ordering: FROZEN (leaf) — never acquired while SCHEDULER is held.
74static FROZEN: Spinlock<alloc::collections::BTreeSet<u64>> =
75    Spinlock::new(alloc::collections::BTreeSet::new());
76
77/// Force-release this module's lock during fault teardown.
78///
79/// # Safety
80/// Single-hart; called only from the fault/panic path with interrupts disabled.
81pub unsafe fn force_unlock_locks() {
82    FROZEN.force_unlock();
83}

85/// Mark `cell_id` as frozen.  Subsequent `sys_send` calls to this cell will
86/// queue the message in the task's pending queue instead of delivering it.
87pub fn freeze(cell_id: CellId) {
88    FROZEN.lock().insert(cell_id.0);
89    log::info!("[hotswap] froze cell {}", cell_id.0);
90}
91
92/// Return true if `cell_id` is currently frozen.
93pub fn is_frozen(cell_id: CellId) -> bool {
94    FROZEN.lock().contains(&cell_id.0)
95}
96
97/// Remove `cell_id` from the freeze set and resume normal message delivery.
98pub fn unfreeze(cell_id: CellId) {
99    FROZEN.lock().remove(&cell_id.0);
100    log::info!("[hotswap] unfroze cell {}", cell_id.0);
101}

104
105/// Called from the `HotSwapReady` syscall handler (syscall 401) to record that
106/// the new cell has finished deserializing state.
107///
108/// Sets `Task::hotswap_ready = true` for `tid` under the SCHEDULER lock.
109pub fn set_task_hotswap_ready(tid: usize) {
110    if let Some(sched) = crate::task::SCHEDULER.lock().as_mut() {
111        if let Some(task) = sched.tasks.get_mut(&tid) {
112            task.hotswap_ready = true;
113            log::info!("[hotswap] task {} signalled HotSwapReady", tid);
114        }
115    }
116}

120/// Resolve the task-id for a live cell, or `ViError::NotFound`.
121fn find_tid_for_cell(cell_id: CellId) -> ViResult<usize> {
122    let guard = crate::task::SCHEDULER.lock();
123    guard
124        .as_ref()
125        .and_then(|s| {
126            s.tasks
127                .values()
128                .find(|t| t.cell_id == cell_id)
129                .map(|t| t.id)
130        })
131        .ok_or(ViError::NotFound)
132}

134/// Transition `tid` to `TaskState::Frozen { swap_id }`.
135///
136/// The task is removed from the scheduler ready queues so it cannot be selected
137/// for execution while the swap is in progress.
138pub(crate) fn set_task_frozen(tid: usize, swap_id: u64) -> ViResult<()> {
139    use crate::task::tcb::TaskState;
140    if let Some(sched) = crate::task::SCHEDULER.lock().as_mut() {
141        if let Some(task) = sched.tasks.get_mut(&tid) {
142            task.state = TaskState::Frozen { swap_id };
143            // Remove from all hart ready queues — a Frozen task must not run.
144            crate::task::hart_local::ready::remove_from_all(tid);
145            return Ok(());
146        }
147    }
148    Err(ViError::NotFound)
149}

151/// Roll back a Frozen task to `TaskState::Ready` and re-queue it.
152///
153/// Called on swap abort so the old cell resumes from where it left off.
154pub(crate) fn unfreeze_task(tid: usize) {
155    use crate::task::tcb::TaskState;
156    if let Some(sched) = crate::task::SCHEDULER.lock().as_mut() {
157        if let Some(task) = sched.tasks.get_mut(&tid) {
158            if matches!(task.state, TaskState::Frozen { .. }) {
159                task.state = TaskState::Ready;
160                sched.push_ready(tid);
161            }
162        }
163    }
164}

166/// Terminate `tid` via the internal path, bypassing the Frozen kill-guard.
167///
168/// Used at the end of a successful swap to terminate the old cell.  The old cell
169/// is Frozen at this point; the regular `ForceExit` syscall would reject the
170/// request with `PermissionDenied`.
171///
172/// Mirrors the cleanup sequence from the `ForceExit` handler — must remain in sync.
173pub(crate) fn exit_task_internal(tid: usize, cell_id: CellId) {
174    // Resource cleanup (same order as ForceExit handler).
175    crate::cell::cap_registry::CAP_TABLE
176        .lock()
177        .revoke_all_for(cell_id);
178    crate::memory::cell_quota::deregister(cell_id);
179    crate::resource_registry::release_for(cell_id);
180    crate::resource_registry::release_bdfs_for(tid);
181    crate::task::drivers::iommu::cleanup_cell(cell_id.0);
182
183    if let Some(sched) = crate::task::SCHEDULER.lock().as_mut() {
184        // 0xAAAA_AAAA = hot-swap sentinel (distinguishes from clean exit 0 or watchdog MAX).
185        sched.exit_task(tid, 0xAAAA_AAAAusize);
186    }
187
188    // Grant pages are freed outside SCHEDULER lock (lock-order safety).
189    // SAFETY: reap_grants_for_task is pub(crate); hotswap.rs is in the same crate.
190    crate::task::syscall::reap_grants_for_task(tid);
191
192    crate::audit::log_event(
193        crate::audit::AuditEvent::CellExit,
194        &crate::audit::encode_u32x2(tid as u32, 0xAA00_0000u32), // hot-swap marker
195    );
196}

198/// Send an `AppEvent::Snapshot { swap_id }` IPC envelope to `tid`.
199///
200/// Envelope: `[0xAC, 0xF0, swap_id_le8]` = 10 bytes.
201fn send_snapshot_event(tid: usize, swap_id: u64) -> ViResult<()> {
202    let mut buf = [0u8; 10];
203    buf[0] = APP_MSG_MAGIC;
204    buf[1] = DISC_SNAPSHOT;
205    buf[2..10].copy_from_slice(&swap_id.to_le_bytes());
206    crate::task::send_to(tid, &buf)
207}

209/// Send an `AppEvent::Restore { key }` IPC envelope to `tid`.
210///
211/// `key_str` is the decimal string of `swap_id`, null-padded to 64 bytes.
212/// Envelope: `[0xAC, 0xF1, key[64]]` = 66 bytes.
213fn send_restore_event(tid: usize, swap_id: u64) -> ViResult<()> {
214    let mut buf = [0u8; 66];
215    buf[0] = APP_MSG_MAGIC;
216    buf[1] = DISC_RESTORE;
217    // Write decimal swap_id as null-terminated ASCII into buf[2..66].
218    let mut tmp = [0u8; 20]; // max u64 decimal = 20 digits
219    let n = fmt_u64_decimal(swap_id, &mut tmp);
220    let key_len = n.min(63);
221    buf[2..2 + key_len].copy_from_slice(&tmp[..key_len]);
222    // remaining bytes default to 0 (null terminator + padding)
223    crate::task::send_to(tid, &buf)
224}

226/// Format `val` as decimal ASCII into `buf`. Returns the number of bytes written.
227///
228/// `buf` must be at least 20 bytes (max u64 decimal length).
229fn fmt_u64_decimal(mut val: u64, buf: &mut [u8; 20]) -> usize {
230    if val == 0 {
231        buf[0] = b'0';
232        return 1;
233    }
234    let mut end = 20usize;
235    while val > 0 {
236        end -= 1;
237        buf[end] = b'0' + (val % 10) as u8;
238        val /= 10;
239    }
240    let len = 20 - end;
241    buf.copy_within(end..20, 0);
242    len
243}

245/// Stash key for a given `swap_id`.
246///
247/// Must match `ostd::hotswap::hotswap_key(swap_id)` exactly — both sides of a
248/// hot-swap derive the key independently: the old cell uses the ostd helper,
249/// the kernel uses this function.  Any divergence causes a stash-miss (timeout).
250///
251/// Layout: `0xA3_0000_0000_0000 | (swap_id & 0x0000_FFFF_FFFF_FFFF)`.
252fn stash_key_for(swap_id: u64) -> u64 {
253    // Namespace tag matches ostd::hotswap::hotswap_key (libs/ostd/src/hotswap.rs:136).
254    0x00A3_0000_0000_0000u64 | (swap_id & 0x0000_FFFF_FFFF_FFFFu64)
255}

257/// Spin-poll until the state-stash contains an entry for `key`, or timeout.
258///
259/// Returns `Ok(())` when the stash entry appears, `Err(ViError::WouldBlock)` if
260/// `HOTSWAP_TIMEOUT_TICKS` ticks elapse without the key appearing.
261///
262/// Precondition: called from a context where the SCHEDULER lock is NOT held
263/// (spin-polls release the CPU between iterations via `core::hint::spin_loop()`).
264fn wait_for_stash_key(key: u64) -> ViResult<()> {
265    let deadline = crate::task::system_ticks() as u64 + HOTSWAP_TIMEOUT_TICKS;
266    loop {
267        {
268            // Peek without consuming — the new cell needs to read it too.
269            let mut probe = [0u8; 1];
270            if crate::cell::state_stash::restore(key, &mut probe) > 0 {
271                return Ok(());
272            }
273        }
274        if crate::task::system_ticks() as u64 >= deadline {
275            return Err(ViError::WouldBlock);
276        }
277        // Yield CPU so the old cell can run its Snapshot handler and call
278        // sys_state_stash.  Without this, a single-hart system deadlocks:
279        // the orchestrator spins forever, the old cell never gets to run.
280        crate::task::yield_cpu();
281    }
282}

284/// Spin-poll until `tid`'s `hotswap_ready` flag is true, or timeout.
285fn wait_for_hotswap_ready(tid: usize) -> ViResult<()> {
286    let deadline = crate::task::system_ticks() as u64 + HOTSWAP_TIMEOUT_TICKS;
287    loop {
288        {
289            let guard = crate::task::SCHEDULER.lock();
290            if guard
291                .as_ref()
292                .and_then(|s| s.tasks.get(&tid))
293                .map(|t| t.hotswap_ready)
294                .unwrap_or(false)
295            {
296                return Ok(());
297            }
298        }
299        if crate::task::system_ticks() as u64 >= deadline {
300            return Err(ViError::WouldBlock);
301        }
302        // Yield CPU so the new cell can run its Restore handler and call
303        // sys_hotswap_ready.  Without this, a single-hart system deadlocks.
304        crate::task::yield_cpu();
305    }
306}

310/// Perform a live hot-swap of `old_cell_id` with the ELF at `new_elf_path`.
311///
312/// Called from `sys_hotswap` (syscall 400) after the SpawnCap gate passes.
313/// `caller_tid` is the tid of the orchestrating cell (e.g. init/shell).
314///
315/// Returns the new task id on success, or an error indicating which step failed.
316///
317/// # Error codes
318/// - `ViError::NotFound` — `old_cell_id` not live or `new_elf_path` missing.
319/// - `ViError::WouldBlock`  — old cell did not serialize / new cell did not signal
320///   ready within `HOTSWAP_TIMEOUT_TICKS` scheduler ticks (≈5 s).
321/// - `ViError::IO`       — IPC send to old or new cell failed.
322///
323/// # Rollback on failure
324/// If any step fails before Step 3 (spawn), the old cell is unfrozen and
325/// resumes normally.  If Step 3 succeeds but a later step fails, the old cell
326/// remains Frozen (manual recovery required); this is the safe-side choice
327/// rather than running two potentially inconsistent instances simultaneously.
328///
329/// # Panics
330/// None — all errors are propagated.
331pub fn hotswap(old_cell_id: CellId, new_elf_path: &str, caller_tid: usize) -> ViResult<usize> {
332    log::info!(
333        "[hotswap] starting swap: cell {} → {} (caller_tid={})",
334        old_cell_id.0,
335        new_elf_path,
336        caller_tid
337    );
338
339    // ── Snapshot caller's capability ceiling for the new cell ─────────────
340    // The replacement must not receive caps the replaced cell didn't hold.
341    let ceiling = crate::task::SCHEDULER
342        .lock()
343        .as_ref()
344        .and_then(|s| {
345            s.tasks
346                .values()
347                .find(|t| t.cell_id == old_cell_id)
348                .map(|t| crate::task::cap::CapSet::of_task(t))
349        })
350        .unwrap_or(crate::task::cap::CapSet::EMPTY);
351
352    let swap_id = next_swap_id();
353    let old_tid = find_tid_for_cell(old_cell_id)?;
354    let stash_key = stash_key_for(swap_id);
355
356    // ── Step 1: FREEZE (soft) ────────────────────────────────────────────
357    // Mark cell in the FROZEN set so IPC routers know it is mid-swap.  We do
358    // NOT set TaskState::Frozen yet — the cell must still be able to receive
359    // the Snapshot IPC in Step 2 (it is typically blocked in Recv).
360    // Clear service registry so new callers retry instead of stale-delivering.
361    freeze(old_cell_id);
362    crate::cell::service_registry::clear_tid(old_tid);
363    log::info!(
364        "[hotswap] step 1 done: old_tid={} FROZEN set (swap_id={})",
365        old_tid,
366        swap_id
367    );
368
369    // ── Step 2: SERIALIZE ────────────────────────────────────────────────
370    // Send AppEvent::Snapshot to the old cell.  The cell must be in Recv state
371    // (idle, waiting for next IPC) for delivery to succeed immediately.  If the
372    // cell is busy (not in Recv) the kernel enqueues the caller in Sending state
373    // temporarily — the cell will pick it up when it next enters Recv.
374    if let Err(e) = send_snapshot_event(old_tid, swap_id) {
375        // Old cell does not exist or IPC is broken; abort.
376        unfreeze(old_cell_id);
377        log::error!("[hotswap] step 2 aborted: snapshot IPC failed: {:?}", e);
378        return Err(e);
379    }
380
381    // Poll until the cell stashes its state.  Tolerate timeout — cells that
382    // do not implement ViStateTransfer simply won't stash anything; the swap
383    // continues with an empty stash (new cell starts with a cold state).
384    match wait_for_stash_key(stash_key) {
385        Ok(()) => {
386            log::info!(
387                "[hotswap] step 2 done: state stashed (key={:#x})",
388                stash_key
389            );
390        }
391        Err(ViError::WouldBlock) => {
392            log::warn!(
393                "[hotswap] step 2 timeout: old cell did not stash (continuing with empty stash)"
394            );
395        }
396        Err(e) => {
397            unfreeze(old_cell_id);
398            log::error!("[hotswap] step 2 error: {:?}", e);
399            return Err(e);
400        }
401    }
402
403    // ── Step 1b: FREEZE (hard) ───────────────────────────────────────────
404    // Now that state is serialized, set TaskState::Frozen to prevent the old
405    // cell from executing again.  From this point no external actor can kill it
406    // (ForceExit gate checks Frozen) and the scheduler will not run it.
407    if let Err(e) = set_task_frozen(old_tid, swap_id) {
408        // Task exited on its own between step 2 and here — that is a valid race
409        // (cell shut down while we were waiting for the stash).  Treat as success
410        // with no old-cell cleanup needed.
411        log::warn!("[hotswap] step 1b: old task gone ({:?}); continuing", e);
412    }
413    log::info!(
414        "[hotswap] step 1b done: old_tid={} TaskState::Frozen",
415        old_tid
416    );
417
418    // ── Step 3: SPAWN ────────────────────────────────────────────────────
419    let new_tid = match crate::loader::spawn_from_path(
420        new_elf_path,
421        crate::task::cap::Spawner::Ceiling(ceiling),
422    ) {
423        Ok(id) => id,
424        Err(e) => {
425            // New ELF failed to load; roll back old cell to Ready.
426            unfreeze(old_cell_id);
427            unfreeze_task(old_tid);
428            log::error!("[hotswap] step 3 failed: spawn {:?}: {:?}", new_elf_path, e);
429            return Err(e);
430        }
431    };
432    log::info!("[hotswap] step 3 done: new_tid={}", new_tid);
433
434    // ── Step 4: DESERIALIZE ──────────────────────────────────────────────
435    // Signal the new cell to restore state.  The new cell starts running its
436    // event loop and will enter Recv immediately, so the Restore IPC delivers.
437    if let Err(e) = send_restore_event(new_tid, swap_id) {
438        log::error!("[hotswap] step 4 aborted: restore IPC failed: {:?}", e);
439        // Old cell stays Frozen — manual recovery; do not run two instances.
440        return Err(e);
441    }
442
443    // Wait for the new cell to call sys_hotswap_ready() (syscall 401).
444    if let Err(e) = wait_for_hotswap_ready(new_tid) {
445        log::error!(
446            "[hotswap] step 4 timeout: new cell did not signal ready: {:?}",
447            e
448        );
449        return Err(e);
450    }
451    log::info!("[hotswap] step 4 done: new cell {} is ready", new_tid);
452
453    // ── Step 5: UNFREEZE ─────────────────────────────────────────────────
454    // Remove from FROZEN set so is_frozen() returns false for the old cell.
455    // Drain pending_msgs to the new cell BEFORE terminating the old cell —
456    // this guarantees the new cell sees all buffered messages in order.
457    unfreeze(old_cell_id);
458
459    // Extract the buffered message queue from the old task under the SCHEDULER
460    // lock, then release the lock before calling ipc_send (which re-acquires it).
461    // Lock order: SCHEDULER acquired → pending_msgs taken → SCHEDULER released →
462    //             ipc_send (re-acquires SCHEDULER per message).
463    let pending: alloc::vec::Vec<crate::task::tcb::PendingMsg> = {
464        let mut guard = crate::task::SCHEDULER.lock();
465        if let Some(sched) = guard.as_mut() {
466            if let Some(old_task) = sched.tasks.get_mut(&old_tid) {
467                core::mem::take(&mut old_task.pending_msgs)
468            } else {
469                alloc::vec::Vec::new()
470            }
471        } else {
472            alloc::vec::Vec::new()
473        }
474    };
475
476    let queued_count = pending.len();
477    for msg in pending {
478        // Deliver each buffered message to the new cell.
479        //
480        // Caller identity: we pass `msg.sender_tid` so the new cell's `sys_recv`
481        // sees the *original* sender as `current_caller` — preserving the IPC
482        // identity contract.  However, the original sender has already resumed
483        // (the Frozen intercept returned `Ok(0)` to them), so we must NOT let
484        // `ipc_send` put them back into `TaskState::Sending` if the new cell is
485        // not yet in Recv.  The new cell is expected to be in Recv at this point
486        // (invariant: `wait_for_hotswap_ready` succeeded, meaning the cell called
487        // `sys_hotswap_ready` and re-entered its recv loop before we reach Step 5).
488        //
489        // If, despite the invariant, the new cell is not in Recv, `ipc_send`
490        // returns `Ok(1)` and sets `sender_tid`'s state to Sending — which would
491        // corrupt its live state.  Guard against this: we only proceed when the
492        // new cell is actually in Recv (fire-and-assert).  Log a warning and skip
493        // if not; the caller already has the returned `Ok(0)` and has moved on.
494        //
495        // SAFETY: msg.data is an owned Box<[u8]> from the Frozen intercept.
496        // The pointer is valid for this loop iteration; ipc_send copies before
497        // returning.
498        let new_cell_in_recv = {
499            let guard = crate::task::SCHEDULER.lock();
500            guard
501                .as_ref()
502                .and_then(|s| s.tasks.get(&new_tid))
503                .map(|t| matches!(t.state, crate::task::tcb::TaskState::Recv { .. }))
504                .unwrap_or(false)
505        };
506
507        if !new_cell_in_recv {
508            log::warn!(
509                "[hotswap] new cell {} not in Recv during drain; dropping msg from tid={}",
510                new_tid,
511                msg.sender_tid
512            );
513            continue;
514        }
515
516        let ptr = msg.data.as_ptr() as usize;
517        let len = msg.data.len();
518        let _ = crate::task::ipc_send(msg.sender_tid, new_tid, ptr, len);
519    }
520
521    if queued_count > 0 {
522        log::info!(
523            "[hotswap] step 5: drained {} buffered msgs to new_tid={}",
524            queued_count,
525            new_tid
526        );
527    }
528
529    log::info!(
530        "[hotswap] step 5 done: old_tid={} → new_tid={}; terminating old cell",
531        old_tid,
532        new_tid
533    );
534
535    // Terminate the old cell via the internal path (bypasses Frozen kill-guard).
536    exit_task_internal(old_tid, old_cell_id);
537
538    // Free the stash slot so it does not count toward MAX_ENTRIES.
539    crate::cell::state_stash::remove(stash_key);
540
541    log::info!(
