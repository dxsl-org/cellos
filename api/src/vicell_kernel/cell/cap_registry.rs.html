1//! Kernel capability registry.
2//!
3//! Every open file (or other resource) in ViCell is represented by a `CapId`
4//! assigned by this registry.  The registry enforces single-ownership:
5//! only the cell that allocated or received a cap may use or revoke it.
6//!
7//! Current implementation is a simple global `BTreeMap` protected by a
8//! Spinlock.  A sharded design (one lock per 256 cap IDs) is deferred
9//! to Phase 13 when concurrent pressure is expected to be higher.
10
11use crate::sync::Spinlock;
12use alloc::collections::BTreeMap;
13use types::{CellId, ViError, ViResult};
14
15/// Capability identifier — re-exported from the API crate so kernel and
16/// user-space share exactly one definition.
17pub use api::cap::CapId;
18
19/// What resource a capability refers to.
20pub enum CapResource {
21    /// An open file backed by the kernel-internal FS (`VIFS1` / ramFS).
22    ///
23    /// The `file` field is `None` while a `ReadCap` syscall is in progress —
24    /// the kernel takes the `Box` out, releases the cap-table lock, performs I/O,
25    /// then re-acquires the lock and puts the `Box` back.  This prevents the
26    /// global cap-table lock from being held across potentially-slow I/O.
27    /// A concurrent `ReadCap` on the same cap while it is parked returns
28    /// `Err(ViError::TryAgain)`.
29    File {
30        file: Option<alloc::boxed::Box<dyn api::fs::ViFile + Send + Sync>>,
31    },
32}
33
34/// One capability entry in the table.
35pub struct CapEntry {
36    /// Cell that currently owns this capability.
37    pub owner: CellId,
38    /// The resource this capability refers to.
39    pub resource: CapResource,
40    /// Permissions this cap grants (`api::cap::CapPerms` bits).
41    pub perms: u32,
42    /// Optional lease expiry in kernel monotonic ticks.
43    ///
44    /// `None` = permanent (revoked only by explicit `close` or cell exit).
45    /// `Some(t)` = auto-revoked on the first `verify` call after tick `t`.
46    pub expires_at: Option<u64>,
47    /// Remaining re-grant depth (default `MAX_GRANT_DEPTH = 4`).
48    ///
49    /// 0 = this cap cannot be delegated further via `grant_to`.  Each
50    /// successful `grant_to` clones the cap with `grant_depth - 1`.
51    pub grant_depth: u8,
52}
53
54/// The global capability table.
55pub struct CapTable {
56    entries: BTreeMap<CapId, CapEntry>,
57    next_id: u64,
58}
59
60/// Maximum number of times a capability can be delegated via `grant_to`.
61const MAX_GRANT_DEPTH: u8 = 4;
62
63impl Default for CapTable {
64    fn default() -> Self {
65        Self::new()
66    }
67}
68
69impl CapTable {
70    pub const fn new() -> Self {
71        Self {
72            entries: BTreeMap::new(),
73            next_id: 1, // 0 is the null sentinel — never assigned
74        }
75    }
76
77    /// Allocate a new capability for `owner` over `resource`.
78    ///
79    /// Returns the new `CapId` (always > 0 for the first 2^64-1 calls).
80    /// Panics in debug builds if `next_id` wraps to 0 (effectively unreachable).
81    pub fn alloc(&mut self, owner: CellId, resource: CapResource, perms: u32) -> CapId {
82        let id = CapId(self.next_id);
83        let next = self.next_id.wrapping_add(1);
84        debug_assert!(
85            next != 0,
86            "CapId counter wrapped to 0 — unreachable in practice"
87        );
88        self.next_id = if next == 0 { 1 } else { next };
89        self.entries.insert(
90            id,
91            CapEntry {
92                owner,
93                resource,
94                perms,
95                expires_at: None,
96                grant_depth: MAX_GRANT_DEPTH,
97            },
98        );
99        id
100    }
101
102    /// Allocate a capability with an automatic lease expiry.
103    ///
104    /// `expires_at` is the absolute kernel tick at which the cap becomes invalid.
105    /// Use `crate::task::system_ticks() + duration` to compute this.
106    pub fn alloc_with_lease(
107        &mut self,
108        owner: CellId,
109        resource: CapResource,
110        perms: u32,
111        expires_at: u64,
112    ) -> CapId {
113        let id = self.alloc(owner, resource, perms);
114        if let Some(e) = self.entries.get_mut(&id) {
115            e.expires_at = Some(expires_at);
116        }
117        id
118    }
119
120    /// Delegate `cap_id` from `from_cell` to `to_cell`, decrementing grant depth.
121    ///
122    /// A new capability (with the same resource type, permissions, and lease) is
123    /// allocated for `to_cell`.  The new cap's `grant_depth` is the current
124    /// depth minus one.  `from_cell` retains ownership of the original cap.
125    ///
126    /// # Errors
127    /// - `ViError::PermissionDenied` — `from_cell` does not own `cap_id`.
128    /// - `ViError::NotSupported` — `grant_depth == 0` on the source cap.
129    /// - `ViError::NotFound` — `cap_id` not in table.
130    pub fn grant_to(
131        &mut self,
132        cap_id: CapId,
133        from_cell: CellId,
134        to_cell: CellId,
135    ) -> ViResult<CapId> {
136        self.verify(cap_id, from_cell)?;
137        let entry = self.entries.get_mut(&cap_id).ok_or(ViError::NotFound)?;
138        if entry.grant_depth == 0 {
139            return Err(ViError::NotSupported); // depth exhausted — cannot delegate further
140        }
141        let new_depth = entry.grant_depth - 1;
142        let new_perms = entry.perms;
143        let new_expires = entry.expires_at;
144        // Allocate a new cap for the target; we cannot clone the resource itself
145        // (it lives inside a Box) so the grant creates a "reference" cap type.
146        // For v1.0 we model this as a File cap pointing to the same underlying file.
147        // Full resource sharing across cells is deferred to Phase 13 VFS caps.
148        let new_id = self.next_id;
149        self.next_id = self.next_id.wrapping_add(1).max(1);
150        // Re-borrow entry after next_id mutation.
151        let parent_owner = self.entries[&cap_id].owner;
152        let _ = parent_owner; // for future ACL audit trail
153                              // Create a shallow grant cap — for now we mark it as non-file to avoid
154                              // moving the Box.  The VFS file-handle grant is a separate op in Phase 07.
155                              // This primarily enables grant-depth bookkeeping for non-file caps.
156        self.entries.insert(
157            CapId(new_id),
158            CapEntry {
159                owner: to_cell,
160                resource: CapResource::File { file: None }, // placeholder; real content in VFS cap
161                perms: new_perms,
162                expires_at: new_expires,
163                grant_depth: new_depth,
164            },
165        );
166        Ok(CapId(new_id))
167    }
168
169    /// Verify that `caller` owns `cap_id` and the lease has not expired.
170    ///
171    /// Lease expiry is checked lazily — the entry is revoked and
172    /// `ViError::PermissionDenied` returned if `expires_at` has passed.
173    ///
174    /// # Errors
175    /// - `ViError::PermissionDenied` — cap owned by a different cell or lease expired.
176    /// - `ViError::NotFound` — cap does not exist.
177    pub fn verify(&mut self, cap_id: CapId, caller: CellId) -> ViResult<()> {
178        let now = crate::task::system_ticks() as u64;
179        match self.entries.get(&cap_id) {
180            Some(e) if e.owner == caller => {
181                // Lazy lease check.
182                if let Some(exp) = e.expires_at {
183                    if now >= exp {
184                        // Expired — revoke and report.
185                        self.entries.remove(&cap_id);
186                        return Err(ViError::PermissionDenied);
187                    }
188                }
189                Ok(())
190            }
191            Some(_) => Err(ViError::PermissionDenied),
192            None => Err(ViError::NotFound),
193        }
194    }
195
196    /// Revoke a capability.  Idempotent — no-op if the cap is not in the table.
197    pub fn revoke(&mut self, cap_id: CapId) {
198        self.entries.remove(&cap_id);
199    }
200
201    /// Revoke all capabilities owned by `owner`.
202    ///
203    /// Called when a cell exits to prevent orphaned caps.
204    pub fn revoke_all_for(&mut self, owner: CellId) {
205        self.entries.retain(|_, e| e.owner != owner);
206    }
207
208    /// Return an immutable reference to an entry if owned by `caller`.
209    pub fn get_if_owner(&self, cap_id: CapId, caller: CellId) -> Option<&CapEntry> {
210        self.entries.get(&cap_id).filter(|e| e.owner == caller)
211    }
212
213    /// Return a mutable reference to an entry (does NOT verify ownership —
214    /// callers must call `verify` first).
215    pub fn get_mut_unchecked(&mut self, cap_id: CapId) -> Option<&mut CapEntry> {
216        self.entries.get_mut(&cap_id)
217    }
218
219    /// Take the `Box<dyn ViFile>` out of a `CapResource::File` entry so that
220    /// I/O can be performed with the cap-table lock released.
221    ///
222    /// Returns `Err` if the cap is not found, already parked (concurrent read),
223    /// or is not a `File` cap.
224    pub fn park_file(
225        &mut self,
226        cap_id: CapId,
227        owner: types::CellId,
228    ) -> ViResult<alloc::boxed::Box<dyn api::fs::ViFile + Send + Sync>> {
229        self.verify(cap_id, owner)?;
230        let entry = self.entries.get_mut(&cap_id).ok_or(ViError::NotFound)?;
231        // Single variant today; match for future exhaustiveness when more cap types land.
232        let CapResource::File { ref mut file } = entry.resource;
233        // None means a concurrent read is in progress (single-core: shouldn't happen).
234        file.take().ok_or(ViError::InvalidInput)
235    }
236
237    /// Return a previously parked `Box<dyn ViFile>` back into the cap entry.
238    ///
239    /// No-ops if the cap was revoked while the file was parked.
240    pub fn unpark_file(
241        &mut self,
242        cap_id: CapId,
243        boxed_file: alloc::boxed::Box<dyn api::fs::ViFile + Send + Sync>,
244    ) {
245        if let Some(CapEntry {
246            resource: CapResource::File { ref mut file },
247            ..
248        }) = self.entries.get_mut(&cap_id)
249        {
250            *file = Some(boxed_file);
251        }
252        // If the cap was revoked while parked, just drop `boxed_file` here.
253    }
254}

256/// Global kernel capability table.
257pub static CAP_TABLE: Spinlock<CapTable> = Spinlock::new(CapTable::new());
