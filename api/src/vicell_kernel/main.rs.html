41/// Signal QEMU to exit with a success (0) or failure (1) code.
42///
43/// Only available under the `test-hooks` feature — never call this in
44/// production builds. The kernel integration-test harness uses this
45/// instead of parsing serial output for "PASS"/"FAIL" banners.
46///
47/// Device addresses: RISC-V = SiFive test 0x100000, ARM64 = semihosting,
48/// x86_64 = isa-debug-exit (iobase 0xF4).
49#[cfg(feature = "test-hooks")]
50pub fn qemu_exit(success: bool) -> ! {
51    use qemu_exit::QEMUExit;
52    #[cfg(target_arch = "riscv64")]
53    {
54        qemu_exit::RISCV64::new(0x100000).exit(if success { 0 } else { 1 });
55    }
56    #[cfg(target_arch = "aarch64")]
57    {
58        qemu_exit::AArch64Semihosting::default().exit(if success { 0 } else { 1 });
59    }
60    #[cfg(target_arch = "x86_64")]
61    {
62        qemu_exit::X86::new(0xF4, 0).exit(if success { 0 } else { 1 });
63    }
64    // Fallback for other arches: spin forever so the test times out clearly.
65    #[allow(clippy::empty_loop)]
66    loop {}
67}

81/// Kernel entry point called from HAL boot code
82#[no_mangle]
83pub extern "C" fn kmain(hartid: usize, dtb: usize) -> ! {
84    let _hartid = hartid;
85    cpu_features::detect(dtb);
86    // Parse DTB for MMIO bases before any driver or paging init.
87    #[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
88    crate::platform::init(dtb);
89    // Set runtime PLIC base before hal::ARCH.init() calls plic::init() internally.
90    #[cfg(target_arch = "riscv64")]
91    crate::platform::with(|p| hal::common::plic::set_plic_base(p.plic_base));
92    // 0. Initialize UART immediately for early logging
93    #[cfg(any(target_arch = "riscv64", target_arch = "riscv32"))]
94    task::drivers::uart::init();
95    #[cfg(all(target_arch = "aarch64", feature = "board-rpi3"))]
96    crate::hal::uart_bcm_mini::init();
97    #[cfg(all(target_arch = "aarch64", not(feature = "board-rpi3")))]
98    crate::hal::uart_pl011::init();
99    #[cfg(target_arch = "arm")]
100    crate::hal::uart_pl011::init();
101    #[cfg(any(target_arch = "x86_64", target_arch = "x86"))]
102    crate::hal::uart_16550::init();
103
104    // Set HHDM base for LAPIC/IOAPIC MMIO access AND for phys_to_virt.
105    // Limine maps RAM at HHDM_BASE+phys (no identity mapping of physical RAM).
106    // This must be called before FrameAllocator::new_from_map.
107    #[cfg(target_arch = "x86_64")]
108    {
109        let hhdm = crate::boot::limine::get_hhdm_offset().unwrap_or(0);
110        crate::hal::apic::set_hhdm_base(hhdm);
111        crate::memory::frame::set_phys_offset(hhdm as usize);
112        // Propagate the HHDM offset to the HAL PML4 walker so walk_create /
113        // walk_read can dereference physical PTE addresses via HHDM virtual ptrs.
114        crate::hal::paging::set_hhdm_offset(hhdm as usize);
115        // Limine maps only usable e820 RAM in the HHDM. The BIOS ROM area
116        // [0x9F000–0x100000) is reserved in e820 and absent from Limine's HHDM
117        // page table. ACPI RSDP is typically there (~0xf52e0 on q35).
118        // Map it now so phys_to_virt(rsdp) doesn't triple-fault before IDT is up.
119        // SAFETY: called after set_hhdm_offset; PML4 walker uses HHDM-mapped RAM.
120        unsafe {
121            crate::hal::paging::map_bios_area();
122        }
123        // Initialise KASLR seed from HHDM entropy + RDTSC.
124        crate::memory::kaslr::init_kaslr(hhdm);
125    }
126
127    // Parse ACPI tables BEFORE paging init so we have the real MMIO addresses
128    // for LAPIC, IOAPIC, HPET, and PCIe ECAM.  Must run after HHDM offset is set
129    // (phys_to_virt requires it) but before init_kernel_paging_x86 maps MMIO.
130    //
131    // On failure or absent RSDP the parser returns QEMU q35 defaults so the
132    // system boots unchanged on emulated hardware.
133    #[cfg(target_arch = "x86_64")]
134    let acpi_info = {
135        use crate::memory::frame::phys_to_virt;
136        let early_puts = |s: &str| {
137            for c in s.bytes() {
138                crate::hal::uart_16550::putchar(c);
139            }
140        };
141        let rsdp = crate::boot::limine::get_rsdp_ptr().unwrap_or(0);
142        // Limine 8.x maps only usable e820 RAM in its HHDM.  The BIOS ROM area
143        // [0x0000–0x100000) and ACPI-reserved regions near top-of-RAM are absent.
144        // Accessing those regions via HHDM before the IDT is live triple-faults.
145        // On QEMU q35 the RSDP is always in the BIOS area (< 1 MiB), and the XSDT
146        // lives in the ACPI-reserved window — neither is reachable this early.
147        // Use hardcoded q35 defaults here; TODO: re-parse after init_kernel_paging
148        // creates a full physical window that covers all e820 regions.
149        if rsdp != 0 && rsdp >= 0x10_0000 {
150            let info = crate::acpi::parse(rsdp, phys_to_virt);
151            early_puts("[INFO] ACPI tables parsed\n");
152            info
153        } else {
154            if rsdp == 0 {
155                early_puts("[INFO] ACPI RSDP absent — using q35 defaults\n");
156            } else {
157                early_puts("[INFO] ACPI RSDP in BIOS area — using q35 defaults\n");
158            }
159            crate::acpi::AcpiInfo::default()
160        }
161    };
162
163    // 1. Initialize HAL (Architecture specific) - Early Trap Setup
164    // x86_64: LAPIC is deferred until after paging sets up the MMIO mapping
165    // (LAPIC phys 0xFEE00000 isn't in Limine's HHDM for MMIO regions).
166    #[cfg(target_arch = "x86_64")]
167    {
168        crate::hal::gdt::init();
169        crate::hal::idt::init();
170        crate::hal::cet::init_kernel_cet(); // LAYER2-CET-INIT
171        crate::hal::pku::init(); // LAYER2-PKU-INIT (requires IBT, checked inside)
172        crate::hal::syscall::init();
173        // apic::init_lapic() deferred — needs MMIO mapped via custom PML4
174    }
175    #[cfg(not(target_arch = "x86_64"))]
176    hal::ARCH.init();
177    // Initialize Goldfish RTC for wall-clock time on QEMU ARM64 virt.
178    // board-rpi3 has no Goldfish RTC (BCM2837); leave BASE=0 (epoch unknown).
179    #[cfg(all(target_arch = "aarch64", not(feature = "board-rpi3")))]
180    hal::rtc::init_default();
181
182    // Define puts helper — arch-specific character output.
183    let puts = |s: &str| {
184        for c in s.bytes() {
185            #[cfg(any(target_arch = "riscv64", target_arch = "riscv32"))]
186            {
187                let _ = crate::hal::sbi::console_putchar(c);
188            }
189            #[cfg(all(target_arch = "aarch64", feature = "board-rpi3"))]
190            {
191                crate::hal::uart_bcm_mini::putchar(c);
192            }
193            #[cfg(all(target_arch = "aarch64", not(feature = "board-rpi3")))]
194            {
195                crate::hal::uart_pl011::putchar(c);
196            }
197            #[cfg(target_arch = "arm")]
198            {
199                crate::hal::uart_pl011::putchar(c);
200            }
201            #[cfg(any(target_arch = "x86_64", target_arch = "x86"))]
202            {
203                crate::hal::uart_16550::putchar(c);
204            }
205        }
206    };
207
208    // Restore log_info helper
209    let log_info = |s: &str| {
210        puts("[INFO] ");
211        puts(s);
212        puts("\n");
213    };
214
215    // Stable banner — CI greps for this exact string.
216    puts("[Cellos] kernel boot v");
217    puts(env!("CARGO_PKG_VERSION"));
218    puts("\n");
219    puts("Kernel started (Hart: 0, DTB: ...)\n");
220    #[cfg(target_arch = "riscv64")]
221    if cpu_features::has_h_ext() {
222        puts("[cpu] H-extension: detected\n");
223    } else {
224        puts("[cpu] H-extension: not present\n");
225    }
226
227    // Parse bootloader information
228    let boot_info_result = boot::parse_bootloader_info();
229
230    // Check if Limine failed, if so, use fallback (SimpleBootInfo)
231    let boot_info: &dyn BootInfo = match &boot_info_result {
232        Ok(info) => info,
233        Err(_) => {
234            log_info("Limine not found, using QEMU/OpenSBI fallback");
235            // Use fallback static instance (defined in boot.rs or created here)
236            // For now, let's just use the fallback function we will create
237            &boot::FALLBACK_BOOT_INFO
238        }
239    };
240    // Log physical base — non-default value confirms KASLR is active.
241    {
242        puts("[boot] kernel_phys_base=0x");
243        let mut base = boot_info.kernel_base();
244        let digits = b"0123456789abcdef";
245        let mut hex_buf = [b'0'; 16];
246        for i in (0..16).rev() {
247            hex_buf[i] = digits[base & 0xf];
248            base >>= 4;
249        }
250        if let Ok(s) = core::str::from_utf8(&hex_buf) {
251            puts(s);
252        }
253        puts("\n");
254    }
255
256    // Initialize kernel subsystems
257
258    // 1. Memory Management
259    // Get memory map from Boot Info (Converted to Cellos format)
260    let mmap_entries = boot_info.memory_map();
261
262    // Initialize frame allocator with the largest usable region
263    let frame_allocator = memory::frame::FrameAllocator::new_from_map(mmap_entries);
264
265    // 2. Frame Allocator (Physical Memory)
266    // The local `frame_allocator` is moved into the global static.
267    // A mutable reference to the global static will be used for paging setup.
268    unsafe {
269        core::ptr::write(
270            &mut *memory::frame::FRAME_ALLOCATOR.lock(),
271            Some(frame_allocator),
272        );
273    }
274    log_info("Frame allocator initialized");
275
276    // 3. Paging (Virtual Memory) Setup
277    // x86_64 bring-up: Limine's PML4 already maps RAM via HHDM and the kernel
278    // at 0xFFFFFFFF80000000. We skip building + activating our own page tables
279    // until the full x86_64 port (Phase 09). init_kernel_paging uses physical
280    // addresses as virtual pointers, which would fault under Limine's paging.
281    #[cfg(not(any(
282        target_arch = "riscv32",
283        target_arch = "x86_64",
284        target_arch = "x86",
285        target_arch = "arm",
286    )))]
287    {
288        log_info("Initializing paging...");
289        let mut locked_frame_allocator = memory::frame::FRAME_ALLOCATOR.lock();
290        let root_table_phys = memory::paging::init_kernel_paging(
291            locked_frame_allocator
292                .as_mut()
293                .expect("Frame allocator not initialized"),
294            mmap_entries,
295        )
296        .expect("Failed to initialize paging");
297        drop(locked_frame_allocator);
298        log_info("Paging initialized");
299        log_info("Activating paging...");
300        unsafe {
301            memory::paging::activate_paging(root_table_phys);
302        }
303        log_info("Paging activated");
304        // Set sstatus.SUM=1 so S-mode (kernel) can access USER-mapped pages throughout
305        // the kernel lifetime. VirtIO/peripheral MMIO is mapped USER=1 for Driver Cells
306        // (U-mode). Without SUM=1 the kernel's tech-debt VirtIO drivers fault at early-boot
307        // MMIO init. In Cellos's SAS+LBI model security comes from Rust type safety, not
308        // hardware USER-bit separation for kernel-vs-cell — SUM=1 is safe and intentional.
309        #[cfg(target_arch = "riscv64")]
310        // SAFETY: csrs modifies sstatus.SUM (bit 18). Safe to set for kernel S-mode code.
311        unsafe {
312            core::arch::asm!("csrs sstatus, {sum}", sum = in(reg) 0x40000_usize, options(nostack));
313        }
314    }
315    #[cfg(target_arch = "x86_64")]
316    {
317        // Set runtime ECAM base from ACPI before PCIe scan.
318        crate::task::drivers::pcie_ecam::set_ecam_base_x86(acpi_info.ecam_base as usize);
319
320        log_info("Initializing x86_64 paging (kernel PML4)...");
321        let root_table_phys = {
322            let mut locked_frame_allocator = memory::frame::FRAME_ALLOCATOR.lock();
323            memory::paging::init_kernel_paging_x86(
324                locked_frame_allocator
325                    .as_mut()
326                    .expect("Frame allocator not initialized"),
327                acpi_info.ioapic_base,
328                acpi_info.hpet_base,
329                acpi_info.lapic_base,
330                acpi_info.ecam_base,
331            )
332            .expect("Failed to initialize x86_64 kernel PML4")
333        };
334        log_info("x86_64 paging initialized");
335        log_info("Activating x86_64 paging (mov cr3)...");
336        // SAFETY: init_kernel_paging_x86 copied higher-half entries from Limine's PML4
337        // (preserving kernel text/data/HHDM) and identity-mapped MMIO, so the kernel
338        // continues executing after this CR3 switch without a triple-fault.
339        unsafe {
340            memory::paging::activate_paging(root_table_phys);
341        }
342        // Immediate port-I/O probe after activate_paging — if 'Q' appears on serial,
343        // the CR3 switch succeeded and execution reached kmain.  Uses direct out instruction
344        // (no Rust function call) so it cannot be affected by any post-switch state issue.
345        // SAFETY: port I/O to COM1 (0x3F8) is always valid from ring-0.
346        unsafe {
347            core::arch::asm!(
348                "99: in al, dx",
349                "test al, 0x20",
350                "jz 99b",
351                "mov dx, {thr}",
352                "mov al, 0x51",   // 'Q'
353                "out dx, al",
354                thr = const 0x3F8u16,
355                in("dx") 0x3FDu16,
356                out("al") _,
357                options(nomem, nostack)
358            );
359        }
360        // Propagate ACPI-parsed MMIO bases to the APIC driver before init_timers.
361        // Defaults are QEMU q35 values; on real hardware these may differ.
362        crate::hal::apic::set_lapic_phys(acpi_info.lapic_base);
363        crate::hal::apic::set_ioapic_phys(acpi_info.ioapic_base);
364        crate::hal::apic::set_irq_overrides(&acpi_info.irq_overrides, acpi_info.ioapic_gsi_base);
365        // Propagate HPET base to HAL timer init (runtime from ACPI, fallback to
366        // QEMU q35 default 0xFED0_0000 when ACPI is absent).
367        crate::hal::set_hpet_base(acpi_info.hpet_base as usize);
368        // HPET + calibrated LAPIC periodic timer: now safe because HPET, IOAPIC,
369        // and LAPIC are identity-mapped in our new PML4 at the ACPI-parsed bases.
370        crate::hal::init_timers();
371        log_info("x86_64 timers initialized (HPET + LAPIC)");
372    }
373    // Bare physical: RV32 Nano (SATP=0), x86_32 (CR0.PG=0), AArch32 (MMU off).
374    #[cfg(any(target_arch = "riscv32", target_arch = "x86", target_arch = "arm"))]
375    {
376        memory::paging::init_bare();
377        log_info("Paging: bare physical");
378    }
379
380    // 4. Heap Allocator (Global) - MUST be after paging but before any allocations
381    // 32 MiB = 8192 frames. Sized to hold:
382    //   - embedded RAM disk copy (~4 MiB), VirtIO GPU framebuffer (~4 MiB), cell ELFs + kernel structures
383    const HEAP_FRAMES: usize = 4_096;
384    let heap_start = {
385        let mut allocator_guard = memory::frame::FRAME_ALLOCATOR.lock();
386        let allocator = allocator_guard
387            .as_mut()
388            .expect("Frame allocator not initialized");
389        let start = allocator.allocate_frame().expect("OOM: Heap start");
390        for _ in 1..HEAP_FRAMES {
391            allocator.allocate_frame().expect("OOM: Heap continuation");
392        }
393        start
394    };
395    let heap_size = HEAP_FRAMES * 4096;
396    // On x86_64, phys_to_virt adds HHDM offset (Limine maps RAM at HHDM+phys).
397    // On RISC-V, phys_to_virt returns phys unchanged (identity-mapped before paging).
398    let heap_virt = memory::frame::phys_to_virt(heap_start);
399    unsafe {
400        memory::heap::init_heap(heap_virt, heap_size);
401    }
402    log_info("Heap initialized");
403
404    memory::rt_heap::init();
405    log_info("RT heap initialized");
406
407    // 5. Hardware Abstraction Layer (HAL) Initialization
408    // GDT/IDT/SYSCALL already done at step 1. Initialize PLIC for RISC-V external IRQs.
409    #[cfg(target_arch = "riscv64")]
410    crate::hal::common::plic::init();
411    log_info("HAL initialized (PLIC enabled)");
412
413    // 6. Logger & Drivers & FS
414    task::drivers::uart::init(); // registers log backend on all arches
415    #[cfg(target_arch = "riscv64")]
416    task::drivers::uart::init_input();
417    // RV32 Nano / x86_64 bring-up: skip VirtIO MMIO probing (PCIe transport not yet ported).
418    // x86_64 gets VirtIO via the PCI path in virtio_pci::init() below.
419    #[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
420    task::drivers::init();
421    // x86_64: load embedded kernel_fs.img into RAM so EarlyLoader can serve ELFs from it.
422    // VirtIO devices on q35 are on PCIe — probed via virtio_pci::init() after ECAM scan.
423    #[cfg(target_arch = "x86_64")]
424    {
425        task::drivers::ramdisk::init_driver();
426        // Wire COM1 RX IRQ → IDT vector 0x24 → shell stdin.
427        // Must run after init_timers() (which initialises the LAPIC + IOAPIC).
428        crate::hal::uart_16550::init_input_irq();
429        // Initialise the RX buffer that vi_handle_uart_irq() writes into.
430        task::drivers::uart::init_input();
431        log_info("x86_64: ramdisk + UART RX IRQ initialised");
432    }
433
434    // PCIe ECAM scan + NVMe + e1000 + VirtIO PCI init.
435    // ARM64 virt uses VirtIO MMIO (not PCIe); accessing 0x3F000000 on QEMU
436    // virt 7+ triggers a Synchronous External Abort — skip on aarch64.
437    // x86_64 q35: ECAM base 0xB000_0000 is identity-mapped by init_kernel_paging_x86;
438    // virtio_pci::init() probes vendor 0x1AF4 PCIe devices for BLK/NET.
439    #[cfg(any(target_arch = "riscv64", target_arch = "x86_64"))]
440    {
441        // PCIe enumeration is now owned by the Platform Cell.
442        // It scans ECAM and calls sys_register_pci_device + sys_register_pcie_bar for each device.
443        // Arm deferred IOMMU init — fires once the IOMMU device entry appears in PCI_DEVICES.
444        task::drivers::iommu::set_deferred_init_pending();
445        // VirtIO PCI block init removed (G2 loader redesign phase 06). x86 block I/O
446        // is served by the NVMe Driver Cell (F4); the kernel drives no block hardware.
447        // activate_isolation() is now called inside iommu::try_deferred_init() once the
448        // IOMMU device has been registered by the Platform Cell. The call below is a no-op
449        // (IOMMU not yet initialized at this point in boot).
450        task::drivers::iommu::activate_isolation();
451    }
452
453    // Attempt warm boot from snapshot before any cell initialization.
454    // RV32 Nano / x86_64 skip: no VirtIO block in bring-up.
455    #[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
456    if snapshot::try_restore() {
457        // try_restore() called yield_cpu() and should not return in a successful
458        // warm boot.  If we reach here, fall through to cold boot as a safety net.
459        log::warn!("[boot] snapshot restore returned unexpectedly → cold boot");
460    }
461
462    // Cross-check the on-disk MBR against the compiled-in partition layout
463    // (warn-only — surfaces image/kernel drift at boot instead of as silent
464    // corruption later).
465    #[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
466    crate::loader::disk_layout::verify_mbr();
467
468    // Probe the cell bootstrap table so SpawnFromPath works during init.
469    // RV32 Nano / x86_64 bring-up: no disk.
470    #[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
471    match crate::loader::early::EarlyLoader::probe() {
472        Ok(()) => puts("[loader] cell bootstrap table loaded\n"),
473        Err(_) => {
474            puts("[loader] WARN: cell table not found — disk image may lack bootstrap section\n")
475        }
476    }
477
478    // RV32 Nano: no FAT32 FS in bring-up.
479    // x86_64 uses the ramdisk-backed embedded FS to serve cell ELFs via VIFS1.
480    #[cfg(any(
481        target_arch = "riscv64",
482        target_arch = "aarch64",
483        target_arch = "x86_64"
484    ))]
485    fs::init();
486
487    // Load + verify the signed operator policy (P5b) NOW: after VIFS1 is mounted,
488    // before any cap-bearing cell spawns. Absent → dev-permissive (this G1 build);
489    // invalid → fail-closed. Phase 04 folds policy::lookup into the spawn grant.
490    #[cfg(any(
491        target_arch = "riscv64",
492        target_arch = "aarch64",
493        target_arch = "x86_64"
494    ))]
495    policy::load_from_vifs1();
496
497    // Phase 20: hot-migration state-transfer self-test.
498    #[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
499    crate::cell::state_stash::self_test();
500
501    log_info("Kernel subsystems initialized successfully.");
502
503    // 7. Initialize Scheduler
504    log_info("Initializing scheduler...");
505    task::init();
506    log_info("Scheduler initialized");
507
508    // 7a. Trust-model self-tests (thread identity inheritance + honest revoke).
509    // Runs HERE — after the scheduler exists but BEFORE secondaries start — so the
510    // synthetic thread it spawns cannot be raced onto another hart before teardown.
511    #[cfg(any(
512        target_arch = "riscv64",
513        target_arch = "aarch64",
514        target_arch = "x86_64"
515    ))]
516    {
517        if task::thread_cap_selftest::self_test() {
518            log_info("thread-cap self-test PASS (thread-inherit + honest-revoke)");
519        } else {
520            log_info("thread-cap self-test FAIL");
521        }
522    }
523
524    // 7b. Bring secondary harts online (riscv64 only; no-op on other arches).
525    // Must run AFTER task::init() so the heap and scheduler are live before
526    // any secondary hart starts running kernel code.
527    #[cfg(target_arch = "riscv64")]
528    task::smp::start_secondaries();
529
530    // 8. Spawn Embedded Init
531    // RV32 Nano bring-up: no init binary — boot to idle loop.
532    // x86_64 now included (Phase 04): embedded init ELF at embedded-x86_64/init.
533    #[cfg(any(
534        target_arch = "riscv64",
535        target_arch = "aarch64",
536        target_arch = "x86_64"
537    ))]
538    {
539        log_info("Spawning Embedded Init...");
540
541        // Enable SUM (Supervisor User Memory access) bit in sstatus (RISC-V only).
542        // ARM64 EL1 can always access EL0 pages — no equivalent bit needed.
543        #[cfg(target_arch = "riscv64")]
544        unsafe {
545            core::arch::asm!("csrs sstatus, {0}", in(reg) 0x40000);
546        }
547
548        // Power-on self-test of the Ed25519 verify primitive (RFC 8032 TEST 1 +
549        // tamper-negative) before it is trusted to authenticate the signed
550        // operator policy (P5). Cheap (~one verify); a FAIL means the crypto path
551        // is broken and signed policy must not be trusted.
552        if crate::ed25519::self_test() {
553            log_info("ed25519 verify self-test PASS (RFC 8032 + tamper)");
554        } else {
555            log_info("ed25519 verify self-test FAIL — signed policy unsafe");
556        }
557        // Power-on self-test of the signed-policy path: verify + parse a known
558        // dev-signed blob, and confirm a tampered blob is rejected.
559        if crate::policy::self_test() {
560            log_info("policy verify+parse self-test PASS (signed blob + tamper)");
561        } else {
562            log_info("policy verify+parse self-test FAIL");
563        }
564        // Cell binary signing self-test: known-good vector verifies + flipped
565        // payload is rejected. Runs before any spawn_from_path is called.
566        if crate::signing::self_test() {
567            log_info("cell signing self-test PASS");
568        } else {
569            log_info("cell signing self-test FAIL — cell signature gate unsafe");
570        }
571        // P-TRUST: privileged path-caps are bounded by the spawn ceiling. Pure
572        // CapSet logic (no scheduler), so it runs here alongside the crypto tests,
573        // before any cap-bearing cell spawns.
574        if crate::task::p_trust_selftest::self_test() {
575            log_info("P-TRUST self-test PASS (privileged path-caps ceiling-bounded)");
576        } else {
577            log_info("P-TRUST self-test FAIL");
578        }
579        // Manifest v2: v1-upcast/v2-parse + the tier-floor invariant. Pure logic,
580        // no scheduler — runs alongside the other crypto/trust self-tests.
581        if crate::task::manifest_v2_selftest::self_test() {
582            log_info("Manifest-v2 self-test PASS (v1 upcast + v2 parse + tier-floor)");
583        } else {
584            log_info("Manifest-v2 self-test FAIL");
585        }
586
587        // Layer-2 hardware security self-tests (test-hooks feature only).
588        // MTE (aarch64) and PKU (x86_64) — each prints PASS or SKIP.
589        // Runs here: after all HW init, before scheduler + first cell spawn.
590        #[cfg(feature = "test-hooks")]
591        {
592            #[cfg(target_arch = "aarch64")]
593            crate::layer2_selftest::run_mte_selftest();
594            #[cfg(target_arch = "x86_64")]
595            crate::layer2_selftest::run_pku_selftest();
596        }
597
598        // Spawn Platform Cell (PCIe ECAM scanner) before init so PCI_DEVICES is
599        // populated before init's NVMe/e1000 Driver Cells call sys_find_pcie_device.
600        // PCIe ECAM only on x86_64 q35 and RISC-V virt — ARM64 virt uses VirtIO MMIO.
601        // Failure is non-fatal: kernel-side PCI_DEVICES stays empty; Driver Cells
602        // that rely on sys_find_pcie_device will simply not find their device.
603        #[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
604        match crate::loader::spawn_from_path("/bin/platform", crate::task::cap::Spawner::Root) {
605            Ok(_) => log_info("Platform Cell spawned (PCIe ECAM scanner)"),
606            Err(_) => log_info("Platform Cell absent — PCIe BARs will not be pre-registered"),
607        }
608
609        // Copy to Vec to ensure alignment (include_bytes! is align 1, parsing needs align 8)
610        // CellId(0) placeholder → fixed up to CellId(init_tid) below, mirroring the
611        // path-spawn convention (loader.rs: cell_id = CellId(tid)). A hardcoded
612        // CellId(1) here would COLLIDE with the Platform Cell, which spawns first
613        // (tid=1 → CellId(1)); the collision commingled their per-cell quota slots
614        // and made fault attribution ambiguous ("Cell 1" meant either).
615        let init_data = alloc::vec::Vec::from(INIT_ELF);
616        match task::spawn_from_mem(&init_data, "init", types::CellId(0), alloc::vec![]) {
617            Ok((init_tid, _load_base)) => {
618                log_info("Successfully spawned init");
619                // Probe 'V': confirms spawn_from_mem succeeded → init is in ready queue.
620                #[cfg(feature = "board-rpi3")]
621                crate::hal::uart_bcm_mini::probe_put(b'V');
622                // init is the ROOT AUTHORITY (P2 monotonic-downgrade): grant the
623                // FULL capability set directly here. init is spawned via
624                // spawn_from_mem (NOT spawn_from_path), so its manifest is never
625                // read — this direct TCB write is the only place its caps come
626                // from. init then delegates subsets to vfs/net/shell/... via the
627                // SpawnFromPath syscall, where each child is intersected against
628                // init's caps. Escalation-oracle bound: init's spawn targets MUST
629                // remain compile-time constants (no data-derived paths).
630                if let Some(sched) = task::SCHEDULER.lock().as_mut() {
631                    if let Some(t) = sched.tasks.get_mut(&init_tid) {
632                        // Unique per-cell identity (see spawn comment above): init
633                        // gets CellId(init_tid), never the placeholder or a value
634                        // shared with an earlier path-spawned cell.
635                        t.cell_id = types::CellId(init_tid as u64);
636                        task::cap::CapSet::ALL.apply_to(t);
637                        // SupervisorCap is NOT in CapSet (not delegatable via intersection).
638                        // Init holds it so it can unfreeze cells if the Supervisor Cell crashes.
639                        t.supervisor_cap = Some(task::cap::SupervisorCap::new());
640                        // init is the restart-tree root — freeze/kill must be refused
641                        // even by the Supervisor Cell (which also holds SupervisorCap).
642                        t.is_critical = true;
643                    }
644                }
645                log_info("init granted root authority (CapSet::ALL + SupervisorCap)");
646            }
647            Err(_e) => {
648                log_info("Failed to spawn init");
649                // Probe 'F': spawn_from_mem failed, init NOT in ready queue.
650                #[cfg(feature = "board-rpi3")]
651                crate::hal::uart_bcm_mini::probe_put(b'F');
652            }
653        }
654    }
655
656    // Ring-3 smoke test: spawn a minimal U-mode task that logs and exits.
657    // RISC-V only — task writes RISC-V machine code directly.
658    // Expected serial output: "Hi from U-mode!\n" followed by task exit.
659    #[cfg(all(target_arch = "riscv64", feature = "test-hooks"))]
660    match task::user_hello::spawn() {
661        Ok(tid) => {
662            puts("[task] spawning user_hello at ");
663            // Print tid as decimal (max 20 digits for usize)
664            let mut buf = [0u8; 20];
665            let mut n = tid;
666            let mut i = 20usize;
667            if n == 0 {
668                i -= 1;
669                buf[i] = b'0';
670            } else {
671                while n > 0 {
672                    i -= 1;
673                    buf[i] = b'0' + (n % 10) as u8;
674                    n /= 10;
675                }
676            }
677            let _ = core::str::from_utf8(&buf[i..]).map(|s| puts(s));
678            puts("\n");
679            let _ = tid; // suppress unused warning
680        }
681        Err(_) => log_info("[task] user_hello spawn failed"),
682    }
683
684    log_info("Kernel initialization complete. Entering idle loop.");
685
686    // 9. Start multitasking
687    log_info("Starting scheduler...");
688
689    // Quiet the shared console for interactive use. Kernel bring-up is done; the
690    // remaining Info chatter is per-spawn noise ([loader] SpawnFromPath, Spawn:,
691    // ELF LOAD) that floods the UART and buries the shell prompt. WARN/ERROR still
692    // surface real problems. Raise back to Info when debugging the spawn path.
693    log::set_max_level(log::LevelFilter::Warn);
694
695    // Enable interrupts before entering the idle loop.
696    // RISC-V: set SPP=1 and SIE=1 in sstatus (0x102).
697    // AArch64: clear DAIF.I bit to unmask IRQs.
698    // x86_64: STI via ARCH.enable_interrupts().
699    #[cfg(any(target_arch = "riscv64", target_arch = "riscv32"))]
700    unsafe {
701        // SAFETY: csrs sstatus SPP|SIE from S-mode — standard interrupt enable.
702        core::arch::asm!("csrs sstatus, {0}", in(reg) 0x102usize);
703    }
704    #[cfg(target_arch = "aarch64")]
705    unsafe {
706        core::arch::asm!("msr daifclr, #2", options(nomem, nostack));
707    }
708    // x86_64, x86_32, AArch32: use the Arch trait's enable_interrupts().
709    #[cfg(any(target_arch = "x86_64", target_arch = "x86", target_arch = "arm"))]
710    crate::hal::ARCH.enable_interrupts();
711
712    // Probe 'Q': fires ONLY if no IRQ preempted the code between daifclr and here.
713    // Q fires  → BCM2835 IRQ is not actually delivered to CPU on QEMU (pending but silent).
714    // Q absent → IRQ fired immediately after daifclr (expected); check if H/G appear.
715    #[cfg(feature = "board-rpi3")]
716    crate::hal::uart_bcm_mini::probe_put(b'Q');
717
718    // board-rpi3: Print IRQ state snapshot immediately after enabling IRQs.
719    // Format: "K<src_bit8><pend_bit1><src_nibble>" where src = CORE0_IRQ_SOURCE,
720    // pend = IRQ_PENDING1 bit 1 (BCM2835 systimer C1 pending).
721    // If G never fires but K shows pend=1 and src=0: QEMU does not route BCM2835→BCM2836.
722    #[cfg(feature = "board-rpi3")]
723    {
724        let src_raw = unsafe { core::ptr::read_volatile(0x4000_0060usize as *const u32) };
725        let pend = unsafe { core::ptr::read_volatile(0x3F00_B204usize as *const u32) };
726        let hex = |n: u32| -> u8 {
727            if n < 10 {
728                b'0' + n as u8
729            } else {
730                b'a' + n as u8 - 10
731            }
732        };
733        crate::hal::uart_bcm_mini::probe_put(b'K');
734        crate::hal::uart_bcm_mini::probe_put(if src_raw & (1 << 8) != 0 { b'1' } else { b'0' });
735        crate::hal::uart_bcm_mini::probe_put(if pend & (1 << 1) != 0 { b'1' } else { b'0' });
736        crate::hal::uart_bcm_mini::probe_put(hex(src_raw & 0xF));
737    }
738
739    // Probe 'L': fires once per idle loop iteration (only first 3 on board-rpi3 to avoid flood).
740    // If 'L' never appears after 'K', the code never reaches the idle loop.
741    #[cfg(feature = "board-rpi3")]
742    {
743        static IDLE_COUNT: core::sync::atomic::AtomicUsize =
744            core::sync::atomic::AtomicUsize::new(0);
745        loop {
746            let n = IDLE_COUNT.fetch_add(1, core::sync::atomic::Ordering::Relaxed);
747            if n < 4 {
748                crate::hal::uart_bcm_mini::probe_put(b'L');
749            }
750            crate::task::yield_cpu();
751            crate::hal::ARCH.wait_for_interrupt();
752        }
753    }
754    #[cfg(not(feature = "board-rpi3"))]
755    loop {
756        crate::task::yield_cpu();
757        crate::hal::ARCH.wait_for_interrupt();
758    }
759}

761/// Panic handler.
762///
763/// If the panic occurs while a Cell is running (`CURRENT_CELL_ID != 0`),
764/// kills the Cell instead of halting the system (e.g. OOM after QuotaAlloc
765/// returns null → Cell's alloc panics → we kill it, kernel continues).
766/// True kernel panics (cell_id == 0) halt as before.
767#[panic_handler]
768fn panic(info: &PanicInfo) -> ! {
769    let cell_id = task::hart_local::current_cell_id();
770
771    if cell_id != 0 {
772        // Cell OOM/panic — kill the Cell, kernel survives. Print the panic
773        // FIRST: this path used to swallow the message entirely, leaving only
774        // a meaningless "scause=0x0 sepc=0x0" fault line to debug from.
775        {
776            #[inline(always)]
777            fn cell_panic_putchar(c: u8) {
778                #[cfg(any(target_arch = "riscv64", target_arch = "riscv32"))]
779                {
780                    let _ = crate::hal::sbi::console_putchar(c);
781                }
782                #[cfg(all(target_arch = "aarch64", feature = "board-rpi3"))]
783                {
784                    crate::hal::uart_bcm_mini::putchar(c);
785                }
786                #[cfg(all(target_arch = "aarch64", not(feature = "board-rpi3")))]
787                {
788                    crate::hal::uart_pl011::putchar(c);
789                }
790                #[cfg(target_arch = "arm")]
791                {
792                    crate::hal::uart_pl011::putchar(c);
793                }
794                #[cfg(any(target_arch = "x86_64", target_arch = "x86"))]
795                {
796                    crate::hal::uart_16550::putchar(c);
797                }
798            }
799            struct CellPanicWriter;
800            impl core::fmt::Write for CellPanicWriter {
801                fn write_str(&mut self, s: &str) -> core::fmt::Result {
802                    for c in s.bytes() {
803                        cell_panic_putchar(c);
804                    }
805                    Ok(())
806                }
807            }
808            use core::fmt::Write;
809            let _ = write!(CellPanicWriter, "\n[panic-in-cell {}] {}\n", cell_id, info);
810        }
811        // SAFETY: panic context, interrupts disabled (abort mode), single-hart.
812        task::terminate_current_cell_on_fault(0, 0, 0);
813        // terminate_current_cell_on_fault calls yield_cpu() which switches away.
814        // In abort mode we never return here, but placate the compiler:
815        #[cfg(any(target_arch = "x86_64", target_arch = "x86"))]
816        loop {
817            unsafe {
818                core::arch::asm!("cli; hlt", options(nomem, nostack));
819            }
820        }
821        #[cfg(not(any(target_arch = "x86_64", target_arch = "x86")))]
822        loop {
823            unsafe {
824                core::arch::asm!("wfi");
825            }
826        }
827    }
828
829    // True kernel panic: print diagnostics and halt.
830    #[inline(always)]
831    fn panic_putchar(c: u8) {
832        #[cfg(any(target_arch = "riscv64", target_arch = "riscv32"))]
833        {
834            let _ = crate::hal::sbi::console_putchar(c);
835        }
836        #[cfg(all(target_arch = "aarch64", feature = "board-rpi3"))]
837        {
838            crate::hal::uart_bcm_mini::putchar(c);
839        }
840        #[cfg(all(target_arch = "aarch64", not(feature = "board-rpi3")))]
841        {
842            crate::hal::uart_pl011::putchar(c);
843        }
844        #[cfg(target_arch = "arm")]
845        {
846            crate::hal::uart_pl011::putchar(c);
847        }
848        #[cfg(any(target_arch = "x86_64", target_arch = "x86"))]
849        {
850            crate::hal::uart_16550::putchar(c);
851        }
852    }
853    let puts = |s: &str| {
854        for c in s.bytes() {
855            panic_putchar(c);
856        }
857    };
858    puts("\n[KERNEL PANIC] ");
859    puts("Critical failure.\n");
860    use core::fmt::Write;
861    struct PanicWriter;
862    impl core::fmt::Write for PanicWriter {
863        fn write_str(&mut self, s: &str) -> core::fmt::Result {
864            for c in s.bytes() {
865                panic_putchar(c);
866            }
867            Ok(())
868        }
869    }
870    let _ = writeln!(PanicWriter, "{}", info);
871
872    // Reboot or spin: RISC-V uses SBI SRST; ARM64 / x86_64 spin.
873    puts("[KERNEL PANIC] halting...\n");
874    #[cfg(any(target_arch = "riscv64", target_arch = "riscv32"))]
875    crate::hal::sbi::system_reset(crate::hal::sbi::SBI_RESET_COLD_REBOOT, 0);
876    #[cfg(any(target_arch = "x86_64", target_arch = "x86"))]
877    loop {
878        unsafe {
879            core::arch::asm!("cli; hlt", options(nomem, nostack));
880        }
881    }
882    // Fallback halt for all non-x86 arches (including riscv — unreachable after system_reset).
883    #[cfg(not(any(target_arch = "x86_64", target_arch = "x86")))]
884    loop {
885        unsafe {
886            core::arch::asm!("wfi", options(nomem, nostack));
887        }
888    }
