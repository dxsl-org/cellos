1//! Ed25519 signature VERIFY (no_std, verify-only) for the signed operator policy
2//! (roadmap §G.2 P5). Thin wrapper over `ed25519-compact` so the rest of the
3//! kernel depends only on `ed25519::verify` — the backend choice is isolated here.
4//!
5//! Verify-only: the kernel never signs (the fleet root private key lives offline).
6//! `ed25519-compact`'s verify enforces canonical encodings (rejects malformed /
7//! non-canonical signatures) — adequate for a public-key signature check where no
8//! secret is involved.
9
10/// Verify an Ed25519 signature. Returns `false` on any malformed input or
11/// mismatch — never panics.
12pub fn verify(pubkey: &[u8; 32], msg: &[u8], sig: &[u8; 64]) -> bool {
13    use ed25519_compact::{PublicKey, Signature};
14    let Ok(pk) = PublicKey::from_slice(pubkey) else {
15        return false;
16    };
17    let Ok(s) = Signature::from_slice(sig) else {
18        return false;
19    };
20    pk.verify(msg, &s).is_ok()
21}

23/// RFC 8032 §7.1 self-test (TEST 1, empty message) + a tamper-negative.
24/// Returns `true` iff the known-good vector verifies AND a flipped signature is
25/// rejected. Used as a boot-time spike check; cheap enough to keep for assurance.
26pub fn self_test() -> bool {
27    const PUBKEY: [u8; 32] = [
28        0xd7, 0x5a, 0x98, 0x01, 0x82, 0xb1, 0x0a, 0xb7, 0xd5, 0x4b, 0xfe, 0xd3, 0xc9, 0x64, 0x07,
29        0x3a, 0x0e, 0xe1, 0x72, 0xf3, 0xda, 0xa6, 0x23, 0x25, 0xaf, 0x02, 0x1a, 0x68, 0xf7, 0x07,
30        0x51, 0x1a,
31    ];
32    const SIG: [u8; 64] = [
33        0xe5, 0x56, 0x43, 0x00, 0xc3, 0x60, 0xac, 0x72, 0x90, 0x86, 0xe2, 0xcc, 0x80, 0x6e, 0x82,
34        0x8a, 0x84, 0x87, 0x7f, 0x1e, 0xb8, 0xe5, 0xd9, 0x74, 0xd8, 0x73, 0xe0, 0x65, 0x22, 0x49,
35        0x01, 0x55, 0x5f, 0xb8, 0x82, 0x15, 0x90, 0xa3, 0x3b, 0xac, 0xc6, 0x1e, 0x39, 0x70, 0x1c,
36        0xf9, 0xb4, 0x6b, 0xd2, 0x5b, 0xf5, 0xf0, 0x59, 0x5b, 0xbe, 0x24, 0x65, 0x51, 0x41, 0x43,
37        0x8e, 0x7a, 0x10, 0x0b,
38    ];
39    // Positive: empty-message vector must verify.
40    if !verify(&PUBKEY, &[], &SIG) {
41        return false;
42    }
43    // Negative: a single flipped signature byte must be rejected.
44    let mut bad = SIG;
45    bad[0] ^= 0x01;
46    if verify(&PUBKEY, &[], &bad) {
47        return false;
48    }
49    true
50}
