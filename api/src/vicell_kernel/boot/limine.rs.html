225/// DTB (Device Tree Blob) request — Limine v8 protocol.
226/// The response provides the physical address of the DTB passed by firmware.
227/// On RISC-V UEFI boots this is the only reliable way to get the DTB address
228/// (the `a1` register contains the Limine boot info pointer, not the DTB).
229///
230/// Request GUID (after LIMINE_COMMON_MAGIC):
231///   0x0b40dca86177520e, 0xc8809c1e7bbbde33
232/// Verify against: https://github.com/limine-bootloader/limine/blob/v8.x-binary/limine.h
233#[repr(C)]
234pub struct LimineDtbRequest {
235    pub id: [u64; 4],
236    pub revision: u64,
237    pub response: *const LimineDtbResponse,
238}
239
240unsafe impl Send for LimineDtbRequest {}
241unsafe impl Sync for LimineDtbRequest {}
242
243#[repr(C)]
244pub struct LimineDtbResponse {
245    pub revision: u64,
246    /// Physical address of the DTB (may be null if no DTB on this platform).
247    pub dtb_ptr: *const u8,
248}
249
250#[used]
251#[link_section = ".requests"]
252static mut DTB_REQUEST: LimineDtbRequest = LimineDtbRequest {
253    id: [
254        LIMINE_COMMON_MAGIC[0],
255        LIMINE_COMMON_MAGIC[1],
256        0x0b40dca86177520e,
257        0xc8809c1e7bbbde33,
258    ],
259    revision: 0,
260    response: core::ptr::null(),
261};

263/// Return the DTB physical address from the Limine DtbResponse, if present.
264///
265/// Returns `None` when:
266/// - not booted via Limine (OpenSBI direct boot), or
267/// - Limine did not populate a DTB response (non-RISC-V or no firmware DTB).
268///
269/// The kernel's `platform::init` calls this first and falls back to the `a1`
270/// register value when it returns `None`.
271pub fn get_dtb_ptr() -> Option<usize> {
272    // SAFETY: DTB_REQUEST is a Limine protocol request static. The response
273    // pointer is written by Limine before the kernel entry point is called,
274    // so no data race exists at the point this function is invoked (single
275    // hart, no other code running). Null check guards against absent response.
276    unsafe {
277        let response = DTB_REQUEST.response;
278        if response.is_null() {
279            return None;
280        }
281        let dtb = (*response).dtb_ptr;
282        if dtb.is_null() {
283            return None;
284        }
285        Some(dtb as usize)
286    }
287}

289/// RSDP (Root System Description Pointer) request — x86_64 only.
290///
291/// Limine provides the RSDP physical address via this request.
292/// On x86_64 UEFI boots, the RSDP lives in EFI memory; Limine locates it
293/// from the UEFI System Table or by scanning the legacy BIOS ROM range.
294///
295/// Request GUID (after LIMINE_COMMON_MAGIC):
296///   0xc5e77b6b397e7b43, 0x27637845accdcf3c
297/// Verify against: https://github.com/limine-bootloader/limine/blob/v8.x-binary/limine.h
298#[cfg(target_arch = "x86_64")]
299#[repr(C)]
300pub struct LimineRsdpRequest {
301    pub id: [u64; 4],
302    pub revision: u64,
303    pub response: *const LimineRsdpResponse,
304}
305

308#[cfg(target_arch = "x86_64")]
309unsafe impl Sync for LimineRsdpRequest {}

333/// Get the RSDP physical address from the Limine RSDP response.
334///
335/// Returns `None` when:
336/// - not x86_64 (compile-time), or
337/// - Limine did not populate an RSDP response (UEFI firmware has no ACPI),
338/// - the response address is null/zero.
339///
340/// The kernel's x86_64 boot block calls this before `init_kernel_paging_x86`
341/// to parse ACPI tables for MMIO base addresses.
342#[cfg(target_arch = "x86_64")]
343pub fn get_rsdp_ptr() -> Option<usize> {
344    // SAFETY: RSDP_REQUEST is a Limine protocol request static. The response
345    // pointer is written by Limine before the kernel entry point is called,
346    // so no data race exists at the point this function is invoked (single
347    // CPU, no other code running). Null check guards against absent response.
348    unsafe {
349        let response = RSDP_REQUEST.response;
350        if response.is_null() {
351            return None;
352        }
353        let addr = (*response).address as usize;
354        if addr == 0 {
355            None
356        } else {
357            Some(addr)
358        }
359    }
360}

362/// Limine memory map entry types
363#[derive(Debug, Clone, Copy, PartialEq, Eq)]
364#[repr(u64)]
365pub enum LimineMemoryType {
366    Usable = 0,
367    Reserved = 1,
368    AcpiReclaimable = 2,
369    AcpiNvs = 3,
370    BadMemory = 4,
371    BootloaderReclaimable = 5,
372    KernelAndModules = 6,
373    Framebuffer = 7,
374}
375
376impl LimineMemoryMapEntry {
377    pub fn memory_type(&self) -> LimineMemoryType {
378        match self.entry_type {
379            0 => LimineMemoryType::Usable,
380            1 => LimineMemoryType::Reserved,
381            2 => LimineMemoryType::AcpiReclaimable,
382            3 => LimineMemoryType::AcpiNvs,
383            4 => LimineMemoryType::BadMemory,
384            5 => LimineMemoryType::BootloaderReclaimable,
385            6 => LimineMemoryType::KernelAndModules,
386            7 => LimineMemoryType::Framebuffer,
387            _ => LimineMemoryType::Reserved,
388        }
389    }
