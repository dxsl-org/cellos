1//! Stack Management for Tasks.
2//!
3//! Handles allocation, deallocation, and guard pages for Kernel and User stacks.
4//! Complies with Rule 2 (Owned Buffers / Memory Safety) and Rule 8 (Resource Management).
5
6use crate::memory::frame::FRAME_ALLOCATOR;
7use crate::memory::paging::{self, Flags, PAGE_SIZE};
8use log::{error, trace};
9use types::{VAddr, ViError};
10
11/// Represents an allocated Stack.
12/// Implements Drop to automatically free pages.
13#[derive(Debug)]
14pub struct Stack {
15    /// Base address (lowest address) of the allocated range.
16    /// This includes the guard page at the bottom if present.
17    pub base: VAddr,
18    /// Number of usable pages (excluding guard page).
19    pub pages: usize,
20    /// Whether this stack has a guard page.
21    pub has_guard: bool,
22    /// Top of the stack (initial SP).
23    pub top: VAddr,
24}
25
26impl Stack {
27    /// Allocate a new Kernel Stack.
28    /// - `pages`: Number of usable pages.
29    /// - Uses `FRAME_ALLOCATOR` to get contiguous physical frames.
30    /// - Maps them as RWX (Kernel).
31    /// - Adds a Guard Page at the bottom (Unmapped).
32    pub fn new_kernel(pages: usize) -> Result<Self, ViError> {
33        Self::allocate(pages, true, false)
34    }
35
36    /// Allocate a new User Stack.
37    /// - `pages`: Number of usable pages.
38    /// - Maps them as USER RWX.
39    /// - Adds a Guard Page at the bottom (Unmapped).
40    pub fn new_user(pages: usize) -> Result<Self, ViError> {
41        Self::allocate(pages, true, true)
42    }
43
44    /// Internal allocation logic.
45    fn allocate(pages: usize, guard: bool, user_mode: bool) -> Result<Self, ViError> {
46        let total_pages = if guard { pages + 1 } else { pages };
47
48        let mut frame_guard = FRAME_ALLOCATOR.lock();
49        let allocator = frame_guard.as_mut().ok_or(ViError::OutOfMemory)?;
50
51        // 1. Allocate contiguous frames
52        // Note: Our simple allocator returns single frames.
53        // We need contiguous VIRTUAL memory.
54        // In Identity Mapping (SAS), Physical Contiguity = Virtual Contiguity.
55        // So we need contiguous physical frames if we rely on simple pointer arithmetic.
56        // However, `paging::map_page` maps arbitrary PAddr to VAddr.
57        // BUT, our current `frame::FrameAllocator` (bitmap) might not guarantee contiguous frames.
58        // And we don't have a virtual memory allocator (VMA) yet.
59        //
60        // TEMPORARY SOLUTION:
61        // We assume we can get contiguous frames OR we are mapping to Identity.
62        // For now, we try to allocate one by one.
63        // Wait, if we use Identity Mapping for Kernel, we MUST have contiguous physical frames.
64        //
65        // If we are mapping User Stack, we can map arbitrary frames to contiguous Virtual Addresses?
66        // No, we don't have a Virtual Address Allocator yet.
67        // We are using Identity Mapping for everything currently (SAS).
68        // So we MUST find a contiguous run of physical pages.
69        //
70        // Our `FrameAllocator` (likely a simple bump or bitmap) needs to support range allocation.
71        // Looking at `kernel/src/memory/frame.rs` (not read yet, but assuming simple).
72        //
73        // Let's assume we can just call `allocate_frame` N times and check contiguity?
74        // No, that's brittle.
75        //
76        // Let's rely on the fact that currently `allocate_frame` is likely linear.
77        //
78        // Refactoring: We will allocate the FIRST frame as base.
79        // Then we hope subsequent calls are contiguous. If not, we panic/fail for now
80        // (until VMA is implemented).
81
82        let base_frame = allocator
83            .allocate_contiguous(total_pages)
84            .ok_or(ViError::OutOfMemory)?;
85
86        let base_addr = base_frame; // Identity Map
87
88        // 2. Map Pages
89        // If Guard Page is requested, the bottom page (base_addr) is NOT mapped (or mapped as invalid).
90        // Ideally unmapped.
91
92        let usable_start_idx = if guard { 1 } else { 0 };
93
94        // SAS identity map: all RAM is already identity-mapped RWX by
95        // init_kernel_paging. The usable pages are re-mapped below, then the guard
96        // frame (base_addr) is unmapped after the loop so an overflow traps.
97
98        // Usable Pages
99        let flags = if user_mode {
100            // User Read/Write (Exec?)
101            Flags::from_bits(
102                Flags::VALID
103                    | Flags::READ
104                    | Flags::WRITE
105                    | Flags::USER
106                    | Flags::ACCESSED
107                    | Flags::DIRTY,
108            )
109        } else {
110            // Kernel Read/Write
111            Flags::from_bits(
112                Flags::VALID | Flags::READ | Flags::WRITE | Flags::ACCESSED | Flags::DIRTY,
113            )
114        };
115
116        for i in usable_start_idx..total_pages {
117            let addr = base_addr + (i * PAGE_SIZE);
118            paging::map_page(allocator, addr, addr, flags).map_err(|_| ViError::OutOfMemory)?;
119        }
120
121        // Guard page: drop the bottom frame's pre-existing identity mapping so a
122        // stack overflow (a write below base_addr+PAGE_SIZE) faults instead of
123        // silently corrupting the neighbouring frame. The spawn paths zero only
124        // the usable pages (skipping base_addr), so nothing legitimately writes to
125        // the guard frame. The frame stays owned by this Stack (freed in Drop);
126        // only its PTE is cleared. unmap_page locks KERNEL_ROOT (not FRAME_ALLOCATOR,
127        // which we still hold) — no deadlock.
128        if guard {
129            if paging::unmap_page(base_addr).is_err() {
130                // Non-fatal: stack is still usable, just unguarded. Loud so a
131                // silently-unprotected stack is never mistaken for a guarded one.
132                error!(
133                    "Stack guard NOT active: unmap of guard frame 0x{:X} failed",
134                    base_addr
135                );
136            } else {
137                paging::tlb_flush_all();
138            }
139        }
140
141        // Calculate Top (Stack grows down)
142        // Top is at the END of the allocated range.
143        let top = base_addr + (total_pages * PAGE_SIZE);
144
145        trace!(
146            "Allocated Stack: Base=0x{:X}, Top=0x{:X}, Pages={}, User={}",
147            base_addr,
148            top,
149            pages,
150            user_mode
151        );
152
153        Ok(Stack {
154            base: base_addr,
155            pages,
156            has_guard: guard,
157            top,
158        })
159    }
160}

162impl Drop for Stack {
163    fn drop(&mut self) {
164        trace!("Dropping Stack at 0x{:X}", self.base);
165
166        let total_pages = if self.has_guard {
167            self.pages + 1
168        } else {
169            self.pages
170        };
171
172        // Restore each frame to the BOOT identity mapping (kernel RWX) before
173        // returning it to the allocator. This is load-bearing in the SAS model:
174        // the cell loader zeroes a freshly-allocated frame through its identity
175        // address (`phys_to_virt(frame)` == frame on RISC-V, elf.rs), so EVERY
176        // free frame must be identity-mapped. Stack::new unmaps the guard frame
177        // (overflow protection) and maps usable frames with USER flags; without
178        // this restore, a freed guard frame stays unmapped → the next owner's
179        // BSS memset store-faults, and a freed USER frame carries stale perms →
180        // wrong-page reads (garbage WAD). Unmap-then-map lands every frame in a
181        // clean, uniform kernel-RWX PTE regardless of its prior state.
182        let kernel_rwx = paging::Flags::from_bits(
183            paging::Flags::VALID
184                | paging::Flags::READ
185                | paging::Flags::WRITE
186                | paging::Flags::EXECUTE
187                | paging::Flags::ACCESSED
188                | paging::Flags::DIRTY,
189        );
190        let mut frame_guard = FRAME_ALLOCATOR.lock();
191        if let Some(allocator) = frame_guard.as_mut() {
192            for i in 0..total_pages {
193                let frame = self.base + (i * PAGE_SIZE);
194                let _ = paging::unmap_page(frame);
195                let _ = paging::map_page(allocator, frame, frame, kernel_rwx);
196                allocator.deallocate_frame(frame);
197            }
198
199            paging::tlb_flush_all();
200        }
201    }
202}

204/// Physical frames mapped for a cell's ELF segments, recorded as `(vaddr, frame)`
205/// at load time so they can be reclaimed when the cell dies.
206///
207/// `Stack::drop` only frees stacks; without this a cell's code/data frames leak
208/// on every death (a supervised service restarted repeatedly would grow to OOM).
209/// Segment frames are allocated exclusively for the cell by `load_segments`
210/// (IPC/shared buffers use separate frames), so freeing them on death is safe.
211#[derive(Debug)]
212pub struct CellSegments {
213    pages: alloc::vec::Vec<(types::VAddr, types::PhysAddr)>,
214    /// VA base allocated by `va_alloc::alloc_cell_va` for PIE cells; `0` for
215    /// fixed-VA cells.  Returned to the allocator's free list on drop.
216    pie_va_base: usize,
217}
218
219impl CellSegments {
220    pub fn new(
221        pages: alloc::vec::Vec<(types::VAddr, types::PhysAddr)>,
222        pie_va_base: usize,
223    ) -> Self {
224        Self { pages, pie_va_base }
225    }
226
227    /// Unmap this cell's segment VAs immediately at death — WITHOUT freeing the
228    /// frames (those are freed lazily when the zombie is reaped, in `drop`).
229    ///
230    /// Frees the address space right away so (a) a respawn can reuse the fixed VA
231    /// and (b) the load-time overwrite guard (`load_segments`) only ever observes
232    /// LIVE cells' (and kernel MMIO) mappings, never a dead-but-unreaped cell's.
233    /// Locks only `KERNEL_ROOT` (a leaf), so it is safe under the SCHEDULER lock.
234    pub fn eager_unmap(&self) {
235        let mut unmapped_any = false;
236        for &(vaddr, frame) in &self.pages {
237            // Only unmap a VA that still resolves to OUR frame (it won't if a
238            // respawn already re-pointed it — leave the new mapping intact).
239            if paging::virt_to_phys(vaddr) == Some(frame) {
240                let _ = paging::unmap_page(vaddr);
241                unmapped_any = true;
242            }
243        }
244        if unmapped_any {
245            paging::tlb_flush_all();
246        }
247    }
248}
249
250impl Drop for CellSegments {
251    fn drop(&mut self) {
252        let mut frame_guard = FRAME_ALLOCATOR.lock();
253        if let Some(allocator) = frame_guard.as_mut() {
254            for &(vaddr, frame) in &self.pages {
255                // Only unmap if this VA still resolves to OUR frame. Cells load at
256                // fixed VAs, so a supervised cell respawned at the same VA before we
257                // are reaped will have re-pointed this VA at the NEW instance's frame
258                // — unmapping it then would crash the new cell. Skip the unmap in
259                // that case; the old frame is still ours to free either way.
260                if paging::virt_to_phys(vaddr) == Some(frame) {
261                    let _ = paging::unmap_page(vaddr);
262                }
263                allocator.deallocate_frame(frame);
264            }
265        }
266        // Return the PIE VA slot to the allocator so it can be reused.
267        if self.pie_va_base != 0 {
268            crate::loader::va_alloc::free_cell_va(self.pie_va_base);
269        }
270    }
271}
