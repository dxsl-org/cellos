42/// Initialize drivers subsystem
43///
44/// Use: Sets up the driver registry and initializes statically linked drivers.
45pub fn init() {
46    registry::init();
47
48    // Init specific drivers
49    input_irq_ack::init_driver(); // ACK-only shim; event routing is in input service Cell
50    console_drv::init();
51    ramdisk::init_driver(); // RAM disk for embedded FAT32 (kernel self-hosted FS)
52                            // Disable global interrupts during VirtIO init to prevent IRQ deadlocks.
53                            // VirtIO block raises an IRQ on init; if the PLIC is enabled and the trap
54                            // handler tries to re-acquire a Spinlock held by this thread, it will spin
55                            // forever.  We re-enable SIE after all drivers are initialised.
56                            // G2 loader redesign (phases 05/06): the kernel drives NO block hardware. The
57                            // virtio-blk Driver Cell (/bin/block) owns the VirtIO block device (MMIO + DMA)
58                            // and registers service::BLOCK_DRIVER; VFS routes all sector I/O to it. Kernel
59                            // boot-time block reads (snapshot restore, verify_mbr, EarlyLoader::probe) degrade
60                            // gracefully (block::read_sector → Err on the null device). MMC (SDHCI) still
61                            // initialises for real boards.
62    mmc::init_driver(); // MMC/SD — no-op on QEMU (no SDHCI); probes SDHCI on real board
63                        // Pre-populate PCIE_BARS with VirtIO MMIO slot addresses so virtio-net Driver Cell
64                        // can claim them via sys_request_mmio (PcieDriverCap path). Must run before
65                        // virtio_net::init_driver so the BAR table is ready before net IPC begins.
66    for slot in virtio_common::virtio_slots() {
67        crate::resource_registry::register_pcie_bar(slot.base, 0x200);
68    }
69    // VirtIO NIC is now served by the virtio-net Driver Cell (P06 complete).
70    // VirtIO RNG init deferred: full MMIO probe hangs on RISC-V when probing
71    // already-claimed slots (block/net). The no-op stub is sufficient until a
72    // safe probe strategy is implemented (skip slots claimed by other drivers).
73
74    // PCIe ECAM scan (pcie_ecam::init() + IOMMU init) is called from main.rs
75    // separately on PCIe arches. NVMe and e1000 are now Driver Cells — no
76    // kernel-side init_driver() calls needed for those devices.
77}
