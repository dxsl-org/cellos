1//! Boot self-test for the P-TRUST spawn trust-model repair.
2//!
3//! Verifies the invariant that closes the live DMA-anywhere / LBI bypass: the
4//! privileged path-triggered caps (`pcie_driver` / `platform` / `supervisor`) are
5//! now carried in `CapSet` and bounded by the spawn-time ceiling intersection,
6//! instead of being minted by a raw `path ==` match after (and blind to) it.
7//!
8//! Exercises the exact functions the loader runs — `CapSet::with_path_caps` (the
9//! request) and `CapSet::intersect` (the ceiling gate) — on the real target/arch.
10//! The positive end-to-end path (init's Root ceiling grants driver cells their
11//! caps) is already covered by boot: the block/NIC/GPU/input Driver Cells only
12//! register if they received `pcie_driver` and could then claim MMIO. This test
13//! adds the negative direction: a non-privileged spawner cannot mint them.
14
15use super::cap::CapSet;
16
17/// One (path, cap-accessor) case exercised by `self_test`.
18type PrivCapCase = (&'static str, fn(&CapSet) -> bool);
19
20/// Returns true iff the ceiling correctly bounds every privileged path-cap.
21pub fn self_test() -> bool {
22    let mut ok = true;
23
24    // A spawner that holds SpawnCap but none of the privileged caps — the actor
25    // the C1 exploit assumed (any SpawnCap holder reaching sys_spawn_from_elf).
26    let non_priv = CapSet {
27        spawn: true,
28        ..CapSet::EMPTY
29    };
30
31    // Each privileged cap: path REQUESTS it, non-priv ceiling DROPS it (C1 closed),
32    // init's ALL ceiling PRESERVES it (the legitimate cell still works).
33    let cases: [PrivCapCase; 3] = [
34        ("/bin/nvme", |c| c.pcie_driver),
35        ("/bin/platform", |c| c.platform),
36        ("/bin/supervisor", |c| c.supervisor),
37    ];
38
39    for (path, sel) in cases {
40        let requested = CapSet::EMPTY.with_path_caps(path);
41        if !sel(&requested) {
42            ok = false;
43            log::error!(
44                "[selftest] P-TRUST: FAIL — path {} did not request its cap",
45                path
46            );
47        }
48        if sel(&requested.intersect(non_priv)) {
49            ok = false;
50            log::error!(
51                "[selftest] P-TRUST: FAIL — {} cap survived a non-privileged ceiling (C1 OPEN)",
52                path
53            );
54        }
55        if !sel(&requested.intersect(CapSet::ALL)) {
56            ok = false;
57            log::error!(
58                "[selftest] P-TRUST: FAIL — {} cap lost under init's Root ceiling (over-tighten)",
59                path
60            );
61        }
62    }
63
64    if ok {
65        log::info!("[selftest] P-TRUST: PASS (privileged path-caps bounded by ceiling)");
66    } else {
67        log::error!("[selftest] P-TRUST: FAIL");
68    }
69    ok
70}
