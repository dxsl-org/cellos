1//! Kernel event waker — IRQ-driven wake for WaitForEvent syscall.
2//!
3//! Architecture: a single `NET_RX_PENDING` AtomicBool is set by the VirtIO NIC
4//! ISR on every RX interrupt.  The global timer sweep (hart 0, `pick_next`) checks
5//! all `WaitEvent` tasks each tick and wakes those whose mask matches a pending bit.
6//! The net cell checks the flag BEFORE parking (lost-wakeup guard): if NET_RX_PENDING
7//! is already set when WaitForEvent is called, it returns immediately without blocking.
8//!
9//! Lock order: SCHEDULER (global) → per-hart ready (leaf).
10//! This module does NOT acquire SCHEDULER itself — callers in the sweep already hold it.
11
12use core::sync::atomic::{AtomicBool, Ordering};
13
14/// Set by the VirtIO NIC ISR whenever at least one RX frame is available.
15/// Cleared by the WaitForEvent handler when the net cell is woken.
16pub static NET_RX_PENDING: AtomicBool = AtomicBool::new(false);
17
18/// Returns true if any event is currently pending.
19pub fn has_any_pending() -> bool {
20    NET_RX_PENDING.load(Ordering::Relaxed)
21}
22
23/// Signal a NIC RX event.  Called from the VirtIO interrupt handler (ISR context).
24///
25/// Sets `NET_RX_PENDING` so the next timer sweep wakes any `WaitEvent(NET_RX)` task.
26/// On RISC-V we also pend local SSIP so the timer handler fires without waiting for
27/// the next mtime tick — sub-millisecond latency on the handling hart.
28///
29/// # Safety contract
30/// Callers must be in S-mode trap context (SIE already cleared by hardware entry).
31pub fn signal_net_rx() {
32    NET_RX_PENDING.store(true, Ordering::Release);
33    // Pend a software interrupt on the current hart so vi_timer_tick fires immediately.
34    // SAFETY: csrsi sip.SSIP is permitted from S-mode (RISC-V priv spec §4.1.3).
35    // SIE is currently cleared by the hardware trap entry, so this is queued and
36    // fires once the ISR returns and sret restores sstatus.SIE.
37    #[cfg(target_arch = "riscv64")]
38    unsafe {
39        core::arch::asm!("csrsi sip, 0x2", options(nomem, nostack))
40    };
41}
42
43/// Check whether event `mask` has any pending bits.  Returns the matching fired bits,
44/// or 0 if none.  Clears the matching bits as a side effect (consume-on-read).
45///
46/// Called by the timer sweep (already under SCHEDULER) and the WaitForEvent syscall
47/// handler before parking the task.
48pub fn consume_pending(mask: u32) -> u32 {
49    let mut fired: u32 = 0;
50    if mask & api::syscall::events::NET_RX != 0
51        && NET_RX_PENDING
52            .compare_exchange(true, false, Ordering::AcqRel, Ordering::Relaxed)
53            .is_ok()
54    {
55        fired |= api::syscall::events::NET_RX;
56    }
57    fired
58}
