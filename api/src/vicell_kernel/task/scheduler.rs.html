34/// CPU-monopoly watchdog budget, in 10 ms scheduler ticks. A task may run this
35/// many consecutive ticks WITHOUT voluntarily blocking before it is deemed a
36/// runaway (infinite loop / livelock) and terminated. 500 ticks = 5 s of
37/// uninterrupted CPU — far beyond any cooperative or real-time cell, which block
38/// (Recv/Send/Sleep) every iteration — so legitimate work never trips it. The
39/// budget is kernel-owned; a cell cannot extend its own.
40const WATCHDOG_BUDGET_TICKS: u32 = 500;
41
42/// CPU-monopoly *warning* threshold (80% of the kill budget). An RT cell that crosses
43/// this without yielding gets a one-shot `RtCpuOverrun` audit event — an early signal
44/// that it is trending toward the hard watchdog kill, so an operator/log analysis can
45/// catch a degrading RT loop before it is terminated. Observability only.
46const WATCHDOG_WARN_TICKS: u32 = WATCHDOG_BUDGET_TICKS * 4 / 5;
47
48/// Sentinel recorded as the "scause" in a `CellFault` audit entry for a watchdog
49/// kill, to distinguish it from a real hardware trap.
50const WATCHDOG_SCAUSE: u32 = 0x0000_DEAD;

52/// Death-notification subscriptions: `watched_tid → [watcher_tid, …]`.
53///
54/// A watcher (a `SpawnCap` holder, e.g. a supervisor) registers via the
55/// `NotifyOnExit` syscall; `exit_task` delivers to each watcher when the watched
56/// task dies (wakes a parked `Recv`, or queues onto `Task::pending_deaths` if the
57/// watcher is busy). One-shot: the subscription is removed on delivery, so a
58/// supervisor re-registers for the respawned child.
59///
60/// Lock order: only ever locked while already holding (or after releasing)
61/// SCHEDULER — never SUBSCRIBERS-then-SCHEDULER — to avoid deadlock.
62static DEATH_SUBSCRIBERS: crate::sync::Spinlock<BTreeMap<usize, Vec<usize>>> =
63    crate::sync::Spinlock::new(BTreeMap::new());
64
65/// Register `watcher` to be notified when `watched` exits or faults.
66pub fn subscribe_death(watched: usize, watcher: usize) {
67    DEATH_SUBSCRIBERS
68        .lock()
69        .entry(watched)
70        .or_default()
71        .push(watcher);
72}

74/// Central task table (Hubris-like).
75///
76/// Ready queues and current_task_id are now PER-HART in `ViHartLocal::ready`
77/// and `ViHartLocal::current_task_id` (Phase 03).  This struct keeps only the
78/// shared state that requires the global SCHEDULER lock: the task table itself,
79/// the zombie list, and the next-id counter.
80pub struct Scheduler {
81    pub tasks: BTreeMap<usize, Box<Task>>,
82    pub zombies: Vec<Box<Task>>,
83    pub next_task_id: usize,
84    /// Task IDs whose grant pages must be reaped outside the SCHEDULER lock.
85    ///
86    /// Watchdog kill paths push here instead of calling reap_grants_for_task directly,
87    /// because free_grant_pages acquires KERNEL_ROOT and FRAME_ALLOCATOR while the
88    /// watchdog runs inside SCHEDULER — inverting the documented lock order.
89    /// yield_cpu() drains this list after dropping SCHEDULER, matching the zombie-reaper pattern.
90    pub(super) pending_grant_reap: Vec<usize>,
91    pub last_global_sweep_tick: usize,
92}

111    /// Push task `id` onto the CALLING hart's local ready queue.
112    ///
113    /// Returns the priority level used so callers can optionally call
114    /// `pend_preempt_if_needed(priority)` to trigger zero-latency RT preemption.
115    ///
116    /// Call while holding SCHEDULER (lock order: SCHEDULER → per-hart ready).
117    pub fn push_ready(&mut self, id: usize) -> u8 {
118        let priority = self
119            .tasks
120            .get(&id)
121            .map(|t| t.priority)
122            .unwrap_or(api::TaskPriority::Normal as u8);
123        // RT tasks target the dedicated RT hart when it is online; fall back to
124        // the current hart on single-hart systems (e.g. QEMU without -smp 2).
125        let target_hart = if priority >= api::TaskPriority::RealTime as u8
126            && crate::task::smp::is_rt_hart_online()
127        {
128            crate::task::smp::HART_RT
129        } else {
130            super::hart_local::current_hart_id()
131        };
132        super::hart_local::ready::push_on_hart(target_hart, id, priority);
133        priority
134    }

136    /// Pend an S-mode software interrupt if `new_priority` exceeds the current
137    /// running task's priority.
138    ///
139    /// Call this after any syscall that transitions a task from blocked → Ready
140    /// so that a newly-runnable RealTime cell preempts a Normal/Background cell
141    /// within the same syscall return, rather than waiting for the next timer tick.
142    ///
143    /// The interrupt fires when the trap handler returns via `sret` and
144    /// `sstatus.SIE` is restored by hardware.
145    #[cfg(target_arch = "riscv64")]
146    pub fn pend_preempt_if_needed(&self, new_priority: u8) {
147        let hart_id = super::hart_local::current_hart_id();
148        let current_tid = super::hart_local::ready::current_task_id_for(hart_id);
149        let current_priority = if current_tid > 0 {
150            self.tasks
151                .get(&current_tid)
152                .map(|t| t.priority)
153                .unwrap_or(0)
154        } else {
155            0
156        };
157
158        if new_priority > current_priority {
159            // RT tasks land on HART_RT when online; fall back to current hart on single-hart systems.
160            let target_hart = if new_priority >= api::TaskPriority::RealTime as u8
161                && crate::task::smp::is_rt_hart_online()
162            {
163                crate::task::smp::HART_RT
164            } else {
165                hart_id
166            };
167            if target_hart == hart_id {
168                // SAFETY: csrsi on sip.SSIP is permitted from S-mode (RISC-V priv spec §4.1.3).
169                // The interrupt fires after sret restores sstatus.SIE.
170                unsafe { core::arch::asm!("csrsi sip, 0x2") };
171            } else {
172                // Cross-hart IPI: SSIP fires on the target hart's next interrupt check.
173                let _ = hal::common::sbi::sbi_send_ipi(1 << target_hart, 0);
174            }
175        }
176    }

183    pub fn spawn(
184        &mut self,
185        name: &str,
186        cell_id: CellId,
187        allowed_drivers: alloc::vec::Vec<usize>,
188    ) -> usize {
189        let mut task = Box::new(Task::new(self.next_task_id, cell_id, name, allowed_drivers));
190        task.state = TaskState::Ready;
191        let id = task.id;
192
193        // Stack Size: 8 Frames (32KB)
194        use crate::task::STACK_PAGES as STACK_FRAMES;
195
196        // Allocate Kernel Stack
197        let kstack = crate::task::stack::Stack::new_kernel(STACK_FRAMES).expect("OOM Stack");
198
199        // Stack grows DOWN. Top is at end of region.
200        let stack_top = kstack.top;
201        let stack_base = kstack.base;
202
203        // Zero the usable stack pages. Skip the guard frame at `stack_base` (it is
204        // unmapped — a write there faults); the usable region starts one page above
205        // base and spans exactly STACK_FRAMES pages. (The old code zeroed from base,
206        // which clobbered the guard AND missed the top usable page.)
207        // SAFETY: we own these freshly-allocated, mapped frames exclusively.
208        unsafe {
209            core::ptr::write_bytes(
210                (stack_base + crate::memory::paging::PAGE_SIZE) as *mut u8,
211                0,
212                STACK_FRAMES * crate::memory::paging::PAGE_SIZE,
213            );
214        }
215
216        let entry = task_entry_point as *const () as usize;
217        let (_gp, _tp) = crate::task::get_kernel_gp_tp();
218
219        // Allocate User Stack
220        let ustack = crate::task::stack::Stack::new_user(STACK_FRAMES).expect("OOM User Stack");
221        let ustack_top = ustack.top;
222
223        task.context.sp = stack_top as _;
224        task.trap_frame.sepc = entry as _;
225        task.trap_frame.sstatus = 0x20_u64 as _; // SPIE enabled, SPP=0 (User Mode)
226        task.trap_frame.regs[2] = ustack_top as _; // sp = x2
227        #[cfg(target_arch = "riscv64")]
228        {
229            task.context.ra = entry;
230            task.context.gp = _gp;
231            task.context.tp = _tp;
232        }
233        #[cfg(target_arch = "aarch64")]
234        {
235            task.context.x30 = entry as u64;
236        }
237        #[cfg(target_arch = "x86_64")]
238        {
239            task.context.rip = entry as u64;
240            task.context.kernel_trap_sp = stack_top as u64;
241        }
242        task.kernel_stack = Some(kstack);
243        task.user_stack = Some(ustack);
244
245        info!(
246            "Task '{}' (ID {}): Stack 0x{:X}-0x{:X}, Entry 0x{:X}",
247            name, id, stack_base, stack_top, entry
248        );
249
250        self.tasks.insert(id, task);
251        self.push_ready(id);
252        self.next_task_id += 1;
253        id
254    }

256    pub fn spawn_thread(
257        &mut self,
258        name: &str,
259        cell_id: CellId,
260        allowed_drivers: alloc::vec::Vec<usize>,
261        entry: usize,
262        arg: usize,
263    ) -> usize {
264        let mut task = Box::new(Task::new(self.next_task_id, cell_id, name, allowed_drivers));
265        task.state = TaskState::Ready;
266        let id = task.id;
267
268        use crate::task::STACK_PAGES as STACK_FRAMES;
269
270        // Allocate Kernel Stack
271        let kstack = crate::task::stack::Stack::new_kernel(STACK_FRAMES).expect("OOM Stack");
272
273        let stack_top = kstack.top;
274        let stack_base = kstack.base;
275
276        // SAFETY: We own the allocated stack memory exclusively. The pointer is valid.
277        // Setting up task context with valid register values for thread initialization.
278        unsafe {
279            // Skip the guard frame at `stack_base` (unmapped); zero only the
280            // STACK_FRAMES usable pages that begin one page above base.
281            core::ptr::write_bytes(
282                (stack_base + crate::memory::paging::PAGE_SIZE) as *mut u8,
283                0,
284                STACK_FRAMES * crate::memory::paging::PAGE_SIZE,
285            );
286
287            let (_gp, _tp) = crate::task::get_kernel_gp_tp();
288            let trampoline = crate::hal::arch::thread_trampoline as *const () as usize;
289
290            task.context.sp = stack_top as _;
291            task.trap_frame.sepc = trampoline as _;
292            task.trap_frame.sstatus = 0x120;
293            #[cfg(target_arch = "riscv64")]
294            {
295                task.context.ra = trampoline;
296                task.context.s0 = arg;
297                task.context.s1 = entry;
298                task.context.gp = _gp;
299                task.context.tp = _tp;
300            }
301            #[cfg(target_arch = "riscv32")]
302            {
303                task.context.ra = trampoline as u32;
304                task.context.s0 = arg as u32;
305                task.context.s1 = entry as u32;
306                task.context.gp = _gp as u32;
307                task.context.tp = _tp as u32;
308            }
309            #[cfg(target_arch = "aarch64")]
310            {
311                task.context.x30 = trampoline as u64;
312                task.context.x19 = arg as u64;
313                task.context.x20 = entry as u64;
314            }
315            #[cfg(target_arch = "x86_64")]
316            {
317                // thread_trampoline reads entry from RBX and arg from R12.
318                // CpuContext::switch restores all callee-saved fields before
319                // jumping to context.rip, so store entry/arg there.
320                task.context.rip = trampoline as u64;
321                task.context.rbx = entry as u64; // thread body fn ptr
322                task.context.r12 = arg as u64; // argument
323                task.context.kernel_trap_sp = stack_top as u64;
324            }
325            task.kernel_stack = Some(kstack);
326
327            info!(
328                "Thread '{}' (ID {}): Stack 0x{:X}-0x{:X}, Entry 0x{:X}, Arg 0x{:X}",
329                name, id, stack_base, stack_top, entry, arg
330            );
331        }
332
333        self.tasks.insert(id, task);
334        self.push_ready(id);
335        self.next_task_id += 1;
336        id
337    }

339    /// Reap a task: move it to the zombie list, purge ready queues, unblock
340    /// senders stuck on it, and wake any `Wait`-ers with `exit_reason`.
341    ///
342    /// `exit_reason` is delivered to waiters as their `reply_value` — the exit
343    /// code for a clean `Exit`, or `usize::MAX` for a fault / force-kill.
344    /// Centralizing the waiter-wake here is the contract that ALL death paths
345    /// (clean `Exit`, `ForceExit`, AND hardware faults) notify waiters uniformly;
346    /// the fault path previously skipped it, so `Wait(tid)` hung forever when the
347    /// target died by fault.
348    pub fn exit_task(&mut self, tid: usize, exit_reason: usize) {
349        info!("Task {} exiting (reason={:#x})...", tid, exit_reason);
350
351        // Capture waiters BEFORE the task is removed from the table.
352        let waiters: Vec<usize> = self
353            .tasks
354            .get_mut(&tid)
355            .map(|t| core::mem::take(&mut t.waiters))
356            .unwrap_or_default();
357
358        // Free the dying cell's address space NOW (unmap its segment VAs) so a
359        // respawn can reuse the fixed VA and the load-time overwrite guard only
360        // ever sees LIVE cells' mappings. Frames are freed lazily at reap.
361        if let Some(t) = self.tasks.get(&tid) {
362            if let Some(seg) = &t.segment_mem {
363                seg.eager_unmap();
364            }
365        }
366
367        // Async-future safety: a task may die while Polling, holding a `pending_future`
368        // that captured a raw pointer into THIS cell's buffer (see fat.rs::read_async).
369        // Removing it from `self.tasks` here — BEFORE its frames are freed at reap — takes
370        // it out of the scheduler poll set (the loop iterates `self.tasks`, gated on
371        // `state == Polling`), so the dangling buffer write can never execute. The future
372        // itself is dropped at reap (outside the SCHEDULER lock) without touching the
373        // buffer, because the inner read is synchronous (no DMA outlives the future).
374        // INVARIANT for future work: if a real async-DMA driver lands (the fat.rs TODO) or
375        // the kernel goes SMP, hardware could write into freed frames — add a descriptor
376        // cancel / frame-unpin point HERE before the frames are reclaimed.
377        if let Some(task) = self.tasks.remove(&tid) {
378            self.zombies.push(task);
379        }
380
381        // Service-registry cleanup: drop any well-known service_id that pointed at this
382        // tid, so a client lookup in the death→respawn window returns "none" (and retries)
383        // rather than a dead provider. The supervisor re-registers the replacement's tid.
384        // Locks only REGISTRY (a leaf), safe under the SCHEDULER lock.
385        crate::cell::service_registry::clear_tid(tid);
386
387        // Input-service registration cleanup: prevent the kernel poll path from pushing
388        // events to a dead/reused TID. Supervisor re-registers after respawn.
389        crate::task::drivers::driver_cell::clear_input_cell_if(tid);
390
391        // Remove from every hart's ready queue if present.
392        super::hart_local::ready::remove_from_all(tid);
393
394        // Best-effort IPC cleanup: unblock tasks stuck sending to the dead task,
395        // and clear stale current_caller references.  Does not handle multi-hop
396        // chains — those require a full state-machine audit (future work).
397        let mut to_wake = Vec::new();
398        for (id, task) in self.tasks.iter_mut() {
399            if let TaskState::Sending { target, .. } = task.state {
400                if target == tid {
401                    task.state = TaskState::Ready;
402                    task.trap_frame.regs[10] = usize::MAX as _; // error return: target gone
403                    to_wake.push(*id);
404                }
405            }
406            if task.current_caller == Some(tid) {
407                task.current_caller = None;
408            }
409        }
410        for id in to_wake {
411            self.push_ready(id);
412        }
413
414        // Wake tasks blocked on Wait(tid).  Last use of `w` ends its borrow of
415        // self.tasks before push_ready re-borrows self (NLL) — mirrors the
416        // former in-handler pattern, now the single source of truth.
417        for wid in waiters {
418            if let Some(w) = self.tasks.get_mut(&wid) {
419                w.state = TaskState::Ready;
420                w.reply_value = Some(exit_reason);
421                self.push_ready(wid);
422            }
423        }
424
425        // Deliver NotifyOnExit death notifications. One-shot: the subscription is
426        // removed here. Wake a watcher parked in Recv (its Recv returns
427        // current_caller = this dead tid), else queue onto pending_deaths so the
428        // watcher gets it on its next Recv (covers a death during respawn).
429        let watchers = DEATH_SUBSCRIBERS.lock().remove(&tid).unwrap_or_default();
430        let mut woken_watchers = Vec::new();
431        for w in watchers {
432            if let Some(wt) = self.tasks.get_mut(&w) {
433                if matches!(wt.state, TaskState::Recv { .. }) {
434                    // Stash the exit reason for delivery as the recv payload (NotifyOnExit
435                    // contract). The actual buffer write happens when the watcher's Recv
436                    // RESUMES, in the watcher's own syscall context — writing a USER buffer
437                    // from here (the trap/fault context) faults (S-mode store to a U page,
438                    // SSTATUS.SUM not set).
439                    wt.current_caller = Some(tid);
440                    wt.pending_exit_reason = Some(exit_reason);
441                    wt.state = TaskState::Ready;
442                    woken_watchers.push(w);
443                } else {
444                    wt.pending_deaths.push((tid, exit_reason));
445                }
446            }
447        }
448        for w in woken_watchers {
449            self.push_ready(w);
450        }
451    }

453    /// Remove and return zombies that have already been switched away from — every
454    /// zombie except the one still set as `current_task_id` (whose context is about
455    /// to be used for the outgoing half of the next switch, so it must stay valid).
456    ///
457    /// The caller MUST drop the returned tasks OUTSIDE the SCHEDULER lock: dropping
458    /// a `Box<Task>` runs `Stack::drop`, which locks `FRAME_ALLOCATOR` and unmaps
459    /// via `KERNEL_ROOT`; doing that while holding `SCHEDULER` would invert the lock
460    /// order. Returning the tasks (cheap pointer moves) keeps the lock window tiny.
461    ///
462    /// This is what actually frees a dead cell's kernel + user stack frames (the
463    /// largest per-cell allocation) — without it, zombies accumulate forever and
464    /// `Stack::drop` never runs (every cell death leaked its stacks).
465    pub fn take_reapable_zombies(&mut self) -> Vec<Box<super::tcb::Task>> {
466        if self.zombies.is_empty() {
467            return Vec::new();
468        }
469        let mut keep = Vec::new();
470        let mut reap = Vec::new();
471        for z in core::mem::take(&mut self.zombies) {
472            // A zombie is reapable only if NO hart is currently context-switching
473            // through its saved Context.  Check all harts' current_task_id.
474            if super::hart_local::ready::any_hart_running(z.id) {
475                keep.push(z);
476            } else {
477                reap.push(z);
478            }
479        }
480        self.zombies = keep;
481        reap
482    }

484    /// Take task IDs whose grant pages must be freed outside the SCHEDULER lock.
485    ///
486    /// The caller MUST call reap_grants_for_task for each returned ID OUTSIDE the lock —
487    /// free_grant_pages locks KERNEL_ROOT and FRAME_ALLOCATOR; holding SCHEDULER inverts order.
488    pub fn take_pending_grant_reap(&mut self) -> Vec<usize> {
489        core::mem::take(&mut self.pending_grant_reap)
490    }

492    /// Picks the next task to run on `hart_id` and returns pointers for context switch.
493    ///
494    /// Hart 0 also runs the global sweep (timer wakes, heartbeat, async-poll, watchdog).
495    /// Other harts only do the per-hart pick + work stealing.
496    ///
497    /// Returns: Option<(current_context_ptr, next_context_ptr)>
498    pub fn pick_next(
499        &mut self,
500        hart_id: usize,
501    ) -> Option<(
502        *mut crate::hal::arch::Context,
503        *const crate::hal::arch::Context,
504    )> {
505        let now = crate::task::system_ticks();
506        // Global sweep (timer wakes, heartbeat, async-poll, watchdog) runs on hart 0 only
507        // to prevent double-wake races on multihart setups.
508        if hart_id != 0 {
509            return self.pick_next_local(hart_id, now);
510        }
511
512        let time_advanced = now > self.last_global_sweep_tick;
513        let events_pending = crate::task::waker::has_any_pending();
514
515        if time_advanced || events_pending {
516            if time_advanced {
517                self.last_global_sweep_tick = now;
518            }
519
520            // 1. Wake tasks whose deadline elapsed: Sleeping (timer) and RecvTimeout
521            //    (a Recv with a deadline). Without the RecvTimeout sweep a cell that
522            //    RecvTimeout's a peer that never replies would block forever — the
523            //    infinite-block-on-dead-peer hazard. Deadlines are absolute
524            //    `system_ticks` (the dispatch stores `system_ticks() + timeout`).
525            let mut waking_tasks = VecDeque::new();
526            for (id, task) in self.tasks.iter_mut() {
527                let mut should_wake = false;
528                let mut timed_out = false;
529                match &task.state {
530                    TaskState::Sleeping { until } if now >= *until => {
531                        should_wake = true;
532                    }
533                    // `deadline` is u64 (mtime-domain field); `now` is usize system
534                    // ticks. On rv64 usize == u64, so the cast is lossless.
535                    TaskState::Recv {
536                        deadline: Some(d), ..
537                    } if now as u64 >= *d => {
538                        should_wake = true;
539                        timed_out = true;
540                    }
541                    TaskState::WaitEvent { mask, deadline } => {
542                        let fired = super::waker::consume_pending(*mask);
543                        if fired != 0 {
544                            // Return fired mask as the syscall result.
545                            task.trap_frame.regs[10] = fired as _;
546                            should_wake = true;
547                        } else if deadline.map(|d| now as u64 >= d).unwrap_or(false) {
548                            task.trap_frame.regs[10] = 0; // timeout — return 0
549                            should_wake = true;
550                            timed_out = true;
551                        }
552                    }
553                    // WaitIrq: IRQ-only wake — no deadline, no timeout.
554                    // ISR sets IRQ_PENDING[irq] atomically (no lock, no scheduler access).
555                    // This sweep is the only place that transitions WaitIrq → Ready.
556                    TaskState::WaitIrq { irq }
557                        if crate::task::drivers::irq_wait::take_pending(*irq) =>
558                    {
559                        crate::task::drivers::irq_wait::clear_waiter(*irq);
560                        should_wake = true;
561                    }
562                    _ => {}
563                }
564                if should_wake {
565                    // ostd `sys_recv_timeout` returns Ok(0) on timeout; the syscall
566                    // return register is regs[10], restored by sret when the task runs.
567                    if timed_out {
568                        task.trap_frame.regs[10] = 0;
569                        // Clear the last sender so the RecvTimeout syscall handler,
570                        // which reads current_caller after yield to build its return
571                        // value, sees "no sender" (0) on a genuine timeout instead of
572                        // the PREVIOUS sender. A stale tid here made every 200 ms
573                        // input-read timeout re-deliver the last keystroke's sender +
574                        // stale buffer, duplicating characters ("wget_out.txt" →
575                        // "wget_ooooo…"). Only the timeout branch clears it, so the
576                        // blocking Recv path (VFS request/reply) is untouched.
577                        task.current_caller = None;
578                        task.deadline_misses = task.deadline_misses.saturating_add(1);
579                        // Observability: an RT cell whose awaited message missed its deadline
580                        // is a missed control-loop cycle — record it (no enforcement). Gated to
581                        // RT priority so the safety-timeout use on Normal cells stays quiet.
582                        if task.priority >= api::TaskPriority::RealTime as u8 {
583                            crate::audit::log_event(
584                                crate::audit::AuditEvent::RtDeadlineMiss,
585                                &crate::audit::encode_u32x2(
586                                    task.cell_id.0 as u32,
587                                    task.deadline_misses,
588                                ),
589                            );
590                        }
591                    }
592                    task.state = TaskState::Ready;
593                    waking_tasks.push_back(*id);
594                }
595            }
596            for id in waking_tasks {
597                self.push_ready(id);
598            }
599
600            // 1b. Heartbeat liveness sweep: terminate any cell that opted into heartbeating
601            //     but missed its deadline — a SILENT hang (deadlock / stuck loop) that the
602            //     CPU-monopoly watchdog cannot see (that only fires on RT compute hogs). The
603            //     death flows through the normal path so the supervisor restarts it. Collect
604            //     first, then `exit_task` outside the iteration (it mutates self.tasks).
605            let mut hung: Vec<(usize, u64)> = Vec::new();
606            for (id, task) in self.tasks.iter() {
607                if let Some(d) = task.heartbeat_deadline {
608                    if now as u64 >= d {
609                        hung.push((*id, task.cell_id.0));
610                    }
611                }
612            }
613            for (tid, cell_raw) in hung {
614                log::error!(
615                    "[heartbeat] task {} (cell {}) missed liveness deadline — terminating (hung)",
616                    tid,
617                    cell_raw
618                );
619                // Dump the hung task's park point (Sending{target}/Recv{mask}/…) —
620                // a silent kill costs hours of triage; the blocked-on peer's state
621                // is the single most useful forensic fact for IPC hangs.
622                if let Some(t) = self.tasks.get(&tid) {
623                    log::error!("[heartbeat] task {} state at kill: {:?}", tid, t.state);
624                    if let TaskState::Sending { target, .. } = t.state {
625                        if let Some(tt) = self.tasks.get(&target) {
626                            log::error!(
627                                "[heartbeat] send target {} state: {:?} pending_msgs={}",
628                                target,
629                                tt.state,
630                                tt.pending_msgs.len()
631                            );
632                        } else {
633                            log::error!("[heartbeat] send target {} no longer exists", target);
634                        }
635                    }
636                }
637                crate::audit::log_event(
638                    crate::audit::AuditEvent::CellHung,
639                    &crate::audit::encode_u32x2(cell_raw as u32, tid as u32),
640                );
641                // Release resources the hung cell owned (each locks its own state, not
642                // SCHEDULER, so they are safe to call inline here — mirrors the watchdog kill).
643                crate::fast_ipc::clear_vfs_if_cell(cell_raw as usize);
644                crate::memory::cell_quota::deregister(CellId(cell_raw));
645                crate::resource_registry::release_for(CellId(cell_raw));
646                // Grant reap deferred: free_grant_pages acquires KERNEL_ROOT + FRAME_ALLOCATOR,
647                // which must not be held under SCHEDULER. yield_cpu() drains this list after unlock.
648                self.pending_grant_reap.push(tid);
649                self.exit_task(tid, usize::MAX);
650                // If this hart was running the hung task, clear its attribution.
651                let hart_id = super::hart_local::current_hart_id();
652                if super::hart_local::ready::current_task_id_for(hart_id) == tid {
653                    super::hart_local::set_current_cell_id(0);
654                    super::hart_local::ready::set_current_task_id(hart_id, 0);
655                }
656            }
657        }
658
659        // 2. Poll Async Tasks
660        let has_polling = self.tasks.values().any(|t| t.state == TaskState::Polling);
661        if has_polling {
662            let mut polled_tasks = Vec::new();
663            let waker = dummy_waker();
664            let mut cx = Context::from_waker(&waker);
665
666            // Iterate keys to avoid borrow check issues
667            let keys: Vec<usize> = self.tasks.keys().cloned().collect();
668            for id in keys {
669                if let Some(task) = self.tasks.get_mut(&id) {
670                    if task.state == TaskState::Polling {
671                        if let Some(ref mut future_enum) = task.pending_future {
672                            match future_enum {
673                                SyscallFuture::FileRead(fd, future) => {
674                                    // Poll the future
675                                    match future.as_mut().poll(&mut cx) {
676                                        Poll::Ready((file, res)) => {
677                                            // Restore file handle
678                                            // file is Box<dyn ViFile>
679                                            task.open_files.insert(*fd, FileHandle::new(file));
680
681                                            // Set return value (a0 / regs[10]); errors use the
682                                            // syscall ABI sentinel usize::MAX (same encoding as
683                                            // ViCell_syscall_dispatch), not a fake 0-byte success.
684                                            task.trap_frame.regs[10] =
685                                                res.unwrap_or(usize::MAX) as _;
686
687                                            // Wake task
688                                            task.state = TaskState::Ready;
689                                            task.pending_future = None;
690                                            polled_tasks.push(id);
691                                        }
692                                        Poll::Pending => {
693                                            // Still waiting
694                                        } //
695                                    }
696                                }
697                            }
698                        }
699                    }
700                }
701            }
702            for id in polled_tasks {
703                self.push_ready(id);
704            }
705        }
706
707        // After global sweep, fall through to per-hart pick.
708        self.pick_next_local(hart_id, now)
709    }
710
711    /// Per-hart task selection: watchdog on current task, then pop from local queue
712    /// (with work-stealing fallback).  Called by `pick_next` for both hart 0
713    /// (after global sweep) and all other harts.
714    fn pick_next_local(
715        &mut self,
716        hart_id: usize,
717        _now: usize,
718    ) -> Option<(
719        *mut crate::hal::arch::Context,
720        *const crate::hal::arch::Context,
721    )> {
722        use super::hart_local::ready as rl;
723
724        // 3. Decide if the current task yields, and run the CPU-monopoly watchdog.
725        let current_id_raw = rl::current_task_id_for(hart_id);
726        let current_id: Option<usize> = if current_id_raw > 0 {
727            Some(current_id_raw)
728        } else {
729            None
730        };
731        if let Some(cid) = current_id {
732            enum WdAction {
733                None,
734                Requeue,
735                Kill(u64),
736            }
737            let mut action = WdAction::None;
738            if let Some(task) = self.tasks.get_mut(&cid) {
739                if task.state == TaskState::Running {
740                    // Only RealTime-priority tasks can livelock the system.
741                    if task.priority >= api::TaskPriority::RealTime as u8 {
742                        task.run_ticks = task.run_ticks.saturating_add(1);
743                        if task.run_ticks >= WATCHDOG_WARN_TICKS && !task.rt_overrun_warned {
744                            task.rt_overrun_warned = true;
745                            crate::audit::log_event(
746                                crate::audit::AuditEvent::RtCpuOverrun,
747                                &crate::audit::encode_u32x2(task.cell_id.0 as u32, task.run_ticks),
748                            );
749                        }
750                        if task.run_ticks > WATCHDOG_BUDGET_TICKS {
751                            action = WdAction::Kill(task.cell_id.0);
752                        } else {
753                            task.state = TaskState::Ready;
754                            action = WdAction::Requeue;
755                        }
756                    } else {
757                        task.state = TaskState::Ready;
758                        action = WdAction::Requeue;
759                    }
760                } else {
761                    task.run_ticks = 0;
762                    task.rt_overrun_warned = false;
763                }
764            }
765            match action {
766                WdAction::Requeue => {
767                    self.push_ready(cid);
768                }
769                WdAction::Kill(cell_raw) => {
770                    log::error!(
771                        "[watchdog] task {} (cell {}) monopolized CPU >{} ticks (~{}s) — terminating",
772                        cid, cell_raw, WATCHDOG_BUDGET_TICKS, WATCHDOG_BUDGET_TICKS / 100
773                    );
774                    crate::audit::log_event(
775                        crate::audit::AuditEvent::CellFault,
776                        &crate::audit::encode_u32x2(cell_raw as u32, WATCHDOG_SCAUSE),
777                    );
778                    crate::fast_ipc::clear_vfs_if_cell(cell_raw as usize);
779                    crate::memory::cell_quota::deregister(CellId(cell_raw));
780                    crate::resource_registry::release_for(CellId(cell_raw));
781                    // Grant reap deferred: free_grant_pages acquires KERNEL_ROOT + FRAME_ALLOCATOR,
782                    // which must not be held under SCHEDULER. yield_cpu() drains this list after unlock.
783                    self.pending_grant_reap.push(cid);
784                    self.exit_task(cid, usize::MAX);
785                    super::hart_local::set_current_cell_id(0);
786                    rl::set_current_task_id(hart_id, 0);
787                }
788                WdAction::None => {}
789            }
790        }
791
792        // 4. Get next task: local queue first, then work-steal from busiest other hart.
793        let next_id = rl::pick_local(hart_id).or_else(|| {
794            super::hart_local::ready::steal_from_busiest(hart_id);
795            rl::pick_local(hart_id)
796        });
797
798        if let Some(nid) = next_id {
799            if let Some(next_task) = self.tasks.get_mut(&nid) {
800                next_task.state = TaskState::Running;
801                super::hart_local::set_current_cell_id(next_task.cell_id.0 as usize);
802                // x86_64 PKU: update CPU_LOCAL.pku_value for the incoming task so
803                // the asm ring-3 exit path restores the correct PKRU. Must run while
804                // we still hold a reference to the task (before releasing the lock).
805                #[cfg(target_arch = "x86_64")]
806                crate::hal::syscall::set_task_pku(next_task.pku_value);
807            }
808            if Some(nid) == current_id {
809                rl::set_current_task_id(hart_id, nid);
810                return None; // No switch needed
811            }
812            // SAFETY: Box<Task> pins the Task on the heap; pointer is valid until reap.
813            let next_ctx: *const crate::hal::arch::Context = self
814                .tasks
815                .get(&nid)
816                .map(|t| &t.context as *const _)
817                .unwrap_or_default();
818            rl::set_current_task_id(hart_id, nid);
819
820            if let Some(cid) = current_id {
821                let curr_ctx: *mut crate::hal::arch::Context =
822                    if let Some(t) = self.tasks.get_mut(&cid) {
823                        &mut t.context as *mut _
824                    } else if let Some(t) = self.zombies.iter_mut().find(|t| t.id == cid) {
825                        &mut t.context as *mut _
826                    } else {
827                        core::ptr::null_mut()
828                    };
829                if !curr_ctx.is_null() && !next_ctx.is_null() {
830                    return Some((curr_ctx, next_ctx));
831                }
832            } else if !next_ctx.is_null() {
833                return Some((core::ptr::null_mut(), next_ctx)); // first switch from boot context
834            }
835        } else {
836            // No ready tasks.
837            if let Some(cid) = current_id {
838                if self.zombies.iter().any(|t| t.id == cid) {
839                    // Zombie with no successor: switch to the idle boot context so
840                    // it can be reaped without holding the SCHEDULER lock.
841                    let curr_ctx = self
842                        .zombies
843                        .iter_mut()
844                        .find(|t| t.id == cid)
845                        .map(|t| &mut t.context as *mut _);
846                    if let Some(c) = curr_ctx {
847                        rl::set_current_task_id(hart_id, 0);
848                        return Some((c, core::ptr::null()));
849                    }
850                } else if let Some(task) = self.tasks.get_mut(&cid) {
851                    // Live blocked task with no peer ready to run.  Suspend it
852                    // cleanly by switching to the idle (boot) context so the CPU
853                    // can enter WFI and wake when a real event unblocks someone.
854                    //
855                    // Without this switch, yield_cpu returns without a context
856                    // change, the SVC handler gets stale Ok(0) results, and
857                    // current_task_id is reset to 0 — causing every subsequent
858                    // SVC to be denied.
859                    //
860                    // BOOT_CONTEXT is valid here: it is saved by __switch on the
861                    // very first boot→task context switch, which always precedes
862                    // any cell SVC.
863                    let curr_ctx = &mut task.context as *mut _;
864                    rl::set_current_task_id(hart_id, 0);
865                    return Some((curr_ctx, core::ptr::null()));
866                } else {
867                    rl::set_current_task_id(hart_id, 0);
868                }
869            } else {
870                rl::set_current_task_id(hart_id, 0);
871            }
872        }
873        None
874    }
875
876    pub fn current_task_mut(&mut self) -> Option<&mut Task> {
877        let tid =
878            super::hart_local::ready::current_task_id_for(super::hart_local::current_hart_id());
879        if tid > 0 {
880            self.tasks.get_mut(&tid).map(|b| &mut **b)
881        } else {
882            None
883        }
884    }
885
886    pub fn current_task_ref(&self) -> Option<&Task> {
887        let tid =
888            super::hart_local::ready::current_task_id_for(super::hart_local::current_hart_id());
889        if tid > 0 {
890            self.tasks.get(&tid).map(|b| &**b)
891        } else {
892            None
893        }
894    }
895
896    pub fn has_ready_tasks(&self) -> bool {
897        super::hart_local::ready::total_ready_count() > 0
898    }
899}

901/// Default entry point for kernel tasks
902#[no_mangle]
903extern "C" fn task_entry_point() {
904    // SAFETY: This is the entry point for new tasks. We need to:
905    // 1. Force unlock the scheduler (safe because we're in a new task context)
906    // 2. Initialize HAL for this task context
907    // 3. Enable interrupts (safe because stack is properly set up)
908    unsafe {
909        crate::task::SCHEDULER.force_unlock();
910        crate::hal::arch::init();
911        // Enable Interrupts MANUALLY now that we're safe and stack is clean
912        crate::hal::arch::enable_interrupts();
913    }
914    info!("Task started!");
915    loop {
916        for _ in 0..10_000_000 {
917            core::hint::spin_loop();
918        }
919        info!("Task tick (ID: {})...", crate::task::current_task_id());
920        crate::task::yield_cpu();
921    }
