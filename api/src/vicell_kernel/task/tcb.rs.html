11/// Maximum number of IPC messages buffered for a Frozen cell during hot-swap.
12///
13/// When a cell is mid-swap (state `Frozen`), incoming `sys_send` calls enqueue
14/// here instead of blocking the caller or returning an error.  If the buffer
15/// fills, additional senders receive `SyscallError::TryAgain` so they can
16/// back off.  Messages are drained to the new cell in Step 5 (UNFREEZE).
17pub const HOTSWAP_MSG_QUEUE_DEPTH: usize = 64;
18
19/// Deeper bound for INPUT key events queued to the focused cell.
20///
21/// A paste-speed serial burst produces 2 events/byte while the focused cell
22/// drains one event per recv loop iteration (each echo is an SBI call per
23/// byte on RISC-V — slow on TCG). Backlog accumulates ACROSS commands, so the
24/// shared 64-slot hotswap bound overflowed mid-line and silently dropped
25/// keystrokes ("vcat /tmp/x" arrived as "vcat /t"). 512 events ≈ 256 typed
26/// characters of headroom — beyond any realistic line — while still bounding
27/// a wedged focused cell to ~50 KiB of queued events.
28pub const INPUT_EVENT_QUEUE_DEPTH: usize = 512;

30/// A message buffered for a `Frozen` cell during a hot-swap sequence.
31///
32/// Invariants:
33/// - `data` holds an owned copy of the original message bytes (Law 2).
34/// - `sender_tid` identifies the originating task; the new cell sees it as
35///   `current_caller` when it next calls `sys_recv`.
36/// - `enqueued_tick` is for diagnostics; not used for ordering.
37pub struct PendingMsg {
38    pub sender_tid: usize,
39    pub data: Box<[u8]>,
40    pub enqueued_tick: u64,
41}

43#[derive(Debug, Clone, PartialEq)]
44pub enum TaskState {
45    Ready,
46    Running,
47    /// Sleeping until a specific monotonic time (ticks/ms)
48    Sleeping {
49        until: usize,
50    },
51    /// Blocked waiting to send a message to `target_id`.
52    /// Stores the message pointer and length temporarily.
53    Sending {
54        target: usize,
55        msg_ptr: VAddr,
56        msg_len: usize,
57    },
58    /// Blocked waiting to receive a message.
59    ///
60    /// `mask`: sender filter (0 = any sender).
61    /// `deadline`: optional monotonic tick count after which the kernel wakes
62    ///   this task with `ViError::Timeout`.  `None` = wait indefinitely.
63    Recv {
64        mask: usize,
65        buf_ptr: VAddr,
66        buf_len: usize,
67        deadline: Option<u64>,
68    },
69    /// This task has finished running.
70    Terminated,
71    /// Blocked on a Futex wait.
72    /// `addr`: The address being waited on.
73    FutexWait {
74        addr: VAddr,
75    },
76    /// Waiting for another task to exit (Join).
77    Waiting {
78        target: usize,
79    },
80    /// Polling an async future (e.g. syscall)
81    Polling,
82    /// Blocked in `WaitForEvent(mask, timeout)`.  Woken by `waker::signal_net_rx()`
83    /// (or equivalent) when any bit in `mask` fires, or when `deadline` ticks pass.
84    /// `deadline = None` means block indefinitely.
85    WaitEvent {
86        mask: u32,
87        deadline: Option<u64>,
88    },
89    /// Cell is suspended during a hot-swap sequence identified by `swap_id`.
90    ///
91    /// Invariants while Frozen:
92    /// - NOT in any scheduler ready queue.
93    /// - Cannot be killed by external signal (returns `ViError::PermissionDenied`).
94    /// - Can only be terminated by the hotswap orchestrator via `exit_task` after a
95    ///   successful swap, or rolled back to `Ready` on failure.
96    /// - Incoming IPC is queued in `crate::cell::hotswap::FROZEN` registry instead
97    ///   of being delivered (Phase 02 will drain the queue; P01 stubs this).
98    Frozen {
99        swap_id: u64,
100    },
101    /// Blocked in `sys_wait_irq(irq, mmio_base)` until hardware IRQ `irq` fires.
102    ///
103    /// Woken by the scheduler sweep (NOT from ISR — ISR only sets `IRQ_PENDING[irq]`).
104    /// No deadline: IRQ-only wake; Driver Cells must implement their own poll-fallback
105    /// if the device can hang without raising an IRQ.
106    WaitIrq {
107        irq: u8,
108    },
109}

148/// Task Control Block (TCB)
149#[allow(dead_code)]
150pub struct Task {
151    pub id: usize,
152    pub cell_id: CellId, // OWNER CELL
153    pub name: String,
154    pub state: TaskState,
155    pub context: Context,
156    pub trap_frame: ViTrapFrame,
157    pub allowed_drivers: Vec<usize>,
158    // Maps LeaseID -> Lease
159    pub leases: alloc::collections::BTreeMap<usize, Lease>,
160    // Next available Lease ID
161    pub next_lease_id: usize,
162
163    // Grant Table (Zero-Copy IPC)
164    // Maps GrantID -> GrantEntry
165    pub grant_table: alloc::collections::BTreeMap<usize, GrantEntry>,
166    pub next_grant_id: usize,
167
168    // Maps FD -> FileHandle
169    pub open_files: alloc::collections::BTreeMap<usize, FileHandle>,
170    // The Task ID that this task is currently handling a request FROM (for Reply).
171    pub current_caller: Option<usize>,
172    // Last Reply Value received
173    pub reply_value: Option<usize>,
174    // Current Working Directory
175    pub cwd: String,
176    // Stack management
177    pub kernel_stack: Option<super::stack::Stack>,
178    pub user_stack: Option<super::stack::Stack>,
179    /// Frames mapped for this cell's ELF segments, freed when the Task is dropped
180    /// (reaped). Without it, a cell's code/data frames leak on every death.
181    pub segment_mem: Option<super::stack::CellSegments>,
182
183    // Lifecycle
184    pub waiters: Vec<usize>,
185    pub exit_code: Option<usize>,
186    /// Death-notification queue (NotifyOnExit): tids of watched tasks that died
187    /// while this watcher was NOT parked in Recv. Drained (highest-priority) by the
188    /// next `Recv` so a supervisor never misses a child death during a respawn.
189    /// Each entry is `(dead_tid, exit_reason)` — the reason is delivered as the recv
190    /// payload (the NotifyOnExit contract) so a supervisor can apply a restart policy
191    /// (e.g. transient = restart only on abnormal exit). `exit_reason` is the exit code
192    /// for a clean exit (0) or `usize::MAX` for a fault / watchdog kill.
193    pub pending_deaths: Vec<(usize, usize)>,
194
195    /// Exit reason for a death notification delivered to a watcher that was PARKED in
196    /// `Recv` (set by `exit_task`). It is written into the watcher's recv buffer when its
197    /// `Recv` resumes — in the watcher's own syscall context, where writing a USER buffer
198    /// is valid (SSTATUS.SUM). It must NOT be written from `exit_task`/the trap context,
199    /// where an S-mode store to a USER page faults. `None` = the wake was a real message.
200    pub pending_exit_reason: Option<usize>,
201
202    // Async Kernel Support
203    pub pending_future: Option<SyscallFuture>,
204
205    /// Raw block-device access (BlkRead/BlkWrite/BlkFlush).  Granted at spawn for `/bin/vfs`.
206    pub block_io_cap: Option<super::cap::BlockIoCap>,
207    /// Network transmit/receive (NetTx/NetRx).  Granted at spawn for `/bin/net`.
208    pub network_cap: Option<super::cap::NetworkCap>,
209    /// Cell spawning + hot-swap (SpawnFromPath/SpawnPinned/HotSwap).
210    /// Granted at spawn for `/bin/init` and `/bin/shell`.
211    pub spawn_cap: Option<super::cap::SpawnCap>,
212    /// RISC-V H-extension CSR access for VMM cells.
213    /// Granted when manifest declares `hypervisor = true` AND the firmware reported H-ext.
214    pub hypervisor_cap: Option<super::cap::HypervisorCap>,
215    /// Supervisor authority: sys_freeze_cell / sys_resume_cell / sys_kill_cell.
216    /// Set ONLY by kernel init (direct TCB write). Never propagated through CapSet.
217    pub supervisor_cap: Option<super::cap::SupervisorCap>,
218    /// PCIe Driver Cell: claim BAR MMIO + authorise DMA via GrantDma.
219    /// Granted when manifest declares `pcie_driver = true`.
220    pub pcie_driver_cap: Option<super::cap::PcieDriverCap>,
221    /// Platform Cell: singleton capability gating `sys_register_pcie_bar`.
222    /// Granted by path match `/bin/platform` in loader.rs; at most one holder ever.
223    pub platform_cap: Option<super::cap::PlatformCap>,
224
225    /// MMIO device-class capability bitmask (`DEV_GPIO` / `DEV_UART` from
226    /// [`crate::resource_registry`]). Set from the ELF manifest's `gpio`/`uart`
227    /// flags. A cell may `sys_request_mmio` only ranges whose device class is
228    /// present here — a GPIO-only cell cannot claim the UART window (this is the
229    /// parameterized form of the old `mmio_cap: bool`). `0` = no MMIO access.
230    pub mmio_devices: u8,
231
232    /// Block-I/O partition range grants (Milestone 2.5 P03) — bitmask:
233    /// bit 0 = P1 FAT32 (`MANIFEST_FLAG_PART_DATA`), bit 1 = P4 littlefs
234    /// (`MANIFEST_FLAG_PART_LFS`). Checked by `check_block_access` for every
235    /// raw block syscall, on top of `block_io_cap`. P2 (cell table) and P3
236    /// (snapshot) have no bit — those ranges are kernel-only by construction.
237    pub block_regions: u8,
238
239    /// x86_64 PKU protection key for this cell's ELF pages (0=trusted, 1=standard, 2=ffi).
240    /// On other architectures this field exists but is always 0 and never consulted.
241    /// Key assignment: 0 = trusted-core cells (block_io/network/spawn/hypervisor),
242    ///                 1 = standard Tier-1 Rust cells,
243    ///                 2 = Tier-1b C/FFI cells (mlibc, DOOM).
244    pub pku_key: u8,
245
246    /// Precomputed PKRU register value for this cell's key domain.
247    /// Written to CPU_LOCAL.pku_value by the scheduler before ring-3 re-entry,
248    /// then loaded into PKRU by the asm exit paths (`__trap_exit` / `syscall_entry`).
249    /// Always 0 on non-x86_64 targets (and for trusted-core cells on x86_64).
250    pub pku_value: u32,
251
252    /// Scheduling priority tier.  Higher value = higher priority.
253    /// See `api::TaskPriority` for the three defined levels.
254    pub priority: u8,
255
256    /// Cluster participation mode parsed from `__ViCell_cluster` ELF section.
257    /// `0` = Isolated (default, no section present).  See `api::cluster::ClusterMode`.
258    ///
259    /// Invariant: a task with `cluster_mode != 0` MUST have `priority != RealTime (2)`.
260    /// The `SpawnPinned` handler enforces this at spawn time.
261    pub cluster_mode: u8,
262
263    /// FNV-1a-64 cluster routing identifier from `__ViCell_cluster`.
264    /// `0` when `cluster_mode == Isolated`.  NOT a credential — routing only.
265    pub cluster_id: u64,
266
267    /// Per-Cell syscall allowlist.  Each bit corresponds to a syscall via
268    /// `api::ViSyscall::allowlist_bit()`.  `u64::MAX` = permit all (default,
269    /// used when the Cell ELF does not embed a `__ViCell_syscalls` section).
270    pub syscall_allowlist: u64,
271
272    /// Watchdog: consecutive 10 ms scheduler ticks this task has been Running
273    /// WITHOUT voluntarily blocking. Incremented each tick it is found Running in
274    /// `pick_next`, reset to 0 the moment it blocks (Recv/Send/Sleep/etc). A
275    /// runaway (infinite loop, never yields) climbs until it crosses the watchdog
276    /// budget and is terminated — preventing livelock ("alive but paralyzed").
277    pub run_ticks: u32,
278
279    /// Cumulative count of `RecvTimeout` deadlines this task has missed (the awaited
280    /// message did not arrive in time). For an RT control loop this is its missed-cycle
281    /// count. Observability only — surfaced via the audit ring ([`crate::audit`]); the
282    /// scheduler does not act on it (RT enforcement is hardware-data-gated).
283    pub deadline_misses: u32,
284
285    /// One-shot latch: set when this task has already emitted an `RtCpuOverrun` warning
286    /// for the current non-yielding episode, so the early-warning audit fires once per
287    /// episode (not every tick). Reset to false whenever the task voluntarily blocks.
288    pub rt_overrun_warned: bool,
289
290    /// Liveness-heartbeat deadline (absolute `system_ticks`). `Some(d)` means the cell
291    /// opted into heartbeating and must call `Heartbeat` again before tick `d`, else the
292    /// scheduler terminates it as HUNG (silent-hang detection — see `pick_next`). `None`
293    /// = heartbeat disabled (the default; most cells don't opt in).
294    pub heartbeat_deadline: Option<u64>,
295
296    /// Per-cell virtual memory area list for demand-paging.
297    ///
298    /// Populated by the ELF loader (Phase 04) with one entry per ELF segment.
299    /// The #PF handler consults this list when a user-mode page fault occurs to
300    /// decide whether to map the page on demand or kill the cell.
301    /// Empty until the ELF loader runs; demand-paging is inert on RISC-V/AArch64
302    /// (those arches use identity-mapped segments today).
303    pub vma: crate::memory::vma::VmaList,
304
305    /// Set by `HotSwapReady` (syscall 401) to signal that the new cell has
306    /// finished deserializing state and is ready to receive IPC.
307    /// Cleared to `false` at spawn; polled by `wait_for_hotswap_ready`.
308    pub hotswap_ready: bool,
309
310    /// When `true`, `FreezeCell` and `KillCell` syscalls are rejected with
311    /// `PermissionDenied`.  Set at spawn for `init` and kernel-owned cells.
312    /// Prevents a compromised Supervisor Cell from disabling the restart tree.
313    pub is_critical: bool,
314
315    /// IPC messages buffered while this task is in `TaskState::Frozen` (hot-swap).
316    ///
317    /// Callers that `sys_send` to a Frozen task have their message copied here
318    /// (owned buffer; Law 2) and return immediately with `Ok(0)` — they are NOT
319    /// blocked in `Sending` state.  Step 5 of the hotswap orchestrator drains
320    /// this queue to the new cell before the old cell is terminated.
321    /// Bounded by `HOTSWAP_MSG_QUEUE_DEPTH`; overflow returns `TryAgain`.
322    pub pending_msgs: Vec<PendingMsg>,
323}

377    pub fn add_lease(&mut self, ptr: VAddr, len: usize, attributes: LeaseAttributes) -> usize {
378        let id = self.next_lease_id;
379        self.next_lease_id += 1;
380
381        let lease = Lease {
382            id,
383            ptr,
384            len,
385            attributes,
386        };
387
388        self.leases.insert(id, lease);
389        id
390    }
391
392    pub fn get_lease(&self, id: usize) -> Option<&Lease> {
393        self.leases.get(&id)
394    }
395
396    pub fn revoke_lease(&mut self, id: usize) {
397        self.leases.remove(&id);
398    }

400    // --- Grant Table Methods ---
401    pub fn add_grant(&mut self, ptr: VAddr, len: usize, flags: u32, sender_id: usize) -> usize {
402        let id = self.next_grant_id;
403        self.next_grant_id += 1;
404        self.grant_table.insert(
405            id,
406            GrantEntry {
407                ptr,
408                len,
409                flags,
410                sender_id,
411            },
412        );
413        id
414    }
415
416    pub fn get_grant(&self, id: usize) -> Option<&GrantEntry> {
417        self.grant_table.get(&id)
418    }
419
420    pub fn remove_grant(&mut self, id: usize) -> Option<GrantEntry> {
421        self.grant_table.remove(&id)
422    }
