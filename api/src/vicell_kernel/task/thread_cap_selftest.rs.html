1//! Boot self-tests for two trust-model fixes (2026-07-13):
2//!   #7 — spawned threads inherit the parent cell's identity (CellId + CapSet +
3//!        syscall_allowlist + PKU domain), closing the `CellId(0)` quota escape.
4//!   #5 — `sys_cap_revoke` refuses ambient-authority bits (HYPERVISOR, MMIO) with
5//!        `NotSupported` instead of silently clearing a field it cannot truly revoke.
6//!
7//! Runs in the single-hart window AFTER `task::init()` but BEFORE
8//! `smp::start_secondaries()` (see `main.rs`), so a synthetic thread inserted into
9//! the boot hart's ready queue can never be picked up by another hart before this
10//! test removes it. All synthetic tasks are created here and torn down before the
11//! function returns — the scheduler is left exactly as it was found.
12//!
13//! Invariant proven, not assumed: after the fix the spawned thread's `cell_id`
14//! equals the parent's (not `CellId(0)`), which is what routes its allocations to
15//! the parent's quota (the charge path keys off `cell_id`; the cell-spawn path in
16//! `loader.rs` already relies on this identity).
17
18use super::cap::CapSet;
19use super::syscall::{handle_syscall, Syscall, SyscallError};
20use super::tcb::Task;
21use api::syscall::cap_mask as CM;
22use types::CellId;
23
24// Synthetic tids outside any range the boot sequence has assigned yet. Removed
25// before return, so they never collide with real cells.
26const PARENT_TID: usize = 9001;
27const TARGET_TID: usize = 9002;
28const CTRL_TID: usize = 9003;
29
30// Distinctive sentinels so an inherited value is unambiguously the parent's.
31const TEST_CELL_ID: u64 = 0x00C0_FFEE;
32// A restricted allowlist that still PERMITS the Spawn syscall (so the global
33// allowlist gate in handle_syscall lets the call through) yet differs from the
34// Task::new default of u64::MAX — bit 63 is unassigned (real syscalls use bits
35// ≤54), so clearing it is behaviourally inert but proves the field was inherited
36// rather than left at the permit-all default.
37const TEST_ALLOWLIST: u64 = !(1u64 << 63);
38const TEST_PKU_KEY: u8 = 2;
39const TEST_PKU_VALUE: u32 = 0xABCD_1234;
40
41/// Build a bare task with the given tid/cell and no caps. Only the fields this
42/// test reads/writes need to be meaningful; the task is never scheduled.
43fn mk_task(tid: usize, cell: u64) -> alloc::boxed::Box<Task> {
44    alloc::boxed::Box::new(Task::new(
45        tid,
46        CellId(cell),
47        "selftest",
48        alloc::vec::Vec::new(),
49    ))
50}
51
52/// Insert a task into the scheduler map (no ready-queue push — never scheduled).
53fn insert(task: alloc::boxed::Box<Task>) {
54    if let Some(sched) = super::SCHEDULER.lock().as_mut() {
55        sched.tasks.insert(task.id, task);
56    }
57}
58
59/// Remove a tid from the scheduler map AND every hart's ready queue.
60fn remove(tid: usize) {
61    if let Some(sched) = super::SCHEDULER.lock().as_mut() {
62        sched.tasks.remove(&tid);
63    }
64    super::hart_local::ready::remove_from_all(tid);
65}
66
67/// Returns true iff both fixes behave as specified. Logs a decisive serial line.
68///
69/// Transparent to the boot sequence: the real thread spawn it performs advances
70/// `next_task_id`, so the counter is snapshotted on entry and restored on exit —
71/// the first real cell (Platform) still gets the tid it would have without this
72/// test, keeping CellId assignment stable whether or not the test is compiled in.
73pub fn self_test() -> bool {
74    let mut ok = true;
75
76    let saved_next_tid = super::SCHEDULER.lock().as_ref().map(|s| s.next_task_id);
77
78    // ── #7: thread identity inheritance ────────────────────────────────────────
79    // Parent cell: a distinctive CellId, a restricted allowlist, a non-zero PKU
80    // domain, and one transferable cap (network) + SpawnCap (needed to spawn).
81    {
82        let mut parent = mk_task(PARENT_TID, TEST_CELL_ID);
83        parent.network_cap = Some(super::cap::NetworkCap::new());
84        parent.spawn_cap = Some(super::cap::SpawnCap::new());
85        parent.syscall_allowlist = TEST_ALLOWLIST;
86        parent.pku_key = TEST_PKU_KEY;
87        parent.pku_value = TEST_PKU_VALUE;
88        let parent_caps = CapSet::of_task(&parent);
89        insert(parent);
90
91        // Invoke the REAL Spawn path as if the parent called it. entry/arg are
92        // placeholders; the thread is removed before it can run.
93        let spawned = handle_syscall(
94            PARENT_TID,
95            Syscall::Spawn {
96                entry: 0x1000,
97                arg: 0,
98            },
99        );
100
101        match spawned {
102            Ok(thread_tid) if thread_tid != 0 => {
103                if let Some(sched) = super::SCHEDULER.lock().as_ref() {
104                    if let Some(t) = sched.tasks.get(&thread_tid) {
105                        let cell_ok = t.cell_id.0 == TEST_CELL_ID; // NOT CellId(0)
106                        let caps_ok = CapSet::of_task(t) == parent_caps;
107                        let allow_ok = t.syscall_allowlist == TEST_ALLOWLIST; // NOT u64::MAX
108                        let pku_ok = t.pku_key == TEST_PKU_KEY && t.pku_value == TEST_PKU_VALUE;
109                        if !(cell_ok && caps_ok && allow_ok && pku_ok) {
110                            ok = false;
111                            log::error!(
112                                "[selftest] THREAD-INHERIT: FAIL \
113                                cell={} caps={} allow={} pku={}",
114                                cell_ok,
115                                caps_ok,
116                                allow_ok,
117                                pku_ok
118                            );
119                        }
120                    } else {
121                        ok = false;
122                        log::error!("[selftest] THREAD-INHERIT: FAIL — thread tid absent");
123                    }
124                }
125                remove(thread_tid);
126            }
127            _ => {
128                ok = false;
129                log::error!(
130                    "[selftest] THREAD-INHERIT: FAIL — spawn returned {:?}",
131                    spawned
132                );
133            }
134        }
135        remove(PARENT_TID);
136    }
137
138    // ── #5: honest revoke (ambient bits refused, lazy bits still work) ──────────
139    {
140        // Caller holds SpawnCap (Gate 1). Target holds an ambient cap (hypervisor)
141        // and an MMIO device bit — neither is truly revocable yet, so revoke must
142        // REFUSE with NotSupported and leave the fields intact.
143        let mut caller = mk_task(PARENT_TID, TEST_CELL_ID);
144        caller.spawn_cap = Some(super::cap::SpawnCap::new());
145        insert(caller);
146
147        let mut target = mk_task(TARGET_TID, 0x1111);
148        target.hypervisor_cap = Some(super::cap::HypervisorCap::new());
149        target.mmio_devices = crate::resource_registry::DEV_GPIO;
150        insert(target);
151
152        // (a) HYPERVISOR bit → NotSupported, cap untouched.
153        let r_hyp = handle_syscall(
154            PARENT_TID,
155            Syscall::CapRevoke {
156                target_tid: TARGET_TID,
157                cap_mask: CM::HYPERVISOR,
158            },
159        );
160        // (b) MMIO bits → NotSupported, device mask untouched.
161        let r_mmio = handle_syscall(
162            PARENT_TID,
163            Syscall::CapRevoke {
164                target_tid: TARGET_TID,
165                cap_mask: CM::MMIO_MASK,
166            },
167        );
168
169        let refused = matches!(r_hyp, Err(SyscallError::NotSupported))
170            && matches!(r_mmio, Err(SyscallError::NotSupported));
171        let intact = if let Some(sched) = super::SCHEDULER.lock().as_ref() {
172            sched
173                .tasks
174                .get(&TARGET_TID)
175                .map(|t| {
176                    t.hypervisor_cap.is_some()
177                        && t.mmio_devices == crate::resource_registry::DEV_GPIO
178                })
179                .unwrap_or(false)
180        } else {
181            false
182        };
183        if !(refused && intact) {
184            ok = false;
185            log::error!(
186                "[selftest] REVOKE-REFUSE: FAIL refused={} intact={} (hyp={:?} mmio={:?})",
187                refused,
188                intact,
189                r_hyp,
190                r_mmio
191            );
192        }
193
194        // (c) Positive control: SPAWN is a lazy (syscall-gated) bit — revoke of a
195        // non-system target must still SUCCEED and clear the field.
196        let mut ctrl = mk_task(CTRL_TID, 0x2222);
197        ctrl.spawn_cap = Some(super::cap::SpawnCap::new());
198        insert(ctrl);
199        let r_spawn = handle_syscall(
200            PARENT_TID,
201            Syscall::CapRevoke {
202                target_tid: CTRL_TID,
203                cap_mask: CM::SPAWN,
204            },
205        );
206        let cleared = if let Some(sched) = super::SCHEDULER.lock().as_ref() {
207            sched
208                .tasks
209                .get(&CTRL_TID)
210                .map(|t| t.spawn_cap.is_none())
211                .unwrap_or(false)
212        } else {
213            false
214        };
215        if !(r_spawn.is_ok() && cleared) {
216            ok = false;
217            log::error!(
218                "[selftest] REVOKE-ALLOW: FAIL r={:?} cleared={}",
219                r_spawn,
220                cleared
221            );
222        }
223
224        remove(PARENT_TID);
225        remove(TARGET_TID);
226        remove(CTRL_TID);
227    }
228
229    // Restore the spawn counter so the test leaves no trace on tid assignment.
230    if let (Some(sched), Some(n)) = (super::SCHEDULER.lock().as_mut(), saved_next_tid) {
231        sched.next_task_id = n;
232    }
233
234    if ok {
235        log::info!("[selftest] THREAD-CAP: PASS (thread-inherit + honest-revoke)");
236    } else {
237        log::error!("[selftest] THREAD-CAP: FAIL");
238    }
239    ok
240}
