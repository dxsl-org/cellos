1//! SMP: secondary hart startup and controlled park loop.
2//!
3//! Phase 01: brings each secondary hart online, installs its trap vector,
4//! then parks it in WFI.  Phase 03 replaces the park loop with a per-hart
5//! scheduler round.
6//!
7//! Invariant: hart 0 calls `start_secondaries()` only AFTER `task::init()`
8//! completes — the SCHEDULER and heap are live before any secondary runs.
9
10use core::sync::atomic::{AtomicBool, Ordering};
11
12/// Maximum number of harts this kernel tracks.  2 covers QEMU virt `-smp 2`
13/// (G2 entry target).  Constant so secondary stacks and HART_ONLINE are
14/// statically sized — no heap allocation during the boot critical path.
15pub const MAX_HARTS: usize = 2;
16
17/// Hart dedicated to RealTime-priority cells.  RT tasks are enqueued here by
18/// `push_ready` and never stolen (Phase 03 steal filter excludes RT).
19pub const HART_RT: usize = 1;
20
21/// Set to `true` by each secondary hart once its trap vector and timer are ready.
22/// Hart 0's bounded wait reads this via `Acquire` to observe all preceding stores.
23pub static HART_ONLINE: [AtomicBool; MAX_HARTS] = [AtomicBool::new(false), AtomicBool::new(false)];
24
25/// How many 10 ms ticks hart 0 waits for each secondary to come online before
26/// logging a warning and continuing single-hart.  500 ms is generous for QEMU.
27/// Only consumed by `start_secondaries`, which is riscv64-only (SBI HSM). Gated
28/// to avoid a dead-code warning on aarch64/x86_64.
29#[cfg(target_arch = "riscv64")]
30const SECONDARY_BOOT_TIMEOUT_TICKS: usize = 50;
31
32/// Called by hart 0 **after** `task::init()` to bring secondary harts online.
33///
34/// Each secondary is started via SBI HSM `hart_start`.  Hart 0 then spins
35/// (bounded) waiting for each secondary to set `HART_ONLINE[hart_id]`.
36/// If a secondary fails to start or times out, a warning is logged and the
37/// system continues single-hart — graceful degradation, never a panic.
38#[cfg(target_arch = "riscv64")]
39pub fn start_secondaries() {
40    use crate::task::stack::Stack;
41    use crate::task::STACK_PAGES;
42    use hal::common::sbi::{sbi_hart_get_status, sbi_hart_start};
43
44    extern "C" {
45        // Physical asm label defined in hal/arch/riscv/src/rv64/boot.rs.
46        // Runs bare (SATP=0); no relocation or BSS clear.
47        fn _secondary_entry();
48    }
49
50    for (hart_id, online) in HART_ONLINE.iter().enumerate().skip(1) {
51        // Allocate a dedicated kernel stack for this hart.  Leak it — it lives
52        // for the entire lifetime of the hart.
53        let stack = match Stack::new_kernel(STACK_PAGES) {
54            Ok(s) => s,
55            Err(e) => {
56                log::warn!("[smp] hart {} stack alloc failed: {:?}", hart_id, e);
57                continue;
58            }
59        };
60        let stack_top = stack.top;
61        core::mem::forget(stack);
62
63        if let Ok(state) = sbi_hart_get_status(hart_id) {
64            log::info!("[smp] hart {} HSM state = {}", hart_id, state);
65        }
66
67        // SAFETY: _secondary_entry is a physical-address asm label; the kernel
68        // is loaded at 0x80200000 with slide=0 so physical == virtual.
69        // stack_top is the usable top of a freshly-allocated kernel stack.
70        // SAFETY: casting function pointer to integer — use double-cast through
71        // *const () to avoid the "direct cast of function item" lint.
72        let entry_paddr = _secondary_entry as *const () as usize;
73        match sbi_hart_start(hart_id, entry_paddr, stack_top) {
74            Ok(()) => log::info!(
75                "[smp] hart {} start requested (entry={:#x})",
76                hart_id,
77                entry_paddr
78            ),
79            Err(e) => {
80                log::warn!("[smp] hart {} SBI hart_start failed: err={}", hart_id, e);
81                continue;
82            }
83        }
84
85        // Bounded spin: wait for the secondary to signal it is online.
86        let deadline = crate::task::system_ticks() + SECONDARY_BOOT_TIMEOUT_TICKS;
87        loop {
88            if online.load(Ordering::Acquire) {
89                log::info!("[smp] hart {} online, parked", hart_id);
90                break;
91            }
92            if crate::task::system_ticks() >= deadline {
93                log::warn!(
94                    "[smp] hart {} did not come online in time — continuing single-hart",
95                    hart_id
96                );
97                break;
98            }
99            core::hint::spin_loop();
100        }
101    }
102}

104/// No-op on non-riscv64 targets.
105#[cfg(not(target_arch = "riscv64"))]
106pub fn start_secondaries() {}

108/// Returns `true` when the RT hart (hart 1) successfully came online.
109///
110/// Used by the scheduler to fall back to hart 0 on single-hart systems
111/// (e.g. QEMU without `-smp 2`) so RT-priority tasks still get scheduled.
112#[inline]
113pub fn is_rt_hart_online() -> bool {
114    HART_ONLINE[HART_RT].load(core::sync::atomic::Ordering::Relaxed)
115}
116
117/// Entry point for secondary harts, called from `_secondary_entry` asm.
118///
119/// a0 = hart_id (set by OpenSBI per SBI HSM §9.1.1).
120///
121/// Installs the trap vector, enables the timer, runs the per-hart scheduler loop.
122#[no_mangle]
123pub extern "C" fn smp_hart_entry(hart_id: usize) -> ! {
124    // Install the trap vector (each hart has its own stvec CSR).
125    // `hal::ARCH.init()` sets stvec + enables SSIE.
126    #[cfg(target_arch = "riscv64")]
127    {
128        use hal::Arch;
129        hal::ARCH.init();
130    }
131
132    // Install per-hart local state (current_cell_id, tp, ready queues).
133    #[cfg(target_arch = "riscv64")]
134    crate::task::hart_local::install(hart_id);
135
136    // Enable S-mode timer interrupt and arm the first tick on this hart.
137    // Each hart has its own mtimecmp register via SBI; arming here starts
138    // the 10ms preemption slice for this hart.
139    #[cfg(target_arch = "riscv64")]
140    {
141        // SAFETY: csrs on sie is always legal from S-mode (RISC-V priv spec §4.1.3).
142        unsafe {
143            core::arch::asm!("csrs sie, {stie}", stie = in(reg) 0x20usize);
144        }
145        let next = hal::common::timer::read_mtime() + hal::common::timer::TICKS_PER_10MS;
146        hal::common::sbi::set_timer(next);
147    }
148
149    // Signal hart 0's bounded wait that we are ready.
150    if hart_id < MAX_HARTS {
151        HART_ONLINE[hart_id].store(true, Ordering::Release);
152    }
153
154    // Per-hart scheduler loop.  The timer ISR (vi_timer_tick) calls yield_cpu()
155    // which runs pick_next for THIS hart (work-stealing from hart 0 if idle).
156    // Between ticks we sit in WFI to save power.  Interrupts are enabled on
157    // entry (ARCH.init() sets sstatus.SIE=1), so WFI fires on the timer ISR.
158    loop {
159        // SAFETY: wfi suspends until the next interrupt; state is unchanged.
160        #[cfg(any(target_arch = "riscv64", target_arch = "riscv32"))]
161        unsafe {
162            core::arch::asm!("wfi", options(nomem, nostack))
163        };
164        core::hint::spin_loop();
165    }
166}
