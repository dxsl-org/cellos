13/// Set of physical frames currently issued via `ShmAlloc`.
14/// `ShmMap` accepts only handles that appear here, preventing a malicious
15/// cell from mapping arbitrary kernel/cell-owned frames into its address
16/// space via a forged handle.
17///
18/// NOTE: This is still a single global pool — any cell that knows a peer's
19/// outstanding handle can map it. A per-owner ACL is the proper fix; this
20/// table is the minimum bar to stop "ShmMap kernel_text_phys" attacks.
21static SHM_HANDLES: Spinlock<Option<BTreeSet<usize>>> = Spinlock::new(None);
22
23fn shm_handles_lock() -> &'static Spinlock<Option<BTreeSet<usize>>> {
24    &SHM_HANDLES
25}
26
27fn shm_register(handle: usize) {
28    let mut guard = shm_handles_lock().lock();
29    if guard.is_none() {
30        *guard = Some(BTreeSet::new());
31    }
32    if let Some(set) = guard.as_mut() {
33        set.insert(handle);
34    }
35}
36
37fn shm_is_valid(handle: usize) -> bool {
38    let guard = shm_handles_lock().lock();
39    guard.as_ref().is_some_and(|set| set.contains(&handle))
40}

44/// Kernel-managed zero-copy memory region.
45///
46/// Distinct from `tcb::GrantEntry` which tracks per-task grants from the kernel.
47/// Owner and grantee are tracked by raw task id (usize) — same as `caller_id`

49/// because `caller_id` IS a task id, not a Cell id (F7).
50struct PageGrant {
51    base: usize,                           // physical address of the first allocated page
52    size: usize,                           // total byte count (multiple of 4096)
53    owner: usize,                          // task id of the GrantAlloc caller
54    shared_to: Option<(usize, GrantPerm)>, // current grantee task id + permission
55}

87/// Allocate `n_pages` contiguous physical frames, map them USER RW, and zero them.
88///
89/// Returns the physical base address on success, or `None` on OOM or partial map.
90/// Lock order: FRAME_ALLOCATOR (alloc) → FRAME_ALLOCATOR (map_page) → release →
91///             FRAME_ALLOCATOR (partial-failure dealloc).
92fn alloc_grant_pages(n_pages: usize) -> Option<usize> {
93    use crate::memory::frame::FRAME_ALLOCATOR;
94    use crate::memory::paging::Flags;
95    const PAGE_SIZE: usize = 4096;
96
97    let user_flags =
98        Flags::VALID | Flags::READ | Flags::WRITE | Flags::USER | Flags::ACCESSED | Flags::DIRTY;
99
100    let paddr = {
101        let mut g = FRAME_ALLOCATOR.lock();
102        g.as_mut().and_then(|a| a.allocate_contiguous(n_pages))?
103    };
104
105    let mut mapped = 0usize;
106    {
107        let mut guard = FRAME_ALLOCATOR.lock();
108        if let Some(alloc) = guard.as_mut() {
109            for i in 0..n_pages {
110                let v = paddr + i * PAGE_SIZE;
111                if crate::memory::paging::map_page(alloc, v, v, Flags::from_bits(user_flags))
112                    .is_ok()
113                {
114                    mapped += 1;
115                } else {
116                    break;
117                }
118            }
119        }
120    }
121
122    if mapped < n_pages {
123        // Partial map: restore the kernel identity mapping for the pages we
124        // re-mapped USER (NOT a bare unmap — every Usable frame must stay
125        // identity-mapped for the loader's identity-address zeroing), then free.
126        let kernel_rwx = Flags::VALID
127            | Flags::READ
128            | Flags::WRITE
129            | Flags::EXECUTE
130            | Flags::ACCESSED
131            | Flags::DIRTY;
132        let mut fa = FRAME_ALLOCATOR.lock();
133        if let Some(a) = fa.as_mut() {
134            for i in 0..mapped {
135                let f = paddr + i * PAGE_SIZE;
136                let _ = crate::memory::paging::unmap_page(f);
137                let _ = crate::memory::paging::map_page(a, f, f, Flags::from_bits(kernel_rwx));
138            }
139            for k in 0..n_pages {
140                a.deallocate_frame(paddr + k * PAGE_SIZE);
141            }
142        }
143        crate::memory::paging::tlb_flush_all();
144        return None;
145    }
146
147    // Zero every mapped page before handing to user: prevents stale data from a
148    // previously-freed grant leaking to a different cell (info-disclosure under G2).
149    // SAFETY: frames are identity-mapped USER RW; SUM=1 allows S-mode writes.
150    unsafe {
151        core::ptr::write_bytes(paddr as *mut u8, 0, n_pages * PAGE_SIZE);
152    }
153
154    Some(paddr)
155}

157/// Restore the boot identity mapping (kernel RWX) and deallocate `n_pages`
158/// physical frames starting at `base`.
159///
160/// Grant frames are identity-mapped at boot (RWX kernel) and re-mapped USER RW by
161/// `alloc_grant_pages`. On free we must RESTORE the kernel identity mapping — NOT
162/// unmap it: in the SAS model every Usable frame must stay identity-mapped so the
163/// cell loader can zero a reused frame through its identity address
164/// (`phys_to_virt(frame)`). Unmapping here left freed grant frames with no PTE →
165/// a later cell load store-faulted while zeroing BSS (and read wrong pages).
166/// Dropping the USER bit also prevents a stale cell from touching a reused frame.
167///
168/// Lock order: FRAME_ALLOCATOR → (map_page/unmap_page → KERNEL_ROOT). Mirrors
169/// `Stack::drop`. Must NOT be called while already holding FRAME_ALLOCATOR.
170fn free_grant_pages(base: usize, n_pages: usize) {
171    use crate::memory::frame::FRAME_ALLOCATOR;
172    use crate::memory::paging::Flags;
173    const PAGE_SIZE: usize = 4096;
174    let kernel_rwx = Flags::from_bits(
175        Flags::VALID | Flags::READ | Flags::WRITE | Flags::EXECUTE | Flags::ACCESSED | Flags::DIRTY,
176    );
177
178    let mut fa = FRAME_ALLOCATOR.lock();
179    if let Some(alloc) = fa.as_mut() {
180        for i in 0..n_pages {
181            let f = base + i * PAGE_SIZE;
182            let _ = crate::memory::paging::unmap_page(f);
183            let _ = crate::memory::paging::map_page(alloc, f, f, kernel_rwx);
184            alloc.deallocate_frame(f);
185        }
186    }
187    crate::memory::paging::tlb_flush_all();
188}

190// ── Grant Reaper ──────────────────────────────────────────────────────────────
191
192/// Reclaim all grant pages owned or held by a dying task.
193///
194/// Called from every task-exit code path (Exit syscall, ForceExit, scheduler watchdog, fault handler).
195/// Two effects:
196///   1. Owner death  — remove entry, unmap pages, return frames to allocator.
197///   2. Grantee death — clear `shared_to` so the owner's grant becomes unshared.
198///
199/// Lock order: PAGE_GRANT_TABLE collect → unmap (KERNEL_ROOT) → FRAME_ALLOCATOR.
200/// Never holds FRAME_ALLOCATOR when calling free_grant_pages.
201pub(crate) fn reap_grants_for_task(dead_tid: usize) {
202    const PAGE_SIZE: usize = 4096;
203
204    // ── PAGE_GRANT_TABLE pass ─────────────────────────────────────────────────
205    let owned: alloc::vec::Vec<PageGrant> = {
206        let mut tbl = grant_table_lock().lock();
207        let Some(map) = tbl.as_mut() else { return };
208        // Clear grantee references (no removal needed — owner keeps the entry).
209        for grant in map.values_mut() {
210            if grant.shared_to.is_some_and(|(tid, _)| tid == dead_tid) {
211                grant.shared_to = None;
212            }
213        }
214        // Collect and remove owned entries.
215        let owned_keys: alloc::vec::Vec<usize> = map
216            .iter()
217            .filter(|(_, g)| g.owner == dead_tid)
218            .map(|(k, _)| *k)
219            .collect();
220        owned_keys.iter().filter_map(|k| map.remove(k)).collect()
221    }; // PAGE_GRANT_TABLE lock released
222
223    for grant in &owned {
224        free_grant_pages(grant.base, grant.size / PAGE_SIZE);
225    }
226
227    // ── REG_GRANT_TABLE pass ──────────────────────────────────────────────────
228    let reg_owned: alloc::vec::Vec<RegGrant> = {
229        let mut tbl = reg_grant_table_lock().lock();
230        let Some(map) = tbl.as_mut() else { return };
231        // Clear grantee references when the grantee dies.
232        for grant in map.values_mut() {
233            if grant.shared_to.is_some_and(|(tid, _)| tid == dead_tid) {
234                grant.shared_to = None;
235            }
236        }
237        let owned_keys: alloc::vec::Vec<usize> = map
238            .iter()
239            .filter(|(_, g)| g.owner == dead_tid)
240            .map(|(k, _)| *k)
241            .collect();
242        owned_keys.iter().filter_map(|k| map.remove(k)).collect()
243    }; // REG_GRANT_TABLE lock released
244
245    for reg in &reg_owned {
246        free_grant_pages(reg.base, reg.size / PAGE_SIZE);
247    }
248}

291/// Per-cell block-I/O range gate (Milestone 2.5 P03).
292///
293/// Replaces the old global `sector >= CELL_TABLE_BASE_LBA` check: the caller's
294/// `block_regions` bitmask (from its manifest PART_* bits, or the legacy VFS
295/// grant) defines which MBR partitions its raw block syscalls may address.
296/// Deny-by-default: a sector outside every granted partition is rejected —
297/// which structurally protects P2 (cell table) and P3 (snapshot), since no
298/// bit exists for them. Logs every denial (silent denials cost us a day once).
299fn check_block_access(caller_id: usize, sector: u64, count: u64) -> bool {
300    use crate::loader::disk_layout as dl;
301    let regions = super::SCHEDULER
302        .lock()
303        .as_ref()
304        .and_then(|s| s.tasks.get(&caller_id))
305        .map(|t| t.block_regions)
306        .unwrap_or(0);
307    let end = match sector.checked_add(count) {
308        Some(e) => e,
309        None => return false,
310    };
311    const GRANTABLE: [(u8, u64, u64); 4] = [
312        (0b001, dl::PART_FAT32_BASE_LBA, dl::PART_FAT32_SECTORS), // P1 (PART_DATA)
313        (0b010, dl::PART_LFS_BASE_LBA, dl::PART_LFS_SECTORS),     // P4 (PART_LFS)
314        (0b100, dl::PART_SRV_BASE_LBA, dl::PART_SRV_SECTORS), // P5 (SRV/RedoxFS, co-granted w/ LFS)
315        (
316            0b1000,
317            dl::PART_CELLSTORE_BASE_LBA,
318            dl::PART_CELLSTORE_SECTORS,
319        ), // P6 (cell-store, read-only; VFS /bin overlay)
320    ];
321    for (bit, base, size) in GRANTABLE {
322        if regions & bit != 0 && sector >= base && end <= base + size {
323            return true;
324        }
325    }
326    log::warn!(
327        "[blk] sector {}..{} denied for tid {} (regions={:#04b})",
328        sector,
329        end,
330        caller_id,
331        regions
332    );
333    false
334}

360/// Validate a user-supplied (ptr, len) buffer descriptor.
361///
362/// Rejects: NULL pointer, zero-length when expected non-empty, lengths above
363/// `max`, and pointer+length arithmetic overflow.
364///
365/// Does NOT walk the page table to confirm the U-bit. The trap handler enables
366/// SUM only for the duration of `handle_syscall`, so a kernel-space `ptr`
367/// supplied by user code will fault on access — but the fault is far more
368/// graceful when we reject obvious garbage up front.
369#[inline]
370fn validate_user_buf(ptr: usize, len: usize, max: usize) -> Result<(), SyscallError> {
371    if ptr == 0 {
372        return Err(SyscallError::InvalidInput);
373    }
374    if len > max {
375        return Err(SyscallError::BufferTooSmall);
376    }
377    if ptr.checked_add(len).is_none() {
378        return Err(SyscallError::InvalidInput);
379    }
380    Ok(())
381}

891/// Dispatches a system call to the appropriate handler.
892///
893/// `caller_id` is the ID of the task invoking the syscall.
894pub fn handle_syscall(caller_id: usize, syscall: Syscall) -> SyscallResult {
895    // Syscall allowlist enforcement: reject if this syscall's bit is not set in
896    // the per-Cell bitset loaded from ELF section `__ViCell_syscalls`.
897    // Cells without that section default to u64::MAX (permit-all, backwards compat).
898    if let Some(vi) = syscall_to_vi(&syscall) {
899        if let Some(bit) = vi.allowlist_bit() {
900            let allowed = get_syscall_allowlist(caller_id);
901            if (allowed >> bit) & 1 == 0 {
902                log::warn!(
903                    "[kernel] syscall {:?} denied for tid {} (allowlist={:#018x})",
904                    vi,
905                    caller_id,
906                    allowed
907                );
908                crate::audit::log_event(
909                    crate::audit::AuditEvent::SyscallDenied,
910                    &crate::audit::encode_u32x2(caller_id as u32, bit as u32),
911                );
912                return Err(SyscallError::PermissionDenied);
913            }
914        }
915    }
916
917    match syscall {
918        // --- Hubris ABI Implementation ---
919        Syscall::Send {
920            target,
921            msg_ptr,
922            msg_len,
923        } => {
924            crate::audit::log_event(
925                crate::audit::AuditEvent::IpcSend,
926                &crate::audit::encode_u32x2(caller_id as u32, target as u32),
927            );
928            let res = super::ipc_send(caller_id, target, msg_ptr, msg_len);
929            match res {
930                Ok(0) => Ok(0),
931                Ok(1) => {
932                    super::yield_cpu(); // Blocked
933                    if let Some(sched) = super::SCHEDULER.lock().as_ref() {
934                        return Ok(sched
935                            .tasks
936                            .get(&caller_id)
937                            .and_then(|t| t.reply_value)
938                            .unwrap_or(0));
939                    }
940                    Ok(0)
941                }
942                Err(_) => Err(SyscallError::InvalidCommand),
943                _ => Ok(0),
944            }
945        }
946        Syscall::TrySend {
947            target,
948            msg_ptr,
949            msg_len,
950        } => {
951            // Non-blocking: deliver if target in Recv, else drop. Never blocks.
952            // (Input-service sends fall back to a bounded queue — see ipc_try_send.)
953            match super::ipc_try_send(caller_id, target, msg_ptr, msg_len) {
954                Ok(()) => Ok(0),           // delivered (or queued for the input service)
955                Err(()) => Ok(usize::MAX), // dropped (target not ready or gone)
956            }
957        }
958        Syscall::Recv {
959            mask,
960            buf_ptr,
961            buf_len,
962        } => {
963            // A NotifyOnExit death that arrived while we were busy (not parked in
964            // Recv) was queued; deliver it now without blocking. The dead tid is
965            // returned as the "sender" so a supervisor never misses a child death.
966            if let Some(sched) = super::SCHEDULER.lock().as_mut() {
967                if let Some(t) = sched.tasks.get_mut(&caller_id) {
968                    if !t.pending_deaths.is_empty() {
969                        let (dead_tid, reason) = t.pending_deaths.remove(0);
970                        // Deliver the exit reason as the recv payload (NotifyOnExit
971                        // contract) so a supervisor can apply a restart policy.
972                        if buf_len >= core::mem::size_of::<u64>()
973                            && validate_user_buf(buf_ptr, core::mem::size_of::<u64>(), MAX_USER_BUF)
974                                .is_ok()
975                        {
976                            // SAFETY: buf_ptr is validated as this caller's user buffer;
977                            // the caller is mid-syscall so the write is exclusive in the SAS.
978                            unsafe {
979                                core::ptr::write_unaligned(buf_ptr as *mut u64, reason as u64);
980                            }
981                        }
982                        return Ok(dead_tid);
983                    }
984                }
985            }
986
987            // Hot-swap pending-message drain — guaranteed delivery path.
988            //
989            // The Step 5 drain in hotswap.rs tries to deliver buffered messages via
990            // ipc_send, which only succeeds when the new cell is already in TaskState::Recv.
991            // But the new cell calls sys_hotswap_ready from inside its Restore handler and
992            // hasn't re-entered sys_recv yet at that moment, so the Step 5 drain races.
993            //
994            // This path is the guaranteed fallback: when the new cell finally calls
995            // sys_recv for the first time (returning from the Restore handler through its
996            // app_entry loop), any messages that slipped through Step 5 are delivered here
997            // before blocking.  FIFO order matches the original sender queue order.
998            //
999            // Invariants:
1000            // - Lock is held for the entire drain + delivery so no concurrent send can
1001            //   observe a partially-drained queue.
1002            // - We copy from msg.data (an owned Box<[u8]>) before releasing the lock,
1003            //   avoiding any lifetime dependency on the task struct.
1004            // - current_caller is set so the cell sees the original sender's TID, not
1005            //   the hotswap orchestrator's TID, preserving the IPC identity contract.
1006            if let Some(sched) = super::SCHEDULER.lock().as_mut() {
1007                if let Some(t) = sched.tasks.get_mut(&caller_id) {
1008                    // The drain MUST honour the recv mask: a masked recv (e.g. the
1009                    // shell awaiting a VFS reply) must NOT consume a queued input
1010                    // key event — that poisoned every VFS request/reply pair after
1011                    // the input-queue change (the shell decoded a keystroke as the
1012                    // VFS reply → "vwrite failed", and the real reply desynced the
1013                    // next conversation). Non-matching messages stay queued for the
1014                    // wildcard read loop that wants them.
1015                    let slot = t
1016                        .pending_msgs
1017                        .iter()
1018                        .position(|m| mask == 0 || m.sender_tid == mask);
1019                    if let Some(i) = slot {
1020                        let msg = t.pending_msgs.remove(i);
1021                        let copy_len = core::cmp::min(msg.data.len(), buf_len);
1022                        if copy_len > 0
1023                            && validate_user_buf(buf_ptr, copy_len, MAX_USER_BUF).is_ok()
1024                        {
1025                            // SAFETY: buf_ptr is the caller's recv buffer, validated above;
1026                            // msg.data is an owned heap allocation from the Frozen intercept;
1027                            // both are exclusive at this point (cell is mid-syscall).
1028                            unsafe {
1029                                core::ptr::copy_nonoverlapping(
1030                                    msg.data.as_ptr(),
1031                                    buf_ptr as *mut u8,
1032                                    copy_len,
1033                                );
1034                            }
1035                        }
1036                        t.current_caller = Some(msg.sender_tid);
1037                        return Ok(msg.sender_tid);
1038                    }
1039                }
1040            }
1041
1042            let res = super::ipc_recv(caller_id, mask, buf_ptr, buf_len);
1043            match res {
1044                Ok(0) => {
1045                    // Blocked
1046                    super::yield_cpu();
1047                    // Resume: return who sent the message (or the dead tid for a death
1048                    // notification). If this wake was a death, a reason was stashed by
1049                    // exit_task — deliver it as the recv payload HERE, in our own syscall
1050                    // context where writing a USER buffer is valid (unlike the trap context).
1051                    let mut sender = 0;
1052                    let mut death_reason = None;
1053                    if let Some(sched) = super::SCHEDULER.lock().as_mut() {
1054                        if let Some(t) = sched.tasks.get_mut(&caller_id) {
1055                            sender = t.current_caller.unwrap_or(0);
1056                            death_reason = t.pending_exit_reason.take();
1057                        }
1058                    }
1059                    if let Some(reason) = death_reason {
1060                        if buf_len >= core::mem::size_of::<u64>()
1061                            && validate_user_buf(buf_ptr, core::mem::size_of::<u64>(), MAX_USER_BUF)
1062                                .is_ok()
1063                        {
1064                            // SAFETY: buf_ptr is validated as this caller's user buffer; we
1065                            // run in the caller's syscall context, so the store is permitted.
1066                            unsafe {
1067                                core::ptr::write_unaligned(buf_ptr as *mut u64, reason as u64);
1068                            }
1069                        }
1070                    }
1071                    Ok(sender)
1072                }
1073                Ok(id) => Ok(id), // Got message instantly
1074                Err(_) => Err(SyscallError::InvalidCommand),
1075            }
1076        }
1077        // ── Scatter/gather IPC ────────────────────────────────────────────────
1078        Syscall::SendGather {
1079            target,
1080            iovec_ptr,
1081            iovec_count,
1082        } => {
1083            // Concatenate all segments into a contiguous kernel buffer, then
1084            // deliver as a single IPC message to `target`.
1085            const MAX_IOVEC: usize = 8;
1086            const IOVEC_ENTRY: usize = core::mem::size_of::<usize>() * 2;
1087            if iovec_count == 0 || iovec_count > MAX_IOVEC {
1088                return Err(SyscallError::InvalidInput);
1089            }
1090            // Allocate a temporary gather buffer.
1091            let mut total = 0usize;
1092            for i in 0..iovec_count {
1093                // SAFETY: iovec_ptr is a valid user-space array of [ptr,len] pairs;
1094                // iovec_count is bounded by MAX_IOVEC; each element is 2×sizeof(usize).
1095                let len = unsafe {
1096                    core::ptr::read_unaligned(
1097                        (iovec_ptr + i * IOVEC_ENTRY + core::mem::size_of::<usize>())
1098                            as *const usize,
1099                    )
1100                };
1101                total = total.saturating_add(len);
1102            }
1103            if total > MAX_USER_BUF {
1104                return Err(SyscallError::BufferTooSmall);
1105            }
1106            let mut gathered: alloc::vec::Vec<u8> = alloc::vec![0u8; total];
1107            let mut pos = 0;
1108            for i in 0..iovec_count {
1109                // SAFETY: bounds validated above; ptr/len come from user-validated iovec.
1110                let (ptr, len) = unsafe {
1111                    let base = iovec_ptr + i * IOVEC_ENTRY;

1181                        );

4044/// Per-cell syscall allowlist gate.
4045///
4046/// Reads the caller's `syscall_allowlist` bitmask and returns
4047/// `Err(SyscallError::PermissionDenied)` if the opcode's bit is not set.
4048/// The SCHEDULER lock is acquired and released here — callers must NOT hold it.
4049fn check_allowlist(syscall_id: usize, caller_id: usize) -> Result<(), SyscallError> {
4050    let sc = ViSyscall::from(syscall_id);
4051    let bit = sc.allowlist_bit();
4052    // Bit 36 gates raw block-I/O opcodes (500/501/503) and BlkReadAsync (212).
4053    let blk_io_bit: Option<u8> = if matches!(syscall_id, 500 | 501 | 503 | 212) {
4054        Some(36)
4055    } else {
4056        None
4057    };
4058
4059    // Raw opcodes with a dedicated `map_syscall` fallback mapping. These are
4060    // intentionally absent from `ViSyscall` (Law 1: keeps experimental ids out
4061    // of the stable ABI) so they decode as `Unknown` — but they are NOT unknown:
4062    // 500/501/503 are gated by bit 36 below + the ZST BlockIoCap at the handler;
4063    // 502 and the legacy FD ops (3/100/106/107/108/110/111) predate the bitmap
4064    // and stay always-permitted, matching their pre-Phase-31b behavior.
4065    let known_raw = matches!(
4066        syscall_id,
4067        3 | 100 | 106 | 107 | 108 | 110 | 111 | 500 | 501 | 502 | 503
4068    );
4069
4070    let allowlist = super::SCHEDULER
4071        .lock()
4072        .as_ref()
4073        .and_then(|s| s.tasks.get(&caller_id))
4074        .map(|t| t.syscall_allowlist)
4075        .unwrap_or(0); // task absent → deny-all for safety
4076
4077    // Deny truly-unknown opcodes that land in the legacy inner-match fallback —
4078    // their allowlist_bit() returns None, so without this guard they bypass the
4079    // check. Exit (60) and Yield (104) are always permitted unconditionally.
4080    // Known-raw ids are exempt: blocking them here made every allowlist-declaring
4081    // cell lose raw block I/O (broke the VFS FAT32 mount silently since Phase 31b).
4082    // Every deny below logs: a silent dispatch-level denial cost a full day of
4083    // triage when the shell's missing `Read` bit bricked serial input with no
4084    // kernel output at all.
4085    if sc == ViSyscall::Unknown
4086        && !known_raw
4087        && !matches!(syscall_id, 60 | 104)
4088        && allowlist != u64::MAX
4089    {
4090        log::warn!(
4091            "[kernel] unknown opcode {} denied for tid {} (allowlist={:#018x})",
4092            syscall_id,
4093            caller_id,
4094            allowlist
4095        );
4096        return Err(SyscallError::PermissionDenied);
4097    }
4098    if let Some(b) = bit {
4099        if allowlist & (1u64 << b) == 0 {
4100            log::warn!(
4101                "[kernel] syscall {:?} (bit {}) denied for tid {} (allowlist={:#018x})",
4102                sc,
4103                b,
4104                caller_id,
4105                allowlist
4106            );
4107            return Err(SyscallError::PermissionDenied);
4108        }
4109    }
4110    if let Some(b) = blk_io_bit {
4111        if allowlist & (1u64 << b) == 0 {
4112            log::warn!(
4113                "[kernel] raw block opcode {} denied for tid {} (no bit 36)",
4114                syscall_id,
4115                caller_id
4116            );
4117            return Err(SyscallError::PermissionDenied);
4118        }
4119    }
4120    Ok(())
4121}
