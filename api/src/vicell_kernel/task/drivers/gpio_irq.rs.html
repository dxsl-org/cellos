1//! GPIO IRQ dispatcher — routes PL061 edge interrupts to the MMIO owner cell.
2//!
3//! When a GPIO pin fires an edge interrupt, the GIC delivers SPI 7 (GIC ID 39)
4//! to the kernel.  This module looks up the cell that currently owns the PL061
5//! MMIO region (0x0903_0000) via the Resource Registry, then sends it a 4-byte
6//! IPC notification: `[0x02, 0x00, 0x00, 0x00]` (opcode 2 = GPIO_IRQ_NOTIFY).
7//!
8//! The receiving cell reads GPIOMIS to determine which pins fired, calls
9//! `clear_irq(mask)` to acknowledge, and re-enables the interrupt as needed.
10//!
11//! # Ownership model
12//! No static registration is needed.  The Resource Registry already tracks which
13//! cell called `RequestMmio(PL061_BASE, …)`, and automatically releases the
14//! entry when the cell exits.  A new cell that re-opens GPIO after a restart
15//! becomes the new owner transparently.
16
17/// GPIO MMIO base — PL061 on QEMU ARM virt, BCM2837 on Raspberry Pi 3.
18/// Must match the allowlist entry in resource_registry.rs for the current target.
19#[cfg(all(target_arch = "aarch64", feature = "board-rpi3"))]
20const GPIO_MMIO_BASE: usize = 0x3F20_0000; // BCM2837 GPIO — RPi 3
21
22#[cfg(not(all(target_arch = "aarch64", feature = "board-rpi3")))]
23const GPIO_MMIO_BASE: usize = 0x0903_0000; // PL061 GPIO — QEMU ARM virt
24
25/// IPC opcode sent to the GPIO owner cell on interrupt.
26/// Chosen to not collide with kernel raw event opcodes 0 (EV_KEY) / 1 (EV_REL) / 2 (EV_ABS).
27const GPIO_IRQ_NOTIFY: u8 = 0xA0;
28
29/// Send a GPIO IRQ notification to the current PL061 MMIO owner.
30///
31/// Called as `vi_gpio_notify_irq` (extern "Rust") from `vi_aarch64_irq_handler`
32/// in `hal/arch/arm/src/aarch64/trap.rs` when GIC ID 39 (GPIO SPI 7) fires.
33///
34/// Safe to call from IRQ context: does not acquire SCHEDULER; `ipc_send` uses
35/// its own internal lock.  Fire-and-forget: if the owner cell is not in `Recv`
36/// state the event is dropped (the cell re-polls GPIOMIS on its next recv).
37#[no_mangle]
38pub extern "Rust" fn vi_gpio_notify_irq() {
39    let Some(tid) = crate::resource_registry::lookup_mmio_owner(GPIO_MMIO_BASE) else {
40        return; // No cell currently owns GPIO — drop the interrupt.
41    };
42    log::info!("[gpio-irq] IRQ fired — notifying GPIO owner tid={}", tid);
43    let msg = [GPIO_IRQ_NOTIFY, 0x00, 0x00, 0x00];
44    // SAFETY: ipc_send copies 4 bytes from a stack buffer to the target cell's
45    // recv page.  On AArch64 EL1 can access EL0 pages in ViCell's SAS layout
46    // (no PAN on cortex-a57/a72 default).
47    let _ = crate::task::ipc_send(0, tid, msg.as_ptr() as usize, 4);
48}
