1use crate::sync::Spinlock;
2use alloc::collections::BTreeMap;
3
4pub struct DriverRegistry {
5    /// Map Name -> Driver ID
6    name_to_id: BTreeMap<&'static str, usize>,
7    next_id: usize,
8}
9
10impl Default for DriverRegistry {
11    fn default() -> Self {
12        Self::new()
13    }
14}
15
16impl DriverRegistry {
17    pub fn new() -> Self {
18        Self {
19            name_to_id: BTreeMap::new(),
20            // ID 0 is reserved for Kernel/System
21            // ID 1 is reserved for Console (for now, or we register it properly)
22            next_id: 2,
23        }
24    }
25
26    pub fn register(&mut self, name: &'static str) -> usize {
27        if let Some(&id) = self.name_to_id.get(name) {
28            return id;
29        }
30
31        let id = self.next_id;
32        self.name_to_id.insert(name, id);
33        self.next_id += 1;
34        id
35    }
36
37    pub fn get_id(&self, name: &str) -> Option<usize> {
38        self.name_to_id.get(name).copied()
39    }
40}
41
42pub static DRIVER_REGISTRY: Spinlock<Option<DriverRegistry>> = Spinlock::new(None);
43
44pub fn init() {
45    let mut reg = DRIVER_REGISTRY.lock();
46    *reg = Some(DriverRegistry::new());
47
48    // Manually register Console as ID 1 for now, to keep compatibility
49    if let Some(r) = reg.as_mut() {
50        r.name_to_id.insert("console", 1);
51    }
52}
53
54pub fn resolve(name: &str) -> Option<usize> {
55    if let Some(reg) = DRIVER_REGISTRY.lock().as_ref() {
56        reg.get_id(name)
57    } else {
58        None
59    }
60}
61
62pub fn register_driver(name: &'static str) -> usize {
63    if let Some(reg) = DRIVER_REGISTRY.lock().as_mut() {
64        reg.register(name)
65    } else {
66        0
67    }
