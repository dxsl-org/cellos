5/// Polling timeout for CMD_COMPLETE and DAT transfers (~500 ms at 1 iteration/µs).
6const POLL_TIMEOUT_US: u32 = 500_000;
7
8/// SDHCI host controller — PIO polling mode, no DMA, no interrupts.
9///
10/// The `base` field is the kernel-mapped virtual address of the SDHCI register block.
11/// All MMIO accesses are `read_volatile` / `write_volatile` to prevent optimisation.
12pub struct SdhciController {
13    base: usize,
14    /// SDHC = true (block-addressed); SDSC = false (byte-addressed).
15    pub is_sdhc: bool,
16    /// SDHCI spec version read from HOST_VERSION[2:0]; affects clock divider encoding.
17    spec_ver: u8,
18}
19
20impl SdhciController {
21    /// Construct a controller from `base`.
22    ///
23    /// # Safety
24    /// `base` must be a valid kernel-mapped MMIO address for the SDHCI register block.
25    /// The address must remain valid for the lifetime of `Self`.
26    pub unsafe fn new(base: usize) -> Self {
27        let mut c = Self {
28            base,
29            is_sdhc: false,
30            spec_ver: 0,
31        };
32        // SAFETY: base is the validated MMIO address passed by the caller.
33        c.spec_ver = (c.read16(SDHCI_HOST_VERSION) & 0xFF) as u8;
34        c
35    }
36
37    // --- volatile MMIO helpers ---
38
39    #[inline]
40    fn read32(&self, off: usize) -> u32 {
41        // SAFETY: base + off is within the SDHCI MMIO block mapped by the kernel.
42        unsafe { core::ptr::read_volatile((self.base + off) as *const u32) }
43    }
44    #[inline]
45    fn read16(&self, off: usize) -> u16 {
46        // SAFETY: same as read32.
47        unsafe { core::ptr::read_volatile((self.base + off) as *const u16) }
48    }
49    #[inline]
50    fn write32(&mut self, off: usize, v: u32) {
51        // SAFETY: same as read32.
52        unsafe { core::ptr::write_volatile((self.base + off) as *mut u32, v) }
53    }
54    #[inline]
55    fn write16(&mut self, off: usize, v: u16) {
56        // SAFETY: same as read32.
57        unsafe { core::ptr::write_volatile((self.base + off) as *mut u16, v) }
58    }
59    #[inline]
60    fn write8(&mut self, off: usize, v: u8) {
61        // SAFETY: same as read32.
62        unsafe { core::ptr::write_volatile((self.base + off) as *mut u8, v) }
63    }
64
65    /// Spin until `(read32(off) & mask) == 0`, or return `Err(Timeout)`.
66    fn poll_clear(&self, off: usize, mask: u32, timeout_us: u32) -> ViResult<()> {
67        let mut i = 0u32;
68        while self.read32(off) & mask != 0 {
69            if i >= timeout_us {
70                return Err(ViError::WouldBlock);
71            }
72            i += 1;
73            // Single iteration ≈ 1 µs on a 1 GHz core with one memory-mapped read.
74            core::hint::spin_loop();
75        }
76        Ok(())
77    }
78
79    /// Spin until `(read32(off) & mask) != 0`, or return `Err(Timeout)`.
80    fn poll_set(&self, off: usize, mask: u32, timeout_us: u32) -> ViResult<()> {
81        let mut i = 0u32;
82        while self.read32(off) & mask == 0 {
83            if i >= timeout_us {
84                return Err(ViError::WouldBlock);
85            }
86            i += 1;
87            core::hint::spin_loop();
88        }
89        Ok(())
90    }
91
92    /// Read the Present State register (offset 0x24) — single MMIO read, no poll.
93    ///
94    /// Used by callers for a fast card-present check before any lengthy poll sequences.
95    pub fn read_present_state(&self) -> u32 {
96        self.read32(SDHCI_PRESENT_STATE)
97    }
98
99    /// Reset the controller (all lines).
100    pub fn reset_all(&mut self) -> ViResult<()> {
101        self.write8(SDHCI_SOFT_RESET, RESET_ALL);
102        self.poll_clear(SDHCI_SOFT_RESET, RESET_ALL as u32, POLL_TIMEOUT_US)?;
103        Ok(())
104    }
105
106    /// Enable 3.3 V power to the card slot.
107    pub fn power_on(&mut self) {
108        self.write8(SDHCI_POWER_CONTROL, PWR_33V);
109    }
110
111    /// Set the SD clock to the requested divider.
112    ///
113    /// Uses spec-v3 10-bit divider encoding when `spec_ver >= SPEC_V3`.
114    fn set_clock_div(&mut self, div: u16) {
115        // Disable SD clock and internal clock first.
116        self.write16(SDHCI_CLOCK_CONTROL, 0);
117
118        let clk = if self.spec_ver >= SPEC_V3 {
119            // 10-bit divider: bits[7:0] in bits[15:8], bits[9:8] in bits[7:6].
120            let lo = div & 0xFF;
121            let hi = (div >> 8) & 0x03;
122            (lo << 8) | (hi << 6) | CLK_INT_EN
123        } else {
124            // 8-bit divider (spec v1/v2): bits[7:0] in bits[15:8].
125            ((div & 0xFF) << 8) | CLK_INT_EN
126        };
127
128        self.write16(SDHCI_CLOCK_CONTROL, clk);
129        // Wait for internal clock to stabilise.
130        let _ = self.poll_set(SDHCI_CLOCK_CONTROL, CLK_INT_STABLE as u32, POLL_TIMEOUT_US);
131        // Enable SD clock to card.
132        self.write16(SDHCI_CLOCK_CONTROL, clk | CLK_SD_EN);
133    }
134
135    /// Read the INT_STATUS register and clear the given bits (w1c).
136    fn clear_int(&mut self, bits: u32) {
137        self.write32(SDHCI_INT_STATUS, bits);
138    }
139
140    /// Wait for CMD_INHIBIT and DAT_INHIBIT to clear before issuing a command.
141    fn wait_cmd_ready(&self, needs_dat: bool) -> ViResult<()> {
142        let mask = if needs_dat {
143            PS_CMD_INHIBIT | PS_DAT_INHIBIT
144        } else {
145            PS_CMD_INHIBIT
146        };
147        self.poll_clear(SDHCI_PRESENT_STATE, mask, POLL_TIMEOUT_US)
148    }
149}

151impl Drop for SdhciController {
152    fn drop(&mut self) {
153        // Power off the card slot on controller teardown.
154        self.write8(SDHCI_POWER_CONTROL, PWR_OFF);
155        self.write16(SDHCI_CLOCK_CONTROL, 0);
156    }
157}

159impl ViMmcHost for SdhciController {
160    fn send_cmd(&mut self, cmd: MmcCmd) -> ViResult<MmcResponse> {
161        self.wait_cmd_ready(cmd.has_data)?;
162
163        // Unmask normal+error interrupts in INT_ENABLE (no CPU IRQ — polling only).
164        self.write32(SDHCI_INT_ENABLE, INT_ALL_NORMAL | INT_ALL_ERROR);
165        self.write32(SDHCI_SIGNAL_ENABLE, 0); // no CPU interrupt
166
167        // Build response flag bits for the COMMAND register.
168        let resp_flags: u16 = match cmd.resp_type {
169            RespType::None => RESP_NONE,
170            RespType::R1 => RESP_R1,
171            RespType::R1b => RESP_R1B,
172            RespType::R2 => RESP_R2,
173            RespType::R3 => RESP_R3,
174            RespType::R6 => RESP_R6,
175            RespType::R7 => RESP_R7,
176        };
177
178        self.write32(SDHCI_ARGUMENT, cmd.arg);
179        // Writing COMMAND fires the command to the card.
180        self.write16(SDHCI_COMMAND, cmd_reg(cmd.index, resp_flags, cmd.has_data));
181
182        // Wait for CMD_COMPLETE (bit 0) or an error.
183        self.poll_set(
184            SDHCI_INT_STATUS,
185            INT_CMD_COMPLETE | INT_ERROR,
186            POLL_TIMEOUT_US,
187        )?;
188
189        let status = self.read32(SDHCI_INT_STATUS);
190        self.clear_int(INT_CMD_COMPLETE | INT_ALL_ERROR);
191
192        if status & INT_ERROR != 0 {
193            log::warn!(
194                "[sdhci] cmd{} error, INT_STATUS=0x{:08x}",
195                cmd.index,
196                status
197            );
198            return Err(ViError::IO);
199        }
200
201        // Read response registers.
202        let r = [
203            self.read32(SDHCI_RESPONSE),
204            self.read32(SDHCI_RESPONSE + 4),
205            self.read32(SDHCI_RESPONSE + 8),
206            self.read32(SDHCI_RESPONSE + 12),
207        ];
208        Ok(r)
209    }
210
211    fn read_block(&mut self, buf: &mut [u8]) -> ViResult<()> {
212        // Caller must pass a 512-byte, 4-byte-aligned buffer (one SDHCI block).
213        if buf.len() != 512 {
214            return Err(ViError::InvalidArgument);
215        }
216
217        // Wait for BUFFER_READ_READY (data available in FIFO).
218        self.poll_set(SDHCI_INT_STATUS, INT_BUF_READ_READY, POLL_TIMEOUT_US)?;
219        self.clear_int(INT_BUF_READ_READY);
220
221        // Read 4 bytes at a time from the BUFFER port.
222        let chunks = buf.len() / 4;
223        for i in 0..chunks {
224            let word = self.read32(SDHCI_BUFFER);
225            let off = i * 4;
226            buf[off] = (word & 0xFF) as u8;
227            buf[off + 1] = ((word >> 8) & 0xFF) as u8;
228            buf[off + 2] = ((word >> 16) & 0xFF) as u8;
229            buf[off + 3] = ((word >> 24) & 0xFF) as u8;
230        }
231
232        // Wait for TRANSFER_COMPLETE.
233        self.poll_set(SDHCI_INT_STATUS, INT_XFER_COMPLETE, POLL_TIMEOUT_US)?;
234        self.clear_int(INT_XFER_COMPLETE);
235        Ok(())
236    }
237
238    fn write_block(&mut self, buf: &[u8]) -> ViResult<()> {
239        if buf.len() != 512 {
240            return Err(ViError::InvalidArgument);
241        }
242
243        // Wait for BUFFER_WRITE_READY (FIFO has space).
244        self.poll_set(SDHCI_INT_STATUS, INT_BUF_WRITE_READY, POLL_TIMEOUT_US)?;
245        self.clear_int(INT_BUF_WRITE_READY);
246
247        let chunks = buf.len() / 4;
248        for i in 0..chunks {
249            let off = i * 4;
250            let word = (buf[off] as u32)
251                | ((buf[off + 1] as u32) << 8)
252                | ((buf[off + 2] as u32) << 16)
253                | ((buf[off + 3] as u32) << 24);
254            self.write32(SDHCI_BUFFER, word);
255        }
256
257        self.poll_set(SDHCI_INT_STATUS, INT_XFER_COMPLETE, POLL_TIMEOUT_US)?;
258        self.clear_int(INT_XFER_COMPLETE);
259        Ok(())
260    }
261
262    fn set_clock_hz(&mut self, hz: u32) -> ViResult<()> {
263        // Base clock frequency (typically 200 MHz on Arasan, 50 MHz on some others).
264        // We assume 200 MHz; the divider is rounded up to the nearest power-of-2 (spec v1/v2)
265        // or any value (spec v3). For boot-time use we target either 400 kHz (ID) or 25 MHz (DS).
266        const BASE_HZ: u32 = 200_000_000;
267        let div = BASE_HZ.checked_div(hz).map_or(0, |q| (q / 2).max(1) as u16);
268        self.set_clock_div(div);
269        Ok(())
270    }
271
272    fn set_bus_width(&mut self, width: BusWidth) -> ViResult<()> {
273        let mut hc = self.read32(SDHCI_HOST_CONTROL) as u8;
274        hc &= !0x26; // clear 4-bit (bit1) and 8-bit (bit5) fields
275        match width {
276            BusWidth::One => {}
277            BusWidth::Four => hc |= 1 << 1,
278            BusWidth::Eight => hc |= 1 << 5,
279        }
280        self.write8(SDHCI_HOST_CONTROL, hc);
281        Ok(())
282    }
283
284    fn card_present(&self) -> bool {
285        self.read32(SDHCI_PRESENT_STATE) & PS_CARD_PRESENT != 0
286    }
287}
