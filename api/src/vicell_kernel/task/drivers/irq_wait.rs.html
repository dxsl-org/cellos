1//! IRQ wait/pending tables for Driver Cell `sys_wait_irq`.
2//!
3//! Provides lock-free, ISR-safe primitives for the deferred-wake pattern:
4//!
5//! - `register_waiter(irq, tid, mmio_base)` — called from syscall context; records
6//!   the TID waiting on this IRQ line and the VirtIO MMIO base for InterruptACK.
7//! - `signal_irq(irq)` — called from ISR; sets `IRQ_PENDING[irq]` atomically.
8//!   NO scheduler access, NO lock; ISR safety is the invariant.
9//! - `take_pending(irq)` — swap-and-read; used by both the lost-wakeup guard in
10//!   `sys_wait_irq` and the scheduler sweep in `pick_next`.
11//! - `clear_waiter(irq)` — scheduler calls after transitioning the task to Ready.
12//!
13//! # ISR safety
14//! `signal_irq` uses only `AtomicBool::store(Release)` — no Spinlock, no allocation,
15//! no scheduler access.  This is the invariant.  Any attempt to wake a task directly
16//! from ISR context (by setting TaskState::Ready inside an interrupt handler) will
17//! deadlock because the scheduler Spinlock is already held by the interrupted task's
18//! scheduler tick.  Model: `kernel/src/task/waker.rs:consume_pending`.
19
20use core::sync::atomic::{AtomicBool, AtomicUsize, Ordering};
21
22/// Maximum IRQ line index supported.
23///
24/// Covers VirtIO MMIO slots 1-8 (IRQs 1-8), the e1000 NIC (IRQ 33 on QEMU ARM/RV),
25/// the NVMe SSD (IRQ 34), and GPIOs (< 64). 64 entries costs 64+64+64 = 192 bytes.
26pub const MAX_IRQ: usize = 64;
27
28/// Sentinel: no task is waiting on this IRQ.
29const NO_WAITER: usize = 0;
30
31/// TID of the Driver Cell waiting on each IRQ line. 0 = no waiter.
32///
33/// Written by `register_waiter` (syscall context, SeqCst).
34/// Read by `get_waiter` (scheduler sweep).
35/// Cleared by `clear_waiter` (scheduler sweep after task transitions to Ready).
36static IRQ_WAITERS: [AtomicUsize; MAX_IRQ] = {
37    // `AtomicUsize::new(0)` is const; arrays of const-init atomics are safe.
38    // `ZERO` is consumed exactly once, by `[ZERO; MAX_IRQ]` below — rustc evaluates
39    // a `const` operand fresh per array slot, so each IRQ line gets its OWN
40    // independent atomic, not a shared one. Switching to `static` would break the
41    // repeat expression (needs a `const` for non-`Copy` element types).
42    #[allow(clippy::declare_interior_mutable_const)]
43    const ZERO: AtomicUsize = AtomicUsize::new(NO_WAITER);
44    [ZERO; MAX_IRQ]
45};
46
47/// Pending flag for each IRQ line.
48///
49/// Set by `signal_irq` (ISR, Release ordering).
50/// Cleared by `take_pending` (scheduler sweep or lost-wakeup guard, AcqRel swap).
51static IRQ_PENDING: [AtomicBool; MAX_IRQ] = {
52    // Same independent-per-slot rationale as `IRQ_WAITERS` above.
53    #[allow(clippy::declare_interior_mutable_const)]
54    const FALSE: AtomicBool = AtomicBool::new(false);
55    [FALSE; MAX_IRQ]
56};
57
58/// VirtIO MMIO slot base for each IRQ line; used by the ISR to write InterruptACK
59/// (offset 0x64) and prevent interrupt storms on level-triggered VirtIO devices.
60/// 0 = non-VirtIO device (PCIe MSI, GPIO, …) — ISR skips the ack write.
61static IRQ_MMIO_BASE: [AtomicUsize; MAX_IRQ] = {
62    // Same independent-per-slot rationale as `IRQ_WAITERS` above.
63    #[allow(clippy::declare_interior_mutable_const)]
64    const ZERO: AtomicUsize = AtomicUsize::new(0);
65    [ZERO; MAX_IRQ]
66};
67
68/// Register `tid` as the waiter for `irq`, storing `mmio_base` for the ISR ack.
69///
70/// Returns `false` if another task is already waiting (single-waiter-per-IRQ
71/// policy). The caller (`sys_wait_irq`) must return `AlreadyClaimed` to the Cell.
72///
73/// SeqCst on the TID store so that a concurrent ISR that reads `has_waiter` after
74/// `register_waiter` returns will always see the new TID.
75pub fn register_waiter(irq: u8, tid: usize, mmio_base: usize) -> bool {
76    let idx = irq as usize;
77    if idx >= MAX_IRQ {
78        return false;
79    }
80    // compare_exchange: only register if slot is empty.
81    let result =
82        IRQ_WAITERS[idx].compare_exchange(NO_WAITER, tid, Ordering::SeqCst, Ordering::SeqCst);
83    if result.is_ok() {
84        IRQ_MMIO_BASE[idx].store(mmio_base, Ordering::Release);
85        true
86    } else {
87        false
88    }
89}
90
91/// Atomically take the pending flag for `irq`.
92///
93/// Returns `true` if an IRQ was pending (and now cleared); `false` if not.
94/// Used as both the lost-wakeup guard in `sys_wait_irq` AND the scheduler sweep.
95pub fn take_pending(irq: u8) -> bool {
96    let idx = irq as usize;
97    if idx >= MAX_IRQ {
98        return false;
99    }
100    // swap(false): clears the flag and returns the old value in one atomic op.
101    IRQ_PENDING[idx].swap(false, Ordering::AcqRel)
102}
103
104/// Signal that IRQ `irq` has fired.
105///
106/// Called ONLY from ISR context (interrupt handler). Sets the pending flag and
107/// optionally writes the VirtIO InterruptACK register to de-assert the IRQ line
108/// before PLIC completion, preventing interrupt storms on level-triggered devices.
109///
110/// # ISR safety
111/// Uses only `AtomicBool::store(Release)` — no Spinlock, no scheduler access.
112/// The actual task state transition (WaitIrq → Ready) happens in `pick_next`.
113pub fn signal_irq(irq: u8) {
114    let idx = irq as usize;
115    if idx >= MAX_IRQ {
116        return;
117    }
118
119    let mmio_base = IRQ_MMIO_BASE[idx].load(Ordering::Acquire);
120    if mmio_base != 0 {
121        // Ack VirtIO InterruptACK (MMIO base + 0x64) to clear the device IRQ line.
122        // Without this, a level-triggered VirtIO interrupt re-fires immediately
123        // after plic_complete, spinning the ISR forever.
124        //
125        // SAFETY: mmio_base was validated by sys_request_mmio before the Cell
126        // stored it; the MMIO window is identity-mapped and within the known device
127        // allowlist. Writing 0x1 acks the first pending interrupt bit.
128        unsafe {
129            core::ptr::write_volatile((mmio_base + 0x64) as *mut u32, 0x1);
130        }
131    }
132
133    // Set pending AFTER the MMIO ack so the scheduler sweep races correctly:
134    // either the cell drains the used ring or the next ISR acks again.
135    IRQ_PENDING[idx].store(true, Ordering::Release);
136}
137
138/// Clear the waiter registration for `irq`.
139///
140/// Called by the scheduler sweep after transitioning the waiting task to Ready,
141/// so the IRQ slot is open for a new `sys_wait_irq` call from the same Cell on
142/// the next iteration.
143pub fn clear_waiter(irq: u8) {
144    let idx = irq as usize;
145    if idx >= MAX_IRQ {
146        return;
147    }
148    IRQ_WAITERS[idx].store(NO_WAITER, Ordering::Release);
149    IRQ_MMIO_BASE[idx].store(0, Ordering::Release);
150}
151
152/// Return `true` if any Driver Cell is currently registered for `irq`.
153///
154/// Called from the IRQ dispatcher to decide whether to route to `signal_irq`
155/// (Cell-claimed) or the kernel-internal VirtIO driver (not yet migrated).
156pub fn has_waiter(irq: u8) -> bool {
157    let idx = irq as usize;
158    if idx >= MAX_IRQ {
159        return false;
160    }
161    IRQ_WAITERS[idx].load(Ordering::Acquire) != NO_WAITER
162}
163
164/// Return the MMIO base stored for `irq`, or 0 if none.
165pub fn get_mmio_base(irq: u8) -> usize {
166    let idx = irq as usize;
167    if idx >= MAX_IRQ {
168        return 0;
169    }
170    IRQ_MMIO_BASE[idx].load(Ordering::Acquire)
171}
