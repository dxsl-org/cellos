1use core::alloc::Layout;
2use core::ptr::NonNull;
3use virtio_drivers::{BufferDirection, Hal, PhysAddr};
4
5/// VirtIO HAL adapter for ViCell.
6///
7/// Assumes identity mapping (VAddr == PAddr) throughout, which holds for the
8/// kernel's current single-address-space model.  Revisit if an HHDM or IOMMU
9/// is introduced: every `paddr = ptr as usize` line must become
10/// `paddr = ptr as usize - HHDM_OFFSET`.
11pub struct VirtioHal;
12
13// SAFETY: VirtioHal holds no state — all methods are stateless function calls.
14unsafe impl Hal for VirtioHal {
15    fn dma_alloc(pages: usize, _direction: BufferDirection) -> (PhysAddr, NonNull<u8>) {
16        let layout = Layout::from_size_align(pages * 4096, 4096)
17            .expect("VirtIO DMA layout must be valid — pages > 0 and align is power-of-two");
18
19        // SAFETY: layout is non-zero size and properly aligned (4096-byte alignment
20        // satisfies VirtIO virtqueue alignment requirement of ≥16 bytes).
21        let ptr = unsafe { alloc::alloc::alloc(layout) };
22
23        if ptr.is_null() {
24            // The Hal trait cannot signal failure (it must return an address),
25            // so we cannot propagate OOM to the caller. Panic rather than spin
26            // silently forever: a kernel panic prints a diagnosable message and
27            // halts, whereas an infinite spin looks like an unexplained hang
28            // (this was exactly the symptom when the GPU framebuffer alloc
29            // outgrew the heap).
30            panic!(
31                "[virtio] DMA alloc OOM: {} pages ({} KB) requested",
32                pages,
33                pages * 4
34            );
35        }
36
37        // SAFETY: ptr is non-null and points to `pages * 4096` bytes — safe to zero.
38        unsafe { core::ptr::write_bytes(ptr, 0, layout.size()) };
39
40        // Identity mapping: physical address == virtual address.
41        let paddr = ptr as usize;
42        log::trace!(
43            "[virtio] DMA alloc {} pages at V:{:p} P:0x{:X}",
44            pages,
45            ptr,
46            paddr
47        );
48
49        // SAFETY: ptr is non-null (checked above).
50        (paddr, unsafe { NonNull::new_unchecked(ptr) })
51    }
52
53    unsafe fn dma_dealloc(_paddr: PhysAddr, vaddr: NonNull<u8>, pages: usize) -> i32 {
54        let layout = Layout::from_size_align(pages * 4096, 4096)
55            .expect("dma_dealloc layout must match dma_alloc — invariant upheld by virtio-drivers");
56        // SAFETY: vaddr was returned by dma_alloc for the same layout; virtio-drivers
57        // guarantees dma_dealloc is called at most once per dma_alloc.  Using vaddr
58        // (not paddr) keeps correctness when HHDM is introduced (paddr != vaddr).
59        unsafe { alloc::alloc::dealloc(vaddr.as_ptr(), layout) };
60        0
61    }
62
63    unsafe fn mmio_phys_to_virt(paddr: PhysAddr, _size: usize) -> NonNull<u8> {
64        // SAFETY: identity mapping — paddr is a valid MMIO address mapped by
65        // init_kernel_paging (CLINT/PLIC/UART/VirtIO explicit block).
66        NonNull::new(paddr as *mut u8).expect("VirtIO MMIO address must be non-zero")
67    }
68
69    unsafe fn share(buffer: NonNull<[u8]>, _direction: BufferDirection) -> PhysAddr {
70        let vaddr = buffer.as_ptr() as *mut u8 as usize;
71        // Translate virtual → physical using the kernel page table.
72        // ViCell SAS does not guarantee VA==PA for all mappings (ELF segments are
73        // mapped at their load VA but allocated to arbitrary physical frames).
74        // Heap-allocated buffers and kernel-stack frames happen to be identity-
75        // mapped, but VFS BSS / static sector buffers are NOT — DMA must use the
76        // real physical address to avoid writing to the wrong physical location.
77        crate::memory::paging::virt_to_phys(vaddr).unwrap_or(vaddr)
78    }
79
80    unsafe fn unshare(_paddr: PhysAddr, _buffer: NonNull<[u8]>, _direction: BufferDirection) {
81        // Identity mapping: no IOMMU to flush, no cache coherency action needed.
82    }
83}
