1//! Minimal VirtIO input IRQ ACK shim.
2//!
3//! Keeps InterruptStatus cleared when the input Cell is not yet running or has
4//! crashed — without this, an unacknowledged input IRQ becomes an interrupt storm.
5//!
6//! This is the ONLY kernel-side remnant of the old virtio_input driver.
7//! All event routing is handled by the input service Cell (cells/services/input).
8//! Full deletion of this shim is a G2 task once input Cell reliability is confirmed.
9
10use crate::sync::Spinlock;
11use crate::task::drivers::virtio_hal::VirtioHal;
12use core::ptr::NonNull;
13use core::sync::atomic::{AtomicU32, Ordering};
14use virtio_drivers::device::input::VirtIOInput;
15use virtio_drivers::transport::mmio::{MmioTransport, VirtIOHeader};
16use virtio_drivers::transport::{DeviceType, Transport};
17
18/// IRQ number of the probed VirtIO input device (0 = not found).
19static INPUT_IRQ: AtomicU32 = AtomicU32::new(0);
20
21struct InputAckDriver {
22    inner: VirtIOInput<VirtioHal, MmioTransport>,
23}
24
25static KEYBOARD_ACK: Spinlock<Option<InputAckDriver>> = Spinlock::new(None);
26
27/// Probe VirtIO MMIO slots for an input device and cache it for IRQ ACK.
28pub fn init_driver() {
29    use crate::task::drivers::virtio_common::virtio_slots;
30    for slot in virtio_slots() {
31        let header = unsafe { NonNull::new_unchecked(slot.base as *mut VirtIOHeader) };
32        if let Ok(transport) = unsafe { MmioTransport::new(header) } {
33            if transport.device_type() == DeviceType::Input {
34                match VirtIOInput::<VirtioHal, MmioTransport>::new(transport) {
35                    Ok(inner) => {
36                        log::info!(
37                            "[input_irq_ack] input device at {:#x} irq={}",
38                            slot.base,
39                            slot.irq
40                        );
41                        INPUT_IRQ.store(slot.irq, Ordering::Release);
42                        *KEYBOARD_ACK.lock() = Some(InputAckDriver { inner });
43                        return;
44                    }
45                    Err(e) => {
46                        log::warn!("[input_irq_ack] init failed at {:#x}: {:?}", slot.base, e)
47                    }
48                }
49            } else {
50                core::mem::forget(transport);
51            }
52        }
53    }
54}
55
56/// Force-release this module's locks during fault teardown.
57///
58/// # Safety
59/// Single-hart; called only from the fault/panic path with interrupts disabled.
60pub unsafe fn force_unlock_locks() {
61    KEYBOARD_ACK.force_unlock();
62}
63
64/// ACK the input IRQ to prevent interrupt storms.
65///
66/// Returns `true` if this IRQ belonged to the input device and was acknowledged.
67/// Only needed while `has_waiter(irq)` is false (input Cell not yet registered).
68pub fn ack_if_input(irq: u32) -> bool {
69    let device_irq = INPUT_IRQ.load(Ordering::Relaxed);
70    if device_irq == 0 || device_irq != irq {
71        return false;
72    }
73    if let Some(drv) = KEYBOARD_ACK.lock().as_mut() {
74        drv.inner.ack_interrupt();
75    }
76    true
77}
