6static NULL_ZST: NullBlock = NullBlock;

8/// Placeholder block device used when no kernel-resident block driver is present.
9///
10/// Since the G2 loader redesign (phases 05/06) the kernel drives NO block hardware
11/// on QEMU — VFS routes all sector I/O to the virtio-blk Driver Cell (`/bin/block`),
12/// and NVMe/e1000 are Driver Cells too. The residual kernel callers of `block::` —
13/// warm-boot snapshot save/restore, `verify_mbr`, and the `EarlyLoader` bootstrap
14/// fallback — all handle a read error gracefully, so this device just fails reads.
15/// MMC (SDHCI) still wins on real boards.
16struct NullBlock;
17
18impl ViBlockDevice for NullBlock {
19    fn read_sector(&self, _sector: u64, _buf: &mut [u8]) -> ViResult<()> {
20        Err(ViError::NotFound)
21    }
22    fn write_sector(&self, _sector: u64, _buf: &[u8]) -> ViResult<()> {
23        Err(ViError::NotFound)
24    }
25    fn sector_count(&self) -> u64 {
26        0
27    }
28    fn sector_size(&self) -> usize {
29        512
30    }
31    fn flush(&self) -> ViResult<()> {
32        Ok(())
33    }
34}
35
36/// Return the active kernel block device.
37///
38/// MMC (SDHCI) on real boards; otherwise a null device — on QEMU the virtio-blk
39/// Driver Cell owns the disk, not the kernel (G2 loader redesign). VFS reaches the
40/// block driver via `sys_lookup_service(service::BLOCK_DRIVER)`, not this path.
41pub fn block_device() -> &'static dyn ViBlockDevice {
42    if super::mmc::is_present() {
43        &MMC_ZST
44    } else {
45        &NULL_ZST
46    }
47}
48
49/// Read one 512-byte sector. Convenience wrapper — no `ViBlockDevice` import required.
50pub fn read_sector(sector: u64, buf: &mut [u8]) -> ViResult<()> {
51    block_device().read_sector(sector, buf)
52}
53
54/// Write one 512-byte sector. Convenience wrapper — no `ViBlockDevice` import required.
55pub fn write_sector(sector: u64, buf: &[u8]) -> ViResult<()> {
56    block_device().write_sector(sector, buf)
57}
58
59/// Flush pending writes. Convenience wrapper — no `ViBlockDevice` import required.
60pub fn flush() -> ViResult<()> {
61    block_device().flush()
