1//! Driver Cell registration — tracks which cells own the block, NIC, and GPU roles.
2//!
3//! When a Tier-1 Driver Cell calls `sys_register_block_driver`,
4//! `sys_register_nic_driver`, or `sys_register_gpu_driver`, the kernel records
5//! its TID here.  Service clients use `sys_lookup_service(service::X)` to find
6//! the provider; these statics are the backing store for that lookup.
7//!
8//! `0` means "no driver cell registered". Block + NIC fall back to kernel-resident drivers
9//! (virtio_blk / MMC for block; no NIC fallback — NIC is always a Driver Cell).
10//! GPU has no kernel fallback; compositor refuses to init until a GPU Cell registers.
11
12use core::sync::atomic::{AtomicUsize, Ordering};
13
14/// TID of the registered block Driver Cell (0 = none; kernel virtio_blk/MMC is the fallback).
15pub static BLOCK_DRIVER_CELL: AtomicUsize = AtomicUsize::new(0);
16
17/// TID of the registered NIC Driver Cell (0 = none; no kernel NIC fallback exists).
18pub static NIC_DRIVER_CELL: AtomicUsize = AtomicUsize::new(0);
19
20/// TID of the registered GPU Driver Cell (0 = none; no kernel GPU fallback).
21pub static GPU_DRIVER_CELL: AtomicUsize = AtomicUsize::new(0);
22
23/// Record `tid` as the active block driver.  Overwrites any previous registration.
24///
25/// Logged at `warn!` for the same reason as `set_input_cell`: Driver Cells are
26/// spawned by init AFTER the kernel drops its log level to Warn, this is a
27/// one-time boot-integrity event (the kernel now routes all sector I/O to this
28/// TID), and it is the marker the x86 nvme/nic integration tests assert on.
29pub fn register_block_driver(tid: usize) {
30    BLOCK_DRIVER_CELL.store(tid, Ordering::Release);
31    log::warn!("[driver_cell] block driver registered: tid={}", tid);
32}
33
34/// Record `tid` as the active NIC driver.  Overwrites any previous registration.
35/// `warn!` — see `register_block_driver`.
36pub fn register_nic_driver(tid: usize) {
37    NIC_DRIVER_CELL.store(tid, Ordering::Release);
38    log::warn!("[driver_cell] NIC driver registered: tid={}", tid);
39}
40
41/// Clear the block driver registration (called on cell exit/kill).
42pub fn deregister_block_driver(tid: usize) {
43    BLOCK_DRIVER_CELL
44        .compare_exchange(tid, 0, Ordering::AcqRel, Ordering::Relaxed)
45        .ok();
46}
47
48/// Clear the NIC driver registration (called on cell exit/kill).
49pub fn deregister_nic_driver(tid: usize) {
50    NIC_DRIVER_CELL
51        .compare_exchange(tid, 0, Ordering::AcqRel, Ordering::Relaxed)
52        .ok();
53}
54
55/// Record `tid` as the active GPU driver.  Overwrites any previous registration.
56pub fn register_gpu_driver(tid: usize) {
57    GPU_DRIVER_CELL.store(tid, Ordering::Release);
58    log::info!("[driver_cell] GPU driver registered: tid={}", tid);
59}
60
61/// Clear the GPU driver registration (called on cell exit/kill).
62pub fn deregister_gpu_driver(tid: usize) {
63    GPU_DRIVER_CELL
64        .compare_exchange(tid, 0, Ordering::AcqRel, Ordering::Relaxed)
65        .ok();
66}
67
68/// TID of the registered input service Cell (0 = unregistered).
69/// Set by the loader after spawning `/bin/input`; cleared on its death.
70pub static INPUT_CELL_TID: AtomicUsize = AtomicUsize::new(0);
71
72/// Register the input service cell.  Called by the loader after spawning `/bin/input`.
73///
74/// Logged at `warn!`, not `info!`: `/bin/input` spawns after the kernel drops
75/// its log level to Warn at end-of-early-boot, so an info line here is
76/// suppressed. This is a one-time boot-integrity event (the kernel now trusts
77/// this TID as the sole keyboard-event sink) — worth surfacing, and the marker
78/// the input-registration integration test asserts on.
79pub fn set_input_cell(tid: usize) {
80    INPUT_CELL_TID.store(tid, Ordering::Release);
81    log::warn!("[input] registered input service TID {}", tid);
82}
83
84/// Clear the input service registration if it matches `tid` (called on cell death).
85pub fn clear_input_cell_if(tid: usize) {
86    INPUT_CELL_TID
87        .compare_exchange(tid, 0, Ordering::AcqRel, Ordering::Relaxed)
88        .ok();
89}
