1//! Shared VirtIO MMIO slot enumeration.
2//!
3//! `virtio_blk`, `input_irq_ack`, and Driver Cells use `virtio_slots()` to
4//! iterate all VirtIO MMIO slots for the current platform.
5//!
6//! AArch64: scans all 32 slots at 0x0a000000, stride 0x200 (QEMU virt layout).
7//! QEMU assigns devices to slots in an implementation-defined order so we must
8//! probe all 32.  The identity map in paging.rs covers the full 0x0a004000 range.
9//!
10//! Other arches: reads DTB-confirmed slots from `platform::PLATFORM`.
11
12extern crate alloc;
13use alloc::vec::Vec;
14
15/// A VirtIO MMIO slot with base address and IRQ.
16pub struct VirtioSlot {
17    pub base: usize,
18    pub irq: u32,
19}
20
21/// Iterator over all VirtIO MMIO slots for the current platform.
22pub fn virtio_slots() -> impl Iterator<Item = VirtioSlot> {
23    #[cfg(target_arch = "aarch64")]
24    {
25        // QEMU ARM virt: 32 VirtIO MMIO slots at 0x0a000000, 512 bytes each, SPI 16+i.
26        // All 32 slots are identity-mapped by init_kernel_paging (0x0a000000..0x0a004000).
27        const BASE: usize = 0x0a00_0000;
28        const STRIDE: usize = 0x200;
29        let slots: Vec<VirtioSlot> = (0..32_usize)
30            .map(|i| VirtioSlot {
31                base: BASE + i * STRIDE,
32                irq: 16 + i as u32,
33            })
34            .collect();
35        slots.into_iter()
36    }
37    #[cfg(not(target_arch = "aarch64"))]
38    {
39        let slots: Vec<VirtioSlot> = crate::platform::with(|p| {
40            p.virtio_mmio
41                .iter()
42                .filter_map(|e| {
43                    e.as_ref().map(|e| VirtioSlot {
44                        base: e.base,
45                        irq: e.irq,
46                    })
47                })
48                .collect()
49        });
50        slots.into_iter()
51    }
52}

54/// VirtIO MMIO IRQ dispatcher — called from the arch trap handlers (riscv64/aarch64)
55/// when any VirtIO MMIO IRQ fires. Routes the IRQ to whichever Driver Cell registered
56/// for it via `sys_wait_irq` (the block / net / gpu cells all rely on this), ACKs the
57/// input slot when the input service Cell isn't up yet, and warns on an unclaimed slot.
58///
59/// Relocated here from the deleted kernel `virtio_blk` driver (G2 loader redesign
60/// phase 06): it is VirtIO-common, not block-specific. The former kernel-block ACK
61/// branch is gone — the virtio-blk Driver Cell now ACKs its own IRQ via the
62/// `sys_wait_irq` path above.
63#[no_mangle]
64pub extern "Rust" fn vi_handle_virtio_irq(irq: u32) {
65    // Driver Cell IRQ routing: a Cell registered for this IRQ via sys_wait_irq —
66    // signal it (sets IRQ_PENDING + writes VirtIO InterruptACK) and return.
67    if crate::task::drivers::irq_wait::has_waiter(irq as u8) {
68        crate::task::drivers::irq_wait::signal_irq(irq as u8);
69        return;
70    }
71    // Input (keyboard) slot: ACK to prevent an interrupt storm before the input
72    // service Cell is up; event routing lives entirely in that Cell.
73    if crate::task::drivers::input_irq_ack::ack_if_input(irq) {
74        return;
75    }
76    // Unknown VirtIO slot — no device registered. InterruptStatus is already cleared
77    // by plic_complete in the trap handler.
78    log::warn!(
79        "[virtio] unhandled IRQ {} — no registered device for this slot",
80        irq
81    );
82}
