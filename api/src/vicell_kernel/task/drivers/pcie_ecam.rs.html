1//! PCIe ECAM (Enhanced Configuration Access Mechanism) walker.
2//!
3//! Scans bus 0, devices 0-31, probing functions per the header-type multi-function
4//! bit. Exposes a kernel-internal snapshot of discovered PCI devices.
5//!
6//! ECAM bases per machine (QEMU defaults — hardcoded for v1; ACPI MCFG parse
7//! and DTB `pci-host-ecam-generic` reg lookup are documented follow-ups):
8//!   x86_64 q35  : 0xB000_0000
9//!   RISC-V virt : 0x3000_0000
10//!   ARM64 virt  : 0x3F00_0000
11//!
12//! Only bus 0 is mapped (1 MiB). Devices on buses > 0 require extending the
13//! MMIO window — documented follow-up (no QEMU NVMe lands on bus > 0).
14//!
15//! Call ordering: `platform::init()` → `init_kernel_paging*()` → `pcie_ecam::init()`.
16
17use crate::sync::Spinlock;
18use alloc::vec::Vec;

20use core::sync::atomic::AtomicUsize;
21use core::sync::atomic::{AtomicBool, Ordering};

25/// PCIe ECAM config-space base for x86_64 q35.
26///
27/// This is the compile-time fallback / QEMU q35 default.
28/// The runtime value (parsed from ACPI MCFG at boot) is stored in
29/// `X86_ECAM_BASE` and takes precedence when non-zero.
30pub const ECAM_BASE_X86: usize = 0xB000_0000;
31
32/// Runtime ECAM base for x86_64, set from ACPI MCFG parse before `init()`.
33///
34/// Zero means "not set"; `ecam_base_x86()` falls back to `ECAM_BASE_X86`.
35#[cfg(target_arch = "x86_64")]
36static X86_ECAM_BASE: AtomicUsize = AtomicUsize::new(0);
37
38/// Store the ECAM base parsed from ACPI MCFG.
39///
40/// Must be called before `pcie_ecam::init()` for the value to take effect.
41/// No-op on non-x86_64 targets (they use compile-time constants).
42#[cfg(target_arch = "x86_64")]
43pub fn set_ecam_base_x86(base: usize) {
44    X86_ECAM_BASE.store(base, Ordering::Relaxed);
45}
46
47/// Return the effective x86_64 ECAM base: runtime value (from ACPI) if set,
48/// otherwise the QEMU q35 compile-time default.
49#[cfg(target_arch = "x86_64")]
50#[inline]
51fn ecam_base_x86() -> usize {
52    let b = X86_ECAM_BASE.load(Ordering::Relaxed);
53    if b != 0 {
54        b
55    } else {
56        ECAM_BASE_X86
57    }
58}

93/// A Base Address Register decoded from PCI config space.
94#[derive(Clone, Copy, Debug)]
95pub enum Bar {
96    /// 32-bit MMIO BAR: base address and probed size.
97    Memory32 { addr: u32, size: u32 },
98    /// 64-bit MMIO BAR: base address (64-bit) and probed size. Consumes two slots.
99    Memory64 { addr: u64, size: u64 },
100    /// I/O port BAR (not mapped by this driver; skipped).
101    Io,
102    /// Unused / empty BAR slot.
103    None,
104}
105
106impl Bar {
107    /// Physical base address of this BAR, or 0 for I/O / empty.
108    pub fn base_addr(&self) -> u64 {
109        match self {
110            Bar::Memory32 { addr, .. } => *addr as u64,
111            Bar::Memory64 { addr, .. } => *addr,
112            _ => 0,
113        }
114    }
115}

180/// Compute MMIO address for a BDF + register offset.
181///
182/// ECAM formula: base + (bus << 20) + (device << 15) + (function << 12) + offset.
183#[inline(always)]
184fn config_addr(ecam_base: usize, bus: u8, dev: u8, fun: u8, off: usize) -> *mut u8 {
185    let addr =
186        ecam_base + ((bus as usize) << 20) + ((dev as usize) << 15) + ((fun as usize) << 12) + off;
187    addr as *mut u8
188}

190/// Read a u32 from PCI config space via ECAM MMIO.
191///
192/// # Safety
193/// `config_addr(ecam_base, bus, dev, fun, off)` must point into the identity-
194/// mapped ECAM window established by `init_kernel_paging*()`.  `off` must be
195/// 4-byte-aligned.
196unsafe fn read32(ecam_base: usize, bus: u8, dev: u8, fun: u8, off: usize) -> u32 {
197    // SAFETY: caller guarantees the ECAM window is identity-mapped.
198    let ptr = config_addr(ecam_base, bus, dev, fun, off) as *const u32;
199    // SAFETY: volatile prevents the compiler from optimising away hardware reads.
200    unsafe { core::ptr::read_volatile(ptr) }
201}

252// ── BAR decode ────────────────────────────────────────────────────────────────
253
254/// Decode all BARs for a type-0 function.
255///
256/// The write-all-ones / read-back size-probe modifies the command register to
257/// disable memory decode before each probe and restores it after, preventing
258/// the device from responding to MMIO accesses during the transient probe window.
259///
260/// # Safety
261/// `ecam_base` must be the identity-mapped ECAM window base.
262unsafe fn decode_bars(ecam_base: usize, bus: u8, dev: u8, fun: u8) -> [Bar; 6] {
263    let mut bars = [Bar::None; 6];
264    let mut i = 0usize;
265    while i < 6 {
266        let off = CFG_BAR0 + i * 4;
267        // SAFETY: ECAM window is identity-mapped; off is within type-0 header.
268        let raw = unsafe { read32(ecam_base, bus, dev, fun, off) };
269        if raw & 1 == 1 {
270            // I/O BAR — skip.
271            bars[i] = Bar::Io;
272            i += 1;
273            continue;
274        }
275        let bar_type = (raw >> 1) & 0x3; // bits [2:1]
276        let addr32 = raw & !0xF;
277
278        if bar_type == 0x2 {
279            // 64-bit MMIO: base spans this slot + next.
280            if i + 1 >= 6 {
281                bars[i] = Bar::None;
282                i += 1;
283                continue;
284            }
285            let off_hi = CFG_BAR0 + (i + 1) * 4;
286            // SAFETY: ECAM window is identity-mapped.
287            let raw_hi = unsafe { read32(ecam_base, bus, dev, fun, off_hi) };
288            let addr64 = (addr32 as u64) | ((raw_hi as u64) << 32);
289            let size64 = unsafe { probe_bar_size64(ecam_base, bus, dev, fun, i) };
290            bars[i] = Bar::Memory64 {
291                addr: addr64,
292                size: size64,
293            };
294            bars[i + 1] = Bar::None; // second slot consumed
295            i += 2;
296        } else {
297            // 32-bit MMIO.
298            let size32 = unsafe { probe_bar_size32(ecam_base, bus, dev, fun, i) };
299            bars[i] = Bar::Memory32 {
300                addr: addr32,
301                size: size32,
302            };
303            i += 1;
304        }
305    }
306    bars
307}

309/// Probe size of a 32-bit MMIO BAR via the write-all-ones / read-back method.
310///
311/// Saves/restores both the command register (disabling memory decode during the
312/// probe) and the BAR value so the device is left in its original state.
313///
314/// # Safety
315/// `ecam_base` must be the identity-mapped ECAM window base.
316unsafe fn probe_bar_size32(ecam_base: usize, bus: u8, dev: u8, fun: u8, bar_idx: usize) -> u32 {
317    let off = CFG_BAR0 + bar_idx * 4;
318    // Save original values.
319    // SAFETY: ECAM window is identity-mapped.
320    let orig_cmd = unsafe { read16(ecam_base, bus, dev, fun, CFG_COMMAND) };
321    // SAFETY: ECAM window is identity-mapped.
322    let orig_bar = unsafe { read32(ecam_base, bus, dev, fun, off) };
323
324    // Disable memory decode before writing all-ones to BAR.
325    // This prevents the device from claiming MMIO during the probe window.
326    // SAFETY: writing command register disables decode only for this function.
327    unsafe {
328        write16(
329            ecam_base,
330            bus,
331            dev,
332            fun,
333            CFG_COMMAND,
334            orig_cmd & !CMD_MEM_SPACE,
335        );
336    }
337
338    // Write all-ones to BAR.
339    // SAFETY: ECAM window is identity-mapped; BAR write is bounded to this function.
340    unsafe {
341        write32(ecam_base, bus, dev, fun, off, 0xFFFF_FFFF);
342    }
343    // SAFETY: ECAM window is identity-mapped.
344    let readback = unsafe { read32(ecam_base, bus, dev, fun, off) };
345
346    // Restore BAR and command register.
347    // SAFETY: ECAM window is identity-mapped.
348    unsafe {
349        write32(ecam_base, bus, dev, fun, off, orig_bar);
350    }
351    unsafe {
352        write16(ecam_base, bus, dev, fun, CFG_COMMAND, orig_cmd);
353    }
354
355    // Size = ~(readback & mask) + 1 where mask = 0xFFFFFFF0 (clear lower 4 bits).
356    let mask = readback & 0xFFFF_FFF0;
357    if mask == 0 {
358        return 0;
359    }
360    (!mask).wrapping_add(1)
361}

363/// Probe size of a 64-bit MMIO BAR (both low and high dwords).
364///
365/// # Safety
366/// `ecam_base` must be the identity-mapped ECAM window base.
367unsafe fn probe_bar_size64(ecam_base: usize, bus: u8, dev: u8, fun: u8, bar_idx: usize) -> u64 {
368    let off_lo = CFG_BAR0 + bar_idx * 4;
369    let off_hi = CFG_BAR0 + (bar_idx + 1) * 4;
370
371    // SAFETY: ECAM window is identity-mapped.
372    let orig_cmd = unsafe { read16(ecam_base, bus, dev, fun, CFG_COMMAND) };
373    // SAFETY: ECAM window is identity-mapped.
374    let orig_lo = unsafe { read32(ecam_base, bus, dev, fun, off_lo) };
375    // SAFETY: ECAM window is identity-mapped.
376    let orig_hi = unsafe { read32(ecam_base, bus, dev, fun, off_hi) };
377
378    // Disable memory decode during probe.
379    // SAFETY: ECAM window is identity-mapped.
380    unsafe {
381        write16(
382            ecam_base,
383            bus,
384            dev,
385            fun,
386            CFG_COMMAND,
387            orig_cmd & !CMD_MEM_SPACE,
388        );
389    }
390
391    // Write all-ones to both halves.
392    // SAFETY: ECAM window is identity-mapped.
393    unsafe {
394        write32(ecam_base, bus, dev, fun, off_lo, 0xFFFF_FFFF);
395    }
396    unsafe {
397        write32(ecam_base, bus, dev, fun, off_hi, 0xFFFF_FFFF);
398    }
399
400    // SAFETY: ECAM window is identity-mapped.
401    let rb_lo = unsafe { read32(ecam_base, bus, dev, fun, off_lo) };
402    // SAFETY: ECAM window is identity-mapped.
403    let rb_hi = unsafe { read32(ecam_base, bus, dev, fun, off_hi) };
404
405    // Restore.
406    // SAFETY: ECAM window is identity-mapped.
407    unsafe {
408        write32(ecam_base, bus, dev, fun, off_lo, orig_lo);
409    }
410    unsafe {
411        write32(ecam_base, bus, dev, fun, off_hi, orig_hi);
412    }
413    unsafe {
414        write16(ecam_base, bus, dev, fun, CFG_COMMAND, orig_cmd);
415    }
416
417    let mask64 = ((rb_hi as u64) << 32) | ((rb_lo & 0xFFFF_FFF0) as u64);
418    if mask64 == 0 {
419        return 0;
420    }
421    (!mask64).wrapping_add(1)
422}

426/// Walk the capability list for a type-0 function; return MSI-X and PM caps.
427///
428/// # Safety
429/// `ecam_base` must be the identity-mapped ECAM window base.
430unsafe fn walk_caps(
431    ecam_base: usize,
432    bus: u8,
433    dev: u8,
434    fun: u8,
435) -> (Option<MsixCap>, Option<PmCap>) {
436    let mut msix = None;
437    let mut pm = None;
438
439    // Capability list is present only when status bit 4 is set.
440    // SAFETY: ECAM window is identity-mapped.
441    let status = unsafe { read16(ecam_base, bus, dev, fun, CFG_STATUS) };
442    if status & (1 << 4) == 0 {
443        return (None, None);
444    }
445
446    // SAFETY: ECAM window is identity-mapped.
447    let mut cap_ptr = unsafe { read8(ecam_base, bus, dev, fun, CFG_CAP_PTR) } & 0xFC;
448    let mut budget = 64u8; // guard against malformed circular lists
449
450    while cap_ptr != 0 && budget > 0 {
451        budget -= 1;
452        // SAFETY: ECAM window is identity-mapped.
453        let cap_id = unsafe { read8(ecam_base, bus, dev, fun, cap_ptr as usize) };
454        // SAFETY: ECAM window is identity-mapped.
455        let next = unsafe { read8(ecam_base, bus, dev, fun, cap_ptr as usize + 1) } & 0xFC;
456
457        match cap_id {
458            CAP_ID_MSIX => {
459                // SAFETY: ECAM window is identity-mapped.
460                let msg_ctrl = unsafe { read16(ecam_base, bus, dev, fun, cap_ptr as usize + 2) };
461                msix = Some(MsixCap {
462                    cap_offset: cap_ptr,
463                    msg_ctrl,
464                });
465            }
466            CAP_ID_PM => {
467                pm = Some(PmCap {
468                    cap_offset: cap_ptr,
469                });
470            }
471            _ => {}
472        }
473        cap_ptr = next;
474    }
475
476    (msix, pm)
477}

481/// Scan bus 0 of the ECAM window and populate the global `PCI_DEVICES` list.
482///
483/// # Safety
484/// `ecam_base` must point to an identity-mapped 1 MiB ECAM bus-0 window.
485unsafe fn scan(ecam_base: usize) {
486    let mut devices = PCI_DEVICES.lock();
487
488    for dev in 0u8..32 {
489        // Check function 0 first. If vendor == 0xFFFF, no device is present.
490        // SAFETY: ECAM window is identity-mapped.
491        let vendor = unsafe { read16(ecam_base, 0, dev, 0, CFG_VENDOR_ID) };
492        if vendor == 0xFFFF {
493            continue;
494        }
495
496        // SAFETY: ECAM window is identity-mapped.
497        let hdr_type = unsafe { read8(ecam_base, 0, dev, 0, CFG_HEADER_TYPE) };
498        // Bit 7 of header_type indicates a multi-function device.
499        let is_multi = hdr_type & 0x80 != 0;
500        let max_fun: u8 = if is_multi { 8 } else { 1 };
501
502        for fun in 0u8..max_fun {
503            // SAFETY: ECAM window is identity-mapped.
504            let vid = unsafe { read16(ecam_base, 0, dev, fun, CFG_VENDOR_ID) };
505            if vid == 0xFFFF {
506                continue;
507            }
508
509            // SAFETY: ECAM window is identity-mapped.
510            let did = unsafe { read16(ecam_base, 0, dev, fun, CFG_DEVICE_ID) };
511            let prog_if = unsafe { read8(ecam_base, 0, dev, fun, CFG_CLASS_PROG) };
512            let subclass = unsafe { read8(ecam_base, 0, dev, fun, CFG_SUBCLASS) };
513            let class = unsafe { read8(ecam_base, 0, dev, fun, CFG_CLASS_CODE) };
514
515            // Decode BARs for type-0 headers (endpoints). Skip bridges (type-1).
516            let hdr = unsafe { read8(ecam_base, 0, dev, fun, CFG_HEADER_TYPE) } & 0x7F;
517            let bars = if hdr == 0 {
518                // SAFETY: ECAM window is identity-mapped.
519                unsafe { decode_bars(ecam_base, 0, dev, fun) }
520            } else {
521                [Bar::None; 6]
522            };
523
524            // Walk capabilities.
525            // SAFETY: ECAM window is identity-mapped.
526            let (msix, pm) = unsafe { walk_caps(ecam_base, 0, dev, fun) };
527
528            let bar0_addr = bars[0].base_addr();
529            log::info!(
530                "[pcie] {:02x}:{:02x}.{} vendor={:04x} device={:04x} \
531                 class={:02x}:{:02x}:{:02x} bar0={:#x}",
532                0u8,
533                dev,
534                fun,
535                vid,
536                did,
537                class,
538                subclass,
539                prog_if,
540                bar0_addr
541            );
542
543            devices.push(PciDevice {
544                bdf: (0, dev, fun),
545                vendor_id: vid,
546                device_id: did,
547                class,
548                subclass,
549                prog_if,
550                bars,
551                msix,
552                pm,
553            });
554        }
555    }
556}

560/// Initialise the ECAM scanner.
561///
562/// Must be called after paging is active (ECAM window mapped) and before
563/// Driver Cells are spawned (they call sys_find_pcie_device to discover their
564/// endpoint). Idempotent; safe to call more than once (later calls are no-ops).
565pub fn init() {
566    // Determine ECAM base for the current architecture.
567    // x86_64: use runtime value from ACPI MCFG (set_ecam_base_x86), falling
568    //         back to the QEMU q35 default if ACPI was not parsed.
569    #[cfg(target_arch = "x86_64")]
570    let ecam_base = ecam_base_x86();
571    #[cfg(target_arch = "riscv64")]
572    let ecam_base = ECAM_BASE_RISCV;
573    #[cfg(target_arch = "aarch64")]
574    let ecam_base = ECAM_BASE_AARCH64;
575    #[cfg(not(any(
576        target_arch = "x86_64",
577        target_arch = "riscv64",
578        target_arch = "aarch64"
579    )))]
580    let ecam_base = 0usize; // bare-physical arches: no PCIe
581
582    if ecam_base == 0 {
583        log::info!("[pcie] ECAM: no PCIe on this architecture");
584        return;
585    }
586
587    // If already populated (e.g. called twice), skip re-scan.
588    if !PCI_DEVICES.lock().is_empty() {
589        log::info!("[pcie] ECAM: already scanned, skipping");
590        return;
591    }
592
593    log::info!("[pcie] ECAM scan bus 0 @ {:#x} (1 MiB window)", ecam_base);
594
595    // SAFETY: The ECAM bus-0 window (1 MiB at ecam_base) is identity-mapped in
596    // `init_kernel_paging` (riscv/arm) or `init_kernel_paging_x86` (x86_64)
597    // before this function is called. Volatile config-space reads/writes are
598    // bounded to that window.
599    unsafe {
600        scan(ecam_base);
601    }
602
603    let count = PCI_DEVICES.lock().len();
604    if count == 0 {
605        log::warn!("[pcie] ECAM scan found no devices on bus 0 — check ECAM base and MMIO mapping");
606    } else {
607        log::info!("[pcie] ECAM scan complete: {} device(s) found", count);
608    }
609
610    // Register all discovered BAR windows in the Resource Registry so Driver
611    // Cells with PcieDriverCap can call sys_request_mmio on them.
612    for dev in PCI_DEVICES.lock().iter() {
613        for bar in &dev.bars {
614            let base = bar.base_addr() as usize;
615            if base == 0 {
616                continue;
617            }
618            // Probe size is stored in Bar::Memory32/Memory64; use a safe default
619            // of 16 KiB if the probed size is 0 (identity entry).
620            let len = match bar {
621                Bar::Memory32 { size, .. } => {
622                    if *size == 0 {
623                        0x4000
624                    } else {
625                        *size as usize
626                    }
627                }
628                Bar::Memory64 { size, .. } => {
629                    if *size == 0 {
630                        0x4000
631                    } else {
632                        *size as usize
633                    }
634                }
635                _ => continue,
636            };
637            crate::resource_registry::register_pcie_bar(base, len);
638        }
639    }
640}

642/// Register a device announced by the Platform Cell via `sys_register_pci_device`.
643///
644/// Decodes `bdf` and `cls` into a `PciDevice` stub with BAR0 populated.
645/// Skips duplicate BDFs (idempotent: kernel ECAM scan may already have found the device).
646/// Also registers BAR0 in the Resource Registry for `sys_request_mmio` validation.
647pub fn register_device(bdf: u32, cls: u32, bar0_base: usize, bar0_size: usize) {
648    // Canonical BDF wire layout everywhere (Platform Cell, FindPcieDevice,
649    // GrantDma, iommu_x86): bus[15:8] | dev[7:3] | fun[2:0]. An earlier decode
650    // here assumed bus<<16|dev<<8|fun, which garbled the stored (bus,dev,fun)
651    // tuple for every device (it only round-tripped through FindPcieDevice's
652    // re-encode by luck on bus 0).
653    let bus = ((bdf >> 8) & 0xFF) as u8;
654    let dev = ((bdf >> 3) & 0x1F) as u8;
655    let fun = (bdf & 0x07) as u8;
656    let class = ((cls >> 16) & 0xFF) as u8;
657    let subclass = ((cls >> 8) & 0xFF) as u8;
658    let prog_if = (cls & 0xFF) as u8;
659
660    {
661        let devices = PCI_DEVICES.lock();
662        if devices.iter().any(|d| d.bdf == (bus, dev, fun)) {
663            return; // already present (kernel ECAM scan ran first)
664        }
665    }
666
667    let bars = if bar0_base != 0 {
668        let bar0 = if bar0_size <= u32::MAX as usize {
669            Bar::Memory32 {
670                addr: bar0_base as u32,
671                size: bar0_size as u32,
672            }
673        } else {
674            Bar::Memory64 {
675                addr: bar0_base as u64,
676                size: bar0_size as u64,
677            }
678        };
679        let mut arr = [Bar::None; 6];
680        arr[0] = bar0;
681        arr
682    } else {
683        [Bar::None; 6]
684    };
685
686    log::info!(
687        "[pcie] platform-reg {:02x}:{:02x}.{} class={:02x}:{:02x}:{:02x} bar0={:#x}",
688        bus,
689        dev,
690        fun,
691        class,
692        subclass,
693        prog_if,
694        bar0_base
695    );
696
697    PCI_DEVICES.lock().push(PciDevice {
698        bdf: (bus, dev, fun),
699        vendor_id: 0, // not probed by Platform Cell (BAR/class is sufficient for find_class)
700        device_id: 0,
701        class,
702        subclass,
703        prog_if,
704        bars,
705        msix: None,
706        pm: None,
707    });
708
709    if bar0_base != 0 && bar0_size != 0 {
710        crate::resource_registry::register_pcie_bar(bar0_base, bar0_size);
711    }
712}
713
714/// Return a cloned snapshot of the discovered device list.
715pub fn devices() -> Vec<PciDevice> {
716    PCI_DEVICES.lock().clone()
717}

719/// Find the first device matching (class, subclass, prog_if), or `None`.
720///
721/// Example: NVMe = `find_class(0x01, 0x08, 0x02)`.
722pub fn find_class(class: u8, subclass: u8, prog_if: u8) -> Option<PciDevice> {
723    PCI_DEVICES
724        .lock()
725        .iter()
726        .find(|d| d.class == class && d.subclass == subclass && d.prog_if == prog_if)
727        .cloned()
728}
