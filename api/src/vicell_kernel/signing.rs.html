1//! Cell binary signing — Ed25519 signature verification for spawned cell ELFs.
2//!
3//! The kernel holds only a public key (fleet trust anchor). Cell binaries are
4//! signed offline with the corresponding private key and carry the 64-byte
5//! signature in an `__ViCell_sig` ELF section (non-loadable, not in any PT_LOAD).
6//!
7//! Canonical signed payload:
8//!   1. PT_LOAD segments sorted by (p_offset, p_filesz, phdr_index) — execution content
9//!   2. `__ViCell_manifest` section bytes — capability claims (signed even if outside PT_LOAD)
10//!
11//! The ELF header is NOT included: `objcopy --add-section` mutates `e_shnum`/`e_shoff` when
12//! embedding `__ViCell_sig`, so including the header would break verification. PT_LOAD covers
13//! all executable code; the manifest covers all capability claims.
14//!
15//! Verify-only: the kernel never signs (private key lives offline).
16
17use alloc::vec::Vec;
18
19/// Dev Ed25519 **public** key — derived from the fixed dev seed in
20/// `scripts/sign-cell.py` (seed `[0x43]*32`, reproducible; never shipped in release).
21const DEV_CELL_SIGNER_PUBKEY: [u8; 32] = [
22    0x22, 0xfc, 0x29, 0x77, 0x92, 0xf0, 0xb6, 0xff, 0xc0, 0xbf, 0xcf, 0xdb, 0x7e, 0xdb, 0x0c, 0x0a,
23    0xa1, 0x4e, 0x02, 0x5a, 0x36, 0x5e, 0xc0, 0xe3, 0x42, 0xe8, 0x6e, 0x38, 0x29, 0xcb, 0x74, 0xb6,
24];
25
26/// Fleet cell-signing trust anchor.
27/// `dev-signing-key` → the reproducible dev key above (matching `scripts/sign-cell.py --seed 0x43*32`).
28/// Otherwise → a zero placeholder that fails every verify (fail-closed until prod key is provisioned).
29#[cfg(feature = "dev-signing-key")]
30const CELL_SIGNER_PUBKEY: [u8; 32] = DEV_CELL_SIGNER_PUBKEY;
31
32#[cfg(not(feature = "dev-signing-key"))]
33const CELL_SIGNER_PUBKEY: [u8; 32] = [0u8; 32]; // TODO(prod): provisioned fleet key
34
35/// Returns `true` when the `signing-required` build feature is set (CI/prod posture).
36/// In dev mode (default) an absent signature is permitted; with `signing-required`
37/// an unsigned cell is denied the same as a tampered one.
38pub const fn signing_required() -> bool {
39    cfg!(feature = "signing-required")
40}

42/// Extract the 64-byte Ed25519 signature from the `__ViCell_sig` ELF section.
43///
44/// Returns `None` if the section is absent or not exactly 64 bytes.
45pub fn extract_sig(elf_bytes: &[u8]) -> Option<[u8; 64]> {
46    use crate::loader::ElfParser;
47    let raw = crate::loader::ElfLoader
48        .get_section(elf_bytes, "__ViCell_sig")
49        .ok()?;
50    if raw.len() != 64 {
51        return None;
52    }
53    let mut sig = [0u8; 64];
54    sig.copy_from_slice(raw);
55    Some(sig)
56}

58/// Verify the Ed25519 signature of a cell ELF binary.
59///
60/// Returns `false` on any malformed input, missing section, or signature
61/// mismatch — never panics. Canonically covers the ELF header, all PT_LOAD
62/// segments, and the `__ViCell_manifest` section.
63pub fn verify_cell(elf_bytes: &[u8], sig: &[u8; 64]) -> bool {
64    verify_cell_with_key(elf_bytes, sig, &CELL_SIGNER_PUBKEY)
65}
66
67/// Inner implementation; accepts an explicit key so `self_test` can use
68/// the precomputed test key without touching `CELL_SIGNER_PUBKEY`.
69fn verify_cell_with_key(elf_bytes: &[u8], sig: &[u8; 64], pubkey: &[u8; 32]) -> bool {
70    use xmas_elf::program::Type;
71    use xmas_elf::ElfFile;
72
73    let elf = match ElfFile::new(elf_bytes) {
74        Ok(e) => e,
75        Err(_) => return false,
76    };
77
78    // Build the signed payload.
79    let mut payload: Vec<u8> = Vec::new();
80
81    // 1. PT_LOAD segments — sort by (p_offset, p_filesz, phdr_index) for a
82    //    deterministic total order even when two segments share an offset.
83    let mut segments: Vec<(u64, u64, usize)> = elf
84        .program_iter()
85        .enumerate()
86        .filter_map(|(i, ph)| {
87            if ph.get_type() == Ok(Type::Load) && ph.file_size() > 0 {
88                Some((ph.offset(), ph.file_size(), i))
89            } else {
90                None
91            }
92        })
93        .collect();
94    segments.sort_by_key(|&(off, fsz, idx)| (off, fsz, idx));
95
96    for (offset, filesz, _) in &segments {
97        let start = *offset as usize;
98        let end = match start.checked_add(*filesz as usize) {
99            Some(e) if e <= elf_bytes.len() => e,
100            _ => return false, // out-of-bounds PT_LOAD — malformed ELF
101        };
102        payload.extend_from_slice(&elf_bytes[start..end]);
103    }
104
105    // 2. __ViCell_manifest section — capability claims. Explicitly included
106    //    because the manifest may fall outside PT_LOAD in some linker layouts.
107    //    A missing manifest means the cell declares no capabilities; we include
108    //    an empty slice (nothing to absorb) so manifest-less cells are signable.
109    if let Ok(manifest) = {
110        use crate::loader::ElfParser;
111        crate::loader::ElfLoader.get_section(elf_bytes, "__ViCell_manifest")
112    } {
113        payload.extend_from_slice(manifest);
114    }
115
116    crate::ed25519::verify(pubkey, &payload, sig)
117}

119/// Boot-time self-test using a precomputed RFC-style test vector.
120///
121/// Uses a separate precomputed `(pubkey, payload, sig)` triple — **no private
122/// key in the kernel**. Returns `true` iff the known-good vector verifies AND
123/// a flipped-byte payload is rejected.
124pub fn self_test() -> bool {
125    // Precomputed vector — seed [0x43]*32, payload b"CellosSigningTest".
126    // Regenerate: `python scripts/sign-cell.py --emit-test-vector`
127    const TEST_PUBKEY: [u8; 32] = [
128        0x22, 0xfc, 0x29, 0x77, 0x92, 0xf0, 0xb6, 0xff, 0xc0, 0xbf, 0xcf, 0xdb, 0x7e, 0xdb, 0x0c,
129        0x0a, 0xa1, 0x4e, 0x02, 0x5a, 0x36, 0x5e, 0xc0, 0xe3, 0x42, 0xe8, 0x6e, 0x38, 0x29, 0xcb,
130        0x74, 0xb6,
131    ];
132    const TEST_SIG: [u8; 64] = [
133        0x22, 0xf6, 0x2e, 0xba, 0x53, 0x9c, 0x66, 0xa0, 0xc1, 0xed, 0x39, 0xc8, 0x90, 0x04, 0xf8,
134        0xfc, 0x46, 0xb0, 0xe5, 0x42, 0xc9, 0x97, 0x22, 0x2d, 0x3f, 0x10, 0x17, 0xf3, 0xa4, 0x56,
135        0x67, 0x58, 0x9b, 0x49, 0x98, 0x2b, 0x4a, 0x48, 0x23, 0x11, 0x90, 0x09, 0x25, 0xe3, 0x9f,
136        0x02, 0x0b, 0x0e, 0x34, 0x70, 0x25, 0xfa, 0x10, 0xe3, 0x7e, 0xac, 0xd4, 0xb1, 0x6c, 0x66,
137        0xcf, 0x7b, 0x1e, 0x0a,
138    ];
139    const TEST_PAYLOAD: &[u8] = b"CellosSigningTest";
140
141    // Positive: known-good vector must verify.
142    if !crate::ed25519::verify(&TEST_PUBKEY, TEST_PAYLOAD, &TEST_SIG) {
143        return false;
144    }
145    // Negative: flipped byte in payload must be rejected.
146    let mut bad_payload = alloc::vec![0u8; TEST_PAYLOAD.len()];
147    bad_payload.copy_from_slice(TEST_PAYLOAD);
148    bad_payload[0] ^= 0x01;
149    if crate::ed25519::verify(&TEST_PUBKEY, &bad_payload, &TEST_SIG) {
150        return false;
151    }
152    true
153}
