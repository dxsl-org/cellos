11/// Global BootFS / initramfs instance — the FAT16 `kernel_fs.img` baked into
12/// the kernel binary. Solves the chicken-and-egg of loading the VFS service
13/// binary before the VFS service exists; the VFS cell also proxies `/bin`
14/// reads here via the FD syscalls (specs/09-vfs.md v0.5 §2).
15///
16/// Naming note: the spec term "viFS1" (a planned RedoxFS fork) was dropped
17/// 2026-06-10 — this static is unrelated to it despite the shared name.
18pub static VIFS1: Spinlock<Option<Box<dyn ViFileSystem>>> = Spinlock::new(None);

20/// Read a complete file from the embedded FAT filesystem into a heap buffer.
21///
22/// Path components are uppercased to match FAT16's all-caps storage convention
23/// (e.g. `/bin/vfs` → opened as `/BIN/VFS`).  Returns `ViError::NotFound`
24/// when VIFS1 is not mounted or the path does not exist.
25pub fn read_file_from_vifs1(path: &str) -> ViResult<Box<[u8]>> {
26    // Build an uppercase copy of the path: FAT16 names are uppercase.
27    let mut upper = Vec::with_capacity(path.len());
28    for b in path.bytes() {
29        upper.push(b.to_ascii_uppercase());
30    }
31    let upper_path = core::str::from_utf8(&upper).map_err(|_| ViError::InvalidInput)?;
32
33    let mut file = {
34        let guard = VIFS1.lock();
35        let fs = guard.as_ref().ok_or(ViError::NotFound)?;
36        fs.open(upper_path, OpenMode::Read)?
37    };
38
39    let mut buf: Vec<u8> = Vec::new();
40    let mut chunk = [0u8; 4096];
41    loop {
42        match file.read(&mut chunk) {
43            Ok(0) => break,
44            Ok(n) => buf.extend_from_slice(&chunk[..n]),
45            Err(ViError::NotFound) => break, // EOF sentinel on some FAT impls
46            Err(e) => return Err(e),
47        }
48    }
49    if buf.is_empty() {
50        return Err(ViError::NotFound);
51    }
52    Ok(buf.into_boxed_slice())
53}

55pub fn init() {
56    log::info!("Filesystem: Initializing...");
57
58    // Attempt to mount the embedded FAT filesystem (FAT16) from the RAM disk.
59    match fat::ViFatFS::new() {
60        Ok(fs) => {
61            log::info!("Filesystem: FAT16 mounted successfully.");
62            *VIFS1.lock() = Some(Box::new(fs));
63        }
64        Err(e) => {
65            log::error!("Filesystem: Failed to mount FAT: {:?}", e);
66        }
67    }
