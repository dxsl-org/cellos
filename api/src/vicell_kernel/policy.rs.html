1//! Signed operator policy (roadmap §G.2 P5b) — the headless "consent" mechanism.
2//!
3//! At boot the kernel reads `/POLICY.BIN` from the kernel-embedded VIFS1, verifies
4//! its Ed25519 signature against the fleet root public key, and parses it into a
5//! `path → CapSet` table. Phase 04 folds `lookup()` into the spawn-time grant so
6//! the effective caps are `manifest ∩ spawner ∩ policy`.
7//!
8//! Security invariants (red-team-driven):
9//! - **Verify-then-parse:** the signature covers `blob[..len-64]`; verify FIRST
10//!   (length-only, no field parsing) so the parser never runs on unverified bytes.
11//! - **Panic-free parser:** every field read is bounds-checked; malformed →
12//!   `Invalid`, never a panic (a boot-path panic = no boot = bricked robot).
13//! - **Fail-safe:** an *invalid* signature/parse is ALWAYS fail-closed. An *absent*
14//!   policy is dev-permissive in G1 (this build) and fail-closed only when the
15//!   `policy-required` feature is set (real-fleet posture). See `lookup`.
16//! - **Domain validation:** parsed cap bytes are masked to known bits; unknown
17//!   bits → `Invalid` (a signed-but-malformed policy is still rejected).
18

25/// Magic "VPOL" as a little-endian u32 (bytes V,P,O,L).
26const MAGIC: u32 = u32::from_le_bytes(*b"VPOL");
27const VERSION: u8 = 1;
28const SIG_LEN: usize = 64;
29const HEADER_LEN: usize = 8; // magic(4) + version(1) + flags(1) + entry_count(2)
30const CAP_BYTES: usize = 6; // block_io, network, spawn, hyp, mmio_devices, block_regions
31/// 8.3-safe, root-level path (VIFS1 uppercases + is FAT16 8.3).
32const POLICY_PATH: &str = "/POLICY.BIN";
33
34/// Valid `mmio_devices` bits and `block_regions` bits (domain-validation masks).
35const MMIO_MASK: u8 = DEV_GPIO | DEV_UART | DEV_PCIE;
36const REGION_MASK: u8 = 0b111;
37
38/// Dev fleet Ed25519 **public** key — derived from the fixed dev seed in
39/// `scripts/sign-policy.py` (reproducible; a dev key, never shipped in release).
40const DEV_FLEET_PUBKEY: [u8; 32] = [
41    0x21, 0x52, 0xf8, 0xd1, 0x9b, 0x79, 0x1d, 0x24, 0x45, 0x32, 0x42, 0xe1, 0x5f, 0x2e, 0xab, 0x6c,
42    0xb7, 0xcf, 0xfa, 0x7b, 0x6a, 0x5e, 0xd3, 0x00, 0x97, 0x96, 0x0e, 0x06, 0x98, 0x81, 0xdb, 0x12,
43];
44
45/// Fleet root Ed25519 **public** key (trust anchor; lives in the kernel TCB, not
46/// in mutable VIFS1 data). `dev-policy-key` feature → the dev key (so a dev-signed
47/// `/POLICY.BIN` verifies); otherwise a placeholder the production provisioning
48/// replaces. A zero/placeholder key fails every verify → any present policy is
49/// `Invalid` (fail-closed), the safe direction; absent policy still boots
50/// (dev-permissive).
51#[cfg(feature = "dev-policy-key")]
52const FLEET_ROOT_PUBKEY: [u8; 32] = DEV_FLEET_PUBKEY;
53#[cfg(not(feature = "dev-policy-key"))]
54const FLEET_ROOT_PUBKEY: [u8; 32] = [0u8; 32]; // TODO(prod): provisioned fleet key

56/// Result of a policy lookup for a given cell path.
57pub enum PolicyDecision {
58    /// Policy explicitly grants this path the given caps (ceiling).
59    Permit(CapSet),
60    /// Policy is present and explicitly denies (or invalid → fail-closed).
61    DenyAll,
62    /// No policy entry for this path (or policy absent). Caller applies the
63    /// fail-safe rule: dev-permissive keeps the spawner-intersected caps;
64    /// `policy-required` treats it as deny.
65    NoEntry,
66}
67
68struct PolicyEntry {
69    path: String,
70    caps: CapSet,
71}
72
73enum PolicyState {
74    Loaded(Vec<PolicyEntry>),
75    Absent,
76    Invalid,
77}
78
79static POLICY: Spinlock<Option<PolicyState>> = Spinlock::new(None);

89/// Load + verify the operator policy from VIFS1. Call once at boot AFTER
90/// `fs::init()` and BEFORE the first cap-bearing cell spawns. Eager-only (no
91/// lazy path — VIFS1 is kernel-embedded and available this early).
92pub fn load_from_vifs1() {
93    let blob = match crate::fs::read_file_from_vifs1(POLICY_PATH) {
94        Ok(b) if !b.is_empty() => b,
95        _ => {
96            log::info!("[policy] no {} in VIFS1 — absent", POLICY_PATH);
97            crate::audit::log_event(
98                crate::audit::AuditEvent::PolicyAbsent,
99                &crate::audit::encode_u32x2(0, 0),
100            );
101            *POLICY.lock() = Some(PolicyState::Absent);
102            return;
103        }
104    };
105
106    // Verify-then-parse: the trailing SIG_LEN bytes are the signature over the body.
107    if blob.len() < HEADER_LEN + SIG_LEN {
108        return mark_invalid(1);
109    }
110    let split = blob.len() - SIG_LEN;
111    let (body, sig) = blob.split_at(split);
112    let mut sig64 = [0u8; SIG_LEN];
113    sig64.copy_from_slice(sig);
114    if !crate::ed25519::verify(&FLEET_ROOT_PUBKEY, body, &sig64) {
115        log::warn!("[policy] signature verification FAILED — fail-closed");
116        return mark_invalid(2);
117    }
118
119    match parse(body) {
120        Some(entries) => {
121            let n = entries.len() as u32;
122            log::info!("[policy] loaded + verified ({} entries)", n);
123            crate::audit::log_event(
124                crate::audit::AuditEvent::PolicyLoaded,
125                &crate::audit::encode_u32x2(n, 0),
126            );
127            *POLICY.lock() = Some(PolicyState::Loaded(entries));
128        }
129        None => {
130            log::warn!("[policy] malformed body — fail-closed");
131            mark_invalid(3);
132        }
133    }
134}

136fn mark_invalid(reason: u32) {
137    crate::audit::log_event(
138        crate::audit::AuditEvent::PolicyInvalid,
139        &crate::audit::encode_u32x2(reason, 0),
140    );
141    *POLICY.lock() = Some(PolicyState::Invalid);
142}

144/// Parse the (already signature-verified) body into entries. Panic-free: every
145/// read is bounds-checked; any malformation or out-of-domain cap bit → `None`.
146fn parse(body: &[u8]) -> Option<Vec<PolicyEntry>> {
147    if body.len() < HEADER_LEN {
148        return None;
149    }
150    let magic = u32::from_le_bytes([body[0], body[1], body[2], body[3]]);
151    if magic != MAGIC || body[4] != VERSION {
152        return None;
153    }
154    let count = u16::from_le_bytes([body[6], body[7]]) as usize;
155
156    let mut entries = Vec::new();
157    let mut off = HEADER_LEN;
158    for _ in 0..count {
159        // path_len
160        let path_len = *body.get(off)? as usize;
161        off += 1;
162        // path bytes
163        let path_bytes = body.get(off..off.checked_add(path_len)?)?;
164        let path = core::str::from_utf8(path_bytes).ok()?;
165        off += path_len;
166        // 6 cap bytes
167        let caps_raw = body.get(off..off.checked_add(CAP_BYTES)?)?;
168        off += CAP_BYTES;
169
170        let mmio_devices = caps_raw[4];
171        let block_regions = caps_raw[5];
172        // Domain validation: reject unknown bits (signed-but-malformed).
173        if mmio_devices & !MMIO_MASK != 0 || block_regions & !REGION_MASK != 0 {
174            return None;
175        }
176        entries.push(PolicyEntry {
177            path: String::from(path),
178            caps: CapSet {
179                block_io: caps_raw[0] != 0,
180                network: caps_raw[1] != 0,
181                spawn: caps_raw[2] != 0,
182                hypervisor: caps_raw[3] != 0,
183                mmio_devices,
184                block_regions,
185                // The operator-policy blob has no fields for the privileged
186                // path-caps; they are never policy-granted (P-TRUST folds them
187                // into the spawner-ceiling intersection, not the policy one).
188                ..CapSet::EMPTY
189            },
190        });
191    }
192    Some(entries)
193}

195/// Self-test of the full signed-policy path: verify + parse a known dev-signed
196/// blob (from `scripts/sign-policy.py`), confirm a known entry parses correctly,
197/// and confirm a tampered blob is REJECTED. Returns `true` iff both hold. Run as
198/// a boot power-on self-test before trusting the policy path.
199pub fn self_test() -> bool {
200    // 135-byte dev-signed blob (4 entries) emitted by scripts/sign-policy.py.
201    const BLOB: [u8; 135] = [
202        0x56, 0x50, 0x4f, 0x4c, 0x01, 0x00, 0x04, 0x00, 0x08, 0x2f, 0x62, 0x69, 0x6e, 0x2f, 0x76,
203        0x66, 0x73, 0x01, 0x00, 0x00, 0x00, 0x00, 0x07, 0x08, 0x2f, 0x62, 0x69, 0x6e, 0x2f, 0x6e,
204        0x65, 0x74, 0x00, 0x01, 0x00, 0x00, 0x00, 0x00, 0x0a, 0x2f, 0x62, 0x69, 0x6e, 0x2f, 0x73,
205        0x68, 0x65, 0x6c, 0x6c, 0x00, 0x00, 0x01, 0x00, 0x00, 0x00, 0x09, 0x2f, 0x62, 0x69, 0x6e,
206        0x2f, 0x69, 0x6e, 0x69, 0x74, 0x01, 0x01, 0x01, 0x00, 0x03, 0x07, 0x44, 0x17, 0x69, 0xc2,
207        0xc9, 0x40, 0x3a, 0x1f, 0x67, 0xcf, 0xfa, 0x4d, 0xa1, 0x23, 0x15, 0x29, 0xa1, 0xa6, 0x62,
208        0x9f, 0xb4, 0xde, 0x48, 0xe1, 0x61, 0x00, 0x0f, 0x83, 0x98, 0x01, 0x00, 0x46, 0x06, 0x6d,
209        0x20, 0xa8, 0xa5, 0xff, 0xd9, 0x05, 0x4f, 0x51, 0x12, 0x46, 0xc6, 0x45, 0x59, 0x7b, 0x15,
210        0xae, 0x1e, 0x22, 0xb6, 0x33, 0xb4, 0x2b, 0xc8, 0x84, 0x28, 0x2d, 0x83, 0x7f, 0xde, 0x00,
211    ];
212    if BLOB.len() < HEADER_LEN + SIG_LEN {
213        return false;
214    }
215    let (body, sig) = BLOB.split_at(BLOB.len() - SIG_LEN);
216    let mut s = [0u8; SIG_LEN];
217    s.copy_from_slice(sig);
218
219    // 1. Valid blob: signature verifies + parses + /bin/vfs has the expected caps.
220    if !crate::ed25519::verify(&DEV_FLEET_PUBKEY, body, &s) {
221        return false;
222    }
223    let Some(entries) = parse(body) else {
224        return false;
225    };
226    let Some(vfs) = entries.iter().find(|e| e.path == "/bin/vfs") else {
227        return false;
228    };
229    if !vfs.caps.block_io || vfs.caps.block_regions != 0b111 {
230        return false;
231    }
232    // 2. Tampered blob: a flipped body byte must FAIL verification.
233    let mut bad = BLOB;
234    bad[10] ^= 0x01;
235    let (bad_body, _) = bad.split_at(bad.len() - SIG_LEN);
236    if crate::ed25519::verify(&DEV_FLEET_PUBKEY, bad_body, &s) {
237        return false;
238    }
239
240    // 3. Phase 04 narrowing rule (decision_to_caps) — default (dev-permissive) posture.
241    let full = CapSet {
242        block_io: true,
243        network: true,
244        spawn: true,
245        hypervisor: false,
246        mmio_devices: 0,
247        block_regions: 0,
248        ..CapSet::EMPTY
249    };
250    let net_only = CapSet {
251        network: true,
252        ..CapSet::EMPTY
253    };
254    // Permit narrows: full ∩ {network} = {network}.
255    if decision_to_caps("/bin/app", full, PolicyDecision::Permit(net_only)) != net_only {
256        return false;
257    }
258    // DenyAll on a non-core path → EMPTY.
259    if decision_to_caps("/bin/app", full, PolicyDecision::DenyAll) != CapSet::EMPTY {
260        return false;
261    }
262    // DenyAll on a trusted-core path → keeps caps (headless recovery hatch).
263    if decision_to_caps("/bin/vfs", full, PolicyDecision::DenyAll) != full {
264        return false;
265    }
266    // NoEntry (dev-permissive default) → keeps caps.
267    if decision_to_caps("/bin/app", full, PolicyDecision::NoEntry) != full {
268        return false;
269    }
270    true
271}

273/// Policy decision for a cell path. See `PolicyDecision`; the caller (Phase 04)
274/// applies the dev-permissive vs `policy-required` fail-safe rule to `NoEntry`.
275pub fn lookup(path: &str) -> PolicyDecision {
276    let guard = POLICY.lock();
277    match guard.as_ref() {
278        Some(PolicyState::Loaded(entries)) => {
279            for e in entries {
280                if e.path == path {
281                    return PolicyDecision::Permit(e.caps);
282                }
283            }
284            PolicyDecision::NoEntry
285        }
286        // Invalid → fail-closed ALWAYS, regardless of posture.
287        Some(PolicyState::Invalid) => PolicyDecision::DenyAll,
288        // Absent / not-yet-loaded → NoEntry; the caller's fail-safe rule decides
289        // (dev-permissive keeps caps; `policy-required` denies).
290        Some(PolicyState::Absent) | None => PolicyDecision::NoEntry,
291    }
292}

294/// The minimal trusted core that is NEVER reduced to no-caps by policy — so a
295/// fail-closed mis-fire (bad/absent policy under `policy-required`) cannot brick
296/// a headless robot by stripping the filesystem/shell/network it needs to recover.
297fn is_trusted_core(path: &str) -> bool {
298    matches!(path, "/bin/vfs" | "/bin/shell" | "/bin/net")
299}
300
301/// Pure narrowing rule: combine spawner-intersected `caps` with a policy
302/// `decision`. Recovery: trusted-core cells keep their caps even under DenyAll /
303/// fail-closed. NoEntry is dev-permissive unless the `policy-required` feature.
304fn decision_to_caps(path: &str, caps: CapSet, decision: PolicyDecision) -> CapSet {
305    match decision {
306        PolicyDecision::Permit(p) => caps.intersect(p),
307        PolicyDecision::DenyAll => {
308            if is_trusted_core(path) {
309                caps
310            } else {
311                CapSet::EMPTY
312            }
313        }
314        PolicyDecision::NoEntry => {
315            #[cfg(feature = "policy-required")]
316            {
317                if is_trusted_core(path) {
318                    caps
319                } else {
320                    CapSet::EMPTY
321                }
322            }
323            #[cfg(not(feature = "policy-required"))]
324            {
325                caps
326            }
327        }
328    }
329}

331/// Apply operator policy to a cell's spawn-time caps (Phase 04): the final grant
332/// is `caps ∩ policy(path)` with trusted-core recovery + fail-safe. Audits when
333/// policy actually narrows the grant. `init` (Spawner::Root) is exempt and must
334/// NOT call this (subjecting the policy loader to the loaded policy is circular).
335pub fn apply(path: &str, tid: usize, caps: CapSet) -> CapSet {
336    // Maintenance image: bypass policy narrowing entirely (field recovery).
337    #[cfg(feature = "maintenance-mode")]
338    {
339        let _ = (path, tid);
340        caps
341    }
342    #[cfg(not(feature = "maintenance-mode"))]
343    {
344        let narrowed = decision_to_caps(path, caps, lookup(path));
345        if narrowed != caps {
346            let dropped = (caps.block_io && !narrowed.block_io) as u32
347                | (((caps.network && !narrowed.network) as u32) << 1)
348                | (((caps.spawn && !narrowed.spawn) as u32) << 2)
349                | (((caps.hypervisor && !narrowed.hypervisor) as u32) << 3);
350            crate::audit::log_event(
351                crate::audit::AuditEvent::CapNarrowedByPolicy,
352                &crate::audit::encode_u32x2(tid as u32, dropped),
353            );
354        }
355        narrowed
356    }
