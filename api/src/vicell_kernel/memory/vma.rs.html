1//! Virtual Memory Area — per-cell region tracker for demand-paging.
2//!
3//! A `VmaList` is embedded in each `Task` (TCB). It records the virtual address
4//! ranges that a cell has reserved, together with the backing physical base and
5//! the PTE flags to use when a page fault triggers on-demand mapping.
6//!
7//! In Phase 01, VMA lists are created empty; the ELF loader (Phase 04) populates
8//! them with `ElfText` and `ElfData` entries when loading a binary. Until then,
9//! any user-mode page fault will find no matching region and panic.
10
11use types::{PhysAddr, VAddr};
12
13/// Semantic classification of a VMA region.
14///
15/// Used by the kernel to apply appropriate fault-handling policy (e.g., ElfText
16/// regions are read-only/executable; Stack regions can grow down).
17#[derive(Debug, Clone, PartialEq)]
18pub enum VmaKind {
19    /// ELF `.text` / read-only executable segment.
20    ElfText,
21    /// ELF `.data` / `.bss` / read-write segment.
22    ElfData,
23    /// User-mode stack (grows downward).
24    Stack,
25    /// Heap (grows upward via `brk`).
26    Heap,
27    /// Shared Grant region mapped into this cell's address space.
28    Grant,
29}
30
31/// A single contiguous virtual memory region with a backing physical base.
32///
33/// `va_start` and `va_end` are page-aligned. `pa_start` is the physical address
34/// of the first page of the backing region (for ELF-backed segments; for
35/// demand-allocated regions `pa_start` is 0 until allocated).
36///
37/// `flags` stores raw architecture PTE flags to install on demand-mapping.
38/// On x86_64 these are the `PTE_*` constants from `hal::paging`; on RISC-V
39/// they are the `PageFlags` bitmask.
40#[derive(Debug, Clone)]
41pub struct VmaRegion {
42    /// Inclusive start of the virtual range (page-aligned).
43    pub va_start: VAddr,
44    /// Exclusive end of the virtual range (page-aligned).
45    pub va_end: VAddr,
46    /// Physical base of the backing memory (0 for demand-allocated regions).
47    pub pa_start: PhysAddr,
48    /// Architecture-specific PTE flags for the on-demand mapping.
49    pub flags: u64,
50    /// Semantic kind of this region.
51    pub kind: VmaKind,
52}
53
54/// Per-cell list of virtual memory areas.
55///
56/// Stored inside `Task` (TCB). All operations are O(n) linear scan — sufficient
57/// for the small number of ELF segments + stack + heap per cell.
58pub struct VmaList(pub alloc::vec::Vec<VmaRegion>);
59
60impl VmaList {
61    /// Create an empty VMA list (zero allocations).
62    pub fn new() -> Self {
63        Self(alloc::vec::Vec::new())
64    }
65
66    /// Append a new region to the list.
67    ///
68    /// The caller is responsible for ensuring regions are non-overlapping.
69    pub fn add(&mut self, r: VmaRegion) {
70        self.0.push(r);
71    }
72
73    /// Find the region that contains virtual address `va`.
74    ///
75    /// Returns `None` if no region covers `va`.  Used by the #PF handler to
76    /// decide whether a fault is a valid demand-page or a true access violation.
77    pub fn find(&self, va: usize) -> Option<&VmaRegion> {
78        self.0.iter().find(|r| va >= r.va_start && va < r.va_end)
79    }
80
81    /// Remove all regions (called on cell teardown).
82    pub fn clear(&mut self) {
83        self.0.clear();
84    }
85
86    /// Number of registered regions.
87    pub fn len(&self) -> usize {
88        self.0.len()
89    }
90
91    /// Returns true if no regions have been registered.
92    pub fn is_empty(&self) -> bool {
93        self.0.is_empty()
94    }
95}
96
97impl Default for VmaList {
98    fn default() -> Self {
99        Self::new()
100    }
101}
102
103impl core::fmt::Debug for VmaList {
104    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
105        write!(f, "VmaList({} regions)", self.0.len())
106    }
107}
