1//! Boot-time KASLR for per-cell virtual base randomisation.
2//!
3//! The seed is derived from the HHDM base (unpredictable to an attacker that
4//! does not know the Limine-randomised physical layout) and the RDTSC counter
5//! on x86_64 (or the mtime CSR on RISC-V). This gives sufficient entropy for
6//! basic ASLR while remaining deterministic within a single boot.
7//!
8//! `kaslr_cell_base(cell_id)` produces a unique, page-aligned virtual base for
9//! each cell in the range `[KASLR_BASE_MIN, KASLR_BASE_MIN + KASLR_RANGE)`.
10//! The ELF loader (Phase 04) calls this function to pick where to load a cell.
11//!
12//! # Limits
13//! - KASLR_BASE_MIN:  0x0000_1000_0000 (256 MiB)
14//! - KASLR_RANGE:     0x0000_6FFF_0000 (~1.75 GiB)
15//! - KASLR_ALIGN:     64 KiB (16 × 4 KiB pages) between cells
16//! - Maximum distinct slots: KASLR_RANGE / KASLR_ALIGN ≈ 28 000
17
18// RV32 lacks native 64-bit atomics; portable-atomic polyfills AtomicU64 there
19// via the critical-section impl hal/arch/riscv registers.
20#[cfg(not(target_arch = "riscv32"))]
21use core::sync::atomic::AtomicU64;
22use core::sync::atomic::Ordering;
23#[cfg(target_arch = "riscv32")]
24use portable_atomic::AtomicU64;
25use types::VAddr;
26
27static KASLR_SEED: AtomicU64 = AtomicU64::new(0);
28
29/// Lowest virtual address a cell may be loaded at.
30const KASLR_BASE_MIN: usize = 0x0000_1000_0000;
31/// Width of the randomisable VA window (must be a power of two or handled as modulo).
32const KASLR_RANGE: usize = 0x0000_6FFF_0000;
33/// Alignment between cell VA bases (64 KiB).
34const KASLR_ALIGN: usize = 0x1000 * 16;
35
36/// Initialise the KASLR seed from boot-time entropy.
37///
38/// Should be called once, early in `kmain`, after the HHDM offset is known.
39///
40/// # Entropy sources
41/// - `hhdm_offset`: Limine randomises the HHDM base; unknown to an attacker.
42/// - x86_64: RDTSC counter at the moment of the call (nanosecond-scale jitter).
43/// - RISC-V: `mtime` lower 32 bits from the platform timer.
44/// - Fallback: a compile-time constant XOR'd with `hhdm_offset`.
45pub fn init_kaslr(hhdm_offset: u64) {
46    let hw_entropy = read_hw_entropy();
47    // Mix HHDM offset with hardware entropy. The golden-ratio constant
48    // (0x9E37_79B9_7F4A_7C15) spreads entropy across all bits.
49    let seed = (hhdm_offset ^ hw_entropy).wrapping_mul(0x9E37_79B9_7F4A_7C15);
50    KASLR_SEED.store(seed, Ordering::Relaxed);
51}
52
53/// Derive a page-aligned virtual base address for `cell_id`.
54///
55/// Each `cell_id` produces a distinct, pseudo-random address in the range
56/// `[KASLR_BASE_MIN, KASLR_BASE_MIN + KASLR_RANGE)`.  The mapping is stable
57/// within a boot but unpredictable across boots (seed changes with HHDM).
58///
59/// # Arguments
60/// * `cell_id` — monotonically increasing cell identifier (0-based).
61pub fn kaslr_cell_base(cell_id: usize) -> VAddr {
62    let seed = KASLR_SEED.load(Ordering::Relaxed);
63    // Per-cell diversification: multiply cell_id by a large odd constant to
64    // scatter cells across the VA range without clustering them sequentially.
65    let mixed = seed ^ (cell_id as u64).wrapping_mul(0x6C62_272E_07BB_0142);
66    // Derive an offset in [0, KASLR_RANGE) aligned to KASLR_ALIGN.
67    let raw_offset = (mixed as usize) % KASLR_RANGE;
68    let aligned_offset = raw_offset & !(KASLR_ALIGN - 1);
69    KASLR_BASE_MIN + aligned_offset
70}

72// ─── Architecture-specific entropy read ──────────────────────────────────────
73
74fn read_hw_entropy() -> u64 {
75    #[cfg(target_arch = "x86_64")]
76    {
77        // RDTSC returns the current timestamp counter. On QEMU TCG this is a
78        // monotonically increasing instruction count — sufficient for ASLR.
79        let lo: u32;
80        let hi: u32;
81        // SAFETY: RDTSC is an unprivileged instruction on x86_64; it reads the
82        // CPU time-stamp counter and has no memory or control-flow side-effects.
83        unsafe {
84            core::arch::asm!(
85                "rdtsc",
86                out("eax") lo,
87                out("edx") hi,
88                options(nomem, nostack)
89            );
90        }
91        ((hi as u64) << 32) | (lo as u64)
92    }
93    #[cfg(target_arch = "riscv64")]
94    {
95        // SAFETY: `csrr time` reads the mtime-backed TIME CSR (unprivileged read
96        // permitted in S-mode when sstatus.TVM=0, which is always the case here).
97        let t: u64;
98        unsafe {
99            core::arch::asm!("csrr {}, time", out(reg) t, options(nomem, nostack));
100        }
101        t
102    }
103    #[cfg(target_arch = "aarch64")]
104    {
105        // SAFETY: CNTVCT_EL0 is the virtual count register; accessible from EL1.
106        let t: u64;
107        unsafe {
108            core::arch::asm!("mrs {}, cntvct_el0", out(reg) t, options(nomem, nostack));
109        }
110        t
111    }
112    #[cfg(not(any(
113        target_arch = "x86_64",
114        target_arch = "riscv64",
115        target_arch = "aarch64"
116    )))]
117    {
118        // No hardware timer available; use a fixed diversifier.
119        // This means KASLR is purely seed-based (hhdm_offset) on these arches.
120        0xDEAD_BEEF_CAFE_1234u64
121    }
122}
