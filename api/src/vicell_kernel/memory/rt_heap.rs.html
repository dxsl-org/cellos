1//! Real-time heap — an O(1) TLSF allocator for RealTime cell stacks.
2//!
3//! Isolated from the global `linked_list_allocator` heap so that RealTime cells
4//! always receive bounded-latency allocation, even when the Normal heap is under
5//! pressure from AI workloads.
6//!
7//! Call `init()` once after the main heap is ready.  Then use `alloc` / `dealloc`
8//! directly; this pool is NOT the `#[global_allocator]`.
9
10use crate::sync::Spinlock;
11use core::alloc::Layout;
12use core::ptr::NonNull;
13
14/// TLSF first-level length: 20 → max block ≈ 1 MiB.
15const FLLEN: usize = 20;
16/// TLSF second-level length: 16 → worst-case internal fragmentation ≈ 6%.
17const SLLEN: usize = 16;
18
19type RtTlsf = rlsf::Tlsf<'static, u32, u16, FLLEN, SLLEN>;
20
21/// Backing memory for the RT pool.
22///
23/// 256 KiB static store — enough for ~4 RealTime cells with 64 KiB stacks each.
24/// `MaybeUninit<u8>` matches `insert_free_block`'s expected slice element type.
25/// `align(8)` satisfies TLSF block-header alignment requirements.
26#[repr(C, align(8))]
27struct RtPoolMem([core::mem::MaybeUninit<u8>; 256 * 1024]);
28
29static mut RT_POOL_MEM: RtPoolMem = RtPoolMem([core::mem::MaybeUninit::uninit(); 256 * 1024]);
30
31static RT_HEAP: Spinlock<Option<RtTlsf>> = Spinlock::new(None);
32
33/// Force-release this module's lock during fault teardown.
34///
35/// # Safety
36/// Single-hart; called only from the fault/panic path with interrupts disabled.
37pub unsafe fn force_unlock_locks() {
38    RT_HEAP.force_unlock();
39}
40
41/// Initialise the RT heap.
42///
43/// Must be called once after `memory::heap::init_heap()`.  Subsequent calls
44/// are no-ops (idempotent).
45pub fn init() {
46    let mut guard = RT_HEAP.lock();
47    if guard.is_some() {
48        return;
49    }
50
51    // SAFETY: RtTlsf (rlsf::Tlsf) is valid when zero-initialised — all-zero
52    // bitmaps represent an empty pool with no free blocks.
53    let mut tlsf: RtTlsf = unsafe { core::mem::zeroed() };
54    // SAFETY: RT_POOL_MEM is a static array initialised to zero.  This function
55    // is called once (guarded by the Spinlock check above) before any RT cell
56    // allocation, so no aliased mutable reference exists.
57    unsafe {
58        tlsf.insert_free_block(&mut RT_POOL_MEM.0[..]);
59    }
60    *guard = Some(tlsf);
61    log::info!("[rt-heap] TLSF pool ready — 256 KiB, O(1) alloc/dealloc");
62}
63
64/// Allocate `layout` bytes from the RT pool.
65///
66/// Returns `None` on OOM; callers should convert this to `ViError::OutOfMemory`.
67///
68/// # Safety
69/// The returned pointer is valid and correctly aligned until `dealloc` is called
70/// with the same pointer and `layout.align()`.
71pub unsafe fn alloc(layout: Layout) -> Option<NonNull<u8>> {
72    RT_HEAP
73        .lock()
74        .as_mut()
75        .expect("rt_heap::alloc called before rt_heap::init")
76        .allocate(layout)
77}
78
79/// Deallocate memory previously returned by `rt_heap::alloc`.
80///
81/// # Safety
82/// `ptr` must have been returned by `rt_heap::alloc` with the given `align`.
83/// Calling this with a pointer from the global heap is undefined behaviour.
84pub unsafe fn dealloc(ptr: NonNull<u8>, align: usize) {
85    if let Some(tlsf) = RT_HEAP.lock().as_mut() {
86        tlsf.deallocate(ptr, align);
87    }
