1//! ARM64 Stage-2 (IPA → PA) page-table builder for the ViCell hypervisor.
2//!
3//! # Layout
4//! 40-bit IPA space, 4 KB granule, 3-level walk starting at Level 1 (VTCR.SL0=1).
5//! Root = 2 × 512-entry Level-1 frames (8 KB, 8 KB-aligned); VMID ≥ 1.
6//!
7//! # Invariants (enforced)
8//! - Guest RAM IPA range maps exclusively to the carved HPA region (SAS isolation).
9//! - Emulated-MMIO IPAs (PL011, GIC, virtio-mmio ×4) are intentionally **unmapped**
10//!   so Alpine probes trap to the hypervisor cell (ISV=0 mitigation).
11//! - `Drop` frees every frame allocated here — no leak on VM teardown (Law 8).
12//!
13//! # Descriptor format (ARM DDI 0487 D8.3)
14//! | Field | Bits | Value used |
15//! |-------|------|-----------|
16//! | Valid | [0] | 1 |
17//! | Table/Block | [1] | 1 = table/page; 0 = block |
18//! | MemAttr | [5:2] | 0b1111 = Normal Inner+Outer WB-WA |
19//! | S2AP | [7:6] | 0b11 = RW, 0b01 = RO |
20//! | SH | [9:8] | 0b11 = Inner-shareable |
21//! | AF | [10] | 1 (suppress AF-fault) |
22//! | Output PA | [47:12] | host physical frame address |
23

30// ── S2 descriptor constants ─────────────────────────────────────────────────
31
32const DESC_VALID: u64 = 1 << 0;
33const DESC_TABLE: u64 = 1 << 1; // At L1/L2 → table pointer; at L3 → page descriptor
34
35// Stage-2 MemAttr bits[5:2] — inline, not a MAIR index.
36const S2_MEMATTR_NORMAL: u64 = 0b1111 << 2; // Normal Inner+Outer WB-WA
37const S2_S2AP_RW: u64 = 0b11 << 6; // Read-write
38const S2_S2AP_RO: u64 = 0b01 << 6; // Read-only
39const S2_SH_INNER: u64 = 0b11 << 8; // Inner-shareable
40const S2_AF: u64 = 1 << 10; // Access Flag (suppress fault)
41
42// Base flags for a Normal-WB-IS entry (MemAttr | S2AP_RW | SH | AF).
43const S2_BASE_RW: u64 = S2_MEMATTR_NORMAL | S2_S2AP_RW | S2_SH_INNER | S2_AF;
44const S2_BASE_RO: u64 = S2_MEMATTR_NORMAL | S2_S2AP_RO | S2_SH_INNER | S2_AF;
45
46// PA mask: bits[47:12].
47const PA_MASK: u64 = 0x0000_FFFF_FFFF_F000;
48
49// ── Address-space constants ─────────────────────────────────────────────────
50
51// 40-bit IPA space: T0SZ=24 → 2^40 = 1 TiB.
52const IPA_LIMIT: u64 = 1 << 40;
53
54// Translation level shifts for 4 KB granule.
55const L1_SHIFT: u32 = 30; // L1: bits[39:30], 1 GB per entry
56const L2_SHIFT: u32 = 21; // L2: bits[29:21], 2 MB per entry
57const L3_SHIFT: u32 = 12; // L3: bits[20:12], 4 KB per entry
58
59// 9-bit index within one 512-entry table.
60const IDX_MASK: u64 = 0x1FF;
61
62// Concatenated L1 index: 10 bits (1024 entries across 2 × L1 frames).
63const L1_CONC_MASK: u64 = 0x3FF;
64
65// ── MMIO holes that must remain unmapped before HCR_EL2.VM=1 (M3) ───────────
66
67/// IPA ranges left **unmapped** so guest device-probe traps reach the hypervisor.
68/// Frozen before `enable_stage2` is called; never remapped post-activation
69/// without a full S2 TLB invalidation.
70pub const MMIO_HOLES: &[(u64, u64)] = &[
71    (0x08000000, 0x08020000), // GIC distributor + CPU interface (GICD/GICC)
72    (0x09000000, 0x09001000), // PL011 UART
73    (0x0a000000, 0x0a004000), // virtio-mmio bus: 4 slots × 0x1000 (P06/P07/P08)
74];
75
76// ── Helpers ──────────────────────────────────────────────────────────────────
77
78#[inline]
79fn l1_idx(ipa: u64) -> usize {
80    ((ipa >> L1_SHIFT) & L1_CONC_MASK) as usize
81}
82#[inline]
83fn l2_idx(ipa: u64) -> usize {
84    ((ipa >> L2_SHIFT) & IDX_MASK) as usize
85}
86#[inline]
87fn l3_idx(ipa: u64) -> usize {
88    ((ipa >> L3_SHIFT) & IDX_MASK) as usize
89}
90
91#[inline]
92fn desc_pa(desc: u64) -> usize {
93    (desc & PA_MASK) as usize
94}
95
96#[inline]
97fn table_desc(next_pa: u64) -> u64 {
98    (next_pa & PA_MASK) | DESC_TABLE | DESC_VALID
99}
100
101#[inline]
102fn page_desc(pa: u64, writable: bool) -> u64 {
103    // At L3, bits[1:0]=0b11 → page descriptor (DESC_TABLE | DESC_VALID).
104    let base = if writable { S2_BASE_RW } else { S2_BASE_RO };
105    (pa & PA_MASK) | base | DESC_TABLE | DESC_VALID
106}
107
108#[inline]
109fn page_desc_device(pa: u64, writable: bool) -> u64 {
110    // Device-nGnRnE: MemAttr[5:2]=0b0000, SH[9:8]=non-shareable=0b00, AF=1.
111    let ap = if writable { S2_S2AP_RW } else { S2_S2AP_RO };
112    (pa & PA_MASK) | ap | S2_AF | DESC_TABLE | DESC_VALID
113}

115// ── Error type ───────────────────────────────────────────────────────────────
116
117#[derive(Debug, PartialEq)]
118pub enum S2MapError {
119    /// IPA or HPA range wraps around (checked_add overflow, C3).
120    Overflow,
121    /// IPA or HPA exceeds the 40-bit PA space limit.
122    OutOfBounds,
123    /// HPA is outside the guest's carved RAM region (SAS isolation).
124    SasViolation,
125    /// A block descriptor already occupies a slot we need to subdivide.
126    BlockConflict,
127    /// Frame allocator exhausted.
128    OutOfMemory,
129    /// Attempt to map a reserved MMIO hole (M3).
130    MmioHole,
131}

133// ── Stage2Table ──────────────────────────────────────────────────────────────
134
135/// ARM64 Stage-2 page table for one VM.
136///
137/// **Root alignment:** the two concatenated Level-1 frames occupy 8 KB starting
138/// at `root_pa`, which must be 8 KB-aligned for VTTBR_EL2.BADDR.
139///
140/// **Safety:** the raw pointer fields are valid for as long as `self` lives.
141/// The pointed-to frames are freed in `Drop`.  Never share across threads without
142/// external synchronisation.
143pub struct Stage2Table {
144    // Root: 2 × 512-entry L1 frames at root_pa (8 KB, 8 KB-aligned).
145    root_pa: u64,
146    root_va: *mut u64, // maps to 1024 entries (2 × 512)
147
148    // L2/L3 sub-table frames allocated on demand.  Tracked for Drop.
149    sub_frames: Vec<usize>, // physical addresses
150
151    // Carved guest-RAM region used for SAS-isolation assertion in map().
152    guest_ram_pa: u64,
153    guest_ram_pages: usize,
154}
155
156// SAFETY: Stage2Table is not Send/Sync by default because of *mut u64.
157// It is only accessed from single-CPU kernel context in QEMU TCG Phase 02/03.
158unsafe impl Send for Stage2Table {}
159
160impl Stage2Table {
161    /// Allocate a new Stage-2 root (2 × 4 KB frames, 8 KB-aligned).
162    ///
163    /// Returns `None` when the frame allocator has fewer than 2 contiguous
164    /// free frames.
165    pub fn new() -> Option<Self> {
166        // allocate_contiguous(2) guarantees contiguous and 4 KB alignment;
167        // for 8 KB alignment we rely on the allocator returning an even-indexed
168        // frame pair.  The FrameAllocator's memory_start is page-aligned, so
169        // every pair at an even index is 8 KB-aligned.
170        let root_pa = {
171            let mut g = FRAME_ALLOCATOR.lock();
172            g.as_mut()?.allocate_contiguous(2)? as u64
173        };
174        debug_assert_eq!(
175            root_pa % (2 * PAGE_SIZE as u64),
176            0,
177            "S2 root not 8KB-aligned"
178        );
179
180        let root_va = phys_to_virt(root_pa as usize) as *mut u64;
181        // Zero both L1 frames (1024 entries).
182        // SAFETY: we just allocated these frames; they are ours exclusively.
183        unsafe {
184            core::ptr::write_bytes(root_va, 0, 1024);
185        }
186
187        Some(Self {
188            root_pa,
189            root_va,
190            sub_frames: Vec::new(),
191            guest_ram_pa: 0,
192            guest_ram_pages: 0,
193        })
194    }
195
196    /// Physical address of the root frame (for VTTBR_EL2.BADDR).
197    #[inline]
198    pub fn root_pa(&self) -> u64 {
199        self.root_pa
200    }
201
202    /// Carve `n_pages` contiguous physical frames for guest RAM.
203    ///
204    /// Records the carved region so `map()` can enforce the SAS-isolation
205    /// invariant (guest Stage-2 may only map into this region).
206    ///
207    /// Uses `allocate_guest_ram` (chunked scan, M2-mitigated).
208    pub fn carve_guest_ram(&mut self, n_pages: usize) -> Option<u64> {
209        let pa = allocate_guest_ram(n_pages)? as u64;
210        self.guest_ram_pa = pa;
211        self.guest_ram_pages = n_pages;
212        Some(pa)
213    }
214
215    // ── Internal table allocator ─────────────────────────────────────────────
216
217    fn alloc_subtable(&mut self) -> Option<(*mut u64, u64)> {
218        let pa = {
219            let mut g = FRAME_ALLOCATOR.lock();
220            g.as_mut()?.allocate_frame()?
221        };
222        let va = phys_to_virt(pa) as *mut u64;
223        // SAFETY: freshly allocated frame is exclusively ours.
224        unsafe {
225            core::ptr::write_bytes(va, 0, 512);
226        }
227        self.sub_frames.push(pa);
228        Some((va, pa as u64))
229    }
230
231    // ── Public map / unmap ───────────────────────────────────────────────────
232
233    /// Map `n_pages` × 4 KB at guest IPA `ipa` → host PA `hpa`.
234    ///
235    /// # C3 — overflow guard
236    /// Both `ipa + n_pages × PAGE_SIZE` and `hpa + n_pages × PAGE_SIZE` are
237    /// checked with `checked_add`; a wrapping range returns `Overflow`.
238    ///
239    /// # M3 — MMIO-hole guard
240    /// Any IPA that overlaps a reserved MMIO hole returns `MmioHole`.
241    ///
242    /// # SAS-isolation guard
243    /// If a guest-RAM region was carved via `carve_guest_ram`, `hpa` must
244    /// lie within it; otherwise `SasViolation` is returned.
245    pub fn map(
246        &mut self,
247        ipa: u64,
248        hpa: u64,
249        n_pages: usize,
250        writable: bool,
251    ) -> Result<(), S2MapError> {
252        let page_bytes = n_pages as u64 * PAGE_SIZE as u64;
253
254        // C3: overflow guards.
255        let ipa_end = ipa.checked_add(page_bytes).ok_or(S2MapError::Overflow)?;
256        let hpa_end = hpa.checked_add(page_bytes).ok_or(S2MapError::Overflow)?;
257
258        if ipa_end > IPA_LIMIT {
259            return Err(S2MapError::OutOfBounds);
260        }
261        if hpa_end > IPA_LIMIT {
262            return Err(S2MapError::OutOfBounds);
263        }
264
265        // M3: reject any mapping that touches a reserved MMIO hole.
266        for &(hole_base, hole_end) in MMIO_HOLES {
267            if ipa < hole_end && ipa_end > hole_base {
268                return Err(S2MapError::MmioHole);
269            }
270        }
271
272        // SAS isolation: if guest RAM is known, HPA must stay within it.
273        if self.guest_ram_pages > 0 {
274            let guest_end = self.guest_ram_pa + (self.guest_ram_pages as u64 * PAGE_SIZE as u64);
275            if hpa < self.guest_ram_pa || hpa_end > guest_end {
276                return Err(S2MapError::SasViolation);
277            }
278        }
279
280        let mut cur_ipa = ipa;
281        let mut cur_hpa = hpa;
282        for _ in 0..n_pages {
283            self.map_single(cur_ipa, cur_hpa, writable)?;
284            cur_ipa += PAGE_SIZE as u64;
285            cur_hpa += PAGE_SIZE as u64;
286        }
287        Ok(())
288    }
289
290    /// Unmap `n_pages` pages starting at guest IPA `ipa`.
291    ///
292    /// Silently skips pages that are not mapped.  Does NOT free L2/L3 sub-tables
293    /// even if they become fully empty — deferred to `Drop`.
294    pub fn unmap(&mut self, ipa: u64, n_pages: usize) {
295        let mut cur = ipa;
296        for _ in 0..n_pages {
297            self.unmap_single(cur);
298            cur += PAGE_SIZE as u64;
299        }
300    }
301
302    /// Map MMIO HPA directly into guest IPA space for hardware passthrough.
303    ///
304    /// Bypasses both the MMIO-hole guard and the SAS-isolation guard.  Use only for
305    /// legitimate hardware passthrough (e.g., GICV HPA → GICC IPA for vGIC).
306    /// Uses Device-nGnRnE memory attributes.
307    ///
308    /// # Errors
309    /// Returns `Overflow` on wraparound; `OutOfBounds` if range exceeds 40-bit IPA limit.
310    pub fn map_mmio_passthrough(
311        &mut self,
312        ipa: u64,
313        hpa: u64,
314        n_pages: usize,
315        writable: bool,
316    ) -> Result<(), S2MapError> {
317        let page_bytes = n_pages as u64 * PAGE_SIZE as u64;
318        let ipa_end = ipa.checked_add(page_bytes).ok_or(S2MapError::Overflow)?;
319        let hpa_end = hpa.checked_add(page_bytes).ok_or(S2MapError::Overflow)?;
320        if ipa_end > IPA_LIMIT {
321            return Err(S2MapError::OutOfBounds);
322        }
323        if hpa_end > IPA_LIMIT {
324            return Err(S2MapError::OutOfBounds);
325        }
326        let mut cur_ipa = ipa;
327        let mut cur_hpa = hpa;
328        for _ in 0..n_pages {
329            self.map_single_device(cur_ipa, cur_hpa, writable)?;
330            cur_ipa += PAGE_SIZE as u64;
331            cur_hpa += PAGE_SIZE as u64;
332        }
333        Ok(())
334    }
335
336    // ── Single-page walk helpers ─────────────────────────────────────────────
337
338    fn map_single(&mut self, ipa: u64, hpa: u64, writable: bool) -> Result<(), S2MapError> {
339        // ── Level 1 ──────────────────────────────────────────────────────────
340        let l1_ptr: *mut u64 =
341            // SAFETY: root_va valid for 1024 entries; l1_idx < 1024.
342            unsafe { self.root_va.add(l1_idx(ipa)) };
343        let l1e = unsafe { *l1_ptr };
344
345        let l2_va: *mut u64 = if l1e & (DESC_VALID | DESC_TABLE) == (DESC_VALID | DESC_TABLE) {
346            // Existing L2 table pointer.
347            phys_to_virt(desc_pa(l1e)) as *mut u64
348        } else if l1e & DESC_VALID == 0 {
349            // Allocate a fresh L2 table.
350            let (va, pa) = self.alloc_subtable().ok_or(S2MapError::OutOfMemory)?;
351            // SAFETY: l1_ptr valid; DESC_TABLE|DESC_VALID cannot overlap PA bits.
352            unsafe {
353                *l1_ptr = table_desc(pa);
354            }
355            va
356        } else {
357            return Err(S2MapError::BlockConflict); // L1 block — refuse to split
358        };
359
360        // ── Level 2 ──────────────────────────────────────────────────────────
361        let l2_ptr: *mut u64 =
362            // SAFETY: l2_va valid for 512 entries; l2_idx < 512.
363            unsafe { l2_va.add(l2_idx(ipa)) };
364        let l2e = unsafe { *l2_ptr };
365
366        let l3_va: *mut u64 = if l2e & (DESC_VALID | DESC_TABLE) == (DESC_VALID | DESC_TABLE) {
367            phys_to_virt(desc_pa(l2e)) as *mut u64
368        } else if l2e & DESC_VALID == 0 {
369            let (va, pa) = self.alloc_subtable().ok_or(S2MapError::OutOfMemory)?;
370            // SAFETY: l2_ptr valid.
371            unsafe {
372                *l2_ptr = table_desc(pa);
373            }
374            va
375        } else {
376            return Err(S2MapError::BlockConflict); // L2 block
377        };
378
379        // ── Level 3 (page descriptor) ─────────────────────────────────────────
380        let l3_ptr: *mut u64 =
381            // SAFETY: l3_va valid for 512 entries; l3_idx < 512.
382            unsafe { l3_va.add(l3_idx(ipa)) };
383        // SAFETY: writing a well-formed page descriptor.
384        unsafe {
385            *l3_ptr = page_desc(hpa, writable);
386        }
387
388        Ok(())
389    }
390
391    /// Same L1→L2→L3 walk as `map_single`, but writes a Device-nGnRnE L3 descriptor.
392    fn map_single_device(&mut self, ipa: u64, hpa: u64, writable: bool) -> Result<(), S2MapError> {
393        let l1_ptr: *mut u64 = unsafe { self.root_va.add(l1_idx(ipa)) };
394        let l1e = unsafe { *l1_ptr };
395        let l2_va: *mut u64 = if l1e & (DESC_VALID | DESC_TABLE) == (DESC_VALID | DESC_TABLE) {
396            phys_to_virt(desc_pa(l1e)) as *mut u64
397        } else if l1e & DESC_VALID == 0 {
398            let (va, pa) = self.alloc_subtable().ok_or(S2MapError::OutOfMemory)?;
399            unsafe {
400                *l1_ptr = table_desc(pa);
401            }
402            va
403        } else {
404            return Err(S2MapError::BlockConflict);
405        };
406        let l2_ptr: *mut u64 = unsafe { l2_va.add(l2_idx(ipa)) };
407        let l2e = unsafe { *l2_ptr };
408        let l3_va: *mut u64 = if l2e & (DESC_VALID | DESC_TABLE) == (DESC_VALID | DESC_TABLE) {
409            phys_to_virt(desc_pa(l2e)) as *mut u64
410        } else if l2e & DESC_VALID == 0 {
411            let (va, pa) = self.alloc_subtable().ok_or(S2MapError::OutOfMemory)?;
412            unsafe {
413                *l2_ptr = table_desc(pa);
414            }
415            va
416        } else {
417            return Err(S2MapError::BlockConflict);
418        };
419        let l3_ptr: *mut u64 = unsafe { l3_va.add(l3_idx(ipa)) };
420        // SAFETY: writing a Device-nGnRnE page descriptor for MMIO passthrough.
421        unsafe {
422            *l3_ptr = page_desc_device(hpa, writable);
423        }
424        Ok(())
425    }
426
427    fn unmap_single(&mut self, ipa: u64) {
428        let l1_ptr: *mut u64 = unsafe { self.root_va.add(l1_idx(ipa)) };
429        let l1e = unsafe { *l1_ptr };
430        if l1e & (DESC_VALID | DESC_TABLE) != (DESC_VALID | DESC_TABLE) {
431            return;
432        }
433
434        let l2_va: *mut u64 = phys_to_virt(desc_pa(l1e)) as *mut u64;
435        let l2_ptr: *mut u64 = unsafe { l2_va.add(l2_idx(ipa)) };
436        let l2e = unsafe { *l2_ptr };
437        if l2e & (DESC_VALID | DESC_TABLE) != (DESC_VALID | DESC_TABLE) {
438            return;
439        }
440
441        let l3_va: *mut u64 = phys_to_virt(desc_pa(l2e)) as *mut u64;
442        let l3_ptr: *mut u64 = unsafe { l3_va.add(l3_idx(ipa)) };
443        // Clear the page descriptor.
444        unsafe {
445            *l3_ptr = 0;
446        }
447    }
448}

450// ── Drop — free all allocated frames ────────────────────────────────────────
451
452impl Drop for Stage2Table {
453    fn drop(&mut self) {
454        let mut g = FRAME_ALLOCATOR.lock();
455        if let Some(alloc) = g.as_mut() {
456            // Free L2/L3 sub-table frames (individually allocated).
457            for &pa in &self.sub_frames {
458                alloc.deallocate_frame(pa);
459            }
460            // Free the 2 × root L1 frames (allocated contiguously).
461            let root = self.root_pa as usize;
462            alloc.deallocate_frame(root);
463            alloc.deallocate_frame(root + PAGE_SIZE);
464            // Free guest RAM pages if they were carved here.
465            if self.guest_ram_pages > 0 {
466                let base = self.guest_ram_pa as usize;
467                for i in 0..self.guest_ram_pages {
468                    alloc.deallocate_frame(base + i * PAGE_SIZE);
469                }
470            }
471        }
472    }
473}

475// ── Kernel probe (test-cfg) ──────────────────────────────────────────────────
476
477/// Smoke-probe that exercises the Stage-2 table builder without running a vCPU.
478///
479/// Call from kernel init under `#[cfg(feature = "test-hooks")]` or a CI boot flag.
480/// On AArch64 the caller can follow up with `at s12e1r` to verify IPA→HPA via
481/// PAR_EL1; on non-AArch64 targets the function tests the software walk only.
482///
483/// # Panics
484/// Panics on any unexpected error — this is a development smoke check.
485pub fn probe_stage2_table() {
486    // Allocate a minimal table.
487    let mut tbl = Stage2Table::new().expect("Stage2Table::new failed");
488
489    // Carve 2 pages of guest RAM (trivial, no RT concern).
490    let guest_pa = tbl.carve_guest_ram(2).expect("carve_guest_ram failed");
491
492    // Map guest IPA 0x40000000 → carved PA.
493    tbl.map(0x40000000, guest_pa, 2, true)
494        .expect("Stage2Table::map failed");
495
496    // Verify the L3 descriptor was written correctly.
497    let l1_ptr: *mut u64 = unsafe { tbl.root_va.add(l1_idx(0x40000000)) };
498    let l1e = unsafe { *l1_ptr };
499    assert!(
500        l1e & (DESC_VALID | DESC_TABLE) == (DESC_VALID | DESC_TABLE),
501        "L1 table entry invalid"
502    );
503
504    let l2_va: *mut u64 = phys_to_virt(desc_pa(l1e)) as *mut u64;
505    let l2e = unsafe { *l2_va.add(l2_idx(0x40000000)) };
506    assert!(
507        l2e & (DESC_VALID | DESC_TABLE) == (DESC_VALID | DESC_TABLE),
508        "L2 table entry invalid"
509    );
510
511    let l3_va: *mut u64 = phys_to_virt(desc_pa(l2e)) as *mut u64;
512    let l3e = unsafe { *l3_va.add(l3_idx(0x40000000)) };
513    assert!(l3e & DESC_VALID != 0, "L3 page descriptor not valid");
514    assert_eq!(desc_pa(l3e), guest_pa as usize, "L3 PA mismatch");
515
516    // Unmap and verify cleared.
517    tbl.unmap(0x40000000, 2);
518    let l3e_after = unsafe { *l3_va.add(l3_idx(0x40000000)) };
519    assert_eq!(l3e_after, 0, "L3 descriptor not cleared after unmap");
520
521    // Drop exercises the frame-free path (Law 8).
522    drop(tbl);
523    // If the frame allocator panics on double-free, the probe would have caught it.
524
525    log::info!("[stage2::probe] Stage-2 table smoke-check passed");
526}
