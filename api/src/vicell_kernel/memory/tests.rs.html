1//! Memory allocator tests — called at boot-time by the kernel test runner.
2//!
3//! All functions are `pub` so the test runner in `kernel/src/main.rs` can
4//! invoke them via `memory::tests::run_all()`.  They use `log::info!` for
5//! output and `assert!` / `assert_eq!` for correctness.
6
7#![allow(dead_code)]
8
9use alloc::boxed::Box;
10use alloc::vec::Vec;
11
12/// Run every memory test and log a summary.
13pub fn run_all() {
14    log::info!("=== Memory Tests ===");
15    test_box_allocation();
16    test_large_vec_allocation();
17    test_sequential_alloc_free();
18    test_stress_10k_alloc_free();
19    test_multiple_sizes();
20    test_vec_push_pop();
21    test_nested_box();
22    log::info!("=== Memory Tests PASSED ===");
23}

25// ─── Basic heap ───────────────────────────────────────────────────────────────
26
27fn test_box_allocation() {
28    let v = Box::new(42u32);
29    assert_eq!(*v, 42, "Box<u32> value mismatch");
30    log::info!("  [ok] Box<u32>");
31}

33fn test_large_vec_allocation() {
34    const N: usize = 1_000;
35    let mut v: Vec<usize> = Vec::with_capacity(N);
36    for i in 0..N {
37        v.push(i);
38    }
39    assert_eq!(v.len(), N);
40    assert_eq!(v[500], 500);
41    log::info!("  [ok] Vec<usize> len={}", N);
42}

44// ─── Stress: 10K alloc/free cycles ───────────────────────────────────────────
45
46/// Allocate 10K `Box<u64>` values sequentially and verify each is readable after
47/// the previous one is dropped.  This exercises the bump/slab allocator's free-
48/// list path and checks that no corruption occurs under repeated small allocations.
49fn test_stress_10k_alloc_free() {
50    const ITERS: usize = 10_000;
51    for i in 0..ITERS {
52        let b = Box::new(i as u64);
53        // Verify the value survived the allocation and any allocator bookkeeping.
54        assert_eq!(*b, i as u64, "stress iter {} corrupted", i);
55        // `b` is dropped here — returns memory to the allocator.
56    }
57    log::info!("  [ok] 10K Box<u64> stress (alloc+drop each iter)");
58}
59
60/// Allocate 100 elements, keep them alive simultaneously, then free in
61/// reverse order to stress the free-list path.
62fn test_sequential_alloc_free() {
63    const N: usize = 100;
64    let mut boxes: Vec<Box<usize>> = Vec::with_capacity(N);
65    for i in 0..N {
66        boxes.push(Box::new(i));
67    }
68    // Verify all values intact while they are all live.
69    for (i, b) in boxes.iter().enumerate() {
70        assert_eq!(**b, i, "live-batch value[{}] mismatch", i);
71    }
72    // Drop in reverse order.
73    while let Some(b) = boxes.pop() {
74        let _ = b; // dropped here
75    }
76    log::info!("  [ok] sequential alloc then reverse-free ({} items)", N);
77}

81/// Allocate objects of several different sizes to exercise size-class paths in
82/// the allocator (if present).
83fn test_multiple_sizes() {
84    // Small (< 16 bytes)
85    let a = Box::new(1u8);
86    assert_eq!(*a, 1);
87
88    // Medium (64 bytes)
89    let b = Box::new([0u8; 64]);
90    assert_eq!(b[0], 0);
91    assert_eq!(b[63], 0);
92
93    // Larger (4 KB)
94    let c = Box::new([0u8; 4096]);
95    assert_eq!(c[0], 0);
96    assert_eq!(c[4095], 0);
97
98    // Very large (64 KB)
99    let d = Box::new([0u8; 65536]);
100    assert_eq!(d[0], 0);
101    assert_eq!(d[65535], 0);
102
103    log::info!("  [ok] multi-size allocations (1B, 64B, 4KB, 64KB)");
104}

106// ─── Vec push/pop stress ─────────────────────────────────────────────────────
107
108/// Push 5 000 u32s and pop them, verifying LIFO order.  This stresses
109/// reallocation-on-growth (Vec::push may call alloc when capacity is exceeded).
110fn test_vec_push_pop() {
111    const N: usize = 5_000;
112    let mut v: Vec<u32> = Vec::new();
113    for i in 0..N {
114        v.push(i as u32);
115    }
116    for i in (0..N).rev() {
117        assert_eq!(v.pop(), Some(i as u32), "pop order mismatch at {}", i);
118    }
119    assert!(v.is_empty());
120    log::info!("  [ok] Vec<u32> push+pop {} items (LIFO order)", N);
121}

123// ─── Nested Box ──────────────────────────────────────────────────────────────

125/// Allocate a Box pointing to another Box.  Tests that the allocator handles
126/// pointer-sized objects and that the heap can survive nested allocations.
127fn test_nested_box() {
128    let inner: Box<u64> = Box::new(0xDEAD_BEEF);
129    let outer: Box<Box<u64>> = Box::new(inner);
130    assert_eq!(**outer, 0xDEAD_BEEF);
131    log::info!("  [ok] nested Box<Box<u64>>");
132}
