1//! Heap allocator for ViCell kernel.
2//!
3//! Wraps `linked_list_allocator::LockedHeap` in a `QuotaAlloc` that charges
4//! every allocation to the currently-executing Cell's quota.  The kernel
5//! itself (CellId = 0) is unlimited.  A Cell that exceeds its quota receives
6//! a null pointer from `alloc()` — no panic, no system halt.
7
8use core::alloc::{GlobalAlloc, Layout};
9use linked_list_allocator::LockedHeap;
10
11struct QuotaAlloc {
12    inner: LockedHeap,
13}
14
15// SAFETY: QuotaAlloc delegates to LockedHeap which handles its own
16// interior mutability safely.  cell_quota operations are also thread-safe.
17unsafe impl GlobalAlloc for QuotaAlloc {
18    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
19        let cell = crate::task::scheduler::current_cell_id();
20        if !crate::memory::cell_quota::charge(cell, layout.size()) {
21            // Cell quota exceeded — return null, no panic.
22            return core::ptr::null_mut();
23        }
24        let ptr = self.inner.alloc(layout);
25        if ptr.is_null() {
26            // Inner heap OOM — refund the charge we already applied.
27            crate::memory::cell_quota::refund(cell, layout.size());
28        }
29        ptr
30    }
31
32    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
33        crate::memory::cell_quota::refund(crate::task::scheduler::current_cell_id(), layout.size());
34        self.inner.dealloc(ptr, layout);
35    }
36}
37
38#[global_allocator]
39static ALLOCATOR: QuotaAlloc = QuotaAlloc {
40    inner: LockedHeap::empty(),
41};
42
43/// Initialise the kernel heap.
44///
45/// # Safety
46/// Must be called exactly once after physical memory is mapped.
47pub unsafe fn init_heap(heap_start: usize, heap_size: usize) {
48    ALLOCATOR
49        .inner
50        .lock()
51        .init(heap_start as *mut u8, heap_size);
52}
53
54/// Allocator error handler
55#[alloc_error_handler]
56fn alloc_error_handler(layout: core::alloc::Layout) -> ! {
57    log::error!("allocation error: {:?}", layout);
58    // Panic recovery is not possible for OOM, but we loop to avoid double-panics
59    // if the panic handler tries to allocate.
60    loop {
61        #[cfg(target_arch = "x86_64")]
62        unsafe {
63            core::arch::asm!("hlt", options(nomem, nostack))
64        };
65        #[cfg(not(target_arch = "x86_64"))]
66        unsafe {
67            core::arch::asm!("wfi")
68        };
69    }
70}
