1//! Per-cell heap quota enforcement.
2//!
3//! Uses a split design to avoid the alloc-inside-alloc deadlock:
4//! - `QUOTA_LIMITS`: `Spinlock<BTreeMap<usize, usize>>` stores the limit per Cell.
5//!   Only locked in `register`/`deregister` — never inside `GlobalAlloc::alloc`.
6//! - `IN_USE`: `[AtomicUsize; MAX_CELLS]` stores the live byte count per Cell.
7//!   Updated atomically without any lock — safe to call from inside `GlobalAlloc::alloc`.
8
9use crate::sync::Spinlock;
10use alloc::collections::BTreeMap;
11use core::sync::atomic::{AtomicUsize, Ordering};
12use types::CellId;
13
14/// Maximum CellId tracked (index into IN_USE array).
15pub const MAX_CELLS: usize = 64;
16
17/// Default heap quota per Cell: 16 MiB.
18///
19/// Raised from 4 MiB to support heavier cells (DOOM needs ~10 MiB for WAD +
20/// zone heap; compositor and VFS also benefit from the headroom).  The quota
21/// is a runtime charge limit — no physical pages are pre-allocated.
22pub const DEFAULT_QUOTA_BYTES: usize = 16 * 1024 * 1024;
23
24/// Limit store — BTreeMap keyed by CellId raw value, stores the byte limit.
25/// Locked only in `register`/`deregister` — NOT inside the allocator hot path.
26static QUOTA_LIMITS: Spinlock<BTreeMap<usize, usize>> = Spinlock::new(BTreeMap::new());
27
28/// Live byte counters — one AtomicUsize per Cell slot, zero-initialized.
29/// Updated lock-free inside `charge`/`refund` to avoid alloc-inside-alloc deadlock.
30///
31/// Uses an inline `const { }` repeat-array seed (not a named `const`) — each
32/// slot is independently zero-initialized at compile time, so there is no
33/// shared-instance footgun (clippy::declare_interior_mutable_const) to worry
34/// about; a named `static` cannot be used here because `AtomicUsize` is not
35/// `Copy`, which the `[expr; N]` repeat form otherwise requires.
36static IN_USE: [AtomicUsize; MAX_CELLS] = [const { AtomicUsize::new(0) }; MAX_CELLS];
37
38/// Force-release this module's locks during fault teardown.
39///
40/// # Safety
41/// Single-hart; called only from the fault/panic path with interrupts disabled.
42pub unsafe fn force_unlock_locks() {
43    QUOTA_LIMITS.force_unlock();
44}
45
46/// Register a new Cell with the given heap quota.
47///
48/// Call this at spawn, OUTSIDE the allocator.
49///
50/// Deadlock contract: `BTreeMap::insert` allocates a new tree node via the
51/// global `QuotaAlloc`, whose `alloc` calls `charge(current_cell_id())`.  When
52/// `register` runs inside a cell's syscall (e.g. init calling `SpawnFromPath`),
53/// `current_cell_id()` is that cell (non-zero), so `charge` would try to RE-LOCK
54/// the `QUOTA_LIMITS` Spinlock we already hold here → self-deadlock.  We pin
55/// `CURRENT_CELL_ID` to 0 (kernel = unlimited, charge short-circuits without
56/// locking) across the insert so the node allocation never re-enters `charge`'s
57/// lock.  The node is kernel bookkeeping, so charging it to the kernel is also
58/// semantically correct.
59pub fn register(cell_id: CellId, limit: usize) {
60    let id = cell_id.0 as usize;
61    if id < MAX_CELLS {
62        IN_USE[id].store(0, Ordering::Relaxed);
63    }
64    // Pin to kernel context (0) so the BTreeMap::insert's node allocation does
65    // not re-enter `charge` (which locks QUOTA_LIMITS) while we hold that lock.
66    // Deadlock contract: see module doc.
67    let prev_cell = crate::task::hart_local::current_cell_id();
68    crate::task::hart_local::set_current_cell_id(0);
69    QUOTA_LIMITS.lock().insert(id, limit);
70    crate::task::hart_local::set_current_cell_id(prev_cell);
71}
72
73/// Deregister a Cell on exit.
74pub fn deregister(cell_id: CellId) {
75    let id = cell_id.0 as usize;
76    if id < MAX_CELLS {
77        IN_USE[id].store(0, Ordering::Relaxed);
78        DMA_IN_USE[id].store(0, Ordering::Relaxed);
79    }
80    QUOTA_LIMITS.lock().remove(&id);
81}

83/// Charge `size` bytes to the Cell.
84///
85/// Returns `false` if the quota would be exceeded — the caller (`QuotaAlloc::alloc`)
86/// must return `null_mut()` in that case.
87///
88/// Lock-ordering: acquires `QUOTA_LIMITS` briefly for a read (no allocation inside),
89/// then updates `IN_USE` atomically without any lock.
90pub fn charge(cell_id_raw: usize, size: usize) -> bool {
91    if cell_id_raw == 0 {
92        return true; // kernel itself: unlimited
93    }
94    // Read the limit — BTreeMap::get does NOT allocate.  Lock released immediately.
95    let limit = QUOTA_LIMITS
96        .lock()
97        .get(&cell_id_raw)
98        .copied()
99        .unwrap_or(usize::MAX);
100    if cell_id_raw >= MAX_CELLS {
101        return true; // no slot in IN_USE — uncapped
102    }
103    // Optimistic add; roll back on breach.
104    // Use saturating_add to prevent wrapping past usize::MAX — a hostile layout.size()
105    // near MAX would otherwise make the comparison read false and bypass the limit.
106    let prev = IN_USE[cell_id_raw].fetch_add(size, Ordering::Relaxed);
107    if prev.saturating_add(size) > limit {
108        IN_USE[cell_id_raw].fetch_sub(size, Ordering::Relaxed);
109        false
110    } else {
111        true
112    }
113}

115/// Refund `size` bytes when the Cell frees memory.  Lock-free.
116///
117/// Uses saturating subtraction to prevent underflow if a deallocation is
118/// attributed to a cell that has already been deregistered (e.g., dealloc
119/// of a Box shared with another cell arrives after the originating cell exited).
120pub fn refund(cell_id_raw: usize, size: usize) {
121    if cell_id_raw == 0 || cell_id_raw >= MAX_CELLS {
122        return;
123    }
124    // fetch_update with saturating_sub prevents wrapping to usize::MAX on spurious frees.
125    let _ = IN_USE[cell_id_raw].fetch_update(Ordering::Relaxed, Ordering::Relaxed, |cur| {
126        Some(cur.saturating_sub(size))
127    });
128}
129
130/// Current byte usage for a Cell (for diagnostics).
131pub fn in_use(cell_id: CellId) -> usize {
132    let id = cell_id.0 as usize;
133    if id < MAX_CELLS {
134        IN_USE[id].load(Ordering::Relaxed)
135    } else {
136        0
137    }
138}

140// ── DMA quota tracking (for sys_grant_dma) ───────────────────────────────────
141
142/// Live DMA-mapped byte counts — one AtomicUsize per Cell slot, zero-initialized.
143/// DMA quota = 1× memory quota (validated 2026-06-22).
144///
145/// See `IN_USE` above for why this uses an inline `const { }` repeat seed
146/// instead of a named `const`.
147static DMA_IN_USE: [AtomicUsize; MAX_CELLS] = [const { AtomicUsize::new(0) }; MAX_CELLS];
148
149/// Returns `true` if Cell `cell_id_raw` can map `size` more bytes into the IOMMU
150/// without exceeding its DMA quota (1× memory quota limit).
151///
152/// Kernel domain (cell_id_raw=0) is always allowed.
153pub fn can_map_dma(cell_id_raw: usize, size: usize) -> bool {
154    if cell_id_raw == 0 || cell_id_raw >= MAX_CELLS {
155        return true;
156    }
157    let limit = QUOTA_LIMITS
158        .lock()
159        .get(&cell_id_raw)
160        .copied()
161        .unwrap_or(DEFAULT_QUOTA_BYTES);
162    let cur = DMA_IN_USE[cell_id_raw].load(Ordering::Relaxed);
163    cur.saturating_add(size) <= limit
164}
165
166/// Record `size` bytes of DMA mapped for Cell `cell_id_raw`. Lock-free.
167pub fn record_dma_mapped(cell_id_raw: usize, size: usize) {
168    if cell_id_raw == 0 || cell_id_raw >= MAX_CELLS {
169        return;
170    }
171    DMA_IN_USE[cell_id_raw].fetch_add(size, Ordering::Relaxed);
172}
173
174/// Record `size` bytes of DMA released for Cell `cell_id_raw`. Lock-free.
175pub fn record_dma_unmapped(cell_id_raw: usize, size: usize) {
176    if cell_id_raw == 0 || cell_id_raw >= MAX_CELLS {
177        return;
178    }
179    let _ = DMA_IN_USE[cell_id_raw].fetch_update(Ordering::Relaxed, Ordering::Relaxed, |cur| {
180        Some(cur.saturating_sub(size))
181    });
