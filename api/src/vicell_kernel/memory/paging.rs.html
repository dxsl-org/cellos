1//! Paging management for ViCell kernel.
2//!
3//! Architecture dispatch:
4//!   riscv64, aarch64   — SV39 / AArch64 page tables via HAL `PageTable`
5//!   x86_64             — 4-level PML4, dedicated path (own walker + HHDM-aware)
6//!   riscv32, x86, arm  — bare physical (no page tables); stubs return NotSupported

15/// Page size (4KB)

23}

25pub type PagingResult<T> = core::result::Result<T, PageTableError>;

42/// Flush all TLB entries. Call after unmapping PTEs to prevent stale entries.
43///
44/// RISC-V:   sfence.vma — full TLB shootdown from S-mode.
45/// AArch64:  tlbi vmalle1is + dsb sy + isb — broadcast flush, all ASIDs, EL1.
46/// x86_64:   write CR3 to itself (reloads PML4 and flushes all non-global TLB entries).
47/// Bare physical (riscv32/x86/arm): no-op.
48#[inline(always)]
49pub fn tlb_flush_all() {
50    #[cfg(target_arch = "riscv64")]
51    // SAFETY: sfence.vma is a privileged S-mode fence; always safe from S-mode.
52    unsafe {
53        core::arch::asm!("sfence.vma", options(nostack));
54    }
55
56    #[cfg(target_arch = "aarch64")]
57    // SAFETY: tlbi vmalle1is invalidates all EL1 TLB entries broadcast across CPUs.
58    // dsb sy ensures the invalidation completes before subsequent memory accesses.
59    // isb serializes the instruction stream so the next fetch sees the clean TLB.
60    unsafe {
61        core::arch::asm!("tlbi vmalle1is", "dsb sy", "isb", options(nostack, nomem));
62    }
63
64    #[cfg(target_arch = "x86_64")]
65    {
66        // SAFETY: writing CR3 with its own value flushes all non-global TLB entries.
67        // This is a standard x86_64 full-TLB flush idiom; safe from Ring 0.
68        unsafe {
69            let cr3 = hal::paging::read_cr3();
70            hal::paging::write_cr3(cr3);
71        }
72    }
73}

77/// Initialize the kernel page table for riscv64 and aarch64.
78///
79/// Not used on x86_64 (separate path below) or bare-physical arches.
80#[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
81pub fn init_kernel_paging(
82    allocator: &mut FrameAllocator,
83    mmap: &[crate::boot::MemoryMapEntry],
84) -> PagingResult<PhysAddr> {
85    // 1. Allocate root page table frame and zero it.
86    let root_frame = allocator
87        .allocate_frame()
88        .ok_or(PageTableError::OutOfMemory)?;
89
90    // SAFETY: root_frame is freshly allocated; we own it exclusively.
91    unsafe {
92        core::ptr::write_bytes(root_frame as *mut u8, 0, PAGE_SIZE);
93    }
94
95    let root_table = unsafe { &mut *(root_frame as *mut PageTable) };
96
97    // 2. Identity-map all usable memory and kernel sections.
98    for entry in mmap.iter() {
99        let flags = match entry.ty {
100            crate::boot::MemoryType::Usable => PageFlags::from_bits(
101                PageFlags::VALID
102                    | PageFlags::READ
103                    | PageFlags::WRITE
104                    | PageFlags::EXECUTE
105                    | PageFlags::ACCESSED
106                    | PageFlags::DIRTY,
107            ),
108            crate::boot::MemoryType::Kernel => PageFlags::from_bits(
109                PageFlags::VALID
110                    | PageFlags::READ
111                    | PageFlags::WRITE
112                    | PageFlags::EXECUTE
113                    | PageFlags::ACCESSED
114                    | PageFlags::DIRTY,
115            ),
116            crate::boot::MemoryType::Bootloader => PageFlags::from_bits(
117                PageFlags::VALID
118                    | PageFlags::READ
119                    | PageFlags::WRITE
120                    | PageFlags::ACCESSED
121                    | PageFlags::DIRTY,
122            ),
123            crate::boot::MemoryType::Framebuffer => PageFlags::from_bits(
124                PageFlags::VALID
125                    | PageFlags::READ
126                    | PageFlags::WRITE
127                    | PageFlags::ACCESSED
128                    | PageFlags::DIRTY,
129            ),
130            crate::boot::MemoryType::MMIO => PageFlags::from_bits(
131                PageFlags::VALID
132                    | PageFlags::READ
133                    | PageFlags::WRITE
134                    | PageFlags::ACCESSED
135                    | PageFlags::DIRTY,
136            ),
137            _ => continue,
138        };
139
140        let mut alloc_closure = || allocator.allocate_frame();
141        root_table
142            .identity_map(
143                entry.base,
144                entry.base + entry.length,
145                flags,
146                &mut alloc_closure,
147            )
148            .map_err(|_| PageTableError::OutOfMemory)?;
149    }
150
151    // 3. Explicitly identity-map arch MMIO regions not in the memory map.
152    //
153    // AArch64: Device-nGnRnE (MAIR index 0) — required for MMIO.  Normal cacheable
154    // (MAIR index 1) causes QEMU to reject EL0 device accesses with a Permission fault
155    // even when the AP bits correctly grant EL0 read/write access.
156    // RISC-V / other: no DEVICE bit; the arch paging code ignores it.
157    #[cfg(target_arch = "aarch64")]
158    let mmio_flags = PageFlags::from_bits(
159        PageFlags::VALID
160            | PageFlags::READ
161            | PageFlags::WRITE
162            | PageFlags::DEVICE
163            | PageFlags::ACCESSED
164            | PageFlags::DIRTY,
165    );
166    #[cfg(not(target_arch = "aarch64"))]
167    let mmio_flags = PageFlags::from_bits(
168        PageFlags::VALID
169            | PageFlags::READ
170            | PageFlags::WRITE
171            | PageFlags::ACCESSED
172            | PageFlags::DIRTY,
173    );
174    let mut alloc_fn = || allocator.allocate_frame();
175
176    #[cfg(target_arch = "riscv64")]
177    {
178        let (clint_base, plic_base, plic_size, uart_region, rtc_region) =
179            crate::platform::with(|p| {
180                (
181                    p.clint_base,
182                    p.plic_base,
183                    p.plic_size,
184                    p.uart_base & !0xFFFF,
185                    p.rtc_base & !0xFFF,
186                )
187            });
188        root_table
189            .identity_map(clint_base, clint_base + 0x10000, mmio_flags, &mut alloc_fn)
190            .map_err(|_| PageTableError::OutOfMemory)?;
191        root_table
192            .identity_map(plic_base, plic_base + plic_size, mmio_flags, &mut alloc_fn)
193            .map_err(|_| PageTableError::OutOfMemory)?;
194        // UART page (0x10000000..0x10001000): kernel-only — S-mode with SUM=0 cannot
195        // access USER pages, so the kernel UART driver (uart::init, poll_rhr, IRQ handler)
196        // needs a non-USER mapping. Cells use sys_log / SBI; they do not touch UART MMIO.
197        if uart_region != 0 {
198            root_table
199                .identity_map(uart_region, uart_region + 0x1000, mmio_flags, &mut alloc_fn)
200                .map_err(|_| PageTableError::OutOfMemory)?;
201        }
202        // VirtIO MMIO (0x10001000–0x10010000): USER-accessible for Driver Cells (U-mode).
203        // The kernel's tech-debt VirtIO drivers (virtio_blk) also access these slots from
204        // S-mode. With sstatus.SUM=1 (set in main.rs right after activate_paging), S-mode
205        // can access USER-mapped pages throughout the kernel lifetime. SUM=1 is safe and
206        // intentional in Cellos's SAS+LBI model — Rust type safety provides isolation.
207        let rv_cell_mmio = PageFlags::from_bits(
208            PageFlags::VALID
209                | PageFlags::READ
210                | PageFlags::WRITE
211                | PageFlags::USER
212                | PageFlags::ACCESSED
213                | PageFlags::DIRTY,
214        );
215        let virtio_start = if uart_region != 0 {
216            uart_region + 0x1000
217        } else {
218            0x10001000
219        };
220        root_table
221            .identity_map(
222                virtio_start,
223                virtio_start + 0xF000,
224                rv_cell_mmio,
225                &mut alloc_fn,
226            )
227            .map_err(|_| PageTableError::OutOfMemory)?;
228        // Goldfish RTC (QEMU virt: 0x101000). Without this, sys_get_wall_secs/
229        // sys_get_wall_time (op 2/3) dereference an unmapped MMIO address in
230        // rtc::now_epoch_ns and the kernel takes a Load Page Fault (scause=13)
231        // while servicing the syscall — first hit by the TLS handshake's cert
232        // validity clock. The RTC window is one 4 KiB page.
233        root_table
234            .identity_map(rtc_region, rtc_region + 0x1000, mmio_flags, &mut alloc_fn)
235            .map_err(|_| PageTableError::OutOfMemory)?;
236        // PCIe ECAM bus-0 window (1 MiB at 0x3000_0000) for RISC-V virt gpex.
237        // Required before pcie_ecam::init() accesses config space.
238        // Only bus 0 is mapped; extend if a PCIe device lands on bus > 0.
239        //
240        // USER-inclusive (rv_cell_mmio, not kernel-only mmio_flags): the Platform
241        // Cell (`/bin/platform`) runs in U-mode and scans ECAM directly (scan.rs →
242        // mmio.rs read_volatile). With kernel-only flags its very first config read
243        // (dev0 vendor-id at ECAM_BASE) took a Load Page Fault every boot, silently
244        // killing the cell-driven PCIe scan (masked by never-die + the kernel S-mode
245        // fallback scanner). The kernel's own pcie_ecam::init S-mode reads still work
246        // against a USER page because SUM=1 (set right after activate_paging).
247        root_table
248            .identity_map(
249                crate::task::drivers::pcie_ecam::ECAM_BASE_RISCV,
250                crate::task::drivers::pcie_ecam::ECAM_BASE_RISCV
251                    + crate::task::drivers::pcie_ecam::ECAM_BUS0_SIZE,
252                rv_cell_mmio,
253                &mut alloc_fn,
254            )
255            .map_err(|_| PageTableError::OutOfMemory)?;
256    }
257    #[cfg(all(target_arch = "aarch64", feature = "board-rpi3"))]
258    {
259        // BCM2837 peripheral space 0x3F000000–0x3FFFFFFF (16 MiB): kernel-only MMIO.
260        // Covers mini UART, GPIO, I2C, SPI, EMMC — kernel maps them all; individual
261        // cells receive sub-regions via sys_request_mmio / resource_registry.
262        let cell_mmio_flags = PageFlags::from_bits(
263            PageFlags::VALID
264                | PageFlags::READ
265                | PageFlags::WRITE
266                | PageFlags::USER
267                | PageFlags::DEVICE
268                | PageFlags::ACCESSED
269                | PageFlags::DIRTY,
270        );
271        root_table
272            .identity_map(0x3F00_0000, 0x4000_0000, cell_mmio_flags, &mut alloc_fn)
273            .map_err(|_| PageTableError::OutOfMemory)?;
274        // BCM2836 local interrupt controller 0x40000000–0x40001000 (kernel-only).
275        root_table
276            .identity_map(0x4000_0000, 0x4000_1000, mmio_flags, &mut alloc_fn)
277            .map_err(|_| PageTableError::OutOfMemory)?;
278    }
279
280    #[cfg(all(target_arch = "aarch64", not(feature = "board-rpi3")))]
281    {
282        // GIC (EL1-only): cells must not modify interrupt routing — keep mmio_flags.
283        root_table
284            .identity_map(0x0800_0000, 0x0900_0000, mmio_flags, &mut alloc_fn)
285            .map_err(|_| PageTableError::OutOfMemory)?;
286        // Peripheral MMIO (GPIO PL061, UART PL011, RTC): cells access directly via
287        // driver rlibs (bit-bang GPIO/I2C/SPI/PWM). LBI guarantees only cells with
288        // the GPIO capability can construct driver objects, so USER access is safe.
289        let cell_mmio_flags = PageFlags::from_bits(
290            PageFlags::VALID
291                | PageFlags::READ
292                | PageFlags::WRITE
293                | PageFlags::USER
294                | PageFlags::DEVICE
295                | PageFlags::ACCESSED
296                | PageFlags::DIRTY,
297        );
298        root_table
299            .identity_map(0x0900_0000, 0x0904_0000, cell_mmio_flags, &mut alloc_fn)
300            .map_err(|_| PageTableError::OutOfMemory)?;
301        root_table
302            .identity_map(0x0A00_0000, 0x0A00_4000, cell_mmio_flags, &mut alloc_fn)
303            .map_err(|_| PageTableError::OutOfMemory)?;
304        root_table
305            .identity_map(0x1000_0000, 0x1001_0000, mmio_flags, &mut alloc_fn)
306            .map_err(|_| PageTableError::OutOfMemory)?;
307        // PCIe ECAM bus-0 window (1 MiB at 0x3F00_0000) for ARM64 virt gpex.
308        // Required before pcie_ecam::init() accesses config space.
309        root_table
310            .identity_map(
311                crate::task::drivers::pcie_ecam::ECAM_BASE_AARCH64,
312                crate::task::drivers::pcie_ecam::ECAM_BASE_AARCH64
313                    + crate::task::drivers::pcie_ecam::ECAM_BUS0_SIZE,
314                mmio_flags,
315                &mut alloc_fn,
316            )
317            .map_err(|_| PageTableError::OutOfMemory)?;
318    }
319
320    *KERNEL_ROOT.lock() = Some(root_frame);
321    Ok(root_frame)
322}

324/// Activate virtual memory for riscv64 / aarch64.
325///
326/// # Safety
327/// The root table MUST map the currently executing code and kernel stack.
328#[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
329pub unsafe fn activate_paging(root_table_phys: PhysAddr) {
330    // SAFETY: upheld by caller contract.
331    let root_table = unsafe { &*(root_table_phys as *const PageTable) };
332    // SAFETY: PageTable::activate writes SATP/TTBR0 using the table's physical address.
333    unsafe {
334        root_table.activate();
335    }
336}

338// ─── riscv64 + aarch64 + x86_64: map_page / unmap_page ──────────────────────
339//
340// On riscv64/aarch64 these delegate to PageTable::map / ::unmap.
341// On x86_64 they use the HHDM-aware walk_create / walk_read walkers.
342
343/// Map a virtual page to a physical frame in the kernel address space.
344///
345/// Returns `NotSupported` on bare-physical arches (riscv32, x86_32, arm32).
346#[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
347pub fn map_page(
348    allocator: &mut FrameAllocator,
349    vaddr: VAddr,
350    paddr: PhysAddr,
351    flags: Flags,
352) -> PagingResult<()> {
353    let root_lock = KERNEL_ROOT.lock();
354    if let Some(root_phys) = *root_lock {
355        let root_table = unsafe { &mut *(root_phys as *mut PageTable) };
356        let mut alloc_closure = || allocator.allocate_frame();
357        root_table
358            .map(vaddr, paddr, flags, &mut alloc_closure)
359            .map_err(|_| PageTableError::OutOfMemory)?;
360        Ok(())
361    } else {
362        Err(PageTableError::NotSupported)
363    }
364}

366/// Unmap a virtual page from the kernel address space (clears the PTE).
367/// Used to create guard pages. Returns Ok(()) if the page was not mapped.
368///
369/// Returns `NotSupported` on bare-physical arches.
370#[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
371pub fn unmap_page(vaddr: VAddr) -> PagingResult<()> {
372    let root_lock = KERNEL_ROOT.lock();
373    if let Some(root_phys) = *root_lock {
374        let root_table = unsafe { &mut *(root_phys as *mut PageTable) };
375        match root_table.unmap(vaddr) {
376            Ok(()) => Ok(()),
377            Err(_) => Ok(()), // Not mapped — that is acceptable for guard-page creation
378        }
379    } else {
380        Err(PageTableError::NotSupported)
381    }
382}

384// ─── x86_64 paging ───────────────────────────────────────────────────────────
385//
386// Limine boot environment:
387//   - Physical RAM is mapped at HHDM_BASE + phys (NOT identity-mapped).
388//   - Kernel text/data live at 0xFFFFFFFF80000000 (higher half).
389//   - MMIO (LAPIC, HPET, IOAPIC) is NOT in the HHDM; must be identity-mapped.
390//
391// Strategy for init_kernel_paging:
392//   1. Allocate a new PML4 frame and zero it.
393//   2. Copy PML4 entries [256..512] from Limine's CR3 — preserves kernel+HHDM.
394//   3. Identity-map MMIO (VA = PA) so init_timers() can access 0xFED0_0000 etc.
395//   4. Store root in KERNEL_ROOT and return phys address.
396//
397// The walk_create / walk_read helpers (in hal/arch/x86/src/x86_64/paging.rs)
398// always use phys_to_virt_ptr() so intermediate tables are dereferenceable
399// before and after the CR3 switch.
400
401/// Build the kernel's own PML4 for x86_64.
402///
403/// Must be called after `set_phys_offset` (HHDM base) is stored in
404/// `memory::frame` and after the HAL paging walker has the HHDM offset
405/// set via `hal::paging::set_hhdm_offset`.
406///
407/// Parameters:
408/// - `ioapic_base`: I/O APIC physical address (from ACPI MADT or default 0xFEC0_0000)
409/// - `hpet_base`:   HPET physical address (from ACPI HPET table or default 0xFED0_0000)
410/// - `lapic_base`:  Local APIC physical address (from ACPI MADT or default 0xFEE0_0000)
411///
412/// All three MMIO regions are identity-mapped (VA == PA) so `init_timers()` can
413/// access them directly after the CR3 switch.
414///
415/// Preconditions:
416/// - The frame allocator is initialised.
417/// - `hal::paging::set_hhdm_offset(hhdm)` has been called.
418///
419/// # Panics
420/// Panics on OOM (any frame allocation fails).
421#[cfg(target_arch = "x86_64")]
422pub fn init_kernel_paging_x86(
423    allocator: &mut FrameAllocator,
424    ioapic_base: u64,
425    hpet_base: u64,
426    lapic_base: u64,
427    ecam_base: u64,
428) -> PagingResult<PhysAddr> {
429    use crate::memory::frame::phys_to_virt;
430    use hal::paging::{pte_flags_mmio, read_cr3, walk_create, walk_read, PTE_PRESENT};
431
432    // 1. Allocate and zero a new PML4 frame.
433    let pml4_phys = allocator
434        .allocate_frame()
435        .ok_or(PageTableError::OutOfMemory)?;
436    // SAFETY: pml4_phys is a freshly allocated 4KB frame; HHDM virt pointer is valid.
437    unsafe {
438        core::ptr::write_bytes(phys_to_virt(pml4_phys) as *mut u8, 0, PAGE_SIZE);
439    }
440
441    let pml4_virt = phys_to_virt(pml4_phys) as *mut u64;
442
443    // 2. Copy higher-half PML4 entries [256..512] from Limine's active PML4.
444    //    These entries cover 0xFFFF_8000_0000_0000..0xFFFF_FFFF_FFFF_FFFF and include:
445    //      - the HHDM window (index 256..384 depending on Limine's choice)
446    //      - the kernel text/data at 0xFFFFFFFF80000000 (index 511)
447    //    Copying them wholesale is safe: they remain read-only during init.
448    {
449        // SAFETY: read_cr3() returns the physical address of the live Limine PML4.
450        let limine_cr3 = unsafe { read_cr3() } as usize;
451        let limine_pml4 = phys_to_virt(limine_cr3) as *const u64;
452        // SAFETY: limine_pml4 points to a valid 512-entry PML4 (4096 bytes).
453        // We only read entries [256..512] from it.
454        unsafe {
455            for i in 256usize..512 {
456                let entry = core::ptr::read_volatile(limine_pml4.add(i));
457                core::ptr::write_volatile(pml4_virt.add(i), entry);
458            }
459        }
460    }
461
462    // 3. Identity-map MMIO regions so the CPU can reach them at VA == PA after
463    //    the CR3 switch.  Physical addresses are below the 4GB boundary and NOT
464    //    in the Limine HHDM window, so they must be explicitly identity-mapped.
465    //
466    //    The bases come from ACPI parse (ioapic_base, hpet_base, lapic_base params)
467    //    with QEMU q35 defaults as fallback so boot works without ACPI.
468    //    VT-d IOMMU (0xFED9_0000) is always q35-specific; keep hardcoded.
469    //
470    //    IOAPIC:  ioapic_base (default 0xFEC0_0000) — 4 KB
471    //    HPET:    hpet_base   (default 0xFED0_0000) — 4 KB
472    //    VT-d:    0xFED9_0000 (q35 Intel IOMMU)     — 4 KB
473    //    LAPIC:   lapic_base  (default 0xFEE0_0000) — 4 KB
474    let ioapic_base_usize = ioapic_base as usize;
475    let hpet_base_usize = hpet_base as usize;
476    let lapic_base_usize = lapic_base as usize;
477    let ecam_base_usize = ecam_base as usize;
478    // PCIe ECAM bus-0 window = 1 MiB (256 devices × 8 fns × 4 KiB config space).
479    const ECAM_BUS0_SIZE: usize = 0x10_0000;
480    let mmio_regions: &[(usize, usize)] = &[
481        (ioapic_base_usize, ioapic_base_usize + PAGE_SIZE), // IOAPIC (from ACPI)
482        (hpet_base_usize, hpet_base_usize + PAGE_SIZE),     // HPET   (from ACPI)
483        (0xFED9_0000, 0xFED9_0000 + PAGE_SIZE),             // VT-d IOMMU (q35)
484        (lapic_base_usize, lapic_base_usize + PAGE_SIZE),   // LAPIC  (from ACPI)
485        (ecam_base_usize, ecam_base_usize + ECAM_BUS0_SIZE), // PCIe ECAM bus 0
486    ];
487    for &(start, end) in mmio_regions {
488        let mut va = start;
489        while va < end {
490            let mut alloc_fn = || allocator.allocate_frame();
491            // SAFETY: pml4_virt points to our new PML4 (writable via HHDM).
492            // walk_create allocates intermediate tables via alloc_fn.
493            let pte_ptr = unsafe { walk_create(pml4_virt, va, &mut alloc_fn) }
494                .ok_or(PageTableError::OutOfMemory)?;
495            // SAFETY: pte_ptr is the leaf PTE slot returned by walk_create; safe to write.
496            unsafe {
497                core::ptr::write_volatile(pte_ptr, va as u64 | pte_flags_mmio());
498            }
499            va += PAGE_SIZE;
500        }
501    }
502
503    // Sanity: verify the IOAPIC PTE was written correctly.
504    // Uses the runtime ioapic_base (parsed from ACPI or default 0xFEC0_0000).
505    // SAFETY: pml4_virt is our valid new PML4.
506    debug_assert!(
507        unsafe { walk_read(pml4_virt as *const u64, ioapic_base_usize) }
508            .map(|e| e & PTE_PRESENT != 0)
509            .unwrap_or(false),
510        "IOAPIC identity-map sanity check failed"
511    );
512
513    log::info!("[kernel] x86_64 paging: kernel PML4 built");
514
515    *KERNEL_ROOT.lock() = Some(pml4_phys);
516    Ok(pml4_phys)
517}

519/// Activate paging on x86_64 by writing the PML4 physical address to CR3.
520///
521/// # Safety
522/// The PML4 at `root_phys` must map the currently executing code, the kernel
523/// stack, and any data the kernel needs immediately after this call.
524/// An invalid mapping causes a triple-fault.
525#[cfg(target_arch = "x86_64")]
526pub unsafe fn activate_paging(root_phys: PhysAddr) {
527    // SAFETY: caller guarantees root_phys is a valid, fully populated PML4 that
528    // keeps the kernel alive after the CR3 switch.
529    unsafe {
530        hal::paging::write_cr3(root_phys as u64);
531    }
532    log::info!("[kernel] paging activated");
533}

535/// Map a 4KB page (x86_64).
536///
537/// `flags` must be raw x86_64 PTE flags (use `pte_flags_*` helpers from
538/// `hal::paging`).  The function inserts the PTE into KERNEL_ROOT's PML4 and
539/// issues `invlpg` to flush the TLB entry.
540#[cfg(target_arch = "x86_64")]
541pub fn map_page_x86(
542    allocator: &mut FrameAllocator,
543    vaddr: VAddr,
544    paddr: PhysAddr,
545    flags: u64,
546) -> PagingResult<()> {
547    use crate::memory::frame::phys_to_virt;
548    use hal::paging::{invlpg, walk_create};
549
550    let root_lock = KERNEL_ROOT.lock();
551    let root_phys = (*root_lock).ok_or(PageTableError::NotSupported)?;
552    let pml4_virt = phys_to_virt(root_phys) as *mut u64;
553
554    let mut alloc_fn = || allocator.allocate_frame();
555    // SAFETY: pml4_virt is the kernel's active PML4 (stored at boot).
556    let pte_ptr = unsafe { walk_create(pml4_virt, vaddr, &mut alloc_fn) }
557        .ok_or(PageTableError::OutOfMemory)?;
558    // SAFETY: pte_ptr is the leaf PTE address; writing it installs the mapping.
559    unsafe {
560        core::ptr::write_volatile(pte_ptr, paddr as u64 | flags);
561    }
562    // SAFETY: invlpg flushes only the single TLB entry for vaddr.
563    unsafe {
564        invlpg(vaddr);
565    }
566    Ok(())
567}

569/// Identity-map a MMIO BAR region [phys, phys+size) at VA == PA (x86_64).
570///
571/// Called by PCIe drivers at init time to make dynamically-assigned BAR windows
572/// accessible as identity-mapped VA == PA. Each page in the range is mapped with
573/// MMIO cache-disabling flags; existing mappings are silently overwritten.
574///
575/// Panics on OOM or if the frame allocator is not yet initialised.
576#[cfg(target_arch = "x86_64")]
577pub fn map_mmio_x86(phys: usize, size: usize) {
578    map_mmio_x86_flags(phys, size, hal::paging::pte_flags_mmio());
579}
580
581/// Identity-map a MMIO region with USER access (x86_64).
582///
583/// Called by the `RequestMmio` syscall when a ring-3 Driver Cell claims a
584/// range: the SAS identity map built at boot is kernel-only, so the claimed
585/// window must be (re)mapped with `PTE_USER` before the cell can touch it.
586/// Mirrors the riscv/aarch64 posture where cell-owned MMIO windows are mapped
587/// user-accessible — exclusivity is enforced by the resource registry + LBI
588/// (`forbid(unsafe_code)` cells cannot fabricate an MMIO dereference), not by
589/// the U/S bit.
590#[cfg(target_arch = "x86_64")]
591pub fn map_mmio_user_x86(phys: usize, size: usize) {
592    map_mmio_x86_flags(phys, size, hal::paging::pte_flags_mmio_user());
593}
594
595#[cfg(target_arch = "x86_64")]
596fn map_mmio_x86_flags(phys: usize, size: usize, flags: u64) {
597    let mut fa = crate::memory::frame::FRAME_ALLOCATOR.lock();
598    let allocator = fa
599        .as_mut()
600        .expect("map_mmio_x86: frame allocator not ready");
601    let page_mask = PAGE_SIZE - 1;
602    let mut va = phys & !page_mask;
603    let end = (phys + size + page_mask) & !page_mask;
604    while va < end {
605        // Errors are logged but not fatal; a broken mapping will fault on first use.
606        if let Err(e) = map_page_x86(allocator, va, va, flags) {
607            log::warn!("[paging] map_mmio_x86: failed to map {:#x} ({:?})", va, e);
608        }
609        va += PAGE_SIZE;
610    }
611}

613/// Unmap a 4KB page (x86_64).
614///
615/// Clears the leaf PTE and flushes the TLB entry. Returns Ok(()) even if the
616/// page was already unmapped (idempotent — safe for guard-page creation).
617#[cfg(target_arch = "x86_64")]
618pub fn unmap_page_x86(vaddr: VAddr) -> PagingResult<()> {
619    use crate::memory::frame::phys_to_virt;
620    use hal::paging::{invlpg, walk_create, walk_read, PTE_PRESENT};
621
622    let root_lock = KERNEL_ROOT.lock();
623    let root_phys = match *root_lock {
624        Some(p) => p,
625        None => return Ok(()), // paging not yet active — no TLB entry to clear
626    };
627    let pml4_virt_const = phys_to_virt(root_phys) as *const u64;
628    // SAFETY: pml4_virt_const is the kernel's active PML4.
629    let existing = unsafe { walk_read(pml4_virt_const, vaddr) };
630    if let Some(pte) = existing {
631        if pte & PTE_PRESENT != 0 {
632            // Walk again for a mutable pointer to the leaf PTE.
633            let pml4_virt_mut = phys_to_virt(root_phys) as *mut u64;
634            // SAFETY: walk_create on an existing path never allocates new frames;
635            // alloc_fn is only called for absent intermediate tables (none here).
636            let mut noop_alloc = || None::<usize>;
637            if let Some(pte_ptr) = unsafe { walk_create(pml4_virt_mut, vaddr, &mut noop_alloc) } {
638                // SAFETY: pte_ptr is the leaf PTE slot; clearing it unmaps the page.
639                unsafe {
640                    core::ptr::write_volatile(pte_ptr, 0u64);
641                }
642                // SAFETY: invlpg flushes the TLB entry for this virtual address.
643                unsafe {
644                    invlpg(vaddr);
645                }
646            }
647        }
648    }
649    Ok(())
650}

652// ─── Generic map_page / unmap_page wrappers (called from kernel code) ────────
653//
654// These forward to the correct arch implementation. x86_64 versions take raw
655// u64 PTE flags; callers that need the generic Flags type should use the
656// riscv64/aarch64 versions above or convert flags themselves.
657//
658// For x86_64, map_page and unmap_page delegate to map_page_x86 /
659// unmap_page_x86 using kernel-RW flags by default.  Callers needing
660// user/exec/mmio flags must call map_page_x86 directly.
661
662/// Map a page: generic entry point.
663///
664/// On riscv64/aarch64: uses the HAL-generic Flags parameter.
665/// On x86_64: converts generic Flags to x86_64 PTE bits.
666/// On bare-physical arches: returns NotSupported.
667#[cfg(target_arch = "x86_64")]
668pub fn map_page(
669    allocator: &mut FrameAllocator,
670    vaddr: VAddr,
671    paddr: PhysAddr,
672    flags: Flags,
673) -> PagingResult<()> {
674    use hal::paging::{PTE_NX, PTE_PRESENT, PTE_USER, PTE_WRITABLE};
675    // Convert generic PageFlags to x86_64 PTE bits.
676    // Note: cache-disable (PCD/MMIO) has no generic PageFlags equivalent;
677    // callers needing MMIO mappings must use map_page_x86 with pte_flags_mmio().
678    let bits = flags.bits();
679    let mut pte_flags: u64 = PTE_PRESENT;
680    if bits & hal::PageFlags::WRITE != 0 {
681        pte_flags |= PTE_WRITABLE;
682    }
683    if bits & hal::PageFlags::USER != 0 {
684        pte_flags |= PTE_USER;
685    }
686    if bits & hal::PageFlags::EXECUTE == 0 {
687        pte_flags |= PTE_NX;
688    }
689    map_page_x86(allocator, vaddr, paddr, pte_flags)
690}
691
692/// Unmap a page: generic entry point.
693///
694/// On x86_64 delegates to unmap_page_x86 (idempotent).
695/// On bare-physical arches: returns NotSupported.
696#[cfg(target_arch = "x86_64")]
697pub fn unmap_page(vaddr: VAddr) -> PagingResult<()> {
698    unmap_page_x86(vaddr)
699}

701// ─── remap_range_user ────────────────────────────────────────────────────────
702
703/// Remap a range of already-allocated pages with user (U/S=1) permissions.
704///
705/// Used for user stacks which sit in identity-mapped usable RAM.
706/// On x86_64 we re-walk and rewrite the PTEs with USER|WRITABLE|NX.
707#[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
708pub fn remap_range_user(start: PhysAddr, pages: usize) {
709    let mut root_guard = KERNEL_ROOT.lock();
710    if let Some(root_addr) = root_guard.as_mut() {
711        let table = unsafe { &mut *(*root_addr as *mut hal::paging::PageTable) };
712        let mut frame_guard = crate::memory::frame::FRAME_ALLOCATOR.lock();
713        if let Some(allocator) = frame_guard.as_mut() {
714            let mut alloc_closure = || allocator.allocate_frame();
715            use hal::traits::PageTableTrait;
716            let flags = PageFlags::from_bits(
717                PageFlags::VALID
718                    | PageFlags::READ
719                    | PageFlags::WRITE
720                    | PageFlags::EXECUTE
721                    | PageFlags::USER
722                    | PageFlags::ACCESSED
723                    | PageFlags::DIRTY,
724            );
725            use hal::paging::PAGE_SIZE;
726            for i in 0..pages {
727                let addr = start + (i * PAGE_SIZE);
728                table
729                    .map(addr, addr, flags, &mut alloc_closure)
730                    .expect("Failed to map user stack page!");
731            }
732        }
733    }
734}

743            // Map VA == PA (identity) with user-rw flags.

745        }

769// ─── virt_to_phys ────────────────────────────────────────────────────────────
770
771/// Translate a virtual address to its physical address by walking the kernel
772/// page table.
773///
774/// Bare-physical arches (riscv32, x86_32, arm32): VA == PA.
775/// Paged arches read the page-table root from the architecture register.
776#[cfg(any(target_arch = "riscv64", target_arch = "aarch64"))]
777pub fn virt_to_phys(vaddr: VAddr) -> Option<PhysAddr> {
778    use hal::traits::PageTableTrait;
779
780    #[cfg(target_arch = "riscv64")]
781    {
782        // SAFETY: `satp` CSR holds the page-table root PPN.
783        let satp: usize;
784        unsafe { core::arch::asm!("csrr {}, satp", out(reg) satp) };
785        let root_ppn = satp & ((1 << 44) - 1);
786        let root_phys = root_ppn << 12;
787        if root_phys == 0 {
788            return None;
789        }
790        // SAFETY: root_phys is the physical address of the active root PageTable.
791        let root_table = unsafe { &*(root_phys as *const hal::PageTable) };
792        return root_table.translate(vaddr);
793    }
794    #[cfg(target_arch = "aarch64")]
795    {
796        // SAFETY: TTBR0_EL1 holds the root page table physical address.
797        let ttbr0: usize;
798        unsafe { core::arch::asm!("mrs {}, ttbr0_el1", out(reg) ttbr0, options(nomem, nostack)) };
799        let root_phys = ttbr0 & !0xFFF;
800        if root_phys == 0 {
801            return None;
802        }
803        let root_table = unsafe { &*(root_phys as *const hal::PageTable) };
804        return root_table.translate(vaddr);
805    }
806    #[allow(unreachable_code)]
807    None
808}
809
810#[cfg(target_arch = "x86_64")]
811pub fn virt_to_phys(vaddr: VAddr) -> Option<PhysAddr> {
812    use crate::memory::frame::phys_to_virt;
813    use hal::paging::{walk_read, PTE_ADDR_MASK, PTE_PRESENT};
814    let root_guard = KERNEL_ROOT.lock();
815    let root_phys = (*root_guard)?;
816    let pml4 = phys_to_virt(root_phys) as *const u64;
817    // SAFETY: pml4 is the kernel's active PML4.
818    let pte = unsafe { walk_read(pml4, vaddr) }?;
819    if pte & PTE_PRESENT == 0 {
820        return None;
821    }
822    // PTE_ADDR_MASK, NOT !0xFFF: user pages carry PTE_NX (bit 63) — keeping it
823    // made the "physical" address non-canonical after the HHDM offset, and the
824    // ELF loader's shared-page path (already_ours) then wrote through a #GP.
825    Some(((pte & PTE_ADDR_MASK) as usize) | (vaddr & 0xFFF))
826}

828/// Bare-physical identity translation.
829#[cfg(any(target_arch = "riscv32", target_arch = "x86", target_arch = "arm"))]
830pub fn virt_to_phys(vaddr: VAddr) -> Option<PhysAddr> {
831    Some(vaddr)
832}

834// ─── Page-fault handler (x86_64) ─────────────────────────────────────────────
835
836/// HAL-visible entry point: called by the IDT error-code handler via the
837/// `extern "Rust" fn vi_handle_page_fault` declaration in `hal/arch/x86/idt.rs`.
838///
839/// Decision tree:
840///   - User-mode fault (error_code bit 2 = U/S set): look up the current task's
841///     VMA list; if a region covers `va`, install the mapping and return so the
842///     faulting instruction is retried.
843///   - Otherwise (kernel fault or no VMA match): panic — this is a true
844///     kernel bug or an unmapped user access.
845///
846/// # Safety
847/// Must only be called from the #PF exception handler while the faulting
848/// context is suspended on the interrupt stack. The caller must NOT hold any
849/// spinlock that this function might try to acquire (KERNEL_ROOT, SCHEDULER,
850/// FRAME_ALLOCATOR).
851#[cfg(target_arch = "x86_64")]
852#[no_mangle]
853pub extern "Rust" fn vi_handle_page_fault(va: usize, error_code: u64, rip: u64, cs: u64, rsp: u64) {
854    use crate::task::SCHEDULER;
855
856    // Bit 2 of the error code = U/S: fault originated from user mode.
857    let user_fault = error_code & (1 << 2) != 0;
858
859    if !user_fault {
860        // NOTE: the IDT routes EVERY error-code vector (#DF/#TS/#NP/#SS/#GP/
861        // #PF/#AC) through this handler, so `va` (CR2) may be STALE — a #GP
862        // with error_code=0 (e.g. a non-canonical address access) shows up
863        // here as "va=0x0 error_code=0x0". rip/cs identify the real faulting
864        // instruction; symbolize rip against the kernel ELF.
865        //
866        // Poor-man's backtrace: scan the faulting stack for kernel-text
867        // return addresses so the memcpy-style builtins (which fault deep
868        // inside compiler_builtins) can be attributed to their caller.
869        let mut callers: [u64; 6] = [0; 6];
870        let mut n = 0;
871        // SAFETY: rsp is the faulting kernel stack (cs=8); reading a few
872        // qwords above it is safe — the stack is mapped or we would have
873        // double-faulted before reaching this handler.
874        if rsp != 0 && rsp.is_multiple_of(8) {
875            for i in 0..64usize {
876                if n >= callers.len() {
877                    break;
878                }
879                let q = unsafe { core::ptr::read_volatile((rsp as *const u64).add(i)) };
880                if (0xffff_ffff_8000_0000..0xffff_ffff_9000_0000).contains(&q) {
881                    callers[n] = q;
882                    n += 1;
883                }
884            }
885        }
886        panic!(
887            "[#PF kernel] va={:#x} error_code={:#x} rip={:#x} cs={:#x} rsp={:#x} ret-candidates={:#x?} — kernel-mode fault (may be #GP: CR2 can be stale)",
888            va, error_code, rip, cs, rsp, &callers[..n]
889        );
890    }
891
892    // Try to find the faulting VA in the current task's VMA list.
893    // We identify the current task via the scheduler's running-task pointer.
894    let result: Option<(PhysAddr, u64)> = {
895        let mut sched = SCHEDULER.lock();
896        if let Some(s) = sched.as_mut() {
897            // `current_task_mut()` returns the task currently marked Running
898            // (the one whose exception context we are inside).
899            s.current_task_mut().and_then(|task| {
900                task.vma.find(va).map(|region| {
901                    let offset = va - region.va_start;
902                    let pa = region.pa_start + offset;
903                    let flags = region.flags;
904                    (pa, flags)
905                })
906            })
907        } else {
908            None
909        }
910    };
911
912    match result {
913        Some((paddr, pte_flags)) => {
914            // Install the mapping on demand. For ELF-backed segments the ELF loader
915            // (Phase 04) sets pa_start to the physical frame it copied the segment
916            // into; we just wire it up here. For demand-allocated regions (Stack/Heap)
917            // the ELF loader leaves pa_start = 0 and the fault handler allocates.
918            //
919            // Phase 01: VMA list is always empty, so this branch is unreachable
920            // in practice — any user fault panics at the None branch above.
921            let mut frame_guard = crate::memory::frame::FRAME_ALLOCATOR.lock();
922            let alloc = frame_guard
923                .as_mut()
924                .unwrap_or_else(|| panic!("[#PF] frame allocator unavailable at va={:#x}", va));
925            let effective_pa = if paddr == 0 {
926                // Demand-allocate a new zeroed frame.
927                alloc
928                    .allocate_frame()
929                    .unwrap_or_else(|| panic!("[#PF] OOM allocating demand page at va={:#x}", va))
930            } else {
931                paddr
932            };
933            let _ = map_page_x86(alloc, va & !0xFFF, effective_pa & !0xFFF, pte_flags);
934        }
935        None => {
936            panic!(
937                "[#PF user] va={:#x} error_code={:#x} rip={:#x} cs={:#x} — no VMA covers this address",
938                va, error_code, rip, cs
939            );
940        }
941    }
942}
