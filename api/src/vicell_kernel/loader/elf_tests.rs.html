52fn expect_invalid(res: ViResult<usize>, label: &str) {
53    match res {
54        Err(ViError::InvalidInput) | Err(ViError::NotFound) => {}
55        other => panic!("expected error for {}, got {:?}", label, other),
56    }
57}

102/// Construct the 24-byte raw encoding of a single Rela64 entry.
103/// Layout (LE): offset:u64, info:u64, addend:i64
104fn make_rela(offset: u64, r_type: u32, addend: i64) -> [u8; 24] {
105    let mut b = [0u8; 24];
106    b[0..8].copy_from_slice(&offset.to_le_bytes());
107    b[8..16].copy_from_slice(&(r_type as u64).to_le_bytes()); // sym=0 in high 32 bits
108    b[16..24].copy_from_slice(&(addend as u64).to_le_bytes());
109    b
110}

163fn test_reloc_relative_patches_memory() {
164    // Allocate a writable buffer; use its address as `base`.
165    // Create R_RISCV_RELATIVE (type=3): offset=0, addend=0x400.
166    // Expected result: *(base + 0) = base + 0x400.
167    let mut buf = alloc::vec![0u8; 64];
168    let base = buf.as_mut_ptr() as usize;
169
170    let entry = make_rela(0, 3, 0x400_i64); // R_RISCV_RELATIVE
171    let res = crate::loader::reloc::apply_relocations(base, &entry);
172    assert!(res.is_ok(), "R_RISCV_RELATIVE should succeed: {:?}", res);
173
174    // Read back the patched value (usize-width, unaligned-safe).
175    // SAFETY: buf is alive for the duration of this test; we wrote exactly
176    // sizeof(usize) bytes at offset 0 via apply_relocations.
177    let patched: usize = unsafe { core::ptr::read_unaligned(buf.as_ptr() as *const usize) };
178    let expected = base.wrapping_add(0x400);
179    assert_eq!(patched, expected, "R_RISCV_RELATIVE patch value mismatch");
180    log::info!(
181        "  [ok] R_RISCV_RELATIVE patched 0x{:X} → 0x{:X}",
182        base,
183        expected
184    );
185}

189/// Build a legacy v1 8-byte manifest record: `{magic, version=1, flags:u8, _pad:[u8;2]}`.
190fn manifest_bytes_v1(magic: u32, version: u8, flags: u8) -> [u8; 8] {
191    let m = magic.to_le_bytes();
192    [m[0], m[1], m[2], m[3], version, flags, 0, 0]
193}
194
195/// Build a native v2 16-byte manifest record: `{magic, version, tier, flags:u16,
196/// cap_args_off:u32, reserved:u32}`.
197fn manifest_bytes_v2(
198    magic: u32,
199    version: u8,
200    tier: u8,
201    flags: u16,
202    cap_args_off: u32,
203    reserved: u32,
204) -> [u8; 16] {
205    let m = magic.to_le_bytes();
206    let f = flags.to_le_bytes();
207    let c = cap_args_off.to_le_bytes();
208    let r = reserved.to_le_bytes();
209    [
210        m[0], m[1], m[2], m[3], version, tier, f[0], f[1], c[0], c[1], c[2], c[3], r[0], r[1],
211        r[2], r[3],
212    ]
213}

442/// Verify that `extract_sig` correctly finds a `__ViCell_sig` section in a
443/// handcrafted minimal RISC-V ELF64 binary.
444///
445/// Layout (408 bytes total):
446///   [0..64]    ELF64 header (e_phnum=1, e_shnum=3, e_shstrndx=2)
447///   [64..120]  PT_LOAD program header  (p_offset=120, p_filesz=8)
448///   [120..128] "code" bytes            (8 zero bytes)
449///   [128..192] Section header 0: NULL
450///   [192..256] Section header 1: __ViCell_sig  (sh_offset=320, sh_size=64)
451///   [256..320] Section header 2: .shstrtab     (sh_offset=384, sh_size=24)
452///   [320..384] Signature bytes         (64 × 0xAB — sentinel value for test)
453///   [384..408] String table            (\0__ViCell_sig\0.shstrtab\0)
454fn test_signing_extract_sig_some_from_constructed_elf() {
455    let elf = build_minimal_signed_elf([0xABu8; 64]);
456    let result = crate::signing::extract_sig(&elf);
457    assert!(
458        result.is_some(),
459        "extract_sig must find __ViCell_sig in minimal ELF"
460    );
461    let extracted = result.unwrap();
462    assert!(
463        extracted.iter().all(|&b| b == 0xAB),
464        "extracted signature bytes must match embedded sentinel (0xAB×64)"
465    );
466    log::info!("  [ok] extract_sig(constructed elf) → Some([0xAB; 64])");
467}

480/// Build a 408-byte minimal RISC-V ELF64 with one PT_LOAD segment and a
481/// `__ViCell_sig` section carrying `sig` as its data. Used only by tests.
482fn build_minimal_signed_elf(sig: [u8; 64]) -> alloc::vec::Vec<u8> {
483    // String table: \0__ViCell_sig\0.shstrtab\0 (24 bytes)
484    //   strtab[1]  = "__ViCell_sig\0"  → sh_name for section 1
485    //   strtab[14] = ".shstrtab\0"     → sh_name for section 2 (itself)
486    const STRTAB: &[u8] = b"\x00__ViCell_sig\x00.shstrtab\x00";
487    let mut v = alloc::vec![0u8; 384 + STRTAB.len()]; // 384 + 24 = 408
488
489    // ── ELF64 header (offset 0, 64 bytes) ────────────────────────────────────
490    v[0..4].copy_from_slice(b"\x7fELF");
491    v[4] = 2; // ELFCLASS64
492    v[5] = 1; // ELFDATA2LSB
493    v[6] = 1; // EV_CURRENT
494              // e_type=2 (ET_EXEC), e_machine=0xF3 (EM_RISCV)
495    v[16..18].copy_from_slice(&2u16.to_le_bytes());
496    v[18..20].copy_from_slice(&0xF3u16.to_le_bytes());
497    // e_version=1
498    v[20..24].copy_from_slice(&1u32.to_le_bytes());
499    // e_entry=0x1000, e_phoff=64, e_shoff=128
500    v[24..32].copy_from_slice(&0x1000u64.to_le_bytes());
501    v[32..40].copy_from_slice(&64u64.to_le_bytes());
502    v[40..48].copy_from_slice(&128u64.to_le_bytes());
503    // e_flags=5 (RVC + double-float ABI), e_ehsize=64, e_phentsize=56, e_phnum=1
504    v[48..52].copy_from_slice(&5u32.to_le_bytes());
505    v[52..54].copy_from_slice(&64u16.to_le_bytes());
506    v[54..56].copy_from_slice(&56u16.to_le_bytes());
507    v[56..58].copy_from_slice(&1u16.to_le_bytes());
508    // e_shentsize=64, e_shnum=3, e_shstrndx=2
509    v[58..60].copy_from_slice(&64u16.to_le_bytes());
510    v[60..62].copy_from_slice(&3u16.to_le_bytes());
511    v[62..64].copy_from_slice(&2u16.to_le_bytes());
512
513    // ── PT_LOAD program header (offset 64, 56 bytes) ─────────────────────────
514    // ELF64 Phdr: p_type(4), p_flags(4), p_offset(8), p_vaddr(8), p_paddr(8),
515    //             p_filesz(8), p_memsz(8), p_align(8)
516    v[64..68].copy_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
517    v[68..72].copy_from_slice(&5u32.to_le_bytes()); // p_flags = R|X
518    v[72..80].copy_from_slice(&120u64.to_le_bytes()); // p_offset = 120
519    v[80..88].copy_from_slice(&0x1000u64.to_le_bytes()); // p_vaddr
520    v[88..96].copy_from_slice(&0x1000u64.to_le_bytes()); // p_paddr
