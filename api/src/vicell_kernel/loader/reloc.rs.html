1//! ELF relocation engine for ViCell cells.
2//!
3//! Supports the relocation types emitted by the RISC-V LLVM/GCC toolchain
4//! when building position-independent cells (`-pie` or `-shared`).
5
6use types::{VAddr, ViError, ViResult};
7
8/// Maximum number of relocations to process per cell.
9/// Bounds parsing time; a legitimate cell has at most tens of thousands.
10const MAX_RELA_ENTRIES: usize = 65_536;
11
12/// RISC-V ELF relocation types (r_type field in Elf64_Rela.r_info).
13#[allow(dead_code)] // reason: table documents all types; only a subset is used
14mod riscv_reloc_type {
15    pub const R_RISCV_NONE: u32 = 0;
16    pub const R_RISCV_RELATIVE: u32 = 3;
17    pub const R_RISCV_64: u32 = 2;
18    pub const R_RISCV_JUMP_SLOT: u32 = 5;
19}
20
21/// AArch64 ELF relocation types.
22#[allow(dead_code)]
23mod aarch64_reloc_type {
24    pub const R_AARCH64_NONE: u32 = 0;
25    /// Copy-relocation with base+addend — the PIE equivalent of R_RISCV_RELATIVE.
26    pub const R_AARCH64_RELATIVE: u32 = 1027; // 0x403
27}
28
29/// x86-64 ELF relocation types.
30#[allow(dead_code)]
31mod x86_64_reloc_type {
32    pub const R_X86_64_NONE: u32 = 0;
33    /// Copy-relocation with base+addend.
34    pub const R_X86_64_RELATIVE: u32 = 8;
35}
36
37/// A single 64-bit RELA entry (Elf64_Rela layout, little-endian).
38#[repr(C)]
39#[derive(Copy, Clone)]
40struct Rela64 {
41    /// Offset within the ELF's virtual address space to patch.
42    offset: u64,
43    /// Relocation type + symbol index packed as (sym<<32 | type).
44    info: u64,
45    /// Addend to use in the relocation formula.
46    addend: i64,
47}
48
49impl Rela64 {
50    fn r_type(self) -> u32 {
51        (self.info & 0xFFFF_FFFF) as u32
52    }
53}
54
55/// Apply all relocations in the `.rela.dyn` section of an already-mapped ELF.
56///
57/// `base` is the load base chosen by the kernel.  For cells compiled at a fixed
58/// VA (non-PIE linker script), `base == 0` and this function is a no-op beyond
59/// bounds checks.  For PIE cells (base-address randomisation or known-base
60/// placement), this patches every `R_RISCV_RELATIVE` entry.
61///
62/// # Errors
63/// Returns `ViError::InvalidInput` on malformed or oversized relocation tables.
64/// Returns `ViError::NotSupported` if an unsupported relocation type is found.
65pub fn apply_relocations(base: VAddr, rela_section: &[u8]) -> ViResult<()> {
66    let entry_size = core::mem::size_of::<Rela64>();
67    if !rela_section.len().is_multiple_of(entry_size) {
68        log::error!(
69            "[reloc] .rela.dyn size {} not a multiple of Rela64 size {}",
70            rela_section.len(),
71            entry_size
72        );
73        return Err(ViError::InvalidInput);
74    }
75
76    let count = rela_section.len() / entry_size;
77    if count > MAX_RELA_ENTRIES {
78        log::error!(
79            "[reloc] too many relocations: {} > {}",
80            count,
81            MAX_RELA_ENTRIES
82        );
83        return Err(ViError::InvalidInput);
84    }
85
86    for i in 0..count {
87        let offset = i * entry_size;
88        // SAFETY: bounds checked (offset + entry_size <= len).  Use
89        // read_unaligned because the ELF buffer may not be 8-byte aligned —
90        // the backing Vec is u8-aligned; section data inherits that.
91        let entry: Rela64 = unsafe {
92            core::ptr::read_unaligned(rela_section.as_ptr().add(offset) as *const Rela64)
93        };
94
95        // Shared helper: applies the base+addend formula used by all
96        // *_RELATIVE types across every supported architecture.
97        macro_rules! apply_relative {
98            ($label:literal) => {{
99                let patch_va = base.wrapping_add(entry.offset as usize);
100                let value = base.wrapping_add(entry.addend as usize);
101
102                // On AArch64 non-VHE (EL2 kernel), AP[1]=1 in the EL2 stage-1 PTE
103                // means the page is read-only from EL2 — the same bit that grants EL0
104                // access. Write through the physical frame's identity-mapped VA instead,
105                // which is always kernel-writable (no AP[1]).
106                // On RISC-V/x86 the SSTATUS.SUM / U-bit semantics allow S-mode writes.
107                #[cfg(target_arch = "aarch64")]
108                let write_ptr: *mut usize = {
109                    let phys = crate::memory::paging::virt_to_phys(patch_va)
110                        .expect("apply_relocations: relocation target VA not mapped");
111                    // phys includes the in-page offset (translate returns full PA).
112                    crate::memory::frame::phys_to_virt(phys) as *mut usize
113                };
114                #[cfg(not(target_arch = "aarch64"))]
115                // SAFETY: patch_va is within the cell's mapped pages (SAS);
116                // SSTATUS.SUM=1 lets S-mode write to U-mode pages on RISC-V.
117                let write_ptr: *mut usize = patch_va as *mut usize;
118
119                unsafe { write_ptr.write_unaligned(value) };
120            }};
121        }
122
123        match entry.r_type() {
124            // ── no-ops ────────────────────────────────────────────────────────
125            // Relocation type 0 is universally "NONE" across every ELF psABI
126            // (RISC-V, AArch64, x86-64 all define it as 0), so a single arm on
127            // the shared literal value covers all three constants.
128            riscv_reloc_type::R_RISCV_NONE => {}
129
130            // ── RISC-V ────────────────────────────────────────────────────────
131            riscv_reloc_type::R_RISCV_RELATIVE => {
132                apply_relative!("R_RISCV_RELATIVE");
133            }
134            riscv_reloc_type::R_RISCV_64 => {
135                // R_RISCV_64 with sym_index == 0 is an absolute-address fixup;
136                // treat identically to R_RISCV_RELATIVE.
137                let sym_index = (entry.info >> 32) as u32;
138                if sym_index != 0 {
139                    log::error!(
140                        "[reloc] R_RISCV_64 with non-zero sym {} not supported",
141                        sym_index
142                    );
143                    return Err(ViError::NotSupported);
144                }
145                apply_relative!("R_RISCV_64");
146            }
147
148            // ── AArch64 ───────────────────────────────────────────────────────
149            aarch64_reloc_type::R_AARCH64_RELATIVE => {
150                apply_relative!("R_AARCH64_RELATIVE");
151            }
152
153            // ── x86-64 ────────────────────────────────────────────────────────
154            x86_64_reloc_type::R_X86_64_RELATIVE => {
155                apply_relative!("R_X86_64_RELATIVE");
156            }
157
158            other => {
159                log::error!("[reloc] unsupported relocation type {}", other);
160                return Err(ViError::NotSupported);
161            }
162        }
163    }
164
165    log::debug!(
166        "[reloc] applied {} relocation(s) with base 0x{:X}",
167        count,
168        base
169    );
170    Ok(())
171}
