1//! Virtual address allocator for PIE cells.
2//!
3//! Assigns each PIE cell a 32 MiB-aligned VA slot within the cell VA region
4//! [CELL_VA_START, CELL_VA_START + MAX_SLOTS * CELL_VA_STRIDE).
5//!
6//! This region lives entirely below the SV39 user-half boundary (256 GiB), so
7//! elf.rs's `USER_VADDR_MAX` guard is never triggered by any allocated slot.
8//!
9//! # Design
10//! Simple two-level allocator: a bump pointer for never-allocated slots, and a
11//! free list (atomic bitset) for slots returned by `free_cell_va`.  Allocations
12//! are O(n_freed) in the worst case, O(1) when the free list is empty.
13
14// RV32 lacks native 64-bit atomics; portable-atomic polyfills AtomicU64 there
15// via the critical-section impl hal/arch/riscv registers.
16#[cfg(not(target_arch = "riscv32"))]
17use core::sync::atomic::AtomicU64;
18use core::sync::atomic::Ordering;
19#[cfg(target_arch = "riscv32")]
20use portable_atomic::AtomicU64;
21
22/// Cell VA region start — 4 GiB (0x1_0000_0000).
23///
24/// Must lie above ALL identity-mapped regions in the SAS:
25///   - QEMU virt RISC-V: UART/VirtIO MMIO at 0x1000_0000–0x1001_0000
26///   - PCIe ECAM at 0x3000_0000 (1 MiB); VirtIO BAR below 0x8000_0000
27///   - RAM identity map: 256 MB at 0x8000_0000 → ends at 0x8FFF_FFFF
28///   - QEMU virt AArch64: GIC/peripheral MMIO below 0x1000_0000
29///     4 GiB clears all of the above with room to spare and remains well inside
30///     the SV39 user-half (256 GiB = 0x40_0000_0000).
31///
32/// RISC-V medany: intra-cell refs are ≤32 MiB apart → always within ±2 GiB.
33///
34/// RV32 Nano boots with SATP=0 (bare physical, no paging — see specs/04), so
35/// PIE cells never occur there and `alloc_cell_va` is never actually called;
36/// this constant only needs to type-check as a 32-bit `usize` on that target.
37#[cfg(not(target_arch = "riscv32"))]
38const CELL_VA_START: usize = 0x1_0000_0000;
39#[cfg(target_arch = "riscv32")]
40const CELL_VA_START: usize = 0x2000_0000;
41
42/// Each cell slot is 32 MiB — same spacing as the old static VA assignments,
43/// large enough for code + data + stack for any current cell.
44const CELL_VA_STRIDE: usize = 0x200_0000;
45
46/// Maximum simultaneous PIE cell slots — 512 × 32 MiB = 16 GiB of cell VA.
47/// Well within SV39 user half (256 GiB) and leaves room for future extensions.
48const MAX_SLOTS: usize = 512;
49
50/// Number of AtomicU64 words needed to cover MAX_SLOTS bits.
51const BITMAP_WORDS: usize = MAX_SLOTS.div_ceil(64);
52
53/// Bump index: the first slot that has NEVER been allocated.
54static BUMP: core::sync::atomic::AtomicUsize = core::sync::atomic::AtomicUsize::new(0);
55
56/// Free-list bitmap: bit N = 1 means slot N is available for reuse.
57/// Uses relaxed CAS loops — no ordering guarantee needed beyond the CAS itself.
58// Each repeat evaluates the inline `const` block independently, giving
59// BITMAP_WORDS distinct AtomicU64 instances (not one instance aliased across
60// slots) — an inline const sidesteps `clippy::declare_interior_mutable_const`,
61// which would otherwise flag a *named* const of this type, without changing
62// the initialization semantics.
63static FREE: [AtomicU64; BITMAP_WORDS] = [const { AtomicU64::new(0) }; BITMAP_WORDS];
64
65/// Allocate a 32 MiB VA base for a new PIE cell.
66///
67/// Returns `None` when all slots are exhausted (unlikely in practice: 512 cells
68/// and the free list recycles dead cells' slots).
69pub fn alloc_cell_va() -> Option<usize> {
70    // 1. Try the free list first.
71    for (word_idx, word) in FREE.iter().enumerate() {
72        let mut val = word.load(Ordering::Relaxed);
73        while val != 0 {
74            let bit = val.trailing_zeros() as usize;
75            let slot = word_idx * 64 + bit;
76            if slot >= MAX_SLOTS {
77                break;
78            }
79            let mask = 1u64 << bit;
80            match word.compare_exchange_weak(val, val & !mask, Ordering::AcqRel, Ordering::Relaxed)
81            {
82                Ok(_) => return Some(CELL_VA_START + slot * CELL_VA_STRIDE),
83                Err(cur) => val = cur, // retry with updated value
84            }
85        }
86    }
87
88    // 2. Bump-allocate a fresh slot.
89    let idx = BUMP.fetch_add(1, Ordering::SeqCst);
90    if idx >= MAX_SLOTS {
91        BUMP.fetch_sub(1, Ordering::SeqCst);
92        return None;
93    }
94    Some(CELL_VA_START + idx * CELL_VA_STRIDE)
95}
96
97/// Return a VA slot to the free list so it can be reused by a future cell.
98///
99/// `base` must be a value previously returned by `alloc_cell_va`.
100/// Silently ignores invalid values (out of range or misaligned).
101pub fn free_cell_va(base: usize) {
102    if base < CELL_VA_START {
103        return;
104    }
105    let offset = base - CELL_VA_START;
106    if !offset.is_multiple_of(CELL_VA_STRIDE) {
107        return;
108    }
109    let slot = offset / CELL_VA_STRIDE;
110    if slot >= MAX_SLOTS {
111        return;
112    }
113    let word = &FREE[slot / 64];
114    let mask = 1u64 << (slot % 64);
115    word.fetch_or(mask, Ordering::Release);
116}
