6/// Maximum user-space virtual address for RISC-V SV39.
7///
8/// SV39 splits the 39-bit VA space into:
9///   0x0000_0000_0000 – 0x003F_FFFF_FFFF  (user / lower half, 256 GB)
10///   0xFFC0_0000_0000 – 0xFFFF_FFFF_FFFF  (kernel / upper half, 256 GB)
11///
12/// 0x8000_0000 was wrong — it only allowed 2 GB and coincidentally matched
13/// the RISC-V physical RAM base (0x8000_0000 on QEMU virt), causing every
14/// cell ELF compiled at 0x8800_0000+ to be rejected.
15///
16/// The real boundary is half the SV39 address space: 2^38 = 0x40_0000_0000.
17/// Cells compiled at 0x0040_0000 (4 MB) are safely within this range.
18// SV39 user-half upper bound (256 GB). On riscv32 the address space is 32-bit
19// so this value is clamped to usize::MAX (0xFFFF_FFFF), which is still correct
20// as a "no address may be >= 4 GB" guard for riscv32 cells.
21#[cfg(not(any(target_arch = "riscv32", target_arch = "x86", target_arch = "arm")))]
22const USER_VADDR_MAX: usize = 0x40_0000_0000; // 256 GB — SV39 user half
23#[cfg(any(target_arch = "riscv32", target_arch = "x86", target_arch = "arm"))]
24const USER_VADDR_MAX: usize = 0xFFFF_FFFF; // 4 GB — full 32-bit address space
25
26pub struct ElfLoader;

28impl ElfLoader {
29    /// Load loadable segments into memory.
30    ///
31    /// `load_base` is added to every segment's `p_vaddr` before mapping:
32    /// - For fixed-VA cells (ET_EXEC, non-PIE): pass `0` — uses p_vaddr directly.
33    /// - For PIE cells (ET_DYN): pass the VA base allocated by `va_alloc::alloc_cell_va`.
34    ///
35    /// Returns the list of mapped (va, frame) pairs so `CellSegments` can reclaim
36    /// them when the cell dies.
37    pub fn load_segments(
38        &self,
39        data: &[u8],
40        frame_allocator: &mut crate::memory::frame::FrameAllocator,
41        load_base: usize,
42    ) -> ViResult<alloc::vec::Vec<(VAddr, PhysAddr)>> {
43        let elf = ElfFile::new(data).map_err(|_| ViError::InvalidInput)?;
44        // Record each mapped (vaddr, frame) so the cell's segment frames can be
45        // reclaimed when it dies (see task::stack::CellSegments) — otherwise they leak.
46        use crate::memory::paging::Flags;
47        let mut mapped: alloc::vec::Vec<(VAddr, PhysAddr, Flags)> = alloc::vec::Vec::new();
48
49        for ph in elf.program_iter() {
50            if let Ok(xmas_elf::program::Type::Load) = ph.get_type() {
51                let file_offset = ph.offset() as usize;
52                // For PIE cells load_base relocates every segment; for fixed-VA
53                // cells load_base == 0 so p_vaddr is used verbatim.
54                let vaddr = (ph.virtual_addr() as usize).wrapping_add(load_base);
55                let mem_size = ph.mem_size() as usize;
56                let file_size = ph.file_size() as usize;
57                let ph_flags = ph.flags();
58
59                // --- Header sanity checks ---
60                // file_size MUST NOT exceed mem_size (the rest is BSS).
61                if file_size > mem_size {
62                    log::error!(
63                        "ELF: rejecting segment with file_size={} > mem_size={}",
64                        file_size,
65                        mem_size
66                    );
67                    return Err(ViError::InvalidInput);
68                }
69                // file_offset + file_size must fit inside the ELF buffer.
70                let file_end = file_offset
71                    .checked_add(file_size)
72                    .ok_or(ViError::InvalidInput)?;
73                if file_end > data.len() {
74                    log::error!(
75                        "ELF: segment file range {}..{} exceeds buffer len {}",
76                        file_offset,
77                        file_end,
78                        data.len()
79                    );
80                    return Err(ViError::InvalidInput);
81                }
82                // vaddr + mem_size must not overflow and must lie below the
83                // kernel VA window — prevents user ELF clobbering kernel maps.
84                let end_addr = vaddr.checked_add(mem_size).ok_or(ViError::InvalidInput)?;
85                if vaddr >= USER_VADDR_MAX || end_addr > USER_VADDR_MAX {
86                    log::error!(
87                        "ELF: segment VA range 0x{:X}-0x{:X} outside user space",
88                        vaddr,
89                        end_addr
90                    );
91                    return Err(ViError::PermissionDenied);
92                }
93
94                let start_addr = vaddr;
95
96                // Align start/end to page boundaries
97                let start_page = start_addr & !(4096 - 1);
98                let end_page =
99                    end_addr.checked_add(4095).ok_or(ViError::InvalidInput)? & !(4096 - 1);
100
101                // --- Translate ELF p_flags to page-table flags ---
102                // p_flags bits: 0x1=X, 0x2=W, 0x4=R. Default deny if all zero.
103                //
104                // All cell pages are mapped WRITE so the kernel can apply PIE
105                // relocations (.rela.dyn) after loading.  SAS/LBI enforces R/W/X
106                // semantics via Rust's type system, not hardware read-only pages;
107                // hardware-enforced W^X (MTE, SMMU, PKU) is a G2 item.
108                use crate::memory::paging::Flags;
109                let mut perm_bits = Flags::VALID
110                    | Flags::USER
111                    | Flags::ACCESSED
112                    | Flags::READ
113                    | Flags::WRITE
114                    | Flags::DIRTY;
115                if ph_flags.is_execute() {
116                    perm_bits |= Flags::EXECUTE;
117                }
118                let flags = Flags::from_bits(perm_bits);
119
120                // Map pages
121                let mut current_page = start_page;
122                while current_page < end_page {
123                    // Overwrite guard: reject VA collision with kernel MMIO or a
124                    // *different* live cell. Allow shared pages from *this* cell's own
125                    // adjacent PT_LOAD segments (already_ours == true).
126                    //
127                    // PIE linker scripts may place adjacent sections (e.g. .text R-X
128                    // and .rodata R--) at non-page-aligned offsets so the two segments
129                    // share the same physical page.  In that case already_ours==true and
130                    // we reuse the frame, merging the new segment's flags into the page
131                    // rather than allocating a fresh one or blindly overwriting flags.
132                    let already_ours = mapped.iter().any(|(va, _, _)| *va == current_page);
133                    if !already_ours && crate::memory::paging::virt_to_phys(current_page).is_some()
134                    {
135                        log::error!(
136                            "ELF: load VA 0x{:X} already mapped — rejecting spawn (VA collision with a live cell or kernel MMIO; fix the cell's linker script)",
137                            current_page
138                        );
139                        for &(va, fr, _) in &mapped {
140                            let _ = crate::memory::paging::unmap_page(va);
141                            frame_allocator.deallocate_frame(fr);
142                        }
143                        return Err(ViError::PermissionDenied);
144                    }
145
146                    // Get the backing frame's kernel-accessible VA for copying.
147                    let frame_virt = if already_ours {
148                        // Shared page from an earlier LOAD segment (e.g. .rodata R-- and
149                        // .data/RELRO RW- sharing a boundary page).  Reuse the existing
150                        // frame but OR the new segment's permission bits in so the page
151                        // satisfies both segments' access requirements.
152                        let phys = crate::memory::paging::virt_to_phys(current_page)
153                            .expect("already_ours but virt_to_phys returned None");
154                        if let Some(entry) =
155                            mapped.iter_mut().find(|(va, _, _)| *va == current_page)
156                        {
157                            let merged = Flags::from_bits(entry.2.bits() | flags.bits());
158                            if merged != entry.2 {
159                                entry.2 = merged;
160                                let _ = crate::memory::paging::map_page(
161                                    frame_allocator,
162                                    current_page,
163                                    phys,
164                                    merged,
165                                );
166                            }
167                        }
168                        crate::memory::frame::phys_to_virt(phys)
169                    } else {
170                        let buf_frame = frame_allocator
171                            .allocate_frame()
172                            .ok_or(ViError::OutOfMemory)?;
173
174                        crate::memory::paging::map_page(
175                            frame_allocator,
176                            current_page,
177                            buf_frame,
178                            flags,
179                        )
180                        .map_err(|_| ViError::OutOfMemory)?;
181
182                        // Track for reclamation on cell death.
183                        mapped.push((current_page, buf_frame, flags));
184
185                        // Zero the frame first (simplifies BSS and padding, and
186                        // prevents info-leak from previous frame owner).
187                        // Use phys_to_virt: on RISC-V it's a no-op (identity map);
188                        // on x86_64 physical RAM is only accessible via HHDM_BASE+phys.
189                        let fv = crate::memory::frame::phys_to_virt(buf_frame);
190                        unsafe {
191                            core::ptr::write_bytes(fv as *mut u8, 0, 4096);
192                        }
193                        fv
194                    };
195
196                    // Intersection of [page, page+4096) AND [vaddr, vaddr+file_size)
197                    let page_start_vaz = current_page;
198                    let page_end_vaz = current_page + 4096;
199                    let copy_start_v = core::cmp::max(page_start_vaz, vaddr);
200                    let copy_end_v = core::cmp::min(page_end_vaz, vaddr + file_size);
201
202                    if copy_start_v < copy_end_v {
203                        let len = copy_end_v - copy_start_v;
204                        let dst_offset = copy_start_v - page_start_vaz;
205                        let src_offset_in_file = file_offset + (copy_start_v - vaddr);
206                        // file_end was already validated above; this guards
207                        // arithmetic on `len` from any rounding surprise.
208                        let src_end = src_offset_in_file
209                            .checked_add(len)
210                            .ok_or(ViError::InvalidInput)?;
211                        if src_end <= data.len() {
212                            let src = &data[src_offset_in_file..src_end];
213                            unsafe {
214                                let dst = (frame_virt as *mut u8).add(dst_offset);
215                                core::ptr::copy_nonoverlapping(src.as_ptr(), dst, len);
216                            }
217                        }
218                    }
219
220                    current_page += 4096;
221                }
222
223                log::info!(
224                    "ELF LOAD: 0x{:X}-0x{:X} flags={}{}{}",
225                    start_addr,
226                    end_addr,
227                    if ph_flags.is_read() { 'R' } else { '-' },
228                    if ph_flags.is_write() { 'W' } else { '-' },
229                    if ph_flags.is_execute() { 'X' } else { '-' },
230                );
231            }
232        }
233        Ok(mapped.into_iter().map(|(va, phys, _)| (va, phys)).collect())
234    }
235}

237impl ElfParser for ElfLoader {
238    fn parse_header(&self, data: &[u8]) -> ViResult<ElfHeader> {
239        let elf = ElfFile::new(data).map_err(|_| ViError::InvalidInput)?;
240
241        // Verify architecture (RISC-V 64)
242        // Header check is implicit in successful new(), but specific machine check?
243        // elf.header.pt2.machine() == xmas_elf::header::Machine::RISC_V
244
245        Ok(ElfHeader {
246            entry: elf.header.pt2.entry_point() as usize,
247            shoff: elf.header.pt2.sh_offset() as usize,
248        })
249    }
250
251    fn get_section<'a>(&self, data: &'a [u8], name: &str) -> ViResult<&'a [u8]> {
252        let elf = ElfFile::new(data).map_err(|_| ViError::InvalidInput)?;
253        match elf.find_section_by_name(name) {
254            Some(section) => Ok(section.raw_data(&elf)),
255            None => Err(ViError::NotFound),
256        }
257    }
258}
