1//! Constants shared between the kernel early loader and the disk-image builder.
2//!
3//! Since Milestone 2.5 Phase 03, `disk_v3.img` carries a real MBR partition
4//! table at LBA 0 (written by `tools/write-mbr.py`; gen_disk.ps1 builds it):
5//!
6//! ```text
7//! P1  0x0C  LBA   2,048 + 524,288   FAT32 interop volume (VFS /data today, /mnt/sd later)
8//! P2  0x7F  LBA 526,336 +  33,664   cell bootstrap table + raw ELF blobs
9//! P3  0x7D  LBA 560,000 + 240,000   kernel heap snapshot (Phase 29)
10//! P4  0x7E  LBA 800,000 + 131,072   littlefs /data (Milestone 2.5 Phase 04)
11//! ```
12//!
13//! The constants below are the contract; the on-disk MBR is parsed at boot by
14//! [`verify_mbr`] to detect drift between image and kernel (warn-only — the
15//! constants stay authoritative so a blank/legacy image still boots).
16//!
17//! The kernel reads the cell table from `CELL_TABLE_BASE_LBA` onwards using
18//! the VirtIO block driver directly, before any userspace VFS Cell is running.
19//!
20//! Layout of the cell bootstrap section (inside P2):
21//!
22//! ```text
23//! LBA CELL_TABLE_BASE_LBA + 0   : CellTableHeader  (one sector = 512 bytes)
24//! LBA CELL_TABLE_BASE_LBA + 1   : CellEntry[0..MAX_CELL_ENTRIES]
25//!                                   (one sector per entry, padded to 512 bytes)
26//! LBA CELL_TABLE_BASE_LBA + 1 + MAX_CELL_ENTRIES : raw ELF data, concatenated
27//!                                   (each ELF starts at its entry's `data_lba`)
28//! ```
29
30// Partition constants live in `api::disk` (the Law-1 contract shared with
31// cells and image tools); re-exported here so kernel code keeps short paths.
32pub use api::disk::{
33    PART_CELLSTORE_BASE_LBA, PART_CELLSTORE_SECTORS, PART_CELLTBL_SECTORS, PART_FAT32_BASE_LBA,
34    PART_FAT32_SECTORS, PART_LFS_BASE_LBA, PART_LFS_SECTORS, PART_SNAPSHOT_BASE_LBA,
35    PART_SNAPSHOT_SECTORS, PART_SRV_BASE_LBA, PART_SRV_SECTORS,
36};
37
38/// Sector offset (from LBA 0) where the cell bootstrap section (P2) begins.
39/// Equals `PART_FAT32_BASE_LBA + PART_FAT32_SECTORS` — the value predates the
40/// MBR and is kept identical so the early loader needs no migration.
41pub const CELL_TABLE_BASE_LBA: u64 = api::disk::PART_CELLTBL_BASE_LBA;
42
43/// Magic bytes at the start of `CellTableHeader`; identifies a valid table.
44pub const CELL_TABLE_MAGIC: u64 = 0x5649_4F53_5F43_454C; // "ViCell_CEL" in ASCII
45
46/// Maximum number of cells that can appear in the bootstrap table.
47pub const MAX_CELL_ENTRIES: usize = 64;
48
49/// Maximum path length (bytes) for a cell path in the bootstrap table.
50pub const CELL_PATH_LEN: usize = 64;
51
52/// Maximum path length accepted by the `SpawnFromPath` syscall.
53/// Must be ≥ `CELL_PATH_LEN`; defines the trust-boundary validation limit.
54pub const MAX_CELL_PATH: usize = 256;
55
56/// Size of one disk sector in bytes.
57pub const SECTOR_SIZE: usize = 512;
58
59/// Header at `CELL_TABLE_BASE_LBA + 0`.
60#[repr(C)]
61#[derive(Debug, Clone, Copy)]
62pub struct CellTableHeader {
63    /// Must equal `CELL_TABLE_MAGIC`; reject the table otherwise.
64    pub magic: u64,
65    /// Number of valid entries in the entry array that follows.
66    pub count: u32,
67    /// Reserved / zero-padded to fill the sector.
68    pub _pad: [u8; 500],
69}
70
71/// One entry in the cell table; stored starting at `CELL_TABLE_BASE_LBA + 1`.
72/// Each entry is padded to exactly `SECTOR_SIZE` bytes.
73#[repr(C)]
74#[derive(Debug, Clone, Copy)]
75pub struct CellEntry {
76    /// Null-terminated path (e.g. `/bin/vfs\0`).
77    pub path: [u8; CELL_PATH_LEN],
78    /// First LBA of the ELF data.
79    pub data_lba: u64,
80    /// Size of the ELF data in bytes (not rounded to sectors).
81    pub data_size: u64,
82    /// Reserved. (512 − 64 − 8 − 8 = 432 bytes)
83    pub _pad: [u8; 432],
84}
85
86// Compile-time size checks: each header/entry must fit in one sector.
87const _: () = assert!(core::mem::size_of::<CellTableHeader>() == SECTOR_SIZE);
88const _: () = assert!(core::mem::size_of::<CellEntry>() == SECTOR_SIZE);
89
90/// Expected MBR partition map: `(slot, type, start_lba, sectors)`.
91/// P5 (SRV/RedoxFS) is outside the legacy 4-partition MBR limit — verified
92/// separately by the VFS service's partition constants.
93const EXPECTED_PARTS: [(usize, u8, u64, u64); 4] = [
94    (0, 0x0C, PART_FAT32_BASE_LBA, PART_FAT32_SECTORS),
95    (1, 0x7F, CELL_TABLE_BASE_LBA, PART_CELLTBL_SECTORS),
96    (2, 0x7D, PART_SNAPSHOT_BASE_LBA, PART_SNAPSHOT_SECTORS),
97    (3, 0x7E, PART_LFS_BASE_LBA, PART_LFS_SECTORS),
98];
99
100/// Parse LBA 0 and compare the MBR against the compiled-in layout.
101///
102/// Warn-only by design: the constants above remain authoritative (a legacy or
103/// blank image must still boot), but any drift between the image builder and
104/// the kernel is surfaced at boot instead of as silent corruption later.
105/// Called once from kernel init after the VirtIO block driver is up.
106pub fn verify_mbr() {
107    let mut sector = [0u8; SECTOR_SIZE];
108    if crate::task::drivers::block::read_sector(0, &mut sector).is_err() {
109        log::warn!("[mbr] LBA 0 unreadable — skipping partition verification");
110        return;
111    }
112    if sector[510] != 0x55 || sector[511] != 0xAA {
113        log::warn!("[mbr] no MBR signature — legacy whole-disk image (pre-P03 layout)");
114        return;
115    }
116    for (slot, ptype, start, size) in EXPECTED_PARTS {
117        let e = 446 + slot * 16;
118        let found_type = sector[e + 4];
119        let found_start =
120            u32::from_le_bytes([sector[e + 8], sector[e + 9], sector[e + 10], sector[e + 11]])
121                as u64;
122        let found_size = u32::from_le_bytes([
123            sector[e + 12],
124            sector[e + 13],
125            sector[e + 14],
126            sector[e + 15],
127        ]) as u64;
128        if found_type != ptype || found_start != start || found_size != size {
129            log::warn!(
130                "[mbr] P{} mismatch: image type={:#04x} start={} size={} — kernel expects type={:#04x} start={} size={}",
131                slot + 1, found_type, found_start, found_size, ptype, start, size
132            );
133        } else {
134            log::info!(
135                "[mbr] P{} ok: type={:#04x} start={} sectors={}",
136                slot + 1,
137                ptype,
138                start,
139                size
140            );
141        }
142    }
143}
