2//! Minimal no_std ACPI table parser for x86_64 kernel.
3//!
4//! Parses RSDP → XSDT/RSDT → MADT (APIC), MCFG (PCIe ECAM), HPET.
5//!
6//! Design invariants:
7//! - NEVER panics; every parse error → warn log → field keeps its default.
8//! - Called before paging, so all table access uses the `phys_to_virt` closure
9//!   (typically HHDM offset). Limine maps all RAM at HHDM+phys.
10//! - Only x86_64 needs this; other arches use DTB via `platform::init`.
11
12/// Parsed addresses from ACPI tables.
13///
14/// Every field has a safe hardware-compatible default so boot succeeds even
15/// when the RSDP is absent or a table is malformed.
16#[derive(Clone, Copy, Debug)]
17pub struct AcpiInfo {
18    /// Local APIC MMIO base (MADT Local APIC Address field or type-5 override).
19    pub lapic_base: u64,
20    /// I/O APIC MMIO base (MADT type-1 entry).
21    pub ioapic_base: u64,
22    /// I/O APIC Global System Interrupt base (MADT type-1 gsi_base).
23    pub ioapic_gsi_base: u32,
24    /// HPET event timer block address (HPET table GAS address field).
25    pub hpet_base: u64,
26    /// PCIe ECAM config space base (MCFG allocation[0].base_address).
27    pub ecam_base: u64,
28    /// ISA IRQ → GSI override table. Index = ISA IRQ (0–15); value = GSI.
29    /// Entries not overridden by MADT type-2 keep identity mapping (IRQ N → GSI N).
30    pub irq_overrides: [u32; 16],
31}
32
33impl Default for AcpiInfo {
34    fn default() -> Self {
35        let mut overrides = [0u32; 16];
36        for (i, v) in overrides.iter_mut().enumerate() {
37            *v = i as u32;
38        }
39        Self {
40            lapic_base: 0xFEE0_0000,
41            ioapic_base: 0xFEC0_0000,
42            ioapic_gsi_base: 0,
43            hpet_base: 0xFED0_0000,
44            ecam_base: 0xB000_0000,
45            irq_overrides: overrides,
46        }
47    }
48}

50// ---------------------------------------------------------------------------
51// Internal layout types — packed structs used only for read_unaligned casts.
52// ---------------------------------------------------------------------------
53
54// NOTE: SdtHeader / MadtHeader / Gas are documented here as layout reference
55// but not instantiated — we read individual fields by explicit byte offset
56// to avoid references to packed struct fields (which Rust forbids).
57// Table offsets are documented in each parser function.
58
59/// RSDP v1 (revision 0) — 20 bytes.
60#[derive(Copy, Clone)]
61#[repr(C, packed)]
62struct Rsdp {
63    signature: [u8; 8],
64    checksum: u8,
65    oem_id: [u8; 6],
66    revision: u8,
67    rsdt_address: u32,
68}
69
70/// RSDP v2 (revision ≥ 2) extension — layout after the v1 struct.
71/// Total size: 36 bytes.
72#[derive(Copy, Clone)]
73#[repr(C, packed)]
74struct RsdpV2 {
75    v1: Rsdp,
76    length: u32,
77    xsdt_address: u64,
78    extended_checksum: u8,
79    _reserved: [u8; 3],
80}

82// ---------------------------------------------------------------------------
83// Safety helpers
84// ---------------------------------------------------------------------------

86/// Read a `T` from a raw virtual address without alignment requirement.
87///
88/// # Safety
89/// `virt` must be a valid, readable virtual address pointing to at least
90/// `size_of::<T>()` bytes. The value is read byte-by-byte to avoid UB from
91/// unaligned packed struct access.
92#[inline]
93unsafe fn read_unaligned<T: Copy>(virt: usize) -> T {
94    // SAFETY: caller guarantees virt is valid.
95    unsafe { core::ptr::read_unaligned(virt as *const T) }
96}
97
98/// Compute the byte checksum of a memory region (ACPI table validation).
99///
100/// # Safety
101/// `[base, base + len)` must be readable virtual memory.
102unsafe fn table_checksum(base: usize, len: usize) -> u8 {
103    let mut sum: u8 = 0;
104    for i in 0..len {
105        // SAFETY: caller guarantees [base, base+len) is readable.
106        let byte = unsafe { core::ptr::read_volatile((base + i) as *const u8) };
107        sum = sum.wrapping_add(byte);
108    }
109    sum
110}

116/// Parse ACPI tables starting from `rsdp_phys`.
117///
118/// `phys_to_virt` converts a physical address to a kernel-accessible virtual
119/// address (typically `phys + HHDM_OFFSET` under Limine).
120///
121/// Returns `AcpiInfo` with parsed values; any field that cannot be parsed keeps
122/// the QEMU q35 default so the system boots even if ACPI is partially broken.
123///
124/// # Panics
125/// Never — all errors produce a log warning and fall through to defaults.
126pub fn parse(rsdp_phys: usize, phys_to_virt: impl Fn(usize) -> usize) -> AcpiInfo {
127    let mut info = AcpiInfo::default();
128
129    if rsdp_phys == 0 {
130        log::warn!("[acpi] RSDP physical address is null — using defaults");
131        return info;
132    }
133
134    let rsdp_virt = phys_to_virt(rsdp_phys);
135
136    // --- Validate RSDP signature ---
137    // SAFETY: Limine guarantees the RSDP response contains a valid physical
138    // address to the RSDP structure, mapped via HHDM. We read cautiously.
139    let sig_bytes: [u8; 8] = unsafe { read_unaligned(rsdp_virt) };
140    if &sig_bytes != b"RSD PTR " {
141        log::warn!("[acpi] RSDP signature mismatch — using defaults");
142        return info;
143    }
144
145    // --- Validate RSDP v1 checksum (bytes 0..19) ---
146    // SAFETY: RSDP is at least 20 bytes (v1 size); Limine guarantees it.
147    let cksum = unsafe { table_checksum(rsdp_virt, 20) };
148    if cksum != 0 {
149        log::warn!(
150            "[acpi] RSDP v1 checksum failed ({}) — using defaults",
151            cksum
152        );
153        return info;
154    }
155
156    // SAFETY: RSDP is at least size_of::<Rsdp>() bytes.
157    let rsdp: Rsdp = unsafe { read_unaligned(rsdp_virt) };
158    let revision = rsdp.revision;
159
160    if revision >= 2 {
161        // XSDT path: 64-bit table pointers.
162        // SAFETY: RSDP v2 is at least size_of::<RsdpV2>() bytes; Limine guarantees it.
163        let rsdp_v2: RsdpV2 = unsafe { read_unaligned(rsdp_virt) };
164        let xsdt_phys = rsdp_v2.xsdt_address as usize;
165        if xsdt_phys == 0 {
166            log::warn!("[acpi] XSDT address is null — using defaults");
167            return info;
168        }
169        // SAFETY: xsdt_phys is a physical address from RSDP v2; phys_to_virt
170        // maps it to a valid HHDM virtual address readable by the kernel.
171        unsafe {
172            parse_xsdt(xsdt_phys, &phys_to_virt, &mut info);
173        }
174    } else {
175        // RSDT path: 32-bit table pointers.
176        let rsdt_phys = rsdp.rsdt_address as usize;
177        if rsdt_phys == 0 {
178            log::warn!("[acpi] RSDT address is null — using defaults");
179            return info;
180        }
181        // SAFETY: rsdt_phys is a physical address from RSDP v1; phys_to_virt
182        // maps it to a valid HHDM virtual address readable by the kernel.
183        unsafe {
184            parse_rsdt(rsdt_phys, &phys_to_virt, &mut info);
185        }
186    }
187
188    info
189}

195/// Validate an SDT header checksum and return `(virt_base, length)` if valid.
196///
197/// Returns `None` and logs a warning if the checksum fails.
198unsafe fn validate_sdt(
199    phys: usize,
200    phys_to_virt: &impl Fn(usize) -> usize,
201) -> Option<(usize, usize)> {
202    let virt = phys_to_virt(phys);
203    // Read the length field at offset 4 (4 bytes into the header).
204    // SAFETY: phys points to a valid SDT; length field is at offset 4.
205    let length = unsafe { core::ptr::read_unaligned((virt + 4) as *const u32) } as usize;
206    if !(36..=0x10_0000).contains(&length) {
207        log::warn!(
208            "[acpi] SDT at {:#x} has implausible length {} — skipping",
209            phys,
210            length
211        );
212        return None;
213    }
214    // SAFETY: [virt, virt+length) is within a valid Limine HHDM-mapped region.
215    let cksum = unsafe { table_checksum(virt, length) };
216    if cksum != 0 {
217        log::warn!(
218            "[acpi] SDT at {:#x} checksum failed ({}) — skipping",
219            phys,
220            cksum
221        );
222        return None;
223    }
224    Some((virt, length))
225}

227/// Read the 4-byte signature at `virt`.
228#[inline]
229unsafe fn read_sig(virt: usize) -> [u8; 4] {
230    // SAFETY: caller guarantees virt is within a valid SDT.
231    unsafe { core::ptr::read_unaligned(virt as *const [u8; 4]) }
232}
233
234/// Iterate XSDT (64-bit pointer array) and dispatch each child SDT.
235unsafe fn parse_xsdt(phys: usize, p2v: &impl Fn(usize) -> usize, info: &mut AcpiInfo) {
236    let Some((virt, length)) = (unsafe { validate_sdt(phys, p2v) }) else {
237        log::warn!("[acpi] XSDT validation failed — using defaults");
238        return;
239    };
240
241    // Entries start at byte 36 (after the 36-byte common header).
242    // Each entry is an 8-byte physical pointer.
243    let entries_start = virt + 36;
244    let entries_len = (length - 36) / 8;
245
246    for i in 0..entries_len {
247        // SAFETY: within validated XSDT body.
248        let child_phys =
249            unsafe { core::ptr::read_unaligned((entries_start + i * 8) as *const u64) } as usize;
250        if child_phys == 0 {
251            continue;
252        }
253        dispatch_sdt(child_phys, p2v, info);
254    }
255}
256
257/// Iterate RSDT (32-bit pointer array) and dispatch each child SDT.
258unsafe fn parse_rsdt(phys: usize, p2v: &impl Fn(usize) -> usize, info: &mut AcpiInfo) {
259    let Some((virt, length)) = (unsafe { validate_sdt(phys, p2v) }) else {
260        log::warn!("[acpi] RSDT validation failed — using defaults");
261        return;
262    };
263
264    // Entries start at byte 36; each entry is a 4-byte physical pointer.
265    let entries_start = virt + 36;
266    let entries_len = (length - 36) / 4;
267
268    for i in 0..entries_len {
269        // SAFETY: within validated RSDT body.
270        let child_phys =
271            unsafe { core::ptr::read_unaligned((entries_start + i * 4) as *const u32) } as usize;
272        if child_phys == 0 {
273            continue;
274        }
275        dispatch_sdt(child_phys, p2v, info);
276    }
277}
278
279/// Validate and dispatch a child SDT by its 4-byte signature.
280fn dispatch_sdt(phys: usize, p2v: &impl Fn(usize) -> usize, info: &mut AcpiInfo) {
281    let Some((virt, length)) = (unsafe { validate_sdt(phys, p2v) }) else {
282        return;
283    };
284    // SAFETY: virt is the start of a validated SDT.
285    let sig = unsafe { read_sig(virt) };
286
287    match &sig {
288        b"APIC" => parse_madt(virt, length, info),
289        b"MCFG" => parse_mcfg(virt, length, info),
290        b"HPET" => parse_hpet(virt, length, info),
291        _ => {
292            // Skip unknown/unneeded tables silently.
293        }
294    }
295}

298// MADT parser
299// ---------------------------------------------------------------------------

301/// Parse MADT (Multiple APIC Description Table) for LAPIC/IOAPIC/IRQ overrides.
302///
303/// Reads:
304///   - MADT header `local_apic_address` (32-bit)
305///   - Type 1 (I/O APIC): ioapic_addr, gsi_base
306///   - Type 2 (Int Source Override): ISA IRQ → GSI mapping
307///   - Type 5 (Local APIC Address Override): 64-bit lapic_addr
308fn parse_madt(virt: usize, length: usize, info: &mut AcpiInfo) {
309    // MADT-specific header starts after the common 36-byte SDT header.
310    // Layout: local_apic_address (u32) at offset 36, flags (u32) at offset 40.
311    // Entry records begin at offset 44.
312    if length < 44 {
313        log::warn!("[acpi] MADT too short ({} bytes)", length);
314        return;
315    }
316
317    // Read Local APIC 32-bit base from MADT header.
318    // SAFETY: virt+36 is within the validated MADT body.
319    let lapic_addr_32 = unsafe { core::ptr::read_unaligned((virt + 36) as *const u32) };
320    if lapic_addr_32 != 0 {
321        info.lapic_base = lapic_addr_32 as u64;
322    }
323
324    let mut offset = 44usize; // start of MADT entry records
325    let end = virt + length;
326
327    while virt + offset + 2 <= end {
328        // Each entry: type (u8), length (u8), then type-specific payload.
329        // SAFETY: offset is within the validated MADT.
330        let entry_type = unsafe { core::ptr::read_volatile((virt + offset) as *const u8) };
331        let entry_len =
332            unsafe { core::ptr::read_volatile((virt + offset + 1) as *const u8) } as usize;
333
334        if entry_len < 2 || virt + offset + entry_len > end {
335            // Malformed entry — stop parsing entries but keep what we got.
336            log::warn!(
337                "[acpi] MADT entry type={} has bad length={}",
338                entry_type,
339                entry_len
340            );
341            break;
342        }
343
344        match entry_type {
345            // Type 0: Processor Local APIC (length 8) — skip (no SMP enum needed).
346            0 => {}
347
348            // Type 1: I/O APIC (length 12).
349            //   offset+2: ioapic_id (u8)
350            //   offset+3: reserved (u8)
351            //   offset+4: ioapic_address (u32)
352            //   offset+8: gsi_base (u32)
353            1 if entry_len >= 12 => {
354                let ioapic_addr =
355                    unsafe { core::ptr::read_unaligned((virt + offset + 4) as *const u32) };
356                let gsi_base =
357                    unsafe { core::ptr::read_unaligned((virt + offset + 8) as *const u32) };
358                if ioapic_addr != 0 {
359                    info.ioapic_base = ioapic_addr as u64;
360                    info.ioapic_gsi_base = gsi_base;
361                }
362            }
363
364            // Type 2: Interrupt Source Override (length 10).
365            //   offset+2: bus (u8) — always 0 for ISA
366            //   offset+3: source (u8) — ISA IRQ number
367            //   offset+4: gsi (u32)
368            //   offset+8: flags (u16)
369            2 if entry_len >= 10 => {
370                let bus = unsafe { core::ptr::read_volatile((virt + offset + 2) as *const u8) };
371                let source = unsafe { core::ptr::read_volatile((virt + offset + 3) as *const u8) };
372                let gsi = unsafe { core::ptr::read_unaligned((virt + offset + 4) as *const u32) };
373                // Only remap ISA bus (bus==0) IRQs that fit in our table.
374                if bus == 0 && (source as usize) < 16 {
375                    info.irq_overrides[source as usize] = gsi;
376                }
377            }
378
379            // Type 4: Non-Maskable Interrupt (NMI) (length 6) — skip.
380            4 => {}
381
382            // Type 5: Local APIC Address Override (length 12).
383            //   offset+2: reserved (u16)
384            //   offset+4: lapic_address (u64)
385            5 if entry_len >= 12 => {
386                let lapic64 =
387                    unsafe { core::ptr::read_unaligned((virt + offset + 4) as *const u64) };
388                if lapic64 != 0 {
389                    info.lapic_base = lapic64;
390                }
391            }
392
393            // Ignore all other entry types (type 3, 6, 7, 9, 0xA, etc.).
394            _ => {}
395        }
396
397        offset += entry_len;
398    }
399}

405/// Parse MCFG (Memory Mapped Configuration Space) table.
406///
407/// Reads the first allocation entry's `base_address` as the PCIe ECAM base.
408/// MCFG body layout (after 36-byte SDT header):
409///   - 8 bytes reserved
410///   - then N × 16-byte allocation entries:
411///     base_address (u64), segment (u16), bus_start (u8), bus_end (u8), _reserved (u32)
412fn parse_mcfg(virt: usize, length: usize, info: &mut AcpiInfo) {
413    // First allocation entry starts at offset 44 (36-byte header + 8-byte reserved).
414    if length < 44 + 16 {
415        log::warn!(
416            "[acpi] MCFG too short ({} bytes) for any allocation entry",
417            length
418        );
419        return;
420    }
421
422    // SAFETY: offset 44 is within the validated MCFG body.
423    let base_addr = unsafe { core::ptr::read_unaligned((virt + 44) as *const u64) };
424    if base_addr != 0 {
425        info.ecam_base = base_addr;
426        log::info!("[acpi] MCFG: PCIe ECAM base = {:#x}", base_addr);
427    } else {
428        log::warn!("[acpi] MCFG: first allocation base_address is 0 — keeping default");
429    }
430}

436/// Parse HPET table for event timer block MMIO address.
437///
438/// HPET table layout (after 36-byte SDT header):
439///   offset 36: event_timer_block_id (u32) — hardware rev, comparators, etc.
440///   offset 40: base_address (GAS, 12 bytes)
441///       GAS[0]:  address_space_id (u8) — 0 = MMIO
442///       GAS[4..12]: address (u64)
443///   offset 52: hpet_number (u8)
444///   offset 53: minimum_tick (u16)
445///   offset 55: page_protection (u8)
446fn parse_hpet(virt: usize, length: usize, info: &mut AcpiInfo) {
447    // GAS starts at offset 40; address field is at GAS offset 4 → table offset 44.
448    if length < 56 {
449        log::warn!("[acpi] HPET table too short ({} bytes)", length);
450        return;
451    }
452
453    // GAS address_space_id at offset 40.
454    // SAFETY: offset 40 is within the validated HPET body.
455    let addr_space = unsafe { core::ptr::read_volatile((virt + 40) as *const u8) };
456    if addr_space != 0 {
457        // Non-zero address space means I/O ports or PCI config — not simple MMIO.
458        log::warn!(
459            "[acpi] HPET GAS address_space_id={} (not MMIO) — keeping default",
460            addr_space
461        );
462        return;
463    }
464
465    // GAS address at GAS offset +4 → table offset 44.
466    // SAFETY: offset 44 is within the validated HPET body.
467    let hpet_addr = unsafe { core::ptr::read_unaligned((virt + 44) as *const u64) };
468    if hpet_addr != 0 {
469        info.hpet_base = hpet_addr;
470        log::info!("[acpi] HPET: event timer block = {:#x}", hpet_addr);
471    } else {
472        log::warn!("[acpi] HPET: GAS address is 0 — keeping default");
473    }
