1//! Focus-based input event dispatcher.
2//!
3//! Maintains a "focused cell" ID and routes translated `InputEvent`s to it
4//! via IPC Send.  Focus changes clear transient modifiers to avoid stuck keys.
5//!
6//! ## Focus default
7//!
8//! `focused` starts at 0 (no focus).  The first cell to call `SetFocus` owns
9//! the keyboard.  Events before any focus claim are silently dropped — this is
10//! preferable to the previous TID-3 shell fallback which consumed events without
11//! the shell ever reading them (shell uses sys_read(0), not the input service).
12//!
13//! ## Death reversion
14//!
15//! `dispatch()` checks the `sys_send` return value.  When the focused cell has
16//! exited, `sys_send` returns `Err(_)` and focus reverts to 0.  The next app
17//! that calls `SetFocus` resumes delivery.
18
19use api::input::{encode_event, InputEvent, INPUT_EVENT_IPC_SIZE};
20use api::syscall::service;
21use ostd::syscall::{sys_lookup_service, sys_try_send};
22
23/// Opcode prefix byte sent to the focused cell's IPC endpoint.
24pub const INPUT_EVENT_OPCODE: u8 = 0x10;
25
26/// Routes translated events to the currently focused cell.
27pub struct Dispatcher {
28    /// Task ID of the currently focused cell (0 = no focus, events dropped).
29    focused: usize,
30    /// Fallback TID on focus-cell death (0 = park until next SetFocus).
31    // reason: current death-reversion policy (see module doc "Death reversion")
32    // hard-reverts to 0 and waits for the next SetFocus; this field predates that
33    // decision and is kept for a planned "revert to shell" fallback path.
34    #[allow(dead_code)]
35    fallback_tid: usize,
36    /// Cached compositor TID for mouse routing (0 = not yet resolved).
37    /// Re-resolved lazily; reset to 0 when a send fails (compositor respawned).
38    compositor_tid: usize,
39}
40
41impl Dispatcher {
42    /// Create a dispatcher with no initial focus (events dropped until SetFocus).
43    pub fn new() -> Self {
44        Self {
45            focused: 0,
46            fallback_tid: 0,
47            compositor_tid: 0,
48        }
49    }
50
51    /// Change which cell receives input events.
52    ///
53    /// Also resets transient modifiers on the modifier-state tracker so that
54    /// Shift/Ctrl/Alt do not appear "stuck" after a focus change.
55    pub fn set_focus(&mut self, cell_endpoint: usize) {
56        self.focused = cell_endpoint;
57    }
58
59    /// Return the currently focused endpoint.
60    #[allow(dead_code)]
61    pub fn focus(&self) -> usize {
62        self.focused
63    }
64
65    /// Send a translated `InputEvent` to the focused cell.
66    ///
67    /// If `sys_send` fails (focused cell has exited), focus reverts to
68    /// `fallback_tid` so subsequent key events reach the shell again.
69    ///
70    /// The IPC message format is:
71    /// ```text
72    /// byte[0]   = INPUT_EVENT_OPCODE (0x10)
73    /// byte[1..] = encode_event() output (see libs/api/src/input.rs)
74    /// ```
75    pub fn dispatch(&mut self, event: &InputEvent) {
76        if self.focused == 0 {
77            return; // no focus — drop silently
78        }
79        let _ = Self::send_event(self.focused, event);
80        // NOTE: no per-dispatch logging — it would print a line on the shared
81        // console for every keystroke, burying the shell prompt the user types at.
82    }
83
84    /// Send a mouse event (`MouseMove`/`MouseButton`/`MouseScroll`) to the
85    /// compositor rather than the keyboard-focused cell.
86    ///
87    /// The compositor owns the cursor and the surface Z-order, so it is the
88    /// only correct recipient for pointer events: it repaints/moves the cursor
89    /// and hit-tests button clicks to the surface under the pointer. Routing
90    /// mouse through the keyboard focus (the historical behaviour) breaks as
91    /// soon as a non-GUI cell like the shell holds focus — the events land in
92    /// a cell that ignores them and the cursor never moves.
93    ///
94    /// The compositor TID is resolved lazily via the service registry and
95    /// re-resolved after a send failure (compositor death/respawn).
96    pub fn dispatch_mouse(&mut self, event: &InputEvent) {
97        if self.compositor_tid == 0 {
98            match sys_lookup_service(service::COMPOSITOR) {
99                Some(tid) => self.compositor_tid = tid,
100                None => return, // compositor not up yet — drop
101            }
102        }
103        if Self::send_event(self.compositor_tid, event).is_err() {
104            self.compositor_tid = 0; // stale TID — re-resolve on next event
105        }
106    }
107
108    /// Encode and try-send one event to `target`.
109    ///
110    /// The IPC message format is:
111    /// ```text
112    /// byte[0]   = INPUT_EVENT_OPCODE (0x10)
113    /// byte[1..] = encode_event() output (see libs/api/src/input.rs)
114    /// ```
115    fn send_event(target: usize, event: &InputEvent) -> Result<(), ()> {
116        let mut buf = [0u8; INPUT_EVENT_IPC_SIZE + 1];
117        buf[0] = INPUT_EVENT_OPCODE;
118        let mut payload = [0u8; INPUT_EVENT_IPC_SIZE];
119        encode_event(event, &mut payload);
120        buf[1..INPUT_EVENT_IPC_SIZE + 1].copy_from_slice(&payload);
121
122        // Non-blocking dispatch: if the target is not receiving (and its
123        // pending_msgs queue is full), the event is dropped.
124        match sys_try_send(target, &buf) {
125            ostd::syscall::SyscallResult::Ok(_) => Ok(()),
126            _ => Err(()),
127        }
128    }
129}
130
131impl Default for Dispatcher {
132    fn default() -> Self {
133        Self::new()
134    }
135}
