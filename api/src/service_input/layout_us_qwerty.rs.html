1//! US QWERTY scancode → (KeySym, unshifted char, shifted char) mapping.
2//!
3//! Covers Linux evdev scancode set 1 (0x01..=0x7F).  Extended scancodes
4//! (0x80..=0xFF, prefixed with 0xE0 on hardware) are handled separately via
5//! `EXTENDED_TABLE`.
6
7use api::input::{KeyState, KeySym, Modifiers};
8
9/// One entry in the layout table.
10pub struct LayoutEntry {
11    pub keysym: KeySym,
12    pub unshifted: u32, // Unicode code point, 0 = non-printable
13    pub shifted: u32,
14}
15
16impl LayoutEntry {
17    const fn new(keysym: KeySym, u: u32, s: u32) -> Self {
18        Self {
19            keysym,
20            unshifted: u,
21            shifted: s,
22        }
23    }
24    const fn ctrl(keysym: KeySym) -> Self {
25        Self {
26            keysym,
27            unshifted: 0,
28            shifted: 0,
29        }
30    }
31}

171/// Scancodes that update modifier state rather than producing a character.
172/// Returns `Some(Modifiers)` for the modifier this scancode controls.
173pub fn modifier_for_scancode(code: u32) -> Option<Modifiers> {
174    match code {
175        0x2A | 0x36 => Some(Modifiers::SHIFT), // Left/Right Shift
176        0x1D | 0x61 => Some(Modifiers::CTRL),  // Left/Right Ctrl
177        0x38 | 0x64 => Some(Modifiers::ALT),   // Left/Right Alt
178        0x7D | 0x7E => Some(Modifiers::META),  // Left/Right Meta
179        _ => None,
180    }
181}

183/// Returns true if this scancode is a sticky-toggle modifier (Caps/Num/Scroll Lock).
184pub fn toggle_modifier_for_scancode(code: u32) -> Option<Modifiers> {
185    match code {
186        0x3A => Some(Modifiers::CAPS_LOCK),
187        0x45 => Some(Modifiers::NUM_LOCK),
188        0x46 => Some(Modifiers::SCROLL_LOCK),
189        _ => None,
190    }
191}

193/// Translate a raw scancode + modifier state → (KeySym, Unicode char).
194///
195/// Returns `(KeySym::Unknown, 0)` for unrecognised scancodes.
196pub fn translate(code: u32, modifiers: Modifiers) -> (KeySym, u32) {
197    let idx = code as usize;
198    if idx >= LAYOUT.len() {
199        return (KeySym::Unknown, 0);
200    }
201    let entry = &LAYOUT[idx];
202    // Effective shift = Shift XOR CapsLock (for letters only; digits unaffected by CapsLock)
203    let shift_active =
204        modifiers.contains(Modifiers::SHIFT) ^ modifiers.contains(Modifiers::CAPS_LOCK);
205    let ch = if shift_active {
206        entry.shifted
207    } else {
208        entry.unshifted
209    };
210    (entry.keysym, ch)
211}

213/// Convert a `KeyState` from raw evdev value (0=release, 1=press, 2=repeat).
214pub fn key_state_from_evdev(value: u32) -> KeyState {
215    match value {
216        0 => KeyState::Released,
217        1 => KeyState::Pressed,
218        _ => KeyState::Repeated,
219    }
