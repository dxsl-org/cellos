6//! Security Silo Service Cell — hosts the bare-metal P-256 key guest.
7//!
8//! Spawns a minimal VM (32 KB guest RAM), loads the embedded silo-guest binary,
9//! seeds the guest key, then serves Sign / Ecdh / GetPub IPC requests.
10//!
11//! # Guest binary
12//! The embedded `silo-guest.bin` at `cells/services/silo/silo-guest.bin` is a
13//! placeholder (0 bytes) at compile time.  Build the real binary first:
14//!
15//!   cargo build --release \
16//!     --manifest-path cells/guests/silo-guest/Cargo.toml \
17//!     --target aarch64-unknown-none
18//!   llvm-objcopy -O binary \
19//!     cells/guests/silo-guest/target/aarch64-unknown-none/release/silo-guest \
20//!     cells/services/silo/silo-guest.bin
21//!
22//! Without the real binary the silo will log "guest binary empty" and return
23//! without entering the IPC loop — all sign/ecdh requests will timeout.
24

56/// Guest RAM: 32 KB (8 pages) — text (8 KB) + bss/stack (4 KB) + mailbox (4 KB) + spare.
57const GUEST_RAM_PAGES: usize = 8;
58/// Guest IPA base — matches the silo-guest linker script (0x40000000).
59const GUEST_IPA_BASE: u64 = 0x4000_0000;
60/// Total guest RAM in bytes.
61const GUEST_RAM_BYTES: usize = GUEST_RAM_PAGES * 4096;
62/// Guest entry PC — first byte of .text at the IPA base.
63const GUEST_ENTRY_PC: u64 = GUEST_IPA_BASE;
64
65// PLACEHOLDER: cells/services/silo/silo-guest.bin must be replaced with the
66// real stripped binary produced from cells/guests/silo-guest before this cell
67// can operate.  See the module doc above for the build command.
68static GUEST_BIN: &[u8] = include_bytes!("../silo-guest.bin");
69
70#[no_mangle]
71pub fn main() {
72    println("[silo] security silo service starting");
73
74    // ── Guard: placeholder binary ─────────────────────────────────────────────
75    if GUEST_BIN.is_empty() {
76        println("[silo] guest binary empty — build silo-guest.bin first (see module doc)");
77        return;
78    }
79    if GUEST_BIN.len() > GUEST_RAM_BYTES {
80        println("[silo] guest binary too large for guest RAM");
81        return;
82    }
83
84    // ── 1. Create VM ──────────────────────────────────────────────────────────
85    let vm_id = vmm::create_vm(GUEST_RAM_PAGES);
86    if vm_id == 0 || vm_id == usize::MAX {
87        println("[silo] create_vm failed — not at EL2 or OOM");
88        return;
89    }
90
91    // ── 2. Map guest RAM ──────────────────────────────────────────────────────
92    let ret = vmm::map_guest_memory(vm_id, GUEST_IPA_BASE, GUEST_RAM_BYTES, true);
93    if ret == usize::MAX {
94        println("[silo] map_guest_memory failed");
95        return;
96    }
97
98    // ── 3. Load flat guest binary ─────────────────────────────────────────────
99    let written = vmm::write_guest_memory(vm_id, GUEST_IPA_BASE, GUEST_BIN);
100    if written == usize::MAX {
101        println("[silo] write_guest_memory failed");
102        return;
103    }
104    println(&alloc::format!(
105        "[silo] guest loaded {} bytes",
106        GUEST_BIN.len()
107    ));
108
109    // ── 4. Create vCPU ────────────────────────────────────────────────────────
110    let vcpu_id = vmm::create_vcpu(vm_id, GUEST_ENTRY_PC);
111    if vcpu_id == 0 || vcpu_id == usize::MAX {
112        println("[silo] create_vcpu failed");
113        return;
114    }
115
116    // ── 5. Wait for guest init (HVC_SILO_READY) ───────────────────────────────
117    // The guest generates its P-256 key and fires HVC_SILO_READY when done.
118    match run_loop::run_until_done(vm_id, vcpu_id) {
119        run_loop::SiloRunResult::Done => {
120            println("[silo] guest initialised — key ready");
121        }
122        run_loop::SiloRunResult::Fault(code) => {
123            println(&alloc::format!("[silo] guest init fault: 0x{:x}", code));
124            return;
125        }
126        run_loop::SiloRunResult::GuestError => {
127            println("[silo] guest init error — aborting");
128            return;
129        }
130    }
131
132    println("[silo] entering IPC loop");
133
134    // ── 6. IPC server loop (never returns under normal operation) ─────────────
135    ipc::run(vm_id, vcpu_id);
136}
