1//! Low-level VMM syscall wrappers (220-226) for the silo service cell.
2//!
3//! ARM64-only at runtime — guarded by the HypervisorCap manifest bit.
4//! RISC-V stubs return a NotSupported sentinel so the code compiles on all targets.
5//!
6//! `unsafe` is required here for inline assembly (hardware syscall ABI).
7//! All other modules in this crate use `#![forbid(unsafe_code)]`.
8#![allow(unsafe_code)]
9
10use api::hypervisor::ViVmExit;
11use api::syscall::ViSyscall;
12
13/// Scheduler tick budget for each RunVcpu call (~10ms in 10 MHz ticks = 100_000 ticks).
14pub const SCHED_TICK_BUDGET_NS: u64 = 10_000_000; // 10ms in nanoseconds
15
16/// Error sentinel returned by VMM syscalls on failure.
17const ERR: usize = usize::MAX;
18
19// SAFETY: inline asm uses the ViCell ARM64 SVC ABI (x0=syscall_nr, x1-x4=args).
20// The kernel validates all pointer arguments before dereference; no Rust
21// invariants are violated by the register-only asm constraints used here.
22#[inline]
23unsafe fn syscall4(id: ViSyscall, a0: usize, a1: usize, a2: usize, a3: usize) -> usize {
24    #[allow(unused_mut)] // mut needed for aarch64 inlateout; not needed on other targets
25    let mut ret: usize;
26    #[cfg(target_arch = "aarch64")]
27    core::arch::asm!(
28        "svc #0",
29        inlateout("x0") id as usize => ret,
30        in("x1") a0, in("x2") a1, in("x3") a2, in("x4") a3,
31        options(nostack, preserves_flags),
32    );
33    #[cfg(not(target_arch = "aarch64"))]
34    {
35        let _ = (id, a0, a1, a2, a3);
36        ret = ERR;
37    }
38    ret
39}

41/// Allocate guest RAM + Stage-2 table; returns vm_id (> 0) or 0 on error.
42pub fn create_vm(guest_pages: usize) -> usize {
43    unsafe { syscall4(ViSyscall::CreateVm, guest_pages, 0, 0, 0) }
44}
45
46/// Create a vCPU with initial PC `entry_pc` in `vm_id`; returns vcpu_id or 0.
47pub fn create_vcpu(vm_id: usize, entry_pc: u64) -> usize {
48    unsafe { syscall4(ViSyscall::CreateVcpu, vm_id, entry_pc as usize, 0, 0) }
49}
50
51/// Map guest IPA range in `vm_id`; returns 0 on success.
52pub fn map_guest_memory(vm_id: usize, ipa: u64, size: usize, writable: bool) -> usize {
53    unsafe {
54        syscall4(
55            ViSyscall::MapGuestMemory,
56            vm_id,
57            ipa as usize,
58            size,
59            writable as usize,
60        )
61    }
62}
63
64/// Copy `src` bytes into guest RAM at `gpa`; returns bytes written or ERR.
65pub fn write_guest_memory(vm_id: usize, gpa: u64, src: &[u8]) -> usize {
66    // SAFETY: src is a valid slice; the kernel copies from it before returning.
67    unsafe {
68        syscall4(
69            ViSyscall::WriteGuestMemory,
70            vm_id,
71            gpa as usize,
72            src.as_ptr() as usize,
73            src.len(),
74        )
75    }
76}
77
78/// Copy `len` bytes from guest RAM at `gpa` into `dst`; returns bytes read or ERR.
79pub fn read_guest_memory(vm_id: usize, gpa: u64, dst: &mut [u8]) -> usize {
80    // SAFETY: dst is a valid mutable slice; the kernel writes into it before returning.
81    unsafe {
82        syscall4(
83            ViSyscall::ReadGuestMemory,
84            vm_id,
85            gpa as usize,
86            dst.as_mut_ptr() as usize,
87            dst.len(),
88        )
89    }
90}

92/// World-switch into `vcpu_id`; writes exit reason to `*exit`. Returns 0 or ERR.
93pub fn run_vcpu(vm_id: usize, vcpu_id: usize, exit: &mut ViVmExit) -> usize {
94    // SAFETY: exit is a valid mutable reference aligned to ViVmExit; the kernel
95    // writes a valid discriminant + fields before returning control to this cell.
96    unsafe {
97        syscall4(
98            ViSyscall::RunVcpu,
99            vm_id,
100            vcpu_id,
101            SCHED_TICK_BUDGET_NS as usize,
102            exit as *mut ViVmExit as usize,
103        )
104    }
105}
