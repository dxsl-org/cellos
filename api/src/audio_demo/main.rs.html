1//! Audio demo cell — proves the VirtIO sound path end-to-end.
2//!
3//! Generates a 3-tone arpeggio (A4–C#5–E5) as a square wave and plays it through
4//! `sys_audio_play`. PCM format is fixed by the ABI: signed 16-bit LE, 2 channels
5//! (interleaved L/R), 44100 Hz. Square wave is used so no `sin()` (libm) is
6//! needed — purely integer generation.
7
8#![no_std]
9#![no_main]
10#![allow(static_mut_refs)] // single-task cell — no data race on the tone buffer
11
12extern crate ostd;
13
14use ostd::io::println;
15use ostd::syscall::{sys_audio_play, sys_exit};
16
17api::declare_manifest!(block_io = false, network = false, spawn = false);
18api::declare_syscalls![Log, AudioPlay];
19
20const RATE: usize = 44100;
21/// 0.4 s per tone, stereo S16 → 4 bytes/frame.
22const FRAMES: usize = RATE * 2 / 5;
23const BUF_LEN: usize = FRAMES * 4;
24const AMP: i16 = 7000; // ~21% of full scale — audible but not harsh
25
26static mut TONE: [u8; BUF_LEN] = [0u8; BUF_LEN];
27
28/// Fill TONE with a `freq` Hz square wave (same sample on L and R).
29unsafe fn gen_square(freq: usize) {
30    let half = (RATE / (freq * 2)).max(1); // samples per half period
31    for i in 0..FRAMES {
32        let s: i16 = if (i / half).is_multiple_of(2) {
33            AMP
34        } else {
35            -AMP
36        };
37        let b = s.to_le_bytes();
38        let off = i * 4;
39        TONE[off] = b[0];
40        TONE[off + 1] = b[1]; // left
41        TONE[off + 2] = b[0];
42        TONE[off + 3] = b[1]; // right
43    }
44}
45
46#[no_mangle]
47pub extern "C" fn main() {
48    println("[audio-demo] playing A4-C#5-E5 arpeggio on VirtIO sound...");
49    for &freq in &[440usize, 554, 659] {
50        unsafe { gen_square(freq) };
51        let played = sys_audio_play(unsafe { &TONE });
52        if played == 0 {
53            println("[audio-demo] no sound device or play failed");
54            sys_exit(1);
55        }
56    }
57    println("[audio-demo] done — audio path OK");
58    sys_exit(0);
59}
