1//! Typed IPC request dispatch — decodes one `VfsRequest` and produces the
2//! `VfsResponse`, routing filesystem ops through the MountTable.
3//!
4//! Cross-cutting policy lives here, not in backends: AccessTable authorization,
5//! quota accounting (net-delta on overwrite), async-read pending table, and
6//! zero-copy grant I/O.
7
8use crate::manager::VfsManager;
9
10/// Handle one decoded request. `resp_buf` backs `VfsResponse::Data` payloads,
11/// so the response borrows it; callers encode before reusing the buffer.
12pub fn handle_request<'a>(
13    vfs: &mut VfsManager,
14    buf: &[u8; api::ipc::IPC_BUF_SIZE],
15    sender: usize,
16    resp_buf: &'a mut [u8; api::ipc::IPC_BUF_SIZE],
17) -> api::ipc::VfsResponse<'a> {
18    // Decode typed request; `take_from_bytes` tolerates trailing zeros in the
19    // 512-byte receive buffer left over from previous messages.
20    let req = match api::ipc::decode::<api::ipc::VfsRequest>(buf) {
21        Ok(r) => r,
22        Err(_) => return api::ipc::VfsResponse::Err(0xFF), // malformed request
23    };
24
25    match req {
26        api::ipc::VfsRequest::GetFile(p) => {
27            if let Some((ptr, len)) = vfs.get_file_ptr(p) {
28                api::ipc::VfsResponse::DataPtr {
29                    ptr: ptr as u64,
30                    len: len as u64,
31                }
32            } else {
33                api::ipc::VfsResponse::Err(1)
34            }
35        }
36
37        api::ipc::VfsRequest::ListDir(p) => {
38            let n = vfs.list_dir(p, resp_buf);
39            api::ipc::VfsResponse::Data(&resp_buf[..n])
40        }
41
42        api::ipc::VfsRequest::Stat(p) => match vfs.stat(p) {
43            Some((size, is_dir)) => api::ipc::VfsResponse::Stat { size, is_dir },
44            None => api::ipc::VfsResponse::Err(1),
45        },
46
47        api::ipc::VfsRequest::Write { path, content } => {
48            let owner = types::CellId(sender as u64);
49            // Access check: only authorized cells may write to this path.
50            if !vfs.access.can_write(owner, path) {
51                return api::ipc::VfsResponse::Err(3); // 3 = PermissionDenied
52            }
53            // Capture size of any existing file to release its quota share.
54            // Overwriting an existing file should charge the delta, not the
55            // full new size — otherwise repeated overwrites inflate usage.
56            let old_size = vfs.file_size(path);
57            let new_size = content.len() as u64;
58            // Net quota delta: may be negative (file shrunk) or positive.
59            let net_charge = new_size.saturating_sub(old_size);
60            if net_charge > 0 && !vfs.quota.can_charge(owner, net_charge) {
61                return api::ipc::VfsResponse::Err(2); // 2 = quota exceeded
62            }
63            if vfs.write(path, content) {
64                // Release old bytes and charge new size.
65                vfs.quota.release(owner, old_size);
66                let _ = vfs.quota.charge(owner, new_size);
67                api::ipc::VfsResponse::Ok
68            } else {
69                api::ipc::VfsResponse::Err(1)
70            }
71        }
72
73        api::ipc::VfsRequest::Append { path, content } => {
74            let owner = types::CellId(sender as u64);
75            if !vfs.access.can_write(owner, path) {
76                return api::ipc::VfsResponse::Err(3);
77            }
78            let append_len = content.len() as u64;
79            if !vfs.quota.can_charge(owner, append_len) {
80                return api::ipc::VfsResponse::Err(2); // quota exceeded
81            }
82            if vfs.append(path, content) {
83                let _ = vfs.quota.charge(owner, append_len);
84                api::ipc::VfsResponse::Ok
85            } else {
86                api::ipc::VfsResponse::Err(1)
87            }
88        }
89
90        api::ipc::VfsRequest::Mkdir(p) => {
91            let owner = types::CellId(sender as u64);
92            if !vfs.access.can_write(owner, p) {
93                api::ipc::VfsResponse::Err(3)
94            } else if vfs.mkdir(p) {
95                api::ipc::VfsResponse::Ok
96            } else {
97                api::ipc::VfsResponse::Err(1)
98            }
99        }
100
101        api::ipc::VfsRequest::Rmdir(p) => {
102            // Verifies the target IS a directory — POSIX ENOTDIR semantics.
103            if vfs.rmdir(p) {
104                api::ipc::VfsResponse::Ok
105            } else {
106                api::ipc::VfsResponse::Err(1)
107            }
108        }
109
110        api::ipc::VfsRequest::Unlink(p) => {
111            // Capture file size before deletion for quota release.
112            let file_size = vfs.file_size(p);
113            if vfs.unlink(p) {
114                // Release the quota that was charged when the file was written.
115                let owner = types::CellId(sender as u64);
116                vfs.quota.release(owner, file_size);
117                api::ipc::VfsResponse::Ok
118            } else {
119                api::ipc::VfsResponse::Err(1)
120            }
121        }
122
123        api::ipc::VfsRequest::RmdirRecursive(p) => {
124            let owner = types::CellId(sender as u64);
125            // Compute bytes to release BEFORE deletion: rmdir_recursive returns
126            // only bool (adding a bytes-freed value to FsBackend would be a Law 1
127            // ABI change).  Walk the subtree via list+file_size while it still exists.
128            let freed = collect_dir_bytes(vfs, p, 32);
129            if vfs.rmdir_recursive(p) {
130                vfs.quota.release(owner, freed);
131                api::ipc::VfsResponse::Ok
132            } else {
133                api::ipc::VfsResponse::Err(1)
134            }
135        }
136
137        api::ipc::VfsRequest::ReadAsync { path } => {
138            // Read file data synchronously (disk is still blocking in this backend).
139            // Store under a handle and return immediately — caller polls.
140            let data = vfs.read_to_vec(path);
141            let handle = vfs.pending.insert(data);
142            api::ipc::VfsResponse::PendingHandle(handle)
143        }
144
145        api::ipc::VfsRequest::Poll { handle } => {
146            // With a synchronous backend data is always ready on first poll.
147            match vfs.pending.poll(handle) {
148                Some(data) => {
149                    // Cap at 480, not resp_buf.len(): the reply must still fit
150                    // the 512-byte IPC frame AFTER the postcard envelope. A
151                    // full 512-byte payload made encode fail and the client
152                    // saw an empty reply (surfaced by /bin ELF reads).
153                    let n = data.len().min(480);
154                    resp_buf[..n].copy_from_slice(&data[..n]);
155                    api::ipc::VfsResponse::Data(&resp_buf[..n])
156                }
157                None => api::ipc::VfsResponse::Err(4), // 4 = stale/unknown handle
158            }
159        }
160
161        // ── Zero-Copy Grant I/O (Storage 2.0, Phase 02) ────────────────
162        api::ipc::VfsRequest::ReadGrant {
163            cap,
164            offset,
165            size,
166            grant,
167        } => {
168            // Validate: VFS must have been GrantShare'd access by the app.
169            match ostd::syscall::sys_grant_slice(grant) {
170                None => api::ipc::VfsResponse::Err(1), // no access
171                Some(ptr) => {
172                    // Look up the cap in the VFS handle table.
173                    let bytes = if let Some(entry) = vfs.handles.get_mut(api::cap::CapId(cap)) {
174                        let avail = entry.data_len.saturating_sub(offset as usize);
175                        let n = size.min(avail).min(4096);
176                        // SAFETY: data_ptr is a valid in-memory VAddr; ptr is a
177                        // kernel-allocated, identity-mapped grant buffer.
178                        unsafe {
179                            core::ptr::copy_nonoverlapping(
180                                (entry.data_ptr + offset as usize) as *const u8,
181                                ptr,
182                                n,
183                            );
184                        }
185                        n
186                    } else {
187                        0 // unknown cap — caller must register handle first
188                    };
189                    // F14: reply AFTER filling the buffer.
190                    api::ipc::VfsResponse::GrantDone { bytes }
191                }
192            }
193        }
194
195        api::ipc::VfsRequest::WriteGrant {
196            cap,
197            offset,
198            grant,
199            bytes,
200        } => {
201            // TODO(Phase 04): wire quota check here (can_charge → charge) using the
202            // same net-delta pattern as the Write arm before routing cap→path.
203            let _ = (cap, offset); // path routing via cap table deferred to Phase 04
204            match ostd::syscall::sys_grant_slice(grant) {
205                None => api::ipc::VfsResponse::Err(1),
206                Some(ptr) => {
207                    let n = bytes.min(4096);
208                    // SAFETY: ptr is a valid identity-mapped grant buffer filled
209                    // by the app before GrantShare + WriteGrant IPC.
210                    let data = unsafe { core::slice::from_raw_parts(ptr as *const u8, n) };
211                    // Phase 02 stub: data available in `data` slice.
212                    // Full routing via cap→path lookup deferred to Phase 04.
213                    let _ = data;
214                    // F14: GrantDone sent only AFTER reading the grant buffer
215                    // (ipc_call blocks, so app cannot free it prematurely).
216                    api::ipc::VfsResponse::GrantDone { bytes: n }
217                }
218            }
219        }
220
221        api::ipc::VfsRequest::ReadFileGrant { path, grant, max } => {
222            match ostd::syscall::sys_grant_slice(grant) {
223                None => api::ipc::VfsResponse::Err(1), // grant not shared to VFS
224                Some(ptr) => {
225                    // Resolve via the mount table (BinOverlay → cell-store for /bin),
226                    // then copy the WHOLE file into the caller's grant in one shot.
227                    let data = vfs.read_to_vec(path);
228                    let n = data.len().min(max);
229                    // SAFETY: ptr is the caller's identity-mapped grant, GrantShare'd
230                    // RW and (per `max`) large enough for n bytes; `data` is a fresh
231                    // owned Vec. The caller's ipc_call blocks until we reply, so it
232                    // cannot free the grant before this copy completes.
233                    unsafe {
234                        core::ptr::copy_nonoverlapping(data.as_ptr(), ptr, n);
235                    }
236                    api::ipc::VfsResponse::GrantDone { bytes: n }
237                }
238            }
239        }
240    }
241}

243/// Walk the subtree rooted at `path` and return the total bytes occupied by all
244/// regular files inside it, bounded to `depth` recursion levels.
245///
246/// Called by `RmdirRecursive` BEFORE the delete so quota can be released after.
247/// Uses 512-byte stack buffers per level (adequate for embedded directory sizes).
248fn collect_dir_bytes(vfs: &VfsManager, path: &str, depth: u8) -> u64 {
249    if depth == 0 {
250        return 0;
251    }
252    let mut scratch = [0u8; 512];
253    let n = vfs.list_dir(path, &mut scratch);
254    let listing = core::str::from_utf8(&scratch[..n]).unwrap_or("");
255    let base = path.trim_end_matches('/');
256    let mut total = 0u64;
257    for line in listing.split('\n') {
258        if let Some(name) = line.strip_prefix("f:") {
259            let mut child = alloc::string::String::with_capacity(base.len() + 1 + name.len());
260            child.push_str(base);
261            child.push('/');
262            child.push_str(name);
263            total += vfs.file_size(&child);
264        } else if let Some(name) = line.strip_prefix("d:") {
265            let mut child = alloc::string::String::with_capacity(base.len() + 1 + name.len());
266            child.push_str(base);
267            child.push('/');
268            child.push_str(name);
269            total += collect_dir_bytes(vfs, &child, depth - 1);
270        }
271    }
272    total
273}
