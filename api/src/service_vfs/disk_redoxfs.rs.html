1//! VirtIO block device adapter for RedoxFS — maps 4 KiB RedoxFS blocks to
2//! 512-byte VirtIO sectors on MBR partition P5 (`PART_SRV_BASE_LBA`).
3//!
4//! BLOCK_SIZE (redoxfs) = 4096 bytes = 8 × 512-byte sectors.
5//! block N on the RedoxFS volume → sectors PART_SRV_BASE_LBA + N*8 on disk.
6
7use crate::blk_router::{blk_read, blk_write};
8use api::disk::{PART_SRV_BASE_LBA, PART_SRV_SECTORS};
9use redox_syscall::error::{Error, Result, EIO};
10use redoxfs::BLOCK_SIZE;
11
12/// Sectors per one 4 KiB RedoxFS block.
13const SECTORS_PER_BLOCK: u64 = BLOCK_SIZE / 512;
14
15/// Zero-sized disk adapter — all filesystem state lives on the P5 partition.
16/// A fresh instance is equivalent to any other (no in-process cache).
17pub struct VicellDisk;
18
19impl redoxfs::Disk for VicellDisk {
20    unsafe fn read_at(&mut self, block: u64, buffer: &mut [u8]) -> Result<usize> {
21        let base = PART_SRV_BASE_LBA + block * SECTORS_PER_BLOCK;
22        let mut tmp = [0u8; 512];
23        for (i, chunk) in buffer.chunks_mut(512).enumerate() {
24            if !blk_read(base + i as u64, &mut tmp) {
25                return Err(Error::new(EIO));
26            }
27            let n = chunk.len();
28            chunk.copy_from_slice(&tmp[..n]);
29        }
30        Ok(buffer.len())
31    }
32
33    unsafe fn write_at(&mut self, block: u64, buffer: &[u8]) -> Result<usize> {
34        let base = PART_SRV_BASE_LBA + block * SECTORS_PER_BLOCK;
35        for (i, chunk) in buffer.chunks(512).enumerate() {
36            let mut tmp = [0u8; 512];
37            tmp[..chunk.len()].copy_from_slice(chunk);
38            if !blk_write(base + i as u64, &tmp) {
39                return Err(Error::new(EIO));
40            }
41        }
42        Ok(buffer.len())
43    }
44
45    fn size(&mut self) -> Result<u64> {
46        Ok(PART_SRV_SECTORS * 512)
47    }
48}
