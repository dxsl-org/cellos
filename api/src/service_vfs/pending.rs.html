1//! Pending async read table for the VFS service.
2//!
3//! Implements the server side of the two-opcode non-blocking read protocol:
4//!   1. `ReadAsync { path }` → VFS reads file data synchronously (disk is still
5//!      blocking), stores it under a handle, returns `PendingHandle(id)`.
6//!   2. `Poll { handle: id }` → returns `Data(bytes)` (always ready with
7//!      synchronous backend) or `Err` if the handle is stale/consumed.
8//!
9//! The protocol is correct regardless of the backend being synchronous: the
10//! caller-side loop with `yield_now()` cooperates correctly with the scheduler,
11//! and the API shape is ready for a future interrupt-driven block driver.
12
13use alloc::collections::BTreeMap;
14use alloc::vec::Vec;
15
16/// One pending async read slot.
17pub struct PendingRead {
18    /// Pre-read file contents.  Data is available immediately with the current
19    /// synchronous VirtIO block backend.
20    pub data: Vec<u8>,
21}
22
23/// Table of pending reads keyed by opaque handle ID.
24pub struct PendingTable {
25    slots: BTreeMap<u32, PendingRead>,
26    next_id: u32,
27}
28
29impl PendingTable {
30    pub fn new() -> Self {
31        Self {
32            slots: BTreeMap::new(),
33            next_id: 1,
34        }
35    }
36
37    /// Insert pre-read data and return the handle.
38    pub fn insert(&mut self, data: Vec<u8>) -> u32 {
39        let id = self.next_id;
40        self.next_id = self.next_id.wrapping_add(1).max(1); // skip 0
41        self.slots.insert(id, PendingRead { data });
42        id
43    }
44
45    /// Consume the data for `handle` — returns `None` if the handle is
46    /// stale (already polled) or was never issued.
47    pub fn poll(&mut self, handle: u32) -> Option<Vec<u8>> {
48        self.slots.remove(&handle).map(|p| p.data)
49    }
50}
51
52impl Default for PendingTable {
53    fn default() -> Self {
54        Self::new()
55    }
56}
