1//! `VfsManager` — mount table plus the cross-cutting service state (quota,
2//! access control, async-read handles, grant handle table).
3//!
4//! Backend routing is fully encapsulated here: dispatch code calls these
5//! delegates and never inspects path prefixes itself.

33impl VfsManager {
34    pub fn new() -> Self {
35        let mut mounts = MountTable::new();
36        let ram = mounts.add_backend(Box::new(RamFsBackend::new()));
37        // FAT32 (P1) is the SD-card/PC interop volume since P04 — /data moved
38        // to littlefs (P4), which survives power cuts (FAT has no journal).
39        let fat = mounts.add_backend(Box::new(FatBackend::mount(
40            "/mnt/sd",
41            api::disk::PART_FAT32_BASE_LBA,
42        )));
43        // /data (littlefs, P4) is gated on the `littlefs` feature: builds without a
44        // bare-metal C toolchain (x86_64/aarch64) omit it — the persistent /data
45        // volume is simply absent there, which boot-to-shell does not require.
46        #[cfg(feature = "littlefs")]
47        let lfs = mounts.add_backend(Box::new(LittlefsBackend::mount("/data")));
48        // /bin overlay: VIFS1 ramdisk (bootstrap cells) unioned with the on-disk
49        // FAT cell-store (non-bootstrap cells migrated off the raw P2 table).
50        let binov = mounts.add_backend(Box::new(BinOverlay::new(
51            api::disk::PART_CELLSTORE_BASE_LBA,
52        )));
53        // Longest prefix wins: the specific mounts shadow the read-only root.
54        mounts.mount("/", ram, false);
55        mounts.mount("/tmp", ram, true);
56        #[cfg(feature = "littlefs")]
57        mounts.mount("/data", lfs, true);
58        mounts.mount("/mnt/sd", fat, true);
59        mounts.mount("/bin", binov, false);
60        // /srv: RedoxFS CoW B-tree filesystem on MBR partition P5.
61        // Degrades gracefully to empty/false if P5 is unformatted (see
62        // docs/specs/09b-vfs-native-fs-adr.md and scripts/mksrv-img.sh).
63        let srv = mounts.add_backend(Box::new(RedoxFsBackend::mount("/srv")));
64        mounts.mount("/srv", srv, true);
65
66        Self {
67            mounts,
68            handles: HandleTable::new(),
69            // test-hooks: 1.1 KiB quota so vfs-test can hit the limit with
70            // 400-byte chunks (must fit within the 512-byte IPC buffer).
71            #[cfg(feature = "test-hooks")]
72            quota: QuotaTracker::with_limit(1100),
73            #[cfg(not(feature = "test-hooks"))]
74            quota: QuotaTracker::new(),
75            access: AccessTable::new(),
76            pending: PendingTable::new(),
77        }
78    }

152// ─── Hot-swap state transfer ───────────────────────────────────────────────────
153//
154// VFS serialises its quota table so per-cell byte-usage accounting survives a
155// live upgrade.  The handle table is NOT serialised — open handles are inherently
156// session-scoped and client cells reopen files after the swap completes.
157//
158// Wire format (little-endian, schema v1):
159//   [version: u32][cell_count: u32]
160//     [cell_id: u64][bytes_used: u64]...
161
162const VFS_SCHEMA_VERSION: u32 = 1;
163
164impl ViStateTransfer for VfsManager {
165    fn state_size(&self) -> usize {
166        4 + 4 + self.quota.entry_count() * 16 // version + count + (id,used) pairs
167    }
168
169    fn serialize_state(&self, buf: &mut [u8]) -> ViResult<usize> {
170        let needed = self.state_size();
171        if buf.len() < needed {
172            return Err(ViError::InvalidArgument);
173        }
174        let mut pos = 0;
175        buf[pos..pos + 4].copy_from_slice(&VFS_SCHEMA_VERSION.to_le_bytes());
176        pos += 4;
177        let entries = self.quota.all_entries();
178        buf[pos..pos + 4].copy_from_slice(&(entries.len() as u32).to_le_bytes());
179        pos += 4;
180        for (id, used) in &entries {
181            buf[pos..pos + 8].copy_from_slice(&id.to_le_bytes());
182            pos += 8;
183            buf[pos..pos + 8].copy_from_slice(&used.to_le_bytes());
184            pos += 8;
185        }
186        Ok(pos)
187    }
188
189    fn deserialize_state(&mut self, buf: &[u8]) -> ViResult<()> {
190        if buf.len() < 8 {
191            return Err(ViError::InvalidInput);
192        }
193        let _version = u32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
194        let count = u32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]) as usize;
195        let mut pos = 8;
196        for _ in 0..count {
197            if pos + 16 > buf.len() {
198                return Err(ViError::InvalidInput);
199            }
200            let id = u64::from_le_bytes(
201                buf[pos..pos + 8]
202                    .try_into()
203                    .map_err(|_| ViError::InvalidInput)?,
204            );
205            let used = u64::from_le_bytes(
206                buf[pos + 8..pos + 16]
207                    .try_into()
208                    .map_err(|_| ViError::InvalidInput)?,
209            );
210            self.quota.restore(types::CellId(id), used);
211            pos += 16;
212        }
213        Ok(())
214    }
