1//! BootFS proxy backend — serves `/bin` from the kernel's embedded initramfs
2//! (`VIFS1`, the FAT16 `kernel_fs.img` baked into the kernel binary).
3//!
4//! Replaces the VFS-side `include_bytes!` copies of the /bin ELFs, which
5//! duplicated every binary already present in `kernel_fs.img` (two copies of
6//! each ELF resident in RAM). The proxy reads through the kernel's FD
7//! syscalls (Open/Read/ReadDir/Close), which resolve against VIFS1 —
8//! see `kernel/src/task.rs::file_open` and `kernel/src/fs.rs`.
9//!
10//! Strictly read-only: every mutating op returns false before any syscall.
11//! `get_file_ptr` returns `None` (no syscalls) so the fast-IPC handler —
12//! which runs with interrupts disabled in the CALLER's context — never
13//! blocks inside this backend; readers fall back to the ReadAsync copy path.
14
15use alloc::vec::Vec;
16
17use crate::backend::FsBackend;
18use ostd::syscall::{sys_close, sys_close_cap, sys_open, sys_open_cap, sys_read_cap, sys_readdir};
19
20pub struct BootFsProxy;

22/// RAII guard: kernel FDs live in the VFS task's TCB; leaking them across
23/// requests would exhaust the per-task handle map.
24struct Fd(usize);
25impl Drop for Fd {
26    fn drop(&mut self) {
27        sys_close(self.0);
28    }
29}
30
31fn open(path: &str) -> Option<Fd> {
32    sys_open(path).ok().map(Fd)
33}
34
35/// Look up one entry of `parent` by name via ReadDir. `(size, is_dir)`.
36fn dir_lookup(parent: &str, name: &str) -> Option<(u64, bool)> {
37    let fd = open(parent)?;
38    while let Ok(Some(e)) = sys_readdir(fd.0) {
39        let len = e.name.iter().position(|&b| b == 0).unwrap_or(e.name.len());
40        if &e.name[..len] == name.as_bytes() {
41            let is_dir = matches!(e.file_type, types::FileType::Directory);
42            return Some((e.size, is_dir));
43        }
44    }
45    None
46}

48impl FsBackend for BootFsProxy {
49    fn get_file_ptr(&self, _path: &str) -> Option<(usize, usize)> {
50        None // no stable user-visible pointer; callers use the ReadAsync copy path
51    }
52
53    fn list(&self, path: &str, out: &mut [u8]) -> usize {
54        let fd = match open(path) {
55            Some(f) => f,
56            None => return 0,
57        };
58        let mut pos = 0;
59        while let Ok(Some(e)) = sys_readdir(fd.0) {
60            let len = e.name.iter().position(|&b| b == 0).unwrap_or(e.name.len());
61            if len == 0 {
62                continue;
63            }
64            let name = &e.name[..len];
65            if name == b"." || name == b".." {
66                continue;
67            }
68            let prefix: &[u8] = if matches!(e.file_type, types::FileType::Directory) {
69                b"d:"
70            } else {
71                b"f:"
72            };
73            let entry_len = 2 + len + 1;
74            if pos + entry_len > out.len() {
75                break;
76            }
77            out[pos..pos + 2].copy_from_slice(prefix);
78            out[pos + 2..pos + 2 + len].copy_from_slice(name);
79            out[pos + 2 + len] = b'\n';
80            pos += entry_len;
81        }
82        pos
83    }
84
85    fn stat(&self, path: &str) -> Option<(u64, bool)> {
86        // Mount roots ("/bin") are directories by definition; children are
87        // resolved through the parent's directory entries (file_fstat is a
88        // kernel stub, and DirEntry already carries size + type).
89        let trimmed = path.trim_end_matches('/');
90        let slash = trimmed.rfind('/')?;
91        let name = &trimmed[slash + 1..];
92        if name.is_empty() {
93            return Some((0, true));
94        }
95        let parent = if slash == 0 { "/" } else { &trimmed[..slash] };
96        match dir_lookup(parent, name) {
97            Some(hit) => Some(hit),
98            // The mount prefix itself may not appear in its parent's listing
99            // (e.g. stat "/bin" when VIFS1 only lists from "/") — probe by open.
100            None if open(trimmed).is_some() => Some((0, true)),
101            None => None,
102        }
103    }
104
105    fn file_size(&self, path: &str) -> u64 {
106        self.stat(path).map(|(s, _)| s).unwrap_or(0)
107    }
108
109    /// File reads go through the SYNCHRONOUS cap path (OpenCap/ReadCap), NOT
110    /// the FD `Read` syscall: `file_read(fd > 2)` is an async transformation —
111    /// it parks the task as `Polling` and the scheduler sweep writes the result
112    /// into the saved trap frame later. That contract requires the caller to
113    /// stop running immediately after the syscall; a busy read loop here kept
114    /// executing, the sweep clobbered a live trap frame, and the VFS cell died
115    /// with a zeroed-frame fault (scause=0).
116    fn read_to_vec(&self, path: &str) -> Vec<u8> {
117        // FAT16 stores 8.3 names uppercase; OpenCap matches the raw path string
118        // (unlike the loader's read_file_from_vifs1, which uppercases first).
119        let upper: alloc::string::String = path.chars().map(|c| c.to_ascii_uppercase()).collect();
120        let cap = match sys_open_cap(&upper) {
121            Ok(c) => c,
122            Err(_) => return Vec::new(),
123        };
124        let mut buf = [0u8; 512];
125        let mut result = Vec::new();
126        loop {
127            match sys_read_cap(cap, &mut buf) {
128                Ok(0) => break,
129                Ok(n) => result.extend_from_slice(&buf[..n]),
130                Err(_) => break,
131            }
132        }
133        sys_close_cap(cap);
134        result
135    }
136
137    fn write(&mut self, _path: &str, _content: &[u8]) -> bool {
138        false
139    }
140    fn append(&mut self, _path: &str, _content: &[u8]) -> bool {
141        false
142    }
143    fn mkdir(&mut self, _path: &str) -> bool {
144        false
145    }
146    fn rmdir(&mut self, _path: &str) -> bool {
147        false
148    }
149    fn unlink(&mut self, _path: &str) -> bool {
150        false
151    }
152    fn rmdir_recursive(&mut self, _path: &str) -> bool {
153        false
154    }
