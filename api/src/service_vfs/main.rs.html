51// Narrow syscall allowlist — kernel enforces this at dispatch (Phase 27).
52// BootFS proxy (/bin via the kernel initramfs VIFS1): Open/Close/ReadDir for
53// listing (all synchronous), OpenCap/ReadCap/CloseCap for file reads — the FD
54// `Read` syscall is deliberately ABSENT: it is an async transformation that
55// requires the caller to park immediately, which a service dispatch loop
56// cannot do (see backend_bootfs.rs::read_to_vec).
57api::declare_syscalls![
58    Send,
59    Recv,
60    TryRecv,
61    Reply,
62    Log,
63    Heartbeat,
64    LookupService,
65    GrantAlloc,
66    GrantShare,
67    GrantSlice,
68    GrantFree,
69    BlkReadAsync,
70    GrantRegister,
71    GrantUnregister,
72    StateStash,
73    StateRestore,
74    Open,
75    Close,
76    ReadDir,
77    OpenCap,
78    ReadCap,
79    CloseCap,
80    // NOTE: deliberately NO SetTimer. VFS never calls it — a "SetTimer (bit 11)
81    // denied for tid <vfs>" kernel warn on x86 is the CANARY for the known x86
82    // syscall-redispatch corruption (syscall number read as user CS = 0x23 = 35
83    // with a pointer as the tick count; see TODO #9 / x86 q35 P02). Allowing it
84    // turns that corruption into an unbounded sleep that hangs the boot.
85];

87// Global VFS manager for the fast-IPC handler (which runs outside the main recv loop).
88// Protected by a spinlock; on single-hart there is no actual contention.
89static GLOBAL_VFS: Mutex<Option<VfsManager>> = Mutex::new(None);
90
91/// Fast-IPC handler: serves VfsRequest::GetFile without ecall overhead.
92///
93/// # Safety
94/// Called with S-mode interrupts disabled (guaranteed by `ostd::fast_ipc::call_vfs`).
95unsafe fn vfs_fast_handler(
96    req: &api::ipc::VfsRequest<'_>,
97    out: &mut [u8; api::ipc::IPC_BUF_SIZE],
98) -> usize {
99    let resp = match req {
100        api::ipc::VfsRequest::GetFile(path) => {
101            if let Some(vfs) = GLOBAL_VFS.lock().as_ref() {
102                if let Some((ptr, len)) = vfs.get_file_ptr(path) {
103                    api::ipc::VfsResponse::DataPtr {
104                        ptr: ptr as u64,
105                        len: len as u64,
106                    }
107                } else {
108                    api::ipc::VfsResponse::Err(1)
109                }
110            } else {
111                api::ipc::VfsResponse::Err(0xFF)
112            }
113        }
114        _ => api::ipc::VfsResponse::Err(0xFE), // other ops must use ecall path
115    };
116    api::ipc::encode(&resp, out).map(|s| s.len()).unwrap_or(0)
117}

119#[no_mangle]
120pub fn main() {
121    println("VFS Service v0.2: RamFS + mkdir/rmdir/unlink IPC (typed postcard)");
122    // VfsManager::new() mounts all backends, including the FAT volume on the
123    // VirtIO disk (which logs its own success/fallback status).
124    let vfs = VfsManager::new();
125    *GLOBAL_VFS.lock() = Some(vfs);
126
127    // Register the fast-IPC handler so trusted Cells can bypass ecall for VFS reads.
128    // The kernel records the VFS cell's ID at spawn time so it can clear this
129    // pointer if VFS crashes — see loader.rs fast_ipc::set_vfs_handler_cell call.
130    ostd::fast_ipc::register_vfs(vfs_fast_handler);
131    let mut buf = [0u8; api::ipc::IPC_BUF_SIZE];
132
133    loop {
134        match ostd::syscall::sys_recv(0, &mut buf) {
135            ostd::syscall::SyscallResult::Ok(sender) if sender > 0 => {
136                // Encode the response into a local buffer while holding the VFS lock,
137                // then DROP the lock before sys_send.  If ipc_send blocks (client not
138                // yet in Recv), yield_cpu switches to another cell.  That cell may call
139                // call_vfs which also acquires GLOBAL_VFS — a deadlock if we still hold
140                // the lock during the send.
141                let mut encoded = [0u8; api::ipc::IPC_BUF_SIZE];
142                let encoded_len: usize;
143                {
144                    let mut resp_buf = [0u8; api::ipc::IPC_BUF_SIZE];
145                    // Acquire VFS state; released at end of this block, before sys_send.
146                    let mut gvfs = GLOBAL_VFS.lock();
147                    let vfs = gvfs
148                        .as_mut()
149                        .expect("VFS initialized before serving requests");
150                    let resp = dispatch::handle_request(vfs, &buf, sender, &mut resp_buf);
151                    // Encode while holding the lock (safe: no sys_send yet).
152                    encoded_len = api::ipc::encode(&resp, &mut encoded)
153                        .map(|s| s.len())
154                        .unwrap_or(0);
155                } // GLOBAL_VFS lock released here — before sys_send
156
157                // Send after releasing the lock so a blocked ipc_send + yield_cpu
158                // cannot switch to a cell that deadlocks on GLOBAL_VFS.
159                ostd::syscall::sys_send(sender, &encoded[..encoded_len]);
160                buf = [0u8; api::ipc::IPC_BUF_SIZE];
161            }
162            _ => {
163                ostd::task::yield_now();
164            }
165        }
166    }
