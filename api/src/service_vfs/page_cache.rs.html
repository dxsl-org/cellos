1//! LRU sector cache for the VFS block stream.
2//!
3//! Lives entirely inside the VFS cell — one owner, no lock contention.
4//! Cache budget is a compile-time constant; change and recompile for G2.
5
6use crate::block_stream::BlockStream;
7use alloc::collections::{BTreeMap, VecDeque};
8
9/// Sector cache budget — 4 MB = 8192 512-byte sectors.
10/// G2 deployments: change this const and recompile. No feature flag needed.
11const MAX_CACHE_BYTES: usize = 4 * 1024 * 1024;
12
13struct CachedSector {
14    data: [u8; 512],
15    dirty: bool,
16}
17
18/// LRU sector cache keyed by sector number.
19///
20/// Write policy: write-through on FAT32 (no journal) — every `write_sector`
21/// flushes dirty entries to disk immediately. Switch to write-back when
22/// viFS2 (WAL) becomes the backend.
23pub struct PageCache {
24    entries: BTreeMap<u64, CachedSector>,
25    lru_order: VecDeque<u64>,
26    total_bytes: usize,
27    max_bytes: usize,
28}
29
30impl PageCache {
31    pub fn new() -> Self {
32        Self {
33            entries: BTreeMap::new(),
34            lru_order: VecDeque::new(),
35            total_bytes: 0,
36            max_bytes: MAX_CACHE_BYTES,
37        }
38    }
39
40    /// Serve `sector` from cache; fall back to a raw disk read on miss.
41    pub fn read_sector(&mut self, dev: &mut BlockStream, sector: u64, buf: &mut [u8; 512]) -> bool {
42        // Copy data out so we don't hold a borrow into self.entries past this point.
43        let cached = self.entries.get(&sector).map(|e| e.data);
44        if let Some(data) = cached {
45            buf.copy_from_slice(&data);
46            self.touch(sector);
47            return true;
48        }
49        // Cache miss: read from disk, then populate cache.
50        if !dev.read_raw_sector(sector, buf) {
51            return false;
52        }
53        self.insert(sector, buf, false);
54        true
55    }
56
57    /// Write `data` to cache (marked dirty) and flush synchronously.
58    ///
59    /// Write-through for FAT32: durability on every write avoids silent data
60    /// loss since FAT has no journal to recover from a torn write.
61    pub fn write_sector(&mut self, dev: &mut BlockStream, sector: u64, data: &[u8; 512]) -> bool {
62        self.insert(sector, data, true);
63        self.flush_dirty(dev)
64    }
65
66    /// Write all dirty entries to disk and clear their dirty flags.
67    pub fn flush_dirty(&mut self, dev: &mut BlockStream) -> bool {
68        for (&sector, entry) in self.entries.iter_mut() {
69            if entry.dirty {
70                if !dev.write_raw_sector(sector, &entry.data) {
71                    return false;
72                }
73                entry.dirty = false;
74            }
75        }
76        true
77    }
78
79    fn insert(&mut self, sector: u64, data: &[u8; 512], dirty: bool) {
80        if self.entries.contains_key(&sector) {
81            // Update in-place: no eviction, no total_bytes change.
82            let e = self.entries.get_mut(&sector).unwrap();
83            e.data.copy_from_slice(data);
84            e.dirty |= dirty;
85            self.touch(sector);
86            return;
87        }
88        // Evict LRU entries until we have room. Eviction kicks in at 90% capacity
89        // (max_bytes * 9/10 ≈ 3.6MB for the default 4MB cache); this leaves headroom
90        // to avoid thrashing on the boundary. Effective live size = ~3.6MB, not 4MB.
91        while self.total_bytes + 512 > self.max_bytes * 9 / 10 {
92            if let Some(lru) = self.lru_order.pop_back() {
93                if let Some(e) = self.entries.remove(&lru) {
94                    // Write-through invariant: dirty entries are always flushed
95                    // in write_sector before we can hit capacity here.
96                    debug_assert!(!e.dirty, "dirty eviction without prior flush");
97                    self.total_bytes -= 512;
98                }
99            } else {
100                break;
101            }
102        }
103        self.entries
104            .insert(sector, CachedSector { data: *data, dirty });
105        self.lru_order.push_front(sector);
106        self.total_bytes += 512;
107    }
108
109    fn touch(&mut self, sector: u64) {
110        // O(n) scan — acceptable for ≤8192 entries (~32 KB of sector numbers).
111        self.lru_order.retain(|&s| s != sector);
112        self.lru_order.push_front(sector);
113    }
114}
