4//! GPIO Driver Cell — ARM PL061 on QEMU ARM virt.
5//!
6//! Base address: 0x0903_0000 (ARM virt machine, 4 KiB region).
7//!
8//! PL061 GPIODATA uses address-masked access:
9//! - Read all 8 pins: offset 0x3FC (mask = 0xFF in addr\[9:2\])
10//! - Write pin N: offset = 1 << (N+2), value = bit_N_high_or_zero
11//!
12//! GPIODIR: offset 0x400 (bit = 1 → output).
13//! Interrupt registers: 0x404–0x41C (deferred to v2).
14
15extern crate alloc;
16
17use hal_gpio::{Edge, PinDir, ViGpio};
18use ostd::mmio::{request_region, MmioRegion};
19use types::{ViError, ViResult};
20
21/// QEMU ARM virt PL061 MMIO base and size.
22pub const PL061_BASE: usize = 0x0903_0000;
23pub const PL061_SIZE: usize = 0x1000;
24
25// Register offsets
26const GPIODATA_ALL: usize = 0x3FC; // read all 8 pins
27const GPIODIR: usize = 0x400; // direction: 1 = output
28const GPIOIE: usize = 0x410; // interrupt enable
29const GPIOIS: usize = 0x404; // interrupt sense: 0=edge, 1=level
30const GPIOIBE: usize = 0x408; // both edges
31const GPIOIEV: usize = 0x40C; // event: 0=falling, 1=rising
32const GPIOIC: usize = 0x41C; // interrupt clear (write 1 to clear)
33const GPIOMIS: usize = 0x418; // masked interrupt status (read-only)
34
35/// PL061 GPIO controller for QEMU ARM virt.
36pub struct Pl061Gpio {
37    mmio: MmioRegion,
38}
39
40impl Pl061Gpio {
41    /// Acquire exclusive GPIO MMIO access from the kernel.
42    ///
43    /// Fails if the manifest did not declare `gpio = true`, or if another
44    /// Cell already owns the PL061 range.
45    pub fn open() -> ViResult<Self> {
46        let mmio = request_region(PL061_BASE, PL061_SIZE)?;
47        Ok(Self { mmio })
48    }
49
50    // ---
51
52    /// Read the raw 8-bit GPIODATA (all pins).
53    fn read_all(&self) -> ViResult<u8> {
54        let v = self.mmio.read_u32(GPIODATA_ALL)?;
55        Ok(v as u8)
56    }
57
58    /// Write a masked subset of GPIODATA.
59    ///
60    /// `pin_mask`: which pins to affect (addr bits [9:2])
61    /// `value`:    new values for those pins (data bits)
62    fn write_masked(&self, pin: u8, high: bool) -> ViResult<()> {
63        // Address-masked write: offset encodes which pin(s) to affect
64        let offset = (1usize << (pin + 2)) & 0x3FF; // addr bits [9:2]
65        let data: u32 = if high { 1u32 << pin } else { 0 };
66        self.mmio.write_u32(offset, data)
67    }
68
69    /// Return the masked interrupt status register (GPIOMIS).
70    ///
71    /// Each set bit indicates a pin whose edge IRQ fired AND was enabled.
72    /// Call this after receiving a `GPIO_IRQ_NOTIFY` (opcode 0xA0) from the kernel
73    /// to discover which pins triggered, then call `clear_irq(mask)` to ACK them.
74    pub fn read_mis(&self) -> ViResult<u8> {
75        Ok(self.mmio.read_u32(GPIOMIS)? as u8)
76    }
77
78    /// Acknowledge (clear) GPIO interrupts for the pins in `mask`.
79    ///
80    /// Writes GPIOIC without touching GPIOIE, so the interrupt remains enabled
81    /// for the next edge.  Call after reading GPIOMIS and processing all set bits.
82    pub fn clear_irq(&mut self, mask: u8) -> ViResult<()> {
83        self.mmio.write_u32(GPIOIC, u32::from(mask))
84    }
85}
86
87impl ViGpio for Pl061Gpio {
88    fn set_direction(&mut self, pin: u8, dir: PinDir) -> ViResult<()> {
89        if pin >= 8 {
90            return Err(ViError::InvalidInput);
91        }
92        let current = self.mmio.read_u32(GPIODIR)? as u8;
93        let new_val: u32 = match dir {
94            PinDir::Output => (current | (1 << pin)) as u32,
95            PinDir::Input => (current & !(1 << pin)) as u32,
96        };
97        self.mmio.write_u32(GPIODIR, new_val)
98    }
99
100    fn read_pin(&self, pin: u8) -> ViResult<bool> {
101        if pin >= 8 {
102            return Err(ViError::InvalidInput);
103        }
104        let all = self.read_all()?;
105        Ok(all & (1 << pin) != 0)
106    }
107
108    fn write_pin(&mut self, pin: u8, high: bool) -> ViResult<()> {
109        if pin >= 8 {
110            return Err(ViError::InvalidInput);
111        }
112        // Verify pin is configured as output
113        let dir = self.mmio.read_u32(GPIODIR)? as u8;
114        if dir & (1 << pin) == 0 {
115            return Err(ViError::InvalidInput); // not an output
116        }
117        self.write_masked(pin, high)
118    }
119
120    fn enable_edge_irq(&mut self, pin: u8, edge: Edge) -> ViResult<()> {
121        if pin >= 8 {
122            return Err(ViError::InvalidInput);
123        }
124        let mask = 1u32 << pin;
125
126        // Edge-triggered (not level): GPIOIS bit = 0
127        let is_val = self.mmio.read_u32(GPIOIS)? & !mask;
128        self.mmio.write_u32(GPIOIS, is_val)?;
129
130        match edge {
131            Edge::Both => {
132                // Both edges: GPIOIBE bit = 1 (overrides GPIOIEV)
133                let ibe = self.mmio.read_u32(GPIOIBE)? | mask;
134                self.mmio.write_u32(GPIOIBE, ibe)?;
135            }
136            Edge::Rising | Edge::Falling => {
137                // Single edge: GPIOIBE bit = 0, GPIOIEV selects polarity
138                let ibe = self.mmio.read_u32(GPIOIBE)? & !mask;
139                self.mmio.write_u32(GPIOIBE, ibe)?;
140                let iev = self.mmio.read_u32(GPIOIEV)?;
141                let iev_new = if matches!(edge, Edge::Rising) {
142                    iev | mask
143                } else {
144                    iev & !mask
145                };
146                self.mmio.write_u32(GPIOIEV, iev_new)?;
147            }
148        }
149
150        // Enable interrupt for this pin
151        let ie = self.mmio.read_u32(GPIOIE)? | mask;
152        self.mmio.write_u32(GPIOIE, ie)
153    }
154
155    fn disable_irq(&mut self, pin: u8) -> ViResult<()> {
156        if pin >= 8 {
157            return Err(ViError::InvalidInput);
158        }
159        // Clear interrupt enable and pending interrupt for this pin
160        let ie = self.mmio.read_u32(GPIOIE)? & !(1u32 << pin);
161        self.mmio.write_u32(GPIOIE, ie)?;
162        self.mmio.write_u32(GPIOIC, 1u32 << pin)
163    }
164}
