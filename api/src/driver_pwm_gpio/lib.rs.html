4//! Bit-bang PWM over a GPIO pin.
5//!
6//! Generic over any `G: ViGpio`. Concrete use:
7//! `BitBangPwm::<Pl061Gpio>::new(gpio)` — caller opens the GPIO device first.
8//!
9//! Timing model: counter-based (no wall-clock dependency). The caller drives
10//! `tick()` in a tight pinned-poll loop (spec 13-peripherals §6). Each call
11//! increments an internal counter; pins are toggled when counter thresholds
12//! cross the configured duty/period boundary.
13//!
14//! `set_frequency(channel, hz)` converts `hz` to a period in tick-counts
15//! assuming the caller drives `tick()` at `ASSUMED_TICKS_PER_SEC` iterations/sec.
16//! On QEMU TCG the actual loop rate varies; `hz` controls the duty ratio only —
17//! not the precise output frequency.
18
19use hal_gpio::{PinDir, ViGpio};
20use hal_pwm::{PwmError, ViPwm};
21
22/// Assumed tick() calls per second for frequency→period conversion.
23/// This only affects how Hz maps to counter counts, not output correctness.
24const ASSUMED_TICKS_PER_SEC: u32 = 10_000;
25
26/// Maximum channels (one per GPIO pin).
27const MAX_CHANNELS: usize = 8;
28
29/// Per-channel waveform state.
30struct Channel {
31    pin: u8,
32    /// Total counter count per waveform period.
33    period_ticks: u32,
34    /// Counter count for the HIGH portion of each period.
35    high_ticks: u32,
36    /// Current position within the period (0..period_ticks).
37    counter: u32,
38    /// Current output level.
39    high: bool,
40    enabled: bool,
41}
42
43impl Channel {
44    const fn idle(pin: u8) -> Self {
45        Self {
46            pin,
47            period_ticks: 1,
48            high_ticks: 0,
49            counter: 0,
50            high: false,
51            enabled: false,
52        }
53    }
54}
55
56/// Bit-bang PWM master backed by a `ViGpio` implementation.
57pub struct BitBangPwm<G: ViGpio> {
58    gpio: G,
59    channels: [Channel; MAX_CHANNELS],
60}
61
62impl<G: ViGpio> BitBangPwm<G> {
63    /// Take ownership of `gpio` and prepare it as a PWM source.
64    pub fn new(gpio: G) -> Self {
65        Self {
66            gpio,
67            channels: [
68                Channel::idle(0),
69                Channel::idle(1),
70                Channel::idle(2),
71                Channel::idle(3),
72                Channel::idle(4),
73                Channel::idle(5),
74                Channel::idle(6),
75                Channel::idle(7),
76            ],
77        }
78    }
79
80    /// Release the underlying GPIO resource back to the caller.
81    pub fn into_gpio(self) -> G {
82        self.gpio
83    }
84}
85
86impl<G: ViGpio> ViPwm for BitBangPwm<G> {
87    type Error = PwmError;
88
89    fn set_frequency(&mut self, channel: u8, hz: u32) -> Result<(), PwmError> {
90        if channel as usize >= MAX_CHANNELS {
91            return Err(PwmError::InvalidChannel);
92        }
93        if hz == 0 || hz > 50_000 {
94            return Err(PwmError::InvalidFrequency);
95        }
96        let ch = &mut self.channels[channel as usize];
97        let period = (ASSUMED_TICKS_PER_SEC / hz).max(1);
98        // Recompute high_ticks to preserve duty ratio across frequency changes.
99        let duty = ch
100            .high_ticks
101            .checked_mul(1000)
102            .and_then(|n| n.checked_div(ch.period_ticks))
103            .unwrap_or(0);
104        ch.period_ticks = period;
105        ch.high_ticks = (period * duty / 1000).max(if duty > 0 { 1 } else { 0 });
106        Ok(())
107    }
108
109    fn set_duty(&mut self, channel: u8, duty_per_mille: u16) -> Result<(), PwmError> {
110        if channel as usize >= MAX_CHANNELS {
111            return Err(PwmError::InvalidChannel);
112        }
113        if duty_per_mille > 1000 {
114            return Err(PwmError::InvalidDuty);
115        }
116        let ch = &mut self.channels[channel as usize];
117        ch.high_ticks = (ch.period_ticks * duty_per_mille as u32 / 1000)
118            .max(if duty_per_mille > 0 { 1 } else { 0 });
119        Ok(())
120    }
121
122    fn enable(&mut self, channel: u8) -> Result<(), PwmError> {
123        if channel as usize >= MAX_CHANNELS {
124            return Err(PwmError::InvalidChannel);
125        }
126        let pin = self.channels[channel as usize].pin;
127        self.gpio
128            .set_direction(pin, PinDir::Output)
129            .map_err(|_| PwmError::InvalidChannel)?;
130        self.channels[channel as usize].counter = 0;
131        self.channels[channel as usize].enabled = true;
132        Ok(())
133    }
134
135    fn disable(&mut self, channel: u8) -> Result<(), PwmError> {
136        if channel as usize >= MAX_CHANNELS {
137            return Err(PwmError::InvalidChannel);
138        }
139        let ch = &mut self.channels[channel as usize];
140        ch.enabled = false;
141        ch.high = false;
142        let pin = ch.pin;
143        // Best-effort drive LOW; ignore if GPIO already released.
144        let _ = self.gpio.write_pin(pin, false);
145        Ok(())
146    }
147
148    fn tick(&mut self) {
149        for ch in self.channels.iter_mut() {
150            if !ch.enabled {
151                continue;
152            }
153            ch.counter = ch.counter.wrapping_add(1);
154            if ch.counter >= ch.period_ticks {
155                ch.counter = 0;
156            }
157            let should_be_high = ch.counter < ch.high_ticks;
158            if should_be_high != ch.high {
159                ch.high = should_be_high;
160                // Best-effort; ignore GPIO errors in hot loop.
161                let _ = self.gpio.write_pin(ch.pin, should_be_high);
162            }
163        }
164    }
165}
