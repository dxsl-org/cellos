1//! Hotswap orchestration ported from kernel/src/cell/hotswap.rs.
2//!
3//! Kernel mechanisms replaced by SupervisorCap syscalls:
4//!   set_task_frozen(tid)   → sys_freeze_cell(tid)
5//!   unfreeze_task(tid)     → sys_resume_cell(tid)
6//!   exit_task_internal(..) → sys_kill_cell(tid, 0xAAAA_AAAA)
7//!
8//! Polling replaced by yield-and-retry loops using sys_yield() + sys_state_restore()
9//! and sys_query_hotswap_ready().  The timeouts are calibrated to scheduler tick rate
10//! (one tick ≈ 10 ms) — HOTSWAP_MAX_ITERS×YIELD_COST_TICKS ≈ 5 seconds.
11
12extern crate alloc;
13
14use crate::error::HotswapError;
15use ostd::syscall::{
16    sys_freeze_cell, sys_kill_cell, sys_lookup_service, sys_query_hotswap_ready,
17    sys_register_service, sys_resume_cell, sys_send, sys_spawn_from_path, sys_state_restore,
18    sys_state_stash_clear, sys_yield,
19};
20
21/// Maximum poll iterations while waiting for stash/ready (≈ 5 s at 10 ms/tick).
22const MAX_ITERS: u32 = 500;
23
24// ── IPC envelope byte constants (must match kernel hotswap.rs) ───────────────
25
26const APP_MSG_MAGIC: u8 = 0xAC;
27const DISC_SNAPSHOT: u8 = 0xF0; // AppEvent::Snapshot { swap_id }
28const DISC_RESTORE: u8 = 0xF1; // AppEvent::Restore  { key[64] }
29
30// ── Hotswap stash key (must match ostd::hotswap::hotswap_key) ────────────────
31
32fn stash_key_for(swap_id: u64) -> u64 {
33    0x00A3_0000_0000_0000_u64 | (swap_id & 0xFFFF_FFFF_FFFF)
34}
35
36// ── Decimal formatter (no std) ────────────────────────────────────────────────
37
38fn fmt_u64_decimal(mut val: u64, buf: &mut [u8; 20]) -> usize {
39    if val == 0 {
40        buf[0] = b'0';
41        return 1;
42    }
43    let mut end = 20usize;
44    while val > 0 {
45        end -= 1;
46        buf[end] = b'0' + (val % 10) as u8;
47        val /= 10;
48    }
49    let len = 20 - end;
50    buf.copy_within(end..20, 0);
51    len
52}

54// ── Poll helpers ──────────────────────────────────────────────────────────────
55
56/// Poll until `sys_state_restore(key)` returns > 0, yielding between iterations.
57///
58/// Returns `Ok(())` when the stash entry appears.
59/// Returns `Err(SnapshotTimeout)` after `MAX_ITERS` yields without success.
60///
61/// A stash miss is tolerated (cell that doesn't implement ViStateTransfer never
62/// stashes anything); the hotswap continues with an empty stash.
63fn wait_for_stash_key(key: u64) -> Result<(), HotswapError> {
64    let mut probe = [0u8; 1];
65    for _ in 0..MAX_ITERS {
66        if sys_state_restore(key, &mut probe) > 0 {
67            return Ok(());
68        }
69        sys_yield();
70    }
71    Err(HotswapError::SnapshotTimeout)
72}
73
74/// Poll until the new cell has called `sys_hotswap_ready()`, yielding between
75/// iterations.
76///
77/// Returns `Ok(())` when the flag is set, `Err(ReadyTimeout)` on timeout.
78fn wait_for_hotswap_ready(new_tid: usize) -> Result<(), HotswapError> {
79    for _ in 0..MAX_ITERS {
80        match sys_query_hotswap_ready(new_tid) {
81            Ok(true) => return Ok(()),
82            Ok(false) => {}
83            Err(_) => break, // tid vanished — treat as timeout
84        }
85        sys_yield();
86    }
87    Err(HotswapError::ReadyTimeout)
88}

90// ── IPC senders ───────────────────────────────────────────────────────────────
91
92/// Send `AppEvent::Snapshot { swap_id }` to `tid`.
93fn send_snapshot_event(tid: usize, swap_id: u64) -> Result<(), HotswapError> {
94    let mut buf = [0u8; 10];
95    buf[0] = APP_MSG_MAGIC;
96    buf[1] = DISC_SNAPSHOT;
97    buf[2..10].copy_from_slice(&swap_id.to_le_bytes());
98    match sys_send(tid, &buf) {
99        ostd::syscall::SyscallResult::Ok(_) => Ok(()),
100        ostd::syscall::SyscallResult::Err(_) => Err(HotswapError::SnapshotIpcFailed),
101    }
102}
103
104/// Send `AppEvent::Restore { key }` to `tid`.
105fn send_restore_event(tid: usize, swap_id: u64) -> Result<(), HotswapError> {
106    let mut buf = [0u8; 66];
107    buf[0] = APP_MSG_MAGIC;
108    buf[1] = DISC_RESTORE;
109    let mut tmp = [0u8; 20];
110    let n = fmt_u64_decimal(swap_id, &mut tmp);
111    let key_len = n.min(63);
112    buf[2..2 + key_len].copy_from_slice(&tmp[..key_len]);
113    match sys_send(tid, &buf) {
114        ostd::syscall::SyscallResult::Ok(_) => Ok(()),
115        ostd::syscall::SyscallResult::Err(_) => Err(HotswapError::RestoreIpcFailed),
116    }
117}

119// ── Swap-ID counter (monotone — wraps after u64::MAX, which is fine) ─────────
120
121use core::sync::atomic::{AtomicU64, Ordering};
122static SWAP_ID_COUNTER: AtomicU64 = AtomicU64::new(1);
123fn next_swap_id() -> u64 {
124    SWAP_ID_COUNTER.fetch_add(1, Ordering::Relaxed)
125}

127// ── Public entry point ────────────────────────────────────────────────────────
128
129/// Execute a 5-phase live hotswap of the service identified by `service_id`
130/// with the new ELF at `new_elf_path`.
131///
132/// Returns the new task tid on success.
133///
134/// # Phases
135/// 1. FREEZE (soft) — lookup service tid, freeze old cell
136/// 2. SERIALIZE — send Snapshot IPC, wait for stash entry
137/// 3. FREEZE (hard) — apply TaskState::Frozen
138/// 4. SPAWN — load new ELF via SpawnCap
139/// 5. DESERIALIZE — send Restore IPC, wait for HotswapReady
140/// 6. UNFREEZE + RE-REGISTER — kill old cell, register new service tid
141pub fn hotswap(service_id: u16, new_elf_path: &str) -> Result<usize, HotswapError> {
142    // ── Resolve target tid ────────────────────────────────────────────────────
143    let old_tid = sys_lookup_service(service_id).ok_or(HotswapError::ServiceNotFound)?;
144
145    let swap_id = next_swap_id();
146    let stash_key = stash_key_for(swap_id);
147
148    // ── Step 1a: FREEZE (soft) ────────────────────────────────────────────────
149    // The old cell must still run to receive the Snapshot IPC.  We freeze the
150    // service registry entry so new callers retry during the swap window.
151    sys_freeze_cell(old_tid).map_err(|_| HotswapError::FreezeFailed)?;
152
153    // ── Step 2: SERIALIZE ─────────────────────────────────────────────────────
154    if let Err(e) = send_snapshot_event(old_tid, swap_id) {
155        sys_resume_cell(old_tid).ok(); // roll back
156        return Err(e);
157    }
158
159    // Wait for old cell to stash state.  Timeout is non-fatal — cells that
160    // don't implement ViStateTransfer never stash; we continue with empty stash.
161    match wait_for_stash_key(stash_key) {
162        Ok(()) | Err(HotswapError::SnapshotTimeout) => {}
163        Err(e) => {
164            sys_resume_cell(old_tid).ok();
165            return Err(e);
166        }
167    }
168
169    // ── Step 3: SPAWN ─────────────────────────────────────────────────────────
170    let new_tid = {
171        let result = sys_spawn_from_path(new_elf_path);
172        match result {
173            ostd::syscall::SyscallResult::Ok(tid) => tid,
174            ostd::syscall::SyscallResult::Err(_) => {
175                sys_resume_cell(old_tid).ok();
176                return Err(HotswapError::SpawnFailed);
177            }
178        }
179    };
180
181    // ── Step 4: DESERIALIZE ───────────────────────────────────────────────────
182    send_restore_event(new_tid, swap_id)?;
183
184    wait_for_hotswap_ready(new_tid)?;
185
186    // ── Step 5: COMMIT ────────────────────────────────────────────────────────
187    // Re-register the service registry entry with the new tid.
188    // Note: Supervisor must hold SpawnCap to call sys_register_service.
189    let _ = sys_register_service(service_id, new_tid);
190
191    // Terminate the old cell (it is Frozen at this point — KillCell bypasses the
192    // Frozen kill-guard in the kernel, same as exit_task_internal in hotswap.rs).
193    sys_kill_cell(old_tid, 0xAAAA_AAAA_u32).ok();
194
195    // Clean up stash slot.
196    sys_state_stash_clear(stash_key);
197
198    Ok(new_tid)
199}
