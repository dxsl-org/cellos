1//! ECAM bus-0 scanner for the Platform Cell.
2//!
3//! Walks all 32 device slots on bus 0 and registers each discovered MMIO BAR
4//! with the kernel via `sys_register_pcie_bar`. This lets Driver Cells
5//! subsequently claim individual BARs through `sys_request_mmio`.
6//!
7//! BAR size probing follows PCI 3.0 §6.2.5.1: write 0xFFFFFFFF, read back,
8//! mask lower 4 bits, compute `~mask + 1`. Memory decode is disabled in the
9//! command register before each probe and restored after to prevent the device
10//! from responding to MMIO accesses during the transient probe window.
11
12use ostd::mmio::MmioRegion;
13use ostd::syscall::{sys_register_pci_device, sys_register_pcie_bar};
14
15// PCI type-0 config space offsets.
16const CFG_VENDOR_ID: usize = 0x00;
17const CFG_COMMAND: usize = 0x04;
18const CFG_CLASS_PROG: usize = 0x09;
19const CFG_SUBCLASS: usize = 0x0A;
20const CFG_CLASS_CODE: usize = 0x0B;
21const CFG_HEADER_TYPE: usize = 0x0E;
22const CFG_BAR0: usize = 0x10;
23
24// Command register bit — disable MMIO decode during BAR size probe.
25const CMD_MEM_SPACE: u16 = 1 << 1;
26
27// ── low-level config-space accessors (all return a fallback on bounds error) ─
28
29fn r32(r: &MmioRegion, dev: u8, fun: u8, off: usize) -> u32 {
30    r.read_u32(cfg_off(dev, fun, off)).unwrap_or(0xFFFF_FFFF)
31}
32
33fn r16(r: &MmioRegion, dev: u8, fun: u8, off: usize) -> u16 {
34    r.read::<u16>(cfg_off(dev, fun, off)).unwrap_or(0xFFFF)
35}
36
37fn r8(r: &MmioRegion, dev: u8, fun: u8, off: usize) -> u8 {
38    r.read::<u8>(cfg_off(dev, fun, off)).unwrap_or(0xFF)
39}
40
41fn w32(r: &MmioRegion, dev: u8, fun: u8, off: usize, v: u32) {
42    let _ = r.write_u32(cfg_off(dev, fun, off), v);
43}
44
45fn w16(r: &MmioRegion, dev: u8, fun: u8, off: usize, v: u16) {
46    let _ = r.write::<u16>(cfg_off(dev, fun, off), v);
47}
48
49/// ECAM formula for bus 0: `(dev << 15) | (fun << 12) | off`
50#[inline(always)]
51fn cfg_off(dev: u8, fun: u8, off: usize) -> usize {
52    ((dev as usize) << 15) | ((fun as usize) << 12) | off
53}
54
55// ── BAR size probing ──────────────────────────────────────────────────────────
56
57/// Probe size of a 32-bit MMIO BAR via the write-all-ones / read-back method.
58fn probe32(r: &MmioRegion, dev: u8, fun: u8, bar_idx: usize) -> u32 {
59    let off = CFG_BAR0 + bar_idx * 4;
60    let orig_cmd = r16(r, dev, fun, CFG_COMMAND);
61    let orig_bar = r32(r, dev, fun, off);
62    // Disable memory decode before touching the BAR.
63    w16(r, dev, fun, CFG_COMMAND, orig_cmd & !CMD_MEM_SPACE);
64    w32(r, dev, fun, off, 0xFFFF_FFFF);
65    let rb = r32(r, dev, fun, off);
66    // Restore BAR and command register.
67    w32(r, dev, fun, off, orig_bar);
68    w16(r, dev, fun, CFG_COMMAND, orig_cmd);
69    let mask = rb & 0xFFFF_FFF0;
70    if mask == 0 {
71        0
72    } else {
73        (!mask).wrapping_add(1)
74    }
75}
76
77/// Probe size of a 64-bit MMIO BAR (low + high dword pair).
78fn probe64(r: &MmioRegion, dev: u8, fun: u8, bar_idx: usize) -> u64 {
79    let off_lo = CFG_BAR0 + bar_idx * 4;
80    let off_hi = CFG_BAR0 + (bar_idx + 1) * 4;
81    let orig_cmd = r16(r, dev, fun, CFG_COMMAND);
82    let orig_lo = r32(r, dev, fun, off_lo);
83    let orig_hi = r32(r, dev, fun, off_hi);
84    w16(r, dev, fun, CFG_COMMAND, orig_cmd & !CMD_MEM_SPACE);
85    w32(r, dev, fun, off_lo, 0xFFFF_FFFF);
86    w32(r, dev, fun, off_hi, 0xFFFF_FFFF);
87    let rb_lo = r32(r, dev, fun, off_lo);
88    let rb_hi = r32(r, dev, fun, off_hi);
89    w32(r, dev, fun, off_lo, orig_lo);
90    w32(r, dev, fun, off_hi, orig_hi);
91    w16(r, dev, fun, CFG_COMMAND, orig_cmd);
92    let mask64 = ((rb_hi as u64) << 32) | ((rb_lo & 0xFFFF_FFF0) as u64);
93    if mask64 == 0 {
94        0
95    } else {
96        (!mask64).wrapping_add(1)
97    }
98}

100// ── Public scanner entry point ────────────────────────────────────────────────
101
102/// Walk bus 0, decode all type-0 MMIO BARs, and register each non-zero BAR
103/// with the kernel via `sys_register_pcie_bar`.
104///
105/// After this call returns, Driver Cells can claim individual BARs through
106/// `sys_request_mmio` backed by `PcieDriverCap`.
107pub fn scan_and_register(region: &MmioRegion) {
108    for dev in 0u8..32 {
109        if r16(region, dev, 0, CFG_VENDOR_ID) == 0xFFFF {
110            continue; // slot empty
111        }
112        let hdr = r8(region, dev, 0, CFG_HEADER_TYPE);
113        let max_f = if hdr & 0x80 != 0 { 8u8 } else { 1u8 };
114
115        for fun in 0u8..max_f {
116            if r16(region, dev, fun, CFG_VENDOR_ID) == 0xFFFF {
117                continue;
118            }
119            // Skip PCI-to-PCI bridges (header type 1) — they have no BARs.
120            if r8(region, dev, fun, CFG_HEADER_TYPE) & 0x7F != 0 {
121                continue;
122            }
123
124            let bdf: u32 = ((dev as u32) << 3) | (fun as u32);
125            let class = r8(region, dev, fun, CFG_CLASS_CODE);
126            let subclass = r8(region, dev, fun, CFG_SUBCLASS);
127            let prog_if = r8(region, dev, fun, CFG_CLASS_PROG);
128            let cls: u32 = ((class as u32) << 16) | ((subclass as u32) << 8) | (prog_if as u32);
129
130            let mut bar0_base: usize = 0;
131            let mut bar0_size: usize = 0;
132
133            let mut i = 0usize;
134            while i < 6 {
135                let raw = r32(region, dev, fun, CFG_BAR0 + i * 4);
136                if raw & 1 == 1 {
137                    // I/O port BAR — skip.
138                    i += 1;
139                    continue;
140                }
141                let bar_type = (raw >> 1) & 0x3;
142                if bar_type == 0x2 && i + 1 < 6 {
143                    // 64-bit MMIO BAR: spans two slots.
144                    let raw_hi = r32(region, dev, fun, CFG_BAR0 + (i + 1) * 4);
145                    let base = ((raw as u64) & !0xF) | ((raw_hi as u64) << 32);
146                    let size = probe64(region, dev, fun, i);
147                    if base != 0 && size != 0 {
148                        let _ = sys_register_pcie_bar(bdf, base as usize, size as usize);
149                        if i == 0 {
150                            bar0_base = base as usize;
151                            bar0_size = size as usize;
152                        }
153                    }
154                    i += 2;
155                } else {
156                    // 32-bit MMIO BAR.
157                    let base = (raw & !0xF) as usize;
158                    let size = probe32(region, dev, fun, i) as usize;
159                    if base != 0 && size != 0 {
160                        let _ = sys_register_pcie_bar(bdf, base, size);
161                        if i == 0 {
162                            bar0_base = base;
163                            bar0_size = size;
164                        }
165                    }
166                    i += 1;
167                }
168            }
169
170            // Register class/BAR0 info so the kernel PCI_DEVICES list is populated
171            // and sys_find_pcie_device queries work without a kernel ECAM scan.
172            let _ = sys_register_pci_device(bdf, cls, bar0_base, bar0_size);
173        }
174    }
