4//! GPU driver Cell — thin wrapper exposing `sys_gpu_flush` to sibling cells.
5//!
6//! This cell has no IPC loop of its own; it provides library functions that
7//! the compositor cell (or any graphics-capable cell) calls directly, since
8//! all cells share the same address space.
9//!
10//! The actual GPU hardware interaction happens in the kernel-side VirtIO GPU
11//! driver (`kernel/src/task/drivers/virtio_gpu.rs`) via the `GpuFlush` syscall.
12
13use api::display::Rect;
14use types::ViResult;
15
16/// Flush a rectangular region of BGRA8888 pixels to the VirtIO GPU.
17///
18/// `pixels` must contain exactly `rect.w * rect.h * 4` bytes.
19///
20/// # Errors
21/// Returns `ViError::IO` if the GPU driver is not initialised.
22pub fn flush_rect(pixels: &[u8], rect: Rect) -> ViResult<()> {
23    ostd::syscall::sys_gpu_flush(pixels, rect.x as u32, rect.y as u32, rect.w, rect.h)
24        .map_err(|_| types::ViError::IO)
25}
26
27/// Fill a solid-colour rectangle on the GPU framebuffer.
28///
29/// `rgba` is a packed `0xRRGGBBAA` value; pixels are written in BGRA order.
30/// Useful for clearing a region before compositing.
31pub fn fill_rect(rect: Rect, rgba: u32) -> ViResult<()> {
32    let b = ((rgba >> 8) & 0xFF) as u8;
33    let g = ((rgba >> 16) & 0xFF) as u8;
34    let r = ((rgba >> 24) & 0xFF) as u8;
35    let a = (rgba & 0xFF) as u8;
36    let n = (rect.w * rect.h) as usize;
37    let mut pixels = alloc::vec![0u8; n * 4];
38    for i in 0..n {
39        let off = i * 4;
40        pixels[off] = b;
41        pixels[off + 1] = g;
42        pixels[off + 2] = r;
43        pixels[off + 3] = a;
44    }
45    flush_rect(&pixels, rect)
46}
