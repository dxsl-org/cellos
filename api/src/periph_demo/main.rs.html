13use ostd::io::println;
14use ostd::syscall::{sys_get_wall_secs, sys_recv_timeout, sys_spawn_pinned, SyscallResult};
15use types::ViError;

17declare_manifest!(
18    block_io = false,
19    network = false,
20    spawn = false,
21    gpio = true,
22    uart = true
23);
24
25const SELF_PATH: &str = "/bin/periph-demo";
26const POLL_PRIORITY: u8 = 200;
27
28#[no_mangle]
29pub fn main() {
30    // ── RTC test ──────────────────────────────────────────────────────────────
31    // sys_get_wall_secs() calls GetTime op=3 → Goldfish RTC epoch seconds.
32    // Passes if the RTC returned a plausible year (> 2020-01-01 = 1577836800).
33    {
34        let epoch_secs = sys_get_wall_secs();
35        let year_approx = 1970u64 + epoch_secs / 31_557_600; // ~365.25 days
36        let msg = if epoch_secs > 1_577_836_800 {
37            alloc::format!(
38                "[periph-demo] RTC OK: epoch={} (~year {})",
39                epoch_secs,
40                year_approx
41            )
42        } else {
43            alloc::format!(
44                "[periph-demo] RTC FAIL: epoch={} (expected >1577836800; RTC not initialized?)",
45                epoch_secs
46            )
47        };
48        println(&msg);
49    }
50
51    // ── GPIO demo ─────────────────────────────────────────────────────────────
52    // robot-demo (supervised, spawned before us) holds PL061 for its 5-cycle
53    // sensor→actuator loop plus MQTT timeouts per cycle.  Retry only on
54    // AlreadyExists (resource busy) — fail immediately on PermissionDenied or
55    // other errors (non-aarch64 target or manifest mismatch).
56    let gpio_result = {
57        let mut res = Pl061Gpio::open();
58        let mut tries = 0u8;
59        while matches!(res, Err(ViError::AlreadyExists)) && tries < 150 {
60            let mut buf = [0u8; 8];
61            let _ = sys_recv_timeout(0, &mut buf, 10); // ~100 ms per attempt
62            tries += 1;
63            res = Pl061Gpio::open();
64        }
65        res
66    };
67    match gpio_result {
68        Ok(mut gpio) => {
69            println("[periph-demo] GPIO PL061 opened");
70
71            // Basic output test: write HIGH → verify HIGH, write LOW → verify LOW.
72            let _ = gpio.set_direction(0, PinDir::Output);
73            let _ = gpio.write_pin(0, true);
74            match gpio.read_pin(0) {
75                Ok(true) => println("[periph-demo] GPIO pin 0: HIGH OK"),
76                Ok(false) => println("[periph-demo] GPIO pin 0: LOW (unexpected)"),
77                Err(_) => println("[periph-demo] GPIO read_pin error"),
78            }
79            let _ = gpio.write_pin(0, false);
80            match gpio.read_pin(0) {
81                Ok(false) => println("[periph-demo] GPIO pin 0: LOW OK"),
82                Ok(true) => println("[periph-demo] GPIO pin 0: HIGH (unexpected)"),
83                Err(_) => println("[periph-demo] GPIO read_pin error"),
84            }
85
86            // ── GPIO IRQ self-test (output-pin loopback) ───────────────────
87            // PL061 edge detection monitors pin state. For output pins the output
88            // value IS the pin state, so writing 0→1 should set GPIORIS bit 0
89            // and (with GPIOIE set) GPIOMIS bit 0 and fire GIC ID 39.
90            //
91            // The kernel vi_gpio_notify_irq IPC fires immediately when the GPIO
92            // write triggers the edge (cell not yet in Recv state → IPC dropped).
93            // But GPIOMIS is a sticky HW register — it persists until we clear it,
94            // so we can read it directly to confirm the edge was detected.
95            let _ = gpio.enable_edge_irq(0, Edge::Rising); // enable on pin 0
96            let _ = gpio.write_pin(0, true); // 0 → 1 : rising edge
97                                             // At this point vi_gpio_notify_irq was called by the kernel (see kernel
98                                             // log "[gpio-irq] IRQ fired") even though the IPC was dropped.
99            let mis = gpio.read_mis().unwrap_or(0xFF);
100            let _ = gpio.clear_irq(mis);
101            let _ = gpio.disable_irq(0);
102            let _ = gpio.write_pin(0, false); // restore LOW
103            {
104                let msg = if mis & 1 != 0 {
105                    alloc::format!(
106                        "[periph-demo] GPIO IRQ OK: GPIOMIS=0x{:02x} (edge detected on pin 0)",
107                        mis
108                    )
109                } else {
110                    alloc::format!(
111                        "[periph-demo] GPIO IRQ: GPIOMIS=0x{:02x} \
112                         (0=edge not detected; check QEMU PL061 output-loopback behavior)",
113                        mis
114                    )
115                };
116                println(&msg);
117            }
118        }
119        Err(_) => println("[periph-demo] GPIO not available (non-aarch64 target)"),
120    }
121
122    // ── UART demo ─────────────────────────────────────────────────────────────
123    match Pl011Uart::open() {
124        Ok(mut uart) => {
125            println("[periph-demo] UART PL011 opened");
126            let _ = uart.init();
127            for b in b"[periph-demo] Hello from PL011\r\n" {
128                let _ = uart.send(*b);
129            }
130            println("[periph-demo] UART TX done");
131        }
132        Err(_) => println("[periph-demo] UART not available (non-aarch64 target)"),
133    }
134
135    // ── Pinned-poll RT loop ───────────────────────────────────────────────────
136    println("[periph-demo] spawning pinned-poll cell on hart 0");
137    match sys_spawn_pinned(SELF_PATH, POLL_PRIORITY, 0) {
138        SyscallResult::Ok(tid) => {
139            let tid_str = alloc::format!("[periph-demo] pinned cell tid={}", tid);
140            println(&tid_str);
141        }
142        SyscallResult::Err(_) => println("[periph-demo] spawn_pinned unavailable"),
143    }
144}
