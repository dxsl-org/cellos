1//! NVMe controller logic ported from kernel/src/task/drivers/blk_nvme.rs.
2//!
3//! In the Driver Cell, MMIO access goes through `ostd::mmio::MmioRegion` instead
4//! of identity-mapped raw pointers.  DMA allocation uses `ostd::dma::DmaBuf`.
5//!
6//! Law 4 exception: Driver Cells may use `unsafe` for MMIO register access.
7//! Every `unsafe` block is annotated with `// SAFETY:`.
8
9use core::sync::atomic::{fence, Ordering};
10use ostd::dma::DmaBuf;
11use ostd::mmio::MmioRegion;
12use types::{ViError, ViResult};
13
14use crate::queue::{Queue, SqEntry};
15
16// ── BAR0 register offsets (NVMe 1.x §3.1) ─────────────────────────────────────
17
18const REG_CAP: usize = 0x00;
19const REG_CC: usize = 0x14;
20const REG_CSTS: usize = 0x1C;
21const REG_AQA: usize = 0x24;
22const REG_ASQ: usize = 0x28;
23const REG_ACQ: usize = 0x30;
24const REG_DB_BASE: usize = 0x1000;
25
26const CC_EN: u32 = 1 << 0;
27const CC_CSS_NVM: u32 = 0;
28const CC_MPS_4K: u32 = 0;
29const CC_AMS_RR: u32 = 0;
30const CC_IOSQES: u32 = 6 << 16;
31const CC_IOCQES: u32 = 4 << 20;
32
33const CSTS_RDY: u32 = 1 << 0;
34const CSTS_CFS: u32 = 1 << 1;
35
36const ADMIN_QUEUE_DEPTH: u16 = 64;
37const IO_QUEUE_DEPTH: u16 = 64;
38
39const ADMIN_OPC_CREATE_SQ: u8 = 0x01;
40const ADMIN_OPC_CREATE_CQ: u8 = 0x05;
41const ADMIN_OPC_IDENTIFY: u8 = 0x06;
42const CNS_IDENTIFY_NS: u32 = 0;
43const CNS_IDENTIFY_CTRL: u32 = 1;
44const NVM_OPC_WRITE: u8 = 0x01;
45const NVM_OPC_READ: u8 = 0x02;
46
47const POLL_WARN_ITERS: u64 = 1_000_000;
48
49pub struct NvmeController {
50    mmio: MmioRegion,
51    admin: Queue,
52    io: Queue,
53    db_stride: usize,
54    pub n_sectors: u64,
55    pub lba_bytes: u32,
56    pub vwc: bool,
57    // reason: only read at construction (to authorize admin/IO queue DMA);
58    // kept on the struct for a planned re-authorization path if read/write_sector
59    // start allocating DmaBufs directly on the controller instead of via Queue.
60    #[allow(dead_code)]
61    bdf: u32,
62}
63
64impl NvmeController {
65    /// Initialise the NVMe controller reachable via `mmio` (BAR0).
66    ///
67    /// Returns `Err(IO)` if the controller fails to reach RDY or Identify fails.
68    pub fn new(mmio: MmioRegion, bdf: u32) -> ViResult<Self> {
69        // 1. Read capabilities — need CAP.DSTRD for doorbell stride.
70        // SAFETY: BAR0 is MMIO; read_volatile is required for hardware registers.
71        let cap = unsafe {
72            let lo = core::ptr::read_volatile((mmio.base() + REG_CAP) as *const u32) as u64;
73            let hi = core::ptr::read_volatile((mmio.base() + REG_CAP + 4) as *const u32) as u64;
74            lo | (hi << 32)
75        };
76        let dstrd = ((cap >> 32) & 0xF) as usize;
77        let db_stride = 4 << dstrd;
78
79        // 2. Reset controller: CC.EN=0, wait CSTS.RDY=0.
80        Self::write32(&mmio, REG_CC, 0)?;
81        let mut spin = 0u64;
82        loop {
83            let csts = Self::read32(&mmio, REG_CSTS)?;
84            if csts & CSTS_RDY == 0 {
85                break;
86            }
87            spin += 1;
88            if spin > POLL_WARN_ITERS {
89                return Err(ViError::IO);
90            }
91            fence(Ordering::SeqCst);
92        }
93
94        // 3. Allocate admin queues.
95        let admin = Queue::new(bdf, ADMIN_QUEUE_DEPTH).ok_or(ViError::OutOfMemory)?;
96
97        // 4. Program AQA, ASQ, ACQ.
98        let aqa = ((ADMIN_QUEUE_DEPTH as u32 - 1) << 16) | (ADMIN_QUEUE_DEPTH as u32 - 1);
99        Self::write32(&mmio, REG_AQA, aqa)?;
100        Self::write64(&mmio, REG_ASQ, admin.sq_phys())?;
101        Self::write64(&mmio, REG_ACQ, admin.cq_phys())?;
102
103        // 5. Enable controller.
104        let cc = CC_EN | CC_CSS_NVM | CC_MPS_4K | CC_AMS_RR | CC_IOSQES | CC_IOCQES;
105        Self::write32(&mmio, REG_CC, cc)?;
106
107        // 6. Wait CSTS.RDY=1.
108        let mut spin = 0u64;
109        loop {
110            let csts = Self::read32(&mmio, REG_CSTS)?;
111            if csts & CSTS_CFS != 0 {
112                return Err(ViError::IO);
113            }
114            if csts & CSTS_RDY != 0 {
115                break;
116            }
117            spin += 1;
118            if spin > POLL_WARN_ITERS {
119                return Err(ViError::IO);
120            }
121            fence(Ordering::SeqCst);
122        }
123
124        let mut ctrl = NvmeController {
125            mmio,
126            admin,
127            io: Queue::new(bdf, IO_QUEUE_DEPTH).ok_or(ViError::OutOfMemory)?,
128            db_stride,
129            n_sectors: 0,
130            lba_bytes: 512,
131            vwc: false,
132            bdf,
133        };
134
135        // 7. Identify Controller → VWC flag.
136        {
137            let id_buf = DmaBuf::alloc(1).ok_or(ViError::OutOfMemory)?;
138            let _ = id_buf.authorize(bdf);
139            let id_phys = id_buf.phys() as u64;
140            ctrl.admin_cmd(
141                ADMIN_OPC_IDENTIFY,
142                0,
143                id_phys,
144                0,
145                CNS_IDENTIFY_CTRL,
146                0,
147                0,
148                0,
149                0,
150                0,
151            )?;
152            // SAFETY: id_buf is DMA memory we own; VWC is byte 525 in the Identify Controller data.
153            let vwc_byte = unsafe { *(id_buf.virt().add(525)) };
154            ctrl.vwc = vwc_byte & 1 != 0;
155            id_buf.free();
156        }
157
158        // 8. Identify Namespace 1 → LBA count + format.
159        {
160            let id_buf = DmaBuf::alloc(1).ok_or(ViError::OutOfMemory)?;
161            let _ = id_buf.authorize(bdf);
162            let id_phys = id_buf.phys() as u64;
163            ctrl.admin_cmd(
164                ADMIN_OPC_IDENTIFY,
165                1,
166                id_phys,
167                0,
168                CNS_IDENTIFY_NS,
169                0,
170                0,
171                0,
172                0,
173                0,
174            )?;
175            // SAFETY: id_buf is a valid 4-KiB Identify Namespace response.
176            unsafe {
177                let ptr = id_buf.virt();
178                ctrl.n_sectors = core::ptr::read_volatile(ptr as *const u64);
179                let flbas = *ptr.add(26); // FLBAS: current LBA format index
180                let lba_fmt_idx = (flbas & 0x0F) as usize;
181                let lba_ds = *ptr.add(128 + lba_fmt_idx * 4 + 1); // LBA data shift
182                ctrl.lba_bytes = 1u32 << lba_ds;
183            }
184            id_buf.free();
185        }
186
187        // 9. Create I/O CQ (admin opcode 0x05).
188        // CDW10 layout (NVMe 1.x §5.3/§5.4): QSIZE[31:16] (0-based) | QID[15:0].
189        // The first port of this file inverted the two fields — QEMU then created
190        // CQ with QID=63 and Create-SQ failed with Invalid CQID (CQ 1 absent).
191        let io_cq_phys = ctrl.io.cq_phys();
192        ctrl.admin_cmd(
193            ADMIN_OPC_CREATE_CQ,
194            0,
195            io_cq_phys,
196            0,
197            ((IO_QUEUE_DEPTH as u32 - 1) << 16) | 1, // CDW10: QSIZE | QID=1
198            0x1, // CDW11: IEN=0 (polled), PC=1 (physically contiguous)
199            0,
200            0,
201            0,
202            0,
203        )?;
204
205        // 10. Create I/O SQ (admin opcode 0x01).
206        let io_sq_phys = ctrl.io.sq_phys();
207        ctrl.admin_cmd(
208            ADMIN_OPC_CREATE_SQ,
209            0,
210            io_sq_phys,
211            0,
212            ((IO_QUEUE_DEPTH as u32 - 1) << 16) | 1, // CDW10: QSIZE | QID=1
213            (1 << 16) | 0x1,                         // CDW11: CQID=1 | PC=1
214            0,
215            0,
216            0,
217            0,
218        )?;
219
220        Ok(ctrl)
221    }

225    fn read32(mmio: &MmioRegion, off: usize) -> ViResult<u32> {
226        // SAFETY: mmio region was granted by kernel; volatile prevents caching.
227        Ok(unsafe { core::ptr::read_volatile((mmio.base() + off) as *const u32) })
228    }
229
230    fn write32(mmio: &MmioRegion, off: usize, val: u32) -> ViResult<()> {
231        // SAFETY: same contract as read32.
232        unsafe { core::ptr::write_volatile((mmio.base() + off) as *mut u32, val) };
233        Ok(())
234    }
235
236    fn write64(mmio: &MmioRegion, off: usize, val: u64) -> ViResult<()> {
237        // SAFETY: same contract as read32.
238        unsafe { core::ptr::write_volatile((mmio.base() + off) as *mut u64, val) };
239        Ok(())
240    }
241
242    fn ring_sq_tail(&self, qid: usize, tail: u16) {
243        let db_off = REG_DB_BASE + qid * 2 * self.db_stride;
244        // SAFETY: doorbell is within the granted BAR0 MMIO region.
245        unsafe { core::ptr::write_volatile((self.mmio.base() + db_off) as *mut u32, tail as u32) };
246    }
247
248    fn ring_cq_head(&self, qid: usize, head: u16) {
249        let db_off = REG_DB_BASE + (qid * 2 + 1) * self.db_stride;
250        // SAFETY: doorbell is within the granted BAR0 MMIO region.
251        unsafe { core::ptr::write_volatile((self.mmio.base() + db_off) as *mut u32, head as u32) };
252    }

254    // ── Admin command submit + poll ───────────────────────────────────────────
255
256    #[allow(clippy::too_many_arguments)]
257    fn admin_cmd(
258        &mut self,
259        opc: u8,
260        nsid: u32,
261        prp1: u64,
262        prp2: u64,
263        cdw10: u32,
264        cdw11: u32,
265        cdw12: u32,
266        cdw13: u32,
267        cdw14: u32,
268        cdw15: u32,
269    ) -> ViResult<()> {
270        let depth = self.admin.depth as usize;
271        let tail = self.admin.sq_tail as usize;
272        let cid = self.admin.next_cid();
273
274        // SAFETY: tail < depth; sq_buf holds depth SqEntry slots.
275        let sqe = unsafe { &mut *self.admin.sq_entry(tail) };
276        *sqe = SqEntry::default();
277        sqe.cdw0 = (opc as u32) | ((cid as u32) << 16);
278        sqe.nsid = nsid;
279        sqe.prp1 = prp1;
280        sqe.prp2 = prp2;
281        sqe.cdw10 = cdw10;
282        sqe.cdw11 = cdw11;
283        sqe.cdw12 = cdw12;
284        sqe.cdw13 = cdw13;
285        sqe.cdw14 = cdw14;
286        sqe.cdw15 = cdw15;
287
288        self.admin.sq_tail = ((tail + 1) % depth) as u16;
289        fence(Ordering::Release);
290        self.ring_sq_tail(0, self.admin.sq_tail);
291
292        let expected_phase = self.admin.cq_phase;
293        let cq_head = self.admin.cq_head as usize;
294        let mut iters = 0u64;
295        loop {
296            // SAFETY: cq_head < depth; cq_buf holds depth CqEntry slots.
297            let phase_status =
298                unsafe { core::ptr::read_volatile(&(*self.admin.cq_entry(cq_head)).phase_status) };
299            if (phase_status & 1 != 0) == expected_phase {
300                let status = phase_status >> 1;
301                let new_head = (cq_head + 1) % depth;
302                if new_head == 0 {
303                    self.admin.cq_phase = !self.admin.cq_phase;
304                }
305                self.admin.cq_head = new_head as u16;
306                self.ring_cq_head(0, self.admin.cq_head);
307                if status != 0 {
308                    return Err(ViError::IO);
309                }
310                return Ok(());
311            }
312            iters += 1;
313            if iters == POLL_WARN_ITERS {
314                return Err(ViError::IO);
315            }
316            fence(Ordering::Acquire);
317        }
318    }

322    fn submit_io(&mut self, opc: u8, nsid: u32, lba: u64, nlb: u16, prp1: u64) -> ViResult<()> {
323        let depth = self.io.depth as usize;
324        let tail = self.io.sq_tail as usize;
325        let cid = self.io.next_cid();
326
327        // SAFETY: tail < depth; sq_buf holds depth SqEntry slots.
328        let sqe = unsafe { &mut *self.io.sq_entry(tail) };
329        *sqe = SqEntry::default();
330        sqe.cdw0 = (opc as u32) | ((cid as u32) << 16);
331        sqe.nsid = nsid;
332        sqe.prp1 = prp1;
333        sqe.cdw10 = (lba & 0xFFFF_FFFF) as u32;
334        sqe.cdw11 = (lba >> 32) as u32;
335        sqe.cdw12 = nlb as u32;
336
337        self.io.sq_tail = ((tail + 1) % depth) as u16;
338        fence(Ordering::Release);
339        self.ring_sq_tail(1, self.io.sq_tail);
340
341        let expected_phase = self.io.cq_phase;
342        let cq_head = self.io.cq_head as usize;
343        let mut iters = 0u64;
344        loop {
345            // SAFETY: cq_head < depth; cq_buf holds depth CqEntry slots.
346            let phase_status =
347                unsafe { core::ptr::read_volatile(&(*self.io.cq_entry(cq_head)).phase_status) };
348            if (phase_status & 1 != 0) == expected_phase {
349                let status = phase_status >> 1;
350                let new_head = (cq_head + 1) % depth;
351                if new_head == 0 {
352                    self.io.cq_phase = !self.io.cq_phase;
353                }
354                self.io.cq_head = new_head as u16;
355                self.ring_cq_head(1, self.io.cq_head);
356                if status != 0 {
357                    return Err(ViError::IO);
358                }
359                return Ok(());
360            }
361            iters += 1;
362            if iters == POLL_WARN_ITERS {
363                return Err(ViError::IO);
364            }
365            fence(Ordering::Acquire);
366        }
367    }

369    /// Read one 512-byte sector into `buf` (must be DMA-capable).
370    pub fn read_sector(&mut self, sector: u64, buf_phys: u64) -> ViResult<()> {
371        self.submit_io(NVM_OPC_READ, 1, sector, 0, buf_phys)
372    }
373
374    /// Write one 512-byte sector from `buf` (must be DMA-capable).
375    pub fn write_sector(&mut self, sector: u64, buf_phys: u64) -> ViResult<()> {
376        self.submit_io(NVM_OPC_WRITE, 1, sector, 0, buf_phys)
377    }
