1//! IPC request dispatcher — translates `DrvRequest` messages into NVMe I/O.
2//!
3//! Wire format (little-endian):
4//!
5//! Read request  (10 B): `[op=0 (2B)] [sector (8B)]`
6//! Write request (522 B): `[op=1 (2B)] [sector (8B)] [data (512B)]`
7//!
8//! Read reply OK  (513 B): `[0x00] [sector_data (512B)]`
9//! Write reply OK   (1 B): `[0x00]`
10//! Error reply      (1 B): `[0x01]`
11
12use crate::controller::NvmeController;
13use ostd::dma::DmaBuf;
14
15/// Total reply buffer size: status byte + one full 512-byte sector.
16pub const REPLY_SIZE: usize = 513;
17
18#[derive(Debug, PartialEq, Eq)]
19pub enum DrvOp {
20    Read = 0,
21    Write = 1,
22}
23
24fn parse_op(data: &[u8]) -> Option<(DrvOp, u64)> {
25    if data.len() < 10 {
26        return None;
27    }
28    let op = match u16::from_le_bytes([data[0], data[1]]) {
29        0 => DrvOp::Read,
30        1 => DrvOp::Write,
31        _ => return None,
32    };
33    let sector = u64::from_le_bytes(data[2..10].try_into().ok()?);
34    Some((op, sector))
35}
36
37/// Encode a Read IPC request (10 bytes). Used by VFS.
38// reason: VFS does not yet call these encoders directly (it builds requests
39// inline); kept as the documented counterpart to parse_op() for that caller.
40#[allow(dead_code)]
41pub fn encode_read(sector: u64) -> [u8; 10] {
42    let mut b = [0u8; 10];
43    b[0..2].copy_from_slice(&0u16.to_le_bytes());
44    b[2..10].copy_from_slice(&sector.to_le_bytes());
45    b
46}
47
48/// Encode a Write IPC request (522 bytes). Used by VFS.
49// reason: see encode_read() above.
50#[allow(dead_code)]
51pub fn encode_write(sector: u64, data: &[u8; 512]) -> [u8; 522] {
52    let mut b = [0u8; 522];
53    b[0..2].copy_from_slice(&1u16.to_le_bytes());
54    b[2..10].copy_from_slice(&sector.to_le_bytes());
55    b[10..522].copy_from_slice(data);
56    b
57}
58
59/// Handle one incoming IPC message. Writes the reply into `out` and returns
60/// the number of bytes to send back.
61///
62/// Read OK:  writes `[0x00] ++ sector_data`, returns 513.
63/// Write OK: writes `[0x00]`, returns 1.
64/// Error:    writes `[0x01]`, returns 1.
65pub fn handle(
66    ctrl: &mut NvmeController,
67    io_buf: &DmaBuf,
68    data: &[u8],
69    out: &mut [u8; REPLY_SIZE],
70) -> usize {
71    let (op, sector) = match parse_op(data) {
72        Some(v) => v,
73        None => {
74            out[0] = 1;
75            return 1;
76        }
77    };
78
79    match op {
80        DrvOp::Read => {
81            match ctrl.read_sector(sector, io_buf.phys() as u64) {
82                Ok(_) => {
83                    out[0] = 0;
84                    // SAFETY: io_buf is identity-mapped (phys == virt) in SAS.
85                    // read_sector completed; DMA data is now stable in io_buf.
86                    unsafe {
87                        core::ptr::copy_nonoverlapping(io_buf.virt(), out[1..].as_mut_ptr(), 512);
88                    }
89                    513
90                }
91                Err(_) => {
92                    out[0] = 1;
93                    1
94                }
95            }
96        }
97
98        DrvOp::Write => {
99            if data.len() < 10 + 512 {
100                out[0] = 1;
101                return 1;
102            }
103            // SAFETY: io_buf is identity-mapped; we write the caller's payload
104            // in before handing the physical address to the NVMe controller.
105            unsafe {
106                core::ptr::copy_nonoverlapping(data[10..].as_ptr(), io_buf.virt(), 512);
107            }
108            match ctrl.write_sector(sector, io_buf.phys() as u64) {
109                Ok(_) => {
110                    out[0] = 0;
111                    1
112                }
113                Err(_) => {
114                    out[0] = 1;
115                    1
116                }
117            }
118        }
119    }
120}
