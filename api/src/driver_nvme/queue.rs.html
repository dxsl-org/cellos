1//! NVMe Submission/Completion queue pair backed by DMA-allocated memory.
2//!
3//! Each queue pair consists of one Submission Queue (SQ) and one Completion
4//! Queue (CQ).  Both are physically contiguous — allocated via `DmaBuf`.
5//!
6//! This file is `unsafe`-annotated because the queue pointers are raw DMA
7//! memory, accessed via volatile reads/writes.
8//!
9// SAFETY invariant: `sq` and `cq` point into DMA-allocated contiguous pages
10// that are valid for the lifetime of the Queue.  All accesses are volatile.
11
12extern crate alloc;
13
14use core::mem::size_of;
15use ostd::dma::DmaBuf;
16
17/// NVMe Submission Queue Entry (64 bytes, NVMe 1.x §4.2).
18#[repr(C, align(64))]
19#[derive(Clone, Copy, Default)]
20pub struct SqEntry {
21    pub cdw0: u32,
22    pub nsid: u32,
23    pub cdw2: u32,
24    pub cdw3: u32,
25    pub mptr: u64,
26    pub prp1: u64,
27    pub prp2: u64,
28    pub cdw10: u32,
29    pub cdw11: u32,
30    pub cdw12: u32,
31    pub cdw13: u32,
32    pub cdw14: u32,
33    pub cdw15: u32,
34}
35
36/// NVMe Completion Queue Entry (16 bytes, NVMe 1.x §4.4).
37#[repr(C, align(16))]
38#[derive(Clone, Copy, Default)]
39pub struct CqEntry {
40    pub dw0: u32,
41    pub dw1: u32,
42    pub sq_hd: u16,
43    pub sq_id: u16,
44    pub cid: u16,
45    pub phase_status: u16, // phase bit[0], status[15:1]
46}
47
48impl CqEntry {
49    // reason: phase()/status_field() decode the CQE per NVMe 1.x §4.4 but the
50    // current completion poll reads phase_status directly inline; kept as the
51    // documented accessor pair for callers that decode a CqEntry outside that loop.
52    #[inline]
53    #[allow(dead_code)]
54    pub fn phase(&self) -> bool {
55        self.phase_status & 1 != 0
56    }
57    #[inline]
58    #[allow(dead_code)]
59    pub fn status_field(&self) -> u16 {
60        self.phase_status >> 1
61    }
62}
63
64pub struct Queue {
65    pub sq_buf: DmaBuf,
66    pub cq_buf: DmaBuf,
67    pub depth: u16,
68    pub sq_tail: u16,
69    pub cq_head: u16,
70    pub cq_phase: bool,
71    pub cid: u16,
72}
73
74impl Queue {
75    /// Allocate a new queue pair for `depth` entries, registering DMA with IOMMU.
76    pub fn new(bdf: u32, depth: u16) -> Option<Self> {
77        let sq_pages = (depth as usize * size_of::<SqEntry>()).div_ceil(4096);
78        let cq_pages = (depth as usize * size_of::<CqEntry>()).div_ceil(4096);
79
80        let sq_buf = DmaBuf::alloc(sq_pages)?;
81        let cq_buf = DmaBuf::alloc(cq_pages)?;
82
83        // Authorize IOMMU for this PCIe device to DMA into these pages.
84        // SAFETY: DmaBuf was just allocated; bdf is the NVMe controller's BDF.
85        let _ = sq_buf.authorize(bdf);
86        let _ = cq_buf.authorize(bdf);
87
88        // Zero queues (DmaBuf is not guaranteed zeroed).
89        unsafe {
90            core::ptr::write_bytes(sq_buf.virt(), 0, sq_buf.size());
91            core::ptr::write_bytes(cq_buf.virt(), 0, cq_buf.size());
92        }
93
94        Some(Self {
95            sq_buf,
96            cq_buf,
97            depth,
98            sq_tail: 0,
99            cq_head: 0,
100            cq_phase: true, // initial phase = 1 after reset
101            cid: 0,
102        })
103    }
104
105    /// Physical base address of the SQ (to program into controller registers).
106    #[inline]
107    pub fn sq_phys(&self) -> u64 {
108        self.sq_buf.phys() as u64
109    }
110
111    /// Physical base address of the CQ.
112    #[inline]
113    pub fn cq_phys(&self) -> u64 {
114        self.cq_buf.phys() as u64
115    }
116
117    /// Mutable pointer to the SQ entry at `idx`.
118    ///
119    /// # Safety
120    /// `idx < depth` must hold.
121    #[inline]
122    pub unsafe fn sq_entry(&mut self, idx: usize) -> *mut SqEntry {
123        // SAFETY: caller guarantees idx < depth; sq_buf covers depth*64 bytes.
124        unsafe { (self.sq_buf.virt() as *mut SqEntry).add(idx) }
125    }
126
127    /// Shared pointer to the CQ entry at `idx`.
128    ///
129    /// # Safety
130    /// `idx < depth` must hold.
131    #[inline]
132    pub unsafe fn cq_entry(&self, idx: usize) -> *const CqEntry {
133        // SAFETY: caller guarantees idx < depth; cq_buf covers depth*16 bytes.
134        unsafe { (self.cq_buf.virt() as *const CqEntry).add(idx) }
135    }
136
137    pub fn next_cid(&mut self) -> u16 {
138        self.cid = self.cid.wrapping_add(1);
139        self.cid
140    }
141}
