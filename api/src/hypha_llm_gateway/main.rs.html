1//! Hypha `llm-gateway` — the LLM network service cell.
2//!
3//! Receives [`LlmRequest`] over IPC, does an HTTP chat-completion round-trip to
4//! an OpenAI-compatible endpoint, and replies [`LlmReply`]. Two transports
5//! (see `transport.rs`):
6//! - **plaintext** (`USE_TLS = false`, default): plain TCP via `NetClient` to a
7//!   plain-HTTP mock (`tools/hypha-mock-llm/mock_proxy.py --plain`, port 8080).
8//!   Easiest to test — no TLS variable.
9//! - **TLS** (`USE_TLS = true`): TLS 1.3 via the net service (port 8443).
10//!
11//! Either way the cell holds **no** network capability — it only does IPC to the
12//! `net` cell. `core` spawns this cell and talks to it by tid (no registry in P1).
13
14#![no_std]
15#![no_main]

32// ── Configuration ────────────────────────────────────────────────────────────
33// Plaintext (default): the NetClient.tcp_send contract bug (os-gap G16) is fixed
34// and the transport retries until the socket is Established. TLS is currently
35// avoided — the net cell's embedded-tls handshake faults on a real server
36// (os-gap G17, load page fault). Flip to true once G17 is resolved.
37const USE_TLS: bool = false;
38// 10.0.2.2 = QEMU user-net gateway = the host (pin the IP — os-gap G3).
39const PROXY_IP: [u8; 4] = [10, 0, 2, 2];
40const PROXY_PORT: u16 = if USE_TLS { 8443 } else { 8080 };
41const PROXY_HOST: &str = "10.0.2.2";
42const MODEL: &str = "claude-sonnet-4-6";
43// Reply text must fit one IPC message (os-gap G5: Grant streaming comes later).
44const IPC_REPLY_MAX: usize = 3800;
45
46#[no_mangle]
47pub fn main() {
48    println("[hypha/llm-gateway] service ready");
49    // no_heartbeat: an LLM round-trip can exceed the default 5 s watchdog.
50    CellRuntime::new().no_heartbeat().run(|ctx, ev| match ev {
51        AppEvent::Message { sender_tid, data } | AppEvent::RawMessage { sender_tid, data } => {
52            handle(ctx, sender_tid, &data);
53        }
54        AppEvent::Shutdown | AppEvent::ShutdownWith { .. } => ostd::syscall::sys_exit(0),
55        _ => {}
56    });
57}
58
59/// Decode one request, run the completion, reply (truncated to one IPC message).
60fn handle(ctx: &AppContext, sender: usize, data: &[u8]) {
61    let reply = match postcard::from_bytes::<LlmRequest>(data) {
62        Ok(LlmRequest::Complete { prompt }) => match complete(MODEL, prompt) {
63            Ok(text) => classify_reply(text),
64            Err(e) => LlmReply::Error(e),
65        },
66        Err(_) => LlmReply::Error(String::from("bad LlmRequest encoding")),
67    };
68    let mut buf = [0u8; 4096];
69    if let Ok(bytes) = postcard::to_slice(&reply, &mut buf) {
70        let _ = ctx.send(sender, bytes);
71    }
72}

74/// One-shot chat completion. Non-streaming, single inline prompt (P1).
75fn complete(model: &str, prompt: &str) -> Result<String, String> {
76    let body = http::build_chat_body(model, prompt);
77    let request = http::build_post(PROXY_HOST, "/v1/chat/completions", &body);
78
79    print("[gw] ");
80    println(if USE_TLS {
81        "TLS mode -> :8443"
82    } else {
83        "plaintext mode -> :8080"
84    });
85
86    let resp = transport::roundtrip(
87        USE_TLS,
88        PROXY_HOST,
89        PROXY_IP,
90        PROXY_PORT,
91        request.as_bytes(),
92    )?;
93
94    print("[gw] response bytes: ");
95    print_usize(resp.len());
96    println("");
97
98    let body = http::http_body(&resp).ok_or_else(|| String::from("no HTTP body in response"))?;
99    let content =
100        http::extract_content(body).ok_or_else(|| String::from("no content field in response"))?;
101    Ok(content)
102}

104/// Inspect the raw completion text: if it starts with `TOOL_CALL:` return
105/// `ToolCalls`; otherwise return `Text` (truncated to the IPC budget).
106fn classify_reply(text: String) -> LlmReply {
107    if let Some(call) = http::extract_tool_call(&text) {
108        LlmReply::ToolCalls(alloc::vec![call])
109    } else {
110        LlmReply::Text(fit(text, IPC_REPLY_MAX))
111    }
112}
113
114/// Truncate `s` to at most `max` bytes on a char boundary, marking truncation.
115fn fit(mut s: String, max: usize) -> String {
116    if s.len() <= max {
117        return s;
118    }
119    let mut end = max;
120    while end > 0 && !s.is_char_boundary(end) {
121        end -= 1;
122    }
123    s.truncate(end);
124    s.push_str(" …[truncated: P1 4KB IPC cap — Grant streaming later]");
125    s
126}
