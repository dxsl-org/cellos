1//! High-resolution timer for benchmark measurements.
2//!
3//! Reads the kernel's monotonic tick counter via `sys_get_time`.  On RV64 QEMU
4//! this corresponds to `mtime` at ~10 MHz.  Convert ticks → ns by dividing by
5//! `ticks_per_ns()` (queries the Config Cell; falls back to 10 MHz assumed).
6
7use ostd::syscall::sys_get_time;
8
9/// Timer frequency assumed when the Config Cell is unavailable (10 MHz).
10const FALLBACK_FREQ_HZ: u64 = 10_000_000;
11
12/// Nanoseconds per tick — derived from `FALLBACK_FREQ_HZ`.
13/// Multiply tick deltas by this to get nanoseconds.
14pub const NS_PER_TICK: u64 = 1_000_000_000 / FALLBACK_FREQ_HZ;
15
16/// Read the current tick counter value.
17///
18/// Calling this twice and subtracting gives a tick delta.  Use `ticks_to_ns`
19/// to convert to nanoseconds.
20#[inline(always)]
21pub fn read_ticks() -> u64 {
22    sys_get_time()
23}
24
25/// Convert a tick delta to nanoseconds using the fallback frequency.
26///
27/// For accurate results on real hardware, replace `NS_PER_TICK` with a value
28/// queried from the Config Cell (`system.timer_freq_hz`).
29#[inline(always)]
30pub fn ticks_to_ns(ticks: u64) -> u64 {
31    ticks.saturating_mul(NS_PER_TICK)
32}
