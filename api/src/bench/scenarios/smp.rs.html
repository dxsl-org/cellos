1//! SMP throughput scenarios — validate 2-hart work-stealing on QEMU.
2//!
3//! All spawn/notify/force_exit helpers run only in the orchestrator context
4//! (which has SpawnCap).  Only `run_worker` is called from bench-probe.
5
6use crate::framework::timer::NS_PER_TICK;
7use alloc::format;
8use api::task::TaskPriority;
9use core::hint::black_box;
10use ostd::{
11    io::println,
12    syscall::{
13        sys_exit, sys_force_exit, sys_get_time, sys_notify_on_exit, sys_recv, sys_send,
14        sys_set_spawn_args, sys_spawn_pinned, SyscallResult,
15    },
16    task::yield_now,
17};
18
19/// Iterations per worker run.  Calibrated for ≥1 ms on QEMU TCG.
20pub const SMP_WORKER_ITERS: u64 = 500_000;
21
22const PROBE_PATH: &str = "/bin/bench-probe";
23const CAVEAT: &str = " [QEMU-TCG: 2 hart-threads; real-HW shows true 2× parallelism]";
24
25/// Optimizer-resistant arithmetic workload shared by worker and orchestrator.
26fn compute(iters: u64) -> u64 {
27    let mut acc = 0u64;
28    for i in 0..iters {
29        acc = acc.wrapping_add(i.wrapping_mul(3));
30    }
31    black_box(acc)
32}
33
34/// `smp-worker` bench-probe role entry — runs compute, prints, exits.
35pub fn run_worker() -> ! {
36    let acc = compute(SMP_WORKER_ITERS);
37    println(&format!("[smp] worker done acc={}", acc));
38    sys_exit(0);
39}

41// ── Orchestrator-only helpers ─────────────────────────────────────────────────
42
43fn spawn_worker() -> Option<usize> {
44    sys_set_spawn_args("smp-worker");
45    match sys_spawn_pinned(PROBE_PATH, TaskPriority::Normal as u8, 0) {
46        SyscallResult::Ok(tid) => Some(tid),
47        _ => None,
48    }
49}
50
51/// Receive the exit notification for `tid` (caller must have already called
52/// `sys_notify_on_exit(tid)`).  Loops past unrelated notifications.
53fn recv_exit(tid: usize) {
54    let mut buf = [0u8; 8];
55    loop {
56        if let SyscallResult::Ok(s) = sys_recv(0, &mut buf) {
57            if s == tid {
58                break;
59            }
60        }
61    }
62}
63
64/// Register and await `tid`'s exit.
65fn wait_exit(tid: usize) {
66    let _ = sys_notify_on_exit(tid);
67    recv_exit(tid);
68}

70// ── Scenario 1: spawn_rate ────────────────────────────────────────────────────
71
72/// Sequential spawn throughput.  PASS iff ≥ 10 tasks/sec on single-hart QEMU TCG.
73///
74/// Tests that spawn+compute+exit overhead stays bounded. The 10/sec floor
75/// accommodates single-hart QEMU TCG (observed ~15/sec); real 2-hart MTTCG
76/// achieves ≥ 30/sec.
77fn measure_spawn_rate() -> Option<bool> {
78    const N: u64 = 8;
79    const TARGET: u64 = 10; // ≥ 100 ms per full lifecycle budget (conservative for TCG)
80
81    let t0 = sys_get_time();
82    for _ in 0..N {
83        match spawn_worker() {
84            Some(tid) => wait_exit(tid),
85            None => {
86                println("[smp] spawn_rate SKIP — bench-probe not at /bin/bench-probe");
87                return None;
88            }
89        }
90    }
91    let dt_ns = sys_get_time()
92        .saturating_sub(t0)
93        .saturating_mul(NS_PER_TICK);
94    let per_sec = N
95        .saturating_mul(1_000_000_000)
96        .checked_div(dt_ns)
97        .unwrap_or(u64::MAX);
98    let pass = per_sec >= TARGET;
99    println(&format!(
100        "[smp] spawn_rate {}: {}/sec (target ≥{}/sec){}",
101        if pass { "PASS" } else { "FAIL" },
102        per_sec,
103        TARGET,
104        CAVEAT
105    ));
106    Some(pass)
107}

109// ── Scenario 2: ipc_throughput ────────────────────────────────────────────────
110
111/// Sustained IPC round-trips (orchestrator ↔ echo worker).
112/// PASS iff ≥ 5 000 msgs/sec.
113fn measure_ipc_throughput() -> Option<bool> {
114    const MSGS: u64 = 1_000;
115    const TARGET: u64 = 5_000;
116
117    sys_set_spawn_args("ipc-echo");
118    let echo_tid = match sys_spawn_pinned(PROBE_PATH, TaskPriority::Normal as u8, 0) {
119        SyscallResult::Ok(t) => t,
120        _ => {
121            println("[smp] ipc_throughput SKIP — bench-probe not at /bin/bench-probe");
122            return None;
123        }
124    };
125    for _ in 0..20 {
126        yield_now();
127    } // let echo reach its recv loop before measurement
128
129    let mut rbuf = [0u8; 8];
130    let t0 = sys_get_time();
131    for _ in 0..MSGS {
132        let _ = sys_send(echo_tid, &[1u8]);
133        let _ = sys_recv(0, &mut rbuf);
134    }
135    let dt_ns = sys_get_time()
136        .saturating_sub(t0)
137        .saturating_mul(NS_PER_TICK);
138    let _ = sys_force_exit(echo_tid);
139
140    let per_sec = MSGS
141        .saturating_mul(1_000_000_000)
142        .checked_div(dt_ns)
143        .unwrap_or(u64::MAX);
144    let pass = per_sec >= TARGET;
145    println(&format!(
146        "[smp] ipc_throughput {}: {}/sec (target ≥{}/sec){}",
147        if pass { "PASS" } else { "FAIL" },
148        per_sec,
149        TARGET,
150        CAVEAT
151    ));
152    Some(pass)
153}

155// ── Scenario 3: work_distribution ─────────────────────────────────────────────
156
157/// 2-hart scale factor: PASS iff 2×T_single / T_parallel ≥ 1.40.
158///
159/// T_single: orchestrator idle while 1 worker runs.
160/// T_parallel: orchestrator compute + 1 worker run concurrently on 2 harts.
161/// On MTTCG harts both run on separate host threads → scale ≈ 2×.
162fn measure_work_distribution() -> Option<bool> {
163    // T_single: spawn + wait; orchestrator does nothing.
164    let t0 = sys_get_time();
165    let tid1 = spawn_worker().or_else(|| {
166        println("[smp] work_distribution SKIP — bench-probe not at /bin/bench-probe");
167        None
168    })?;
169    wait_exit(tid1);
170    let t_single = sys_get_time().saturating_sub(t0);
171
172    // T_parallel: spawn + register notify before orchestrator compute to prevent
173    // a race where the worker exits before notify_on_exit is registered.
174    let t1 = sys_get_time();
175    let tid2 = spawn_worker().or_else(|| {
176        println("[smp] work_distribution SKIP — bench-probe unavailable for T_parallel run");
177        None
178    })?;
179    let _ = sys_notify_on_exit(tid2); // must register before compute loop
180    let _ = compute(SMP_WORKER_ITERS); // orchestrator's contribution (hart A)
181    recv_exit(tid2); // worker ran on hart B (or was stolen by it)
182    let t_parallel = sys_get_time().saturating_sub(t1);
183
184    // scale_x100 = 2 × T_single × 100 / T_parallel
185    // (zero denominator → report 2.00×)
186    let scale_x100 = t_single
187        .saturating_mul(200)
188        .checked_div(t_parallel)
189        .unwrap_or(200);
190    let pass = scale_x100 >= 140;
191    println(&format!(
192        "[smp] work_distribution {}: scale={}.{:02}x T1={}t Tp={}t (target ≥1.40x){}",
193        if pass { "PASS" } else { "FAIL" },
194        scale_x100 / 100,
195        scale_x100 % 100,
196        t_single,
197        t_parallel,
198        CAVEAT
199    ));
200    Some(pass)
201}

203// ── Suite entry ───────────────────────────────────────────────────────────────
204
205/// Run all 3 SMP throughput scenarios; returns (passed, failed).
206/// SKIP scenarios count as neither.
207pub fn run_smp_suite() -> (u32, u32) {
208    let mut passed = 0u32;
209    let mut failed = 0u32;
210    for result in [
211        measure_spawn_rate(),
212        measure_ipc_throughput(),
213        measure_work_distribution(),
214    ] {
215        match result {
216            Some(true) => passed += 1,
217            Some(false) => failed += 1,
218            None => {} // SKIP
219        }
220    }
221    (passed, failed)
222}
