1//! IPC send/recv round-trip latency benchmark.
2//!
3//! Spawns a private echo peer (another instance of this binary in "ipc-echo"
4//! role) and measures the round-trip time of a 64-byte ping. This avoids any
5//! dependency on VFS TID, which can vary depending on the boot sequence.
6//! PDR target: < 50 µs per round-trip.
7
8use api::benchmark::ViBenchmark;
9use ostd::syscall::{sys_recv, sys_send, sys_set_spawn_args, sys_spawn_pinned, SyscallResult};
10
11// Use bench-probe (VA 0x19000000) so the echo peer doesn't collide with the
12// orchestrator's pages (VA 0x18000000) in the shared SAS page table.
13const SELF_PATH: &str = "/bin/bench-probe";
14
15pub struct IpcSendRecvBench {
16    echo_tid: usize,
17    msg: [u8; 64],
18    buf: [u8; 64],
19}
20
21impl IpcSendRecvBench {
22    pub fn new() -> Self {
23        // Spawn echo peer at Normal priority (1) so it isn't starved when the
24        // RT under_load scenario has 3 Normal-priority load cells spinning.
25        sys_set_spawn_args("ipc-echo");
26        let echo_tid = match sys_spawn_pinned(SELF_PATH, api::TaskPriority::Normal as u8, 0) {
27            SyscallResult::Ok(tid) => tid,
28            _ => {
29                ostd::io::println("FATAL: Failed to spawn bench-probe. Missing in disk image?");
30                // sys_exit diverges (-> !); no trailing value needed to satisfy the match arm type.
31                ostd::syscall::sys_exit(1);
32            }
33        };
34        // Yield a few times so the echo peer reaches its recv loop.
35        for _ in 0..20 {
36            ostd::task::yield_now();
37        }
38        let mut msg = [0u8; 64];
39        msg[0] = 0x42; // ping
40        Self {
41            echo_tid,
42            msg,
43            buf: [0u8; 64],
44        }
45    }
46}
47
48impl ViBenchmark for IpcSendRecvBench {
49    fn name(&self) -> &'static str {
50        "ipc_send_recv"
51    }
52
53    fn run_once(&mut self) -> api::ViResult<u64> {
54        if self.echo_tid == 0 {
55            return Ok(0); // echo not spawned — bench not at /bin/bench
56        }
57        let r1 = sys_send(self.echo_tid, &self.msg);
58        let r2 = sys_recv(0, &mut self.buf);
59        if !matches!(r1, ostd::syscall::SyscallResult::Ok(_))
60            || !matches!(r2, ostd::syscall::SyscallResult::Ok(_))
61        {
62            ostd::io::println(&alloc::format!("IPC err: send={:?} recv={:?}", r1, r2));
63        }
64        Ok(0)
65    }
66}
67
68impl Drop for IpcSendRecvBench {
69    fn drop(&mut self) {
70        // Kill the echo peer so its VA (0x19000000) is freed before RT tests
71        // try to spawn bench-probe in a different role. Without this, the
72        // ipc-echo instance stays alive → VA collision → RT scenarios all fail.
73        if self.echo_tid != 0 {
74            let _ = ostd::syscall::sys_force_exit(self.echo_tid);
75        }
76    }
77}
78
79impl Default for IpcSendRecvBench {
80    fn default() -> Self {
81        Self::new()
82    }
83}
