3/// ADC conversion error kinds.
4#[derive(Debug, Clone, Copy, PartialEq, Eq)]
5pub enum AdcError {
6    /// Channel index out of range.
7    InvalidChannel,
8    /// Hardware did not complete conversion in time.
9    ConversionTimeout,
10    /// Hardware fault or peripheral error.
11    HardwareError,
12}

14/// Analog-to-digital converter trait.
15///
16/// # Contract
17/// - `channel` is a zero-based index. Implementations define the number of channels.
18/// - Raw values are in the range `[0, max_value()]` inclusive.
19/// - `to_millivolts` is a provided default method; implementations may override it
20///   for higher precision or non-linear calibration.
21pub trait ViAdc {
22    type Error: core::fmt::Debug;
23
24    /// Read a raw ADC sample from the given channel.
25    fn read_raw(&mut self, channel: u8) -> Result<u16, Self::Error>;
26
27    /// Maximum raw value (e.g., 4095 for a 12-bit ADC).
28    fn max_value(&self) -> u16;
29
30    /// Number of available channels.
31    fn num_channels(&self) -> u8;
32
33    /// Convert a raw reading to millivolts given the reference supply voltage.
34    ///
35    /// Uses linear scaling: `mV = supply_mv × raw / max_value()`.
36    fn to_millivolts(&self, raw: u16, supply_mv: u32) -> u32 {
37        (supply_mv as u64 * raw as u64 / self.max_value() as u64) as u32
38    }
39}
