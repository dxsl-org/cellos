1//! ARM AArch32 (ARMv7-A) Hardware Abstraction Layer.
2//!
3//! Targets `armv7a-none-eabi` bare-metal.
4//! Sub-modules gated on `#[cfg(target_arch = "arm")]` to avoid AArch32 inline
5//! asm bleeding into non-ARM builds (e.g. AArch64 host compilations).

20// ── Arch trait stub for non-AArch32 hosts ────────────────────────────────────
21#[cfg(not(target_arch = "arm"))]
22impl Arch for AArch32Arch {
23    type Context = usize;
24    fn init(&self) {}
25    unsafe fn switch_context(&self, _: *mut Self::Context, _: *const Self::Context) {}
26    fn enable_interrupts(&self) {}
27    fn disable_interrupts(&self) {}
28    fn wait_for_interrupt(&self) {}
29    fn interrupts_enabled(&self) -> bool {
30        false
31    }
32}

34// ── Full Arch implementation for AArch32 ─────────────────────────────────────
35#[cfg(target_arch = "arm")]
36impl Arch for AArch32Arch {
37    type Context = context::Arm32Context;
38
39    /// Nano bringup: ARM virt GIC init deferred; UART already initialised in kmain.
40    fn init(&self) {}
41
42    unsafe fn switch_context(&self, old: *mut Self::Context, new: *const Self::Context) {
43        // SAFETY: callers guarantee valid, non-aliasing, aligned pointers.
44        context::switch(old, new);
45    }
46
47    fn enable_interrupts(&self) {
48        // SAFETY: cpsie i — enables IRQ in CPSR; standard ARMv7 SVC-mode instruction.
49        unsafe {
50            core::arch::asm!("cpsie i", options(nomem, nostack));
51        }
52    }
53
54    fn disable_interrupts(&self) {
55        // SAFETY: cpsid i — disables IRQ in CPSR.
56        unsafe {
57            core::arch::asm!("cpsid i", options(nomem, nostack));
58        }
59    }
60
61    fn wait_for_interrupt(&self) {
62        // SAFETY: wfi is a standard ARMv7 idle instruction.
63        unsafe {
64            core::arch::asm!("wfi", options(nomem, nostack));
65        }
66    }
67
68    fn interrupts_enabled(&self) -> bool {
69        let cpsr: u32;
70        // SAFETY: mrs reads CPSR without modifying any state.
71        unsafe {
72            core::arch::asm!("mrs {}, cpsr", out(reg) cpsr, options(nomem, nostack));
73        }
74        cpsr & (1 << 7) == 0 // I bit clear = IRQ enabled
75    }
76}

89// ── `hal::arch` public API expected by kernel/src/task.rs ────────────────────
90#[cfg(target_arch = "arm")]
91pub mod arch {
92    /// Minimal ARMv7-A trap frame (nano profile — never populated at runtime).
93    ///
94    /// Fields `regs`, `sepc`, `sstatus` mirror the RV64 layout so scheduler
95    /// code that accesses them compiles. On AArch32 they are zero-initialised
96    /// and never written by hardware.
97    #[repr(C)]
98    #[derive(Debug, Clone, Copy, Default)]
99    pub struct ViTrapFrame {
100        /// General-purpose register file (r0-r15 + padding to 32 slots).
101        pub regs: [usize; 32],
102        /// Program counter (equiv. of RV64 sepc / ELR_EL1).
103        pub sepc: usize,
104        /// Status register (equiv. of RV64 sstatus / SPSR).
105        pub sstatus: usize,
106    }
107
108    /// Context type alias used by `task.rs` BOOT_CONTEXT.
109    pub use super::context::Arm32Context as Context;
110
111    /// GP and TP registers (RISC-V concept; ARM has neither → (0, 0)).
112    pub fn get_gp_tp() -> (usize, usize) {
113        (0, 0)
114    }
115
116    /// Set kernel stack (ARM uses banked registers; stub for nano profile).
117    pub fn set_kernel_stack(_kernel_stack_top: usize) {}
118
119    /// Architecture init — nano profile: GIC/timer deferred; no-op here.
120    pub fn init() {}
121
122    /// Enable IRQs in CPSR (cpsie i).
123    pub fn enable_interrupts() {
124        // SAFETY: standard ARMv7 SVC-mode instruction; only called after boot init.
125        unsafe {
126            core::arch::asm!("cpsie i", options(nomem, nostack));
127        }
128    }
129
130    // Thread trampoline — symbol provided by boot.rs global_asm.
131    extern "C" {
132        pub fn thread_trampoline();
133    }
134}
