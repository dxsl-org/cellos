1//! ARM64 Stage-2 EL2 system-register helpers.
2//!
3//! All functions require the kernel to be running at EL2 (`el2::is_el2()` == true)
4//! and must NOT be called before the Stage-2 page table is fully built and visible
5//! to the CPU (all writes flushed to memory before the `isb`).
6//!
7//! # Register layout
8//!
9//! ## VTCR_EL2 (40-bit IPA, 4 KB granule, SL0=1 start at Level 1)
10//! | Field | Bits | Value |
11//! |-------|------|-------|
12//! | T0SZ  | [5:0] | 24 (IPA = 2^40 = 1 TiB) |
13//! | SL0   | [7:6] | 0b01 (start at L1) |
14//! | IRGN0 | [9:8] | 0b01 (Normal WB-WA inner) |
15//! | ORGN0 | [11:10] | 0b01 (Normal WB-WA outer) |
16//! | SH0   | [13:12] | 0b11 (Inner-shareable) |
17//! | PS    | [18:16] | 0b010 (40-bit PA size) |
18//! | RES1  | [31] | 1 |
19//!
20//! Computed: `0x80023558`
21//!
22//! ## VTTBR_EL2
23//! bits[63:48] = VMID (≥ 1; host is VMID 0)
24//! bits[47:1]  = BADDR (root PA[47:13] for 8 KB-aligned root)
25//!
26//! ## HCR_EL2.VM (bit 0)
27//! Setting this bit enables Stage-2 translation for EL0/EL1.
28//! Must be cleared before tearing down the Stage-2 table.
29
30/// VTCR_EL2 value for 40-bit IPA, 4 KB granule, Level-1 start (SL0=1),
31/// Normal-WB-WA Inner+Outer-shareable, bit[31]=RES1.
32///
33/// T0SZ=24 | SL0=1 | IRGN0=01 | ORGN0=01 | SH0=11 | PS=010 | RES1
34#[cfg(target_arch = "aarch64")]
35const VTCR_VALUE: u64 = 0x8002_3558;
36
37/// Enable Stage-2 translation for the given `vmid` and root page-table at `root_pa`.
38///
39/// Programs VTCR_EL2 and VTTBR_EL2, sets HCR_EL2.VM=1, then issues a full
40/// Stage-2 TLB invalidation to discard any stale cached translations.
41///
42/// # Safety
43/// - Must only be called from EL2 (after `el2::el2_mark_active()`).
44/// - `root_pa` must be 8 KB-aligned (concatenated 2 × L1 tables, VTCR.SL0=1).
45/// - The Stage-2 page table must be fully populated and all writes flushed to
46///   RAM before this function is called.
47/// - `vmid` must be ≥ 1 (VMID 0 is reserved for the EL2 host).
48/// - This function must NOT be called while a vCPU is running (HCR_EL2.VM races).
49#[cfg(target_arch = "aarch64")]
50pub unsafe fn enable_stage2(vmid: u16, root_pa: u64) {
51    debug_assert_eq!(root_pa % (2 * 4096), 0, "S2 root must be 8 KB-aligned");
52    debug_assert!(vmid >= 1, "VMID 0 is reserved for the EL2 host");
53
54    let vttbr = ((vmid as u64) << 48) | root_pa;
55
56    // SAFETY: caller guarantees EL2 context and well-formed root PA.
57    unsafe {
58        core::arch::asm!(
59            // 1. Program VTCR_EL2 (S2 translation parameters).
60            "msr vtcr_el2,  {vtcr}",
61            // 2. Program VTTBR_EL2 (VMID + S2 root PA).
62            "msr vttbr_el2, {vttbr}",
63            "isb",
64            // 3. Set HCR_EL2.VM = bit[0] to enable S2 translation.
65            "mrs {tmp}, hcr_el2",
66            "orr {tmp}, {tmp}, #1",
67            "msr hcr_el2, {tmp}",
68            "isb",
69            // 4. Invalidate all Stage-1 + Stage-2 TLB entries for this VMID.
70            "tlbi vmalls12e1is",
71            "dsb ish",
72            "isb",
73            vtcr  = in(reg) VTCR_VALUE,
74            vttbr = in(reg) vttbr,
75            tmp   = out(reg) _,
76            options(nomem, nostack),
77        );
78    }
79}

81/// Disable Stage-2 translation (clear HCR_EL2.VM=0) and flush the S2 TLB.
82///
83/// Call during VM teardown before freeing the Stage-2 page table frames.
84///
85/// # Safety
86/// Must be called from EL2 with no vCPU currently running.
87#[cfg(target_arch = "aarch64")]
88pub unsafe fn disable_stage2() {
89    // SAFETY: EL2 context; clearing HCR_EL2.VM disables guest translation.
90    unsafe {
91        core::arch::asm!(
92            "mrs {tmp}, hcr_el2",
93            "bic {tmp}, {tmp}, #1",   // clear VM bit
94            "msr hcr_el2, {tmp}",
95            "isb",
96            "tlbi vmalls12e1is",
97            "dsb ish",
98            "isb",
99            tmp = out(reg) _,
100            options(nomem, nostack),
101        );
102    }
103}

105/// Invalidate all Stage-1 + Stage-2 TLB entries for all VMIDs (broadcast).
106///
107/// Use after bulk Stage-2 table changes or during VM teardown.
108///
109/// # Safety
110/// Must be called from EL2.
111#[cfg(target_arch = "aarch64")]
112pub unsafe fn s2_tlb_flush_all() {
113    // SAFETY: EL2 broadcast TLB invalidation; no memory mutation.
114    unsafe {
115        core::arch::asm!(
116            "tlbi vmalls12e1is",
117            "dsb ish",
118            "isb",
119            options(nomem, nostack),
120        );
121    }
122}

124/// Invalidate the Stage-2 TLB entry for a single guest IPA.
125///
126/// Encodes `ipa >> 12` into the `TLBI IPAS2E1IS` operand as required by the
127/// ARM architecture (D8-7204: Xt = IPA[47:12]).  Also issues `TLBI VMALLE1IS`
128/// to flush any Stage-1 entries cached during the S2 walk.
129///
130/// # Safety
131/// Must be called from EL2 after updating the Stage-2 descriptor for `ipa`.
132#[cfg(target_arch = "aarch64")]
133pub unsafe fn s2_tlb_flush_ipa(ipa: u64) {
134    let encoded = ipa >> 12; // TLBI IPAS2E1IS encodes IPA[47:12] in Xt
135                             // SAFETY: EL2 TLB maintenance; encoded IPA operand is correct per ARM DDI.
136    unsafe {
137        core::arch::asm!(
138            "tlbi ipas2e1is, {x}",  // invalidate Stage-2 entry for this IPA
139            "dsb ish",
140            "tlbi vmalle1is",       // flush Stage-1 entries cached during S2 walk
141            "dsb ish",
142            "isb",
143            x = in(reg) encoded,
144            options(nomem, nostack),
145        );
146    }
147}
