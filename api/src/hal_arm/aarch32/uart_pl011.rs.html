1//! PL011 UART driver for ARM32 — MMIO at 0x09000000 (QEMU virt machine).
2//!
3//! Register accesses use 32-bit volatile MMIO writes (ARM32 has no 64-bit registers).
4//! Base address and baud settings are identical to the AArch64 version.
5
6const PL011_BASE: u32 = 0x0900_0000; // QEMU ARM virt machine PL011
7
8const UARTDR: u32 = PL011_BASE; // Data Register (TX/RX)
9const UARTFR: u32 = PL011_BASE + 0x018; // Flag Register
10const UARTIBRD: u32 = PL011_BASE + 0x024; // Integer Baud Rate Divisor
11const UARTFBRD: u32 = PL011_BASE + 0x028; // Fractional Baud Rate Divisor
12const UARTLCR: u32 = PL011_BASE + 0x02C; // Line Control Register
13const UARTCR: u32 = PL011_BASE + 0x030; // Control Register
14
15#[inline(always)]
16unsafe fn mmio_write(addr: u32, val: u32) {
17    // SAFETY: caller guarantees addr is a valid PL011 MMIO register.
18    core::ptr::write_volatile(addr as *mut u32, val);
19}
20
21#[inline(always)]
22unsafe fn mmio_read(addr: u32) -> u32 {
23    // SAFETY: caller guarantees addr is a valid PL011 MMIO register.
24    core::ptr::read_volatile(addr as *const u32)
25}
26
27/// Initialise PL011 at 115200 baud (48 MHz reference clock), 8N1, FIFO enabled.
28pub fn init() {
29    // SAFETY: MMIO addresses are fixed QEMU virt device addresses.
30    unsafe {
31        mmio_write(UARTCR, 0); // disable UART
32        mmio_write(UARTIBRD, 26); // integer divisor  (115200 baud @ 48 MHz)
33        mmio_write(UARTFBRD, 3); // fractional divisor
34        mmio_write(UARTLCR, 0x70); // 8N1 + FIFO enable (FEN bit)
35        mmio_write(UARTCR, 0x301); // TXE | RXE | UARTEN
36    }
37}
38
39/// Write a single byte, blocking until the TX FIFO is not full.
40pub fn putchar(c: u8) {
41    // SAFETY: MMIO addresses are valid PL011 registers; blocking spin is safe.
42    unsafe {
43        while mmio_read(UARTFR) & (1 << 5) != 0 {
44            core::hint::spin_loop();
45        }
46        mmio_write(UARTDR, c as u32);
47    }
48}
