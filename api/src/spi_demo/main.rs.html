15// Gate on gpio manifest flag (Option A: bit-bang SPI is GPIO in disguise).
16// This grants PL061 MMIO access at spawn without requiring a dedicated SPI flag.
17declare_manifest!(
18    block_io = false,
19    network = false,
20    spawn = false,
21    gpio = true,
22    uart = false
23);
24
25#[no_mangle]
26pub fn main() {
27    println("[spi-demo] SPI bit-bang demo (MOSI=2, MISO=3, SCK=4, CS=5)");
28
29    match Pl061Gpio::open() {
30        Ok(gpio) => run_spi_demo(gpio),
31        Err(ViError::PermissionDenied) => {
32            println("[spi-demo] SPI unavailable (gpio cap not granted — non-aarch64 target)");
33        }
34        Err(_) => {
35            println("[spi-demo] SPI unavailable (GPIO open failed)");
36        }
37    }
38}
39
40fn run_spi_demo(gpio: Pl061Gpio) {
41    let mut spi = BitBangSpi::new(gpio);
42
43    // ── TX-only write: primary assertion ─────────────────────────────────────
44    // write() clocks out bytes via MOSI/SCK/CS without sampling MISO.
45    // On QEMU this validates the full GPIO MMIO path.
46    match spi.write(&[0xA5, 0x3C, 0x00]) {
47        Ok(()) => println("[spi-demo] SPI TX OK (0xA5 0x3C 0x00)"),
48        Err(_) => {
49            println("[spi-demo] SPI BusError — TX failed");
50            return;
51        }
52    }
53
54    // ── Full-duplex transfer: MISO floats to 0x00 in QEMU ────────────────────
55    // transfer() clocks out tx bytes and simultaneously clocks in rx bytes.
56    // QEMU PL061 has no MOSI→MISO loopback; rx will be 0x00 — expected.
57    let mut rx = [0xFFu8; 2];
58    match spi.transfer(&[0xAA, 0x55], &mut rx) {
59        Ok(()) => {
60            let msg = alloc::format!(
61                "[spi-demo] SPI transfer OK: sent 0xAA 0x55, recv 0x{:02X} 0x{:02X} (QEMU MISO=0)",
62                rx[0],
63                rx[1]
64            );
65            println(&msg);
66        }
67        Err(_) => {
68            println("[spi-demo] SPI BusError — transfer failed");
69            return;
70        }
71    }
72
73    println("[spi-demo] SPI demo complete");
74}
