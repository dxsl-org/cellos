4//! SiFive GPIO driver — QEMU `sifive_u` machine (FU540/FU740 GPIO0).
5//!
6//! GPIO0 base: 0x1001_2000, 32 pins, 32-bit registers (one bit per pin).
7//! The kernel allowlist must include this range for `open()` to succeed;
8//! on QEMU virt (riscv64gc) the allowlist is empty, so `open()` returns
9//! `PermissionDenied` and callers should skip or fall back.
10
11use hal_gpio::{Edge, PinDir, ViGpio};
12use ostd::mmio::MmioRegion;
13use types::{ViError, ViResult};
14
15const GPIO_BASE: usize = 0x1001_2000;
16const GPIO_SIZE: usize = 0x1000;
17
18// Register offsets (byte addresses, 32-bit wide).
19const INPUT_VAL: usize = 0x00;
20const INPUT_EN: usize = 0x04;
21const OUTPUT_EN: usize = 0x08;
22const OUTPUT_VAL: usize = 0x0C;
23const RISE_IE: usize = 0x18;
24const FALL_IE: usize = 0x20;
25
26/// SiFive GPIO controller (32-pin bank, QEMU `sifive_u` machine).
27pub struct SiFiveGpio {
28    mmio: MmioRegion,
29}
30
31impl SiFiveGpio {
32    /// Request exclusive MMIO ownership from the kernel resource registry.
33    ///
34    /// Returns `Err(PermissionDenied)` when not running on a `sifive_u` QEMU target
35    /// (allowlist empty for virt/aarch64/x86_64).
36    pub fn open() -> ViResult<Self> {
37        let mmio = ostd::mmio::request_region(GPIO_BASE, GPIO_SIZE)?;
38        Ok(Self { mmio })
39    }
40
41    /// Set or clear bit `pin` in a 32-bit register at `reg`.
42    fn rw_bit(mmio: &MmioRegion, reg: usize, pin: u8, set: bool) -> ViResult<()> {
43        let cur = mmio.read_u32(reg)?;
44        let bit = 1u32 << pin;
45        mmio.write_u32(reg, if set { cur | bit } else { cur & !bit })
46    }
47
48    /// Test bit `pin` in a 32-bit register at `reg`.
49    fn test_bit(mmio: &MmioRegion, reg: usize, pin: u8) -> ViResult<bool> {
50        Ok((mmio.read_u32(reg)? >> pin) & 1 != 0)
51    }
52}
53
54impl ViGpio for SiFiveGpio {
55    fn set_direction(&mut self, pin: u8, dir: PinDir) -> ViResult<()> {
56        match dir {
57            PinDir::Input => {
58                // Disable output first to avoid glitch, then enable input.
59                Self::rw_bit(&self.mmio, OUTPUT_EN, pin, false)?;
60                Self::rw_bit(&self.mmio, INPUT_EN, pin, true)
61            }
62            PinDir::Output => {
63                Self::rw_bit(&self.mmio, INPUT_EN, pin, false)?;
64                Self::rw_bit(&self.mmio, OUTPUT_EN, pin, true)
65            }
66        }
67    }
68
69    fn read_pin(&self, pin: u8) -> ViResult<bool> {
70        Self::test_bit(&self.mmio, INPUT_VAL, pin)
71    }
72
73    fn write_pin(&mut self, pin: u8, high: bool) -> ViResult<()> {
74        // SiFive does not enforce direction in hardware — ViGpio contract says
75        // callers must set Output direction before write_pin.
76        if !Self::test_bit(&self.mmio, OUTPUT_EN, pin)? {
77            return Err(ViError::InvalidInput);
78        }
79        Self::rw_bit(&self.mmio, OUTPUT_VAL, pin, high)
80    }
81
82    fn enable_edge_irq(&mut self, pin: u8, edge: Edge) -> ViResult<()> {
83        if matches!(edge, Edge::Rising | Edge::Both) {
84            Self::rw_bit(&self.mmio, RISE_IE, pin, true)?;
85        }
86        if matches!(edge, Edge::Falling | Edge::Both) {
87            Self::rw_bit(&self.mmio, FALL_IE, pin, true)?;
88        }
89        Ok(())
90    }
91
92    fn disable_irq(&mut self, pin: u8) -> ViResult<()> {
93        Self::rw_bit(&self.mmio, RISE_IE, pin, false)?;
94        Self::rw_bit(&self.mmio, FALL_IE, pin, false)
95    }
96}
