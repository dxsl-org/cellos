1//! HKDF-SHA256 (RFC 5869) — extract + expand, hand-rolled over the local
2//! [`super::sha256`] implementation.
3//!
4//! Not the `hkdf` crate on purpose (dossier-4 Decision 1 / YAGNI+KISS): SHA-256 is
5//! already local, HKDF-SHA256 is ~40 lines on top of it, and this must build in
6//! kernel-adjacent `no_std` contexts where a new dependency is unwanted.
7
8use super::sha256::sha256;
9
10const BLOCK_SIZE: usize = 64; // SHA-256 block size
11const HASH_LEN: usize = 32; // SHA-256 output size
12
13/// HMAC-SHA256(key, message).
14fn hmac_sha256(key: &[u8], message: &[u8]) -> [u8; HASH_LEN] {
15    // Key preparation: hash if longer than the block size, zero-pad otherwise.
16    // An empty key zero-pads to a full block of zero bytes — this is exactly
17    // equivalent to RFC 5869's "salt defaults to HashLen zero bytes" rule, since
18    // padding HashLen zero bytes out to BLOCK_SIZE with more zeros is still all
19    // zeros (see `extract`'s doc comment).
20    let mut key_block = [0u8; BLOCK_SIZE];
21    if key.len() > BLOCK_SIZE {
22        let hashed = sha256(key);
23        key_block[..HASH_LEN].copy_from_slice(&hashed);
24    } else {
25        key_block[..key.len()].copy_from_slice(key);
26    }
27
28    let mut ipad = [0x36u8; BLOCK_SIZE];
29    let mut opad = [0x5cu8; BLOCK_SIZE];
30    for i in 0..BLOCK_SIZE {
31        ipad[i] ^= key_block[i];
32        opad[i] ^= key_block[i];
33    }
34
35    // inner = SHA256(ipad || message) — computed incrementally via a scratch
36    // buffer would need alloc for arbitrary-length `message`; instead hash ipad
37    // and message as two logically-concatenated inputs by re-using `sha256`
38    // over a single buffer built from bounded stack space is not possible for
39    // unbounded `message`, so we hash via two calls is NOT how SHA-256 works
40    // (it is not a streaming API here) — build the exact concatenation on the
41    // heap-free path by allocating on the stack sized to ipad+message. Since
42    // `message` length is caller-controlled but always small in this crate
43    // (32-byte IKM or a 145-byte token body), a fixed generous stack buffer is
44    // acceptable and avoids alloc.
45    let mut inner_buf = [0u8; BLOCK_SIZE + MAX_MESSAGE_LEN];
46    assert!(
47        message.len() <= MAX_MESSAGE_LEN,
48        "hmac_sha256: message exceeds crate-internal bound"
49    );
50    inner_buf[..BLOCK_SIZE].copy_from_slice(&ipad);
51    inner_buf[BLOCK_SIZE..BLOCK_SIZE + message.len()].copy_from_slice(message);
52    let inner = sha256(&inner_buf[..BLOCK_SIZE + message.len()]);
53
54    let mut outer_buf = [0u8; BLOCK_SIZE + HASH_LEN];
55    outer_buf[..BLOCK_SIZE].copy_from_slice(&opad);
56    outer_buf[BLOCK_SIZE..].copy_from_slice(&inner);
57    sha256(&outer_buf)
58}
59
60/// The largest `message` this crate's HMAC ever hashes: HKDF-Expand's `T(i-1) ||
61/// info || counter` where `T(i-1)` is 32 bytes, `info` is at most
62/// `MAX_INFO_LEN`, and the counter is 1 byte. Bounded so `hmac_sha256` can use a
63/// fixed stack buffer instead of `alloc`.
64const MAX_INFO_LEN: usize = 32;
65const MAX_MESSAGE_LEN: usize = HASH_LEN + MAX_INFO_LEN + 1;
66
67/// HKDF-Extract: `PRK = HMAC-Hash(salt, IKM)`. An empty `salt` is RFC
68/// 5869-compliant (see `hmac_sha256`'s key-padding note) — do not special-case it.
69pub fn extract(salt: &[u8], ikm: &[u8]) -> [u8; HASH_LEN] {
70    hmac_sha256(salt, ikm)
71}
72
73/// HKDF-Expand: fills `okm` (any length up to `255 * HASH_LEN`, RFC 5869 §2.3)
74/// with `T(1) || T(2) || ...`, `T(i) = HMAC-Hash(PRK, T(i-1) || info || i)`.
75/// `info` must fit in `MAX_INFO_LEN` bytes (32) — generous for this crate's only
76/// use (empty info in `cdi::derive_cdi`); asserts rather than truncates.
77pub fn expand(prk: &[u8; HASH_LEN], info: &[u8], okm: &mut [u8]) {
78    assert!(
79        info.len() <= MAX_INFO_LEN,
80        "hkdf::expand: info exceeds crate-internal bound"
81    );
82    assert!(
83        okm.len() <= 255 * HASH_LEN,
84        "hkdf::expand: requested length exceeds RFC 5869 bound"
85    );
86
87    let mut t_prev: [u8; HASH_LEN] = [0u8; HASH_LEN];
88    let mut t_prev_len = 0usize;
89    let mut counter: u8 = 0;
90    let mut written = 0usize;
91
92    while written < okm.len() {
93        counter += 1;
94        let mut msg = [0u8; MAX_MESSAGE_LEN];
95        msg[..t_prev_len].copy_from_slice(&t_prev[..t_prev_len]);
96        let mut off = t_prev_len;
97        msg[off..off + info.len()].copy_from_slice(info);
98        off += info.len();
99        msg[off] = counter;
100        off += 1;
101
102        let t = hmac_sha256(prk, &msg[..off]);
103        let take = core::cmp::min(HASH_LEN, okm.len() - written);
104        okm[written..written + take].copy_from_slice(&t[..take]);
105        written += take;
106
107        t_prev = t;
108        t_prev_len = HASH_LEN;
109    }
110}

116    // RFC 5869 §A.1 — Test Case 1: Basic test case with SHA-256.
117    #[test]
118    fn rfc5869_test_case_1() {
119        let ikm = [0x0bu8; 22];
120        let salt: [u8; 13] = [
121            0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09, 0x0a, 0x0b, 0x0c,
122        ];
123        let info: [u8; 10] = [0xf0, 0xf1, 0xf2, 0xf3, 0xf4, 0xf5, 0xf6, 0xf7, 0xf8, 0xf9];
124
125        let prk = extract(&salt, &ikm);
126        assert_eq!(
127            prk,
128            [
129                0x07, 0x77, 0x09, 0x36, 0x2c, 0x2e, 0x32, 0xdf, 0x0d, 0xdc, 0x3f, 0x0d, 0xc4, 0x7b,
130                0xba, 0x63, 0x90, 0xb6, 0xc7, 0x3b, 0xb5, 0x0f, 0x9c, 0x31, 0x22, 0xec, 0x84, 0x4a,

133            "PRK mismatch (RFC 5869 A.1)"
134        );

136        let mut okm = [0u8; 42];

147    }
