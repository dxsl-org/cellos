1//! Trap-emulated GICv2 Distributor (GICD) shadow register model.
2//!
3//! Security contract (M5, Red Team): this module ONLY maintains a shadow register
4//! file. It NEVER issues direct MMIO writes to the physical GICD at 0x08000000.
5//! Physical GIC access stays exclusively with the HAL GIC driver in the kernel.
6//!
7//! Linux probes GICD during `gic_init_bases` to read TYPER (IRQ count) and IIDR
8//! (implementer). It then programs ISENABLER/IPRIORITYR/ITARGETSR for each IRQ.
9//! We emulate just enough for a clean init: writes are accepted silently; reads
10//! return safe/plausible values.
11
12/// GICD base IPA.
13pub const GICD_BASE_IPA: u64 = 0x0800_0000;
14pub const GICD_SIZE: u64 = 0x0001_0000; // 64 KiB
15
16/// GICC base IPA (virtual CPU interface — also emulated as pass-through silent).
17pub const GICC_BASE_IPA: u64 = 0x0801_0000;
18pub const GICC_SIZE: u64 = 0x0001_0000;
19
20/// GICD register offsets.
21mod reg {
22    pub const GICD_CTLR: u64 = 0x000;
23    pub const GICD_TYPER: u64 = 0x004; // read: (ITLinesNumber-1) | ...
24    pub const GICD_IIDR: u64 = 0x008; // implementer ID
25    pub const GICD_ISENABLER: u64 = 0x100; // [0..31]: enable-set per IRQ word
26    pub const GICD_ICENABLER: u64 = 0x180; // clear-enable
27    pub const GICD_ISPENDR: u64 = 0x200; // set-pending
28    pub const GICD_ICPENDR: u64 = 0x280; // clear-pending
29    pub const GICD_IPRIORITYR: u64 = 0x400; // priority byte per IRQ
30    pub const GICD_ITARGETSR: u64 = 0x800; // target CPU mask byte per IRQ
31    pub const GICD_ICFGR: u64 = 0xC00; // level/edge config
32    pub const GICD_SGIR: u64 = 0xF00; // software generated IRQ
33}
34
35/// Maximum IRQ lines emulated (GICv2 supports up to 1020).
36const MAX_IRQS: usize = 256;
37const WORD_COUNT: usize = MAX_IRQS / 32; // 8 words for enable/pending
38
39/// GICv2 Distributor shadow state.
40pub struct Gicd {
41    ctlr: u32,
42    isenabler: [u32; WORD_COUNT],
43    // ICENABLER writes clear bits directly in `isenabler` (matching real GICv2
44    // semantics: ISENABLER/ICENABLER are two write-only views onto the same
45    // underlying enable-bit state). This field is therefore write-only shadow
46    // state at the current emulation depth — never read back.
47    #[allow(dead_code)]
48    icenabler: [u32; WORD_COUNT],
49    ispendr: [u32; WORD_COUNT],
50    ipriorityr: [u8; MAX_IRQS],
51    itargetsr: [u8; MAX_IRQS],
52    icfgr: [u32; MAX_IRQS / 16],
53}
54
55impl Gicd {
56    pub const fn new() -> Self {
57        Self {
58            ctlr: 0,
59            isenabler: [0u32; WORD_COUNT],
60            icenabler: [0u32; WORD_COUNT],
61            ispendr: [0u32; WORD_COUNT],
62            ipriorityr: [0xa0u8; MAX_IRQS], // default: low priority
63            itargetsr: [0x01u8; MAX_IRQS],  // default: CPU 0
64            icfgr: [0u32; MAX_IRQS / 16],
65        }
66    }
67
68    /// Handle a GICD MMIO write.
69    pub fn write(&mut self, offset: u64, val: u64, size: u8) {
70        let val32 = val as u32;
71        match offset {
72            reg::GICD_CTLR => {
73                self.ctlr = val32;
74            }
75            o if (reg::GICD_ISENABLER..reg::GICD_ICENABLER).contains(&o) => {
76                let idx = ((o - reg::GICD_ISENABLER) / 4) as usize;
77                if idx < WORD_COUNT {
78                    self.isenabler[idx] |= val32;
79                }
80            }
81            o if (reg::GICD_ICENABLER..reg::GICD_ISPENDR).contains(&o) => {
82                let idx = ((o - reg::GICD_ICENABLER) / 4) as usize;
83                if idx < WORD_COUNT {
84                    self.isenabler[idx] &= !val32;
85                }
86            }
87            o if (reg::GICD_ISPENDR..reg::GICD_ICPENDR).contains(&o) => {
88                let idx = ((o - reg::GICD_ISPENDR) / 4) as usize;
89                if idx < WORD_COUNT {
90                    self.ispendr[idx] |= val32;
91                }
92            }
93            o if (reg::GICD_ICPENDR..reg::GICD_IPRIORITYR).contains(&o) => {
94                let idx = ((o - reg::GICD_ICPENDR) / 4) as usize;
95                if idx < WORD_COUNT {
96                    self.ispendr[idx] &= !val32;
97                }
98            }
99            o if (reg::GICD_IPRIORITYR..reg::GICD_ITARGETSR).contains(&o) => {
100                let base = (o - reg::GICD_IPRIORITYR) as usize;
101                self.write_bytes(&mut self.ipriorityr.clone(), base, val, size);
102                // Workaround: update in place
103                let bytes = val.to_le_bytes();
104                let count = size as usize;
105                for (i, &b) in bytes.iter().enumerate().take(count) {
106                    if base + i < MAX_IRQS {
107                        self.ipriorityr[base + i] = b;
108                    }
109                }
110            }
111            o if (reg::GICD_ITARGETSR..reg::GICD_ICFGR).contains(&o) => {
112                let base = (o - reg::GICD_ITARGETSR) as usize;
113                let bytes = val.to_le_bytes();
114                let count = size as usize;
115                for (i, &b) in bytes.iter().enumerate().take(count) {
116                    if base + i < MAX_IRQS {
117                        self.itargetsr[base + i] = b;
118                    }
119                }
120            }
121            o if (reg::GICD_ICFGR..reg::GICD_SGIR).contains(&o) => {
122                let idx = ((o - reg::GICD_ICFGR) / 4) as usize;
123                if idx < self.icfgr.len() {
124                    self.icfgr[idx] = val32;
125                }
126            }
127            _ => {} // SGIR and unknown: accept silently
128        }
129    }
130
131    /// Handle a GICD MMIO read. Returns a plausible value.
132    pub fn read(&self, offset: u64, size: u8) -> u64 {
133        match offset {
134            reg::GICD_CTLR => self.ctlr as u64,
135            // TYPER: ITLinesNumber=7 → (7+1)*32=256 IRQs; 1 CPU; SecurityExtn=0.
136            reg::GICD_TYPER => 0x0000_0007u64,
137            // IIDR: ARM implementer, GICv2 product.
138            reg::GICD_IIDR => 0x0200_143Bu64,
139            o if (reg::GICD_ISENABLER..reg::GICD_ICENABLER).contains(&o) => {
140                let idx = ((o - reg::GICD_ISENABLER) / 4) as usize;
141                if idx < WORD_COUNT {
142                    self.isenabler[idx] as u64
143                } else {
144                    0
145                }
146            }
147            o if (reg::GICD_IPRIORITYR..reg::GICD_ITARGETSR).contains(&o) => {
148                self.read_bytes(&self.ipriorityr, (o - reg::GICD_IPRIORITYR) as usize, size)
149            }
150            o if (reg::GICD_ITARGETSR..reg::GICD_ICFGR).contains(&o) => {
151                self.read_bytes(&self.itargetsr, (o - reg::GICD_ITARGETSR) as usize, size)
152            }
153            _ => 0,
154        }
155    }
156
157    /// True if `ipa` falls within the GICD MMIO window.
158    pub fn owns_gicd(ipa: u64) -> bool {
159        (GICD_BASE_IPA..GICD_BASE_IPA + GICD_SIZE).contains(&ipa)
160    }
161
162    /// True if `ipa` falls within the GICC MMIO window.
163    pub fn owns_gicc(ipa: u64) -> bool {
164        (GICC_BASE_IPA..GICC_BASE_IPA + GICC_SIZE).contains(&ipa)
165    }
166
167    fn write_bytes(&self, _buf: &mut [u8], _base: usize, _val: u64, _size: u8) {}
168
169    fn read_bytes(&self, buf: &[u8], base: usize, size: u8) -> u64 {
170        let mut out = 0u64;
171        let count = (size as usize).min(8);
172        for i in 0..count {
173            if base + i < buf.len() {
174                out |= (buf[base + i] as u64) << (i * 8);
175            }
176        }
177        out
178    }
