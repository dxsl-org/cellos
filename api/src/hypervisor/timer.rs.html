1//! Virtual timer + VI-bit interrupt injection policy.
2//!
3//! The ARM64 virtual timer (CNTV / PPI 27) is what Linux uses for scheduling.
4//! The kernel sets CNTVOFF_EL2=0 at each vcpu entry (run_vcpu_impl, P05 addition),
5//! so the guest counter matches the host physical counter.
6//!
7//! When the guest issues WFI (trapped by HCR_EL2.TWI), we check if the virtual
8//! timer deadline has passed. If so, we inject PPI 27 via sys_inject_irq, which
9//! sets HCR_EL2.VI (one pending virtual IRQ). On re-entry the guest sees the IRQ
10//! and its timer handler fires.
11//!
12//! This is the "2 exits per timer IRQ" model: WFI → check → inject VI → re-enter →
13//! guest IRQ handler runs → clears CNTV_CTL.ISTATUS → next WFI fires normally.
14
15#[cfg(target_arch = "aarch64")]
16use api::syscall::ViSyscall;
17
18/// GICv2 PPI 27 = virtual timer interrupt.
19///
20/// Only referenced from the `target_arch = "aarch64"` branch of
21/// `inject_timer_irq` below; on other build targets (e.g. this riscv64gc
22/// build) it is unused, so allow dead_code rather than cfg-gating the
23/// constant itself out of the public API.
24#[allow(dead_code)]
25pub const VIRT_TIMER_PPI: u32 = 27;
26
27/// Check if the guest virtual timer has fired (CNTV_CTL.ENABLE && CNTV_CVAL ≤ now).
28///
29/// `cntv_ctl` and `cntv_cval` are the guest shadow values read from vcpu registers
30/// via `sys_vcpu_regs`. `cntpct` is the host physical counter value from `sys_get_time`.
31///
32/// Not yet wired to a caller — scaffolding ahead of full vCPU timer
33/// virtualization (the WFI-trap → check-deadline → inject-VI loop described
34/// in the module doc above is not yet assembled into a driving loop).
35#[allow(dead_code)]
36pub fn is_timer_due(cntv_ctl: u64, cntv_cval: u64, cntpct: u64) -> bool {
37    const ENABLE: u64 = 1 << 0;
38    const IMASK: u64 = 1 << 1;
39    let enabled = cntv_ctl & ENABLE != 0;
40    let not_masked = cntv_ctl & IMASK == 0;
41    enabled && not_masked && cntpct >= cntv_cval
42}
43
44/// Inject PPI 27 into vcpu (vm_id, vcpu_id) to signal a virtual timer expiry.
45///
46/// The kernel sets HCR_EL2.VI on the next vcpu entry, causing the guest to take
47/// a virtual IRQ at EL1. This is the "VI-bit injection" path (sufficient for P05;
48/// proper GICH LR injection is P09).
49pub fn inject_timer_irq(vm_id: usize, vcpu_id: usize) {
50    #[cfg(target_arch = "aarch64")]
51    {
52        // sys_inject_irq(vm_id, vcpu_id, intid=27)
53        unsafe {
54            let mut ret: usize;
55            core::arch::asm!(
56                "svc #0",
57                inlateout("x0") ViSyscall::InjectIrq as usize => ret,
58                in("x1") vm_id,
59                in("x2") vcpu_id,
60                in("x3") VIRT_TIMER_PPI as usize,
61                in("x4") 0usize,
62                options(nostack, preserves_flags),
63            );
64            let _ = ret;
65        }
66    }
67    #[cfg(not(target_arch = "aarch64"))]
68    {
69        let _ = (vm_id, vcpu_id);
70        unimplemented!("inject_timer_irq: ARM64 only");
71    }
72}
