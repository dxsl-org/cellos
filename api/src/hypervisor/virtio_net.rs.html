1//! virtio-net device model (DeviceID=1, virtio-mmio slot 2 → SPI 18).
2//!
3//! TX: guest writes TX avail ring → VMM reads chain → L2Send to Net Cell → kernel NIC TX.
4//! RX: Net Cell L2Recv delivers frame → VMM fills RX avail descriptor → guest SPI 18.
5//!
6//! The 10-byte virtio_net_hdr is prepended on RX and stripped on TX.
7
8extern crate alloc;
9use crate::virtio_mmio::{QueueCfg, VirtioDevice, VirtioMmio};
10use crate::virtqueue::{process_notify, DescBuf};
11use alloc::vec;
12
13/// MAC address presented to the guest virtio-net device.
14pub const GUEST_MAC: [u8; 6] = [0x52, 0x54, 0x00, 0xAA, 0xBB, 0xCC];
15
16/// GICv2 SPI for virtio-mmio slot 2 (intid 18, matches DTB and MMIO_HOLES pre-map).
17const NET_SPI: u32 = 18;
18
19/// virtio-net feature bit: device provides a MAC address in config space.
20const VIRTIO_NET_F_MAC: u32 = 1 << 5;
21
22pub struct NetDev {
23    /// Net Cell TID for L2 frame IPC; 0 = net service unavailable.
24    pub net_tid: usize,
25    rx_last_avail: u16,
26    rx_used_idx: u16,
27    tx_last_avail: u16,
28    tx_used_idx: u16,
29}
30
31impl NetDev {
32    pub fn new(net_tid: usize) -> Self {
33        Self {
34            net_tid,
35            rx_last_avail: 0,
36            rx_used_idx: 0,
37            tx_last_avail: 0,
38            tx_used_idx: 0,
39        }
40    }
41
42    /// Inject one received Ethernet frame into the guest RX virtqueue.
43    ///
44    /// Prepends a 10-byte zero virtio_net_hdr, fills one avail descriptor chain,
45    /// advances the used ring, and injects SPI 18 to notify the guest.
46    /// Silently drops the frame if the avail ring is empty (no guest buffers ready).
47    pub fn push_rx_frame(
48        &mut self,
49        frame: &[u8],
50        vm_id: usize,
51        vcpu_id: usize,
52        net_vmio: &VirtioMmio,
53    ) {
54        let qcfg = net_vmio.queue_cfg(0); // RX queue (queue 0)
55        if qcfg.num == 0 || !qcfg.ready {
56            return;
57        }
58
59        // Read avail.idx to check if the guest has empty buffers ready.
60        let mut b2 = [0u8; 2];
61        if crate::vmm::read_guest_memory(vm_id, qcfg.avail_gpa + 2, &mut b2) != 2 {
62            return;
63        }
64        let avail_idx = u16::from_le_bytes(b2);
65        if self.rx_last_avail == avail_idx {
66            return;
67        } // No RX buffers available
68
69        // Get the head descriptor index from avail.ring[last_avail_idx % q_size].
70        let ring_off = 4 + (self.rx_last_avail as usize % qcfg.num as usize) * 2;
71        if crate::vmm::read_guest_memory(vm_id, qcfg.avail_gpa + ring_off as u64, &mut b2) != 2 {
72            return;
73        }
74        let head = u16::from_le_bytes(b2) as usize;
75        self.rx_last_avail = self.rx_last_avail.wrapping_add(1);
76
77        // Build payload: 10-byte zero virtio_net_hdr followed by the raw Ethernet frame.
78        let mut payload = vec![0u8; 10 + frame.len()];
79        payload[10..].copy_from_slice(frame);
80
81        // Walk descriptor chain and write payload into writable (VRING_DESC_F_WRITE) buffers.
82        let mut pos = 0usize;
83        let mut cur = head;
84        for _ in 0..64 {
85            let mut raw = [0u8; 16]; // VirtqDesc: addr(8) + len(4) + flags(2) + next(2)
86            let desc_gpa = qcfg.desc_gpa + (cur as u64) * 16;
87            if crate::vmm::read_guest_memory(vm_id, desc_gpa, &mut raw) != 16 {
88                break;
89            }
90
91            let addr = u64::from_le_bytes([
92                raw[0], raw[1], raw[2], raw[3], raw[4], raw[5], raw[6], raw[7],
93            ]);
94            let len = u32::from_le_bytes([raw[8], raw[9], raw[10], raw[11]]) as usize;
95            let flags = u16::from_le_bytes([raw[12], raw[13]]);
96            let next = u16::from_le_bytes([raw[14], raw[15]]) as usize;
97
98            if flags & 2 != 0 {
99                // VRING_DESC_F_WRITE
100                let n = payload.len().saturating_sub(pos).min(len);
101                if n > 0 {
102                    crate::vmm::write_guest_memory(vm_id, addr, &payload[pos..pos + n]);
103                    pos += n;
104                }
105            }
106            if flags & 1 == 0 {
107                break;
108            } // No VRING_DESC_F_NEXT
109            cur = next;
110        }
111
112        // Write used ring entry {id: head, len: bytes_written} and advance used.idx.
113        let written = pos as u32;
114        let elem_off = 4 + (self.rx_used_idx as usize % qcfg.num as usize) * 8;
115        let mut elem = [0u8; 8];
116        elem[0..4].copy_from_slice(&(head as u32).to_le_bytes());
117        elem[4..8].copy_from_slice(&written.to_le_bytes());
118        crate::vmm::write_guest_memory(vm_id, qcfg.used_gpa + elem_off as u64, &elem);
119        self.rx_used_idx = self.rx_used_idx.wrapping_add(1);
120        crate::vmm::write_guest_memory(vm_id, qcfg.used_gpa + 2, &self.rx_used_idx.to_le_bytes());
121
122        crate::vmm::inject_irq(vm_id, vcpu_id, NET_SPI);
123    }
124}

126impl VirtioDevice for NetDev {
127    fn device_id(&self) -> u32 {
128        1
129    } // DeviceID=1 = virtio-net
130
131    fn device_features_lo(&self) -> u32 {
132        VIRTIO_NET_F_MAC
133    }
134
135    /// Config space: MAC[0..5] at bytes 0-5, status at bytes 6-7.
136    fn config_read(&self, offset: usize) -> u32 {
137        match offset {
138            0 => u32::from_le_bytes([GUEST_MAC[0], GUEST_MAC[1], GUEST_MAC[2], GUEST_MAC[3]]),
139            4 => u32::from_le_bytes([GUEST_MAC[4], GUEST_MAC[5], 0, 0]),
140            _ => 0,
141        }
142    }
143
144    fn notify(&mut self, q: usize, qcfg: &QueueCfg, vm_id: usize, vcpu_id: usize) {
145        match q {
146            0 => {} // RX queue notify — guest added empty buffers; no action until frame arrives.
147            1 => {
148                // TX queue — drain guest TX descriptors and forward to the Net Cell.
149                let net_tid = self.net_tid;
150                process_notify(
151                    vm_id,
152                    qcfg,
153                    &mut self.tx_last_avail,
154                    &mut self.tx_used_idx,
155                    |bufs| {
156                        handle_tx(bufs, vm_id, net_tid);
157                        0
158                    },
159                );
160                crate::vmm::inject_irq(vm_id, vcpu_id, NET_SPI);
161            }
162            _ => {}
163        }
164    }
165}
166
167/// Read all device-readable descriptor bytes, skip 10-byte virtio_net_hdr, send remainder.
168fn handle_tx(bufs: &[DescBuf], vm_id: usize, net_tid: usize) {
169    if net_tid == 0 {
170        return;
171    }
172    let mut payload: alloc::vec::Vec<u8> = alloc::vec::Vec::new();
173    for buf in bufs {
174        if buf.writable {
175            continue;
176        } // Skip device-write buffers (not present in TX)
177        let n = buf.len as usize;
178        let mut tmp = vec![0u8; n];
179        let got = crate::vmm::read_guest_memory(vm_id, buf.gpa, &mut tmp);
180        if got == 0 || got == usize::MAX {
181            return;
182        }
183        payload.extend_from_slice(&tmp[..got]);
184    }
185    // The first 10 bytes are the virtio_net_hdr; the rest is the raw Ethernet frame.
186    if payload.len() <= 10 {
187        return;
188    }
189    crate::net_backend::transmit(net_tid, &payload[10..]);
190}
