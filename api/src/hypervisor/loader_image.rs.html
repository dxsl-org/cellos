1//! ARM64 Linux Image header parser + guest RAM placement.
2//!
3//! Parses the 64-byte ARM64 Image header, places vmlinuz at the correct GPA
4//! (guest_ram_base + 2MB-aligned + text_offset), initramfs within 1GB of kernel,
5//! and the DTB at an 8-byte-aligned GPA below the kernel load address.
6//!
7//! # ARM64 Image header layout (from Linux Documentation/arm64/booting.rst)
8//! ```text
9//! Offset  Size  Field
10//!  0x00    4    code0 (executable instruction or MZ)
11//!  0x04    4    code1
12//!  0x08    8    text_offset (image load offset from start of RAM, LE)
13//!  0x10    8    image_size  (effective image size, LE)
14//!  0x18    8    flags
15//!  0x20    8    res2
16//!  0x28    8    res3
17//!  0x30    8    res4
18//!  0x38    4    magic ("ARM\x64" = 0x644D5241)
19//!  0x3C    4    res5
20//! ```
21
22extern crate alloc;
23use alloc::vec::Vec;
24use types::{ViError, ViResult};
25
26/// ARM64 Image header magic.
27const ARM64_IMAGE_MAGIC: u32 = 0x644D5241;
28
29/// Loaded guest image addresses.
30pub struct LoadedGuest {
31    /// GPA of the kernel entry point (= RAM base + text_offset).
32    pub kernel_entry_gpa: u64,
33    /// GPA of the initramfs blob.
34    pub initrd_gpa: u64,
35    /// Size of the initramfs in bytes.
36    pub initrd_size: u64,
37    /// GPA of the DTB (passed in x0 on entry).
38    pub dtb_gpa: u64,
39}
40
41/// Parse the ARM64 Image header from `kernel_bytes`.
42///
43/// Returns `(text_offset, image_size)` or an error if the magic is wrong.
44pub fn parse_image_header(kernel_bytes: &[u8]) -> ViResult<(u64, u64)> {
45    if kernel_bytes.len() < 0x40 {
46        return Err(ViError::InvalidInput);
47    }
48    let magic = u32::from_le_bytes(kernel_bytes[0x38..0x3C].try_into().unwrap());
49    if magic != ARM64_IMAGE_MAGIC {
50        return Err(ViError::InvalidInput);
51    }
52    let text_offset = u64::from_le_bytes(kernel_bytes[0x08..0x10].try_into().unwrap());
53    let image_size = u64::from_le_bytes(kernel_bytes[0x10..0x18].try_into().unwrap());
54    Ok((text_offset, image_size))
55}
56
57/// Place vmlinuz, initramfs, and DTB into guest RAM by writing via the VMM.
58///
59/// # Layout
60/// ```text
61///   GUEST_RAM_BASE = 0x4000_0000
62///   dtb_gpa        = GUEST_RAM_BASE + 0x0000 (DTB at start, ≤ 2MB)
63///   kernel_gpa     = GUEST_RAM_BASE + 2MB-align(text_offset) [typically 0x4020_0000]
64///   initrd_gpa     = kernel_gpa + round_up(image_size, 2MB) [within 1GB of kernel]
65/// ```
66///
67/// All writes go through `write_fn(gpa, bytes)` which calls `sys_write_guest_memory`.
68pub fn place_images<W>(
69    kernel_bytes: &[u8],
70    initrd_bytes: &[u8],
71    dtb_bytes: &[u8],
72    guest_ram_base: u64,
73    mut write_fn: W,
74) -> ViResult<LoadedGuest>
75where
76    W: FnMut(u64, &[u8]) -> ViResult<()>,
77{
78    let (text_offset, image_size) = parse_image_header(kernel_bytes)?;
79
80    // Align text_offset up to 2MB boundary (ARM64 boot protocol requirement).
81    const MB2: u64 = 2 * 1024 * 1024;
82    let aligned_offset = (text_offset + MB2 - 1) & !(MB2 - 1);
83
84    let kernel_gpa = guest_ram_base + aligned_offset;
85
86    // DTB: 8-byte-aligned, placed at guest_ram_base (before the kernel).
87    // Must be within 2MB of start of RAM per ARM64 boot spec.
88    let dtb_gpa = guest_ram_base; // 0x4000_0000
89
90    // initramfs: after kernel, 2MB-aligned, within 1GB window.
91    let initrd_offset = aligned_offset + ((image_size + MB2 - 1) & !(MB2 - 1));
92    let initrd_gpa = guest_ram_base + initrd_offset;
93
94    // Write DTB first (small, fits before kernel).
95    write_fn(dtb_gpa, dtb_bytes)?;
96
97    // Write kernel.
98    write_fn(kernel_gpa, kernel_bytes)?;
99
100    // Write initramfs.
101    write_fn(initrd_gpa, initrd_bytes)?;
102
103    Ok(LoadedGuest {
104        kernel_entry_gpa: kernel_gpa,
105        initrd_gpa,
106        initrd_size: initrd_bytes.len() as u64,
107        dtb_gpa,
108    })
109}

111/// Read a file from the ViCell VFS into a `Vec<u8>`.
112pub fn read_file_from_vfs(path: &str) -> ViResult<Vec<u8>> {
113    let mut f = ostd::fs::File::open(path)?;
114    let mut buf = Vec::new();
115    f.read_to_end(&mut buf)?;
116    Ok(buf)
