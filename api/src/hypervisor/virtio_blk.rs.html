1//! virtio-blk device model (DeviceID=2, virtio-mmio slot 1 → SPI 17).
2//!
3//! 16 MiB in-memory backing; reads and writes are volatile (not persisted).
4//!
5//! Chain layout (virtio-blk spec §5.2.6.1):
6//!   [0]      outhdr   16B device-readable: { type:u32, _:u32, sector:u64 }
7//!   [1..n-1] data     device-readable (OUT) or device-writable (IN)
8//!   [last]   status   1B  device-writable: 0=OK 1=IOERR 2=UNSUPP
9
10extern crate alloc;
11use crate::virtio_mmio::{QueueCfg, VirtioDevice};
12use crate::virtqueue::{process_notify, DescBuf};
13use alloc::vec;
14
15const DISK_SIZE: usize = 16 * 1024 * 1024; // 16 MiB
16const SECTOR_SIZE: usize = 512;
17const NUM_SECTORS: u64 = (DISK_SIZE / SECTOR_SIZE) as u64;
18
19const BLK_T_IN: u32 = 0; // read  — device → driver
20const BLK_T_OUT: u32 = 1; // write — driver → device
21const BLK_T_FLUSH: u32 = 4;
22
23const DISK_SPI: u32 = 17; // SPI line for virtio-mmio slot 1
24
25pub struct BlkDisk {
26    data: alloc::vec::Vec<u8>,
27    last_avail: u16,
28    used_idx: u16,
29}
30
31impl BlkDisk {
32    pub fn new() -> Self {
33        Self {
34            data: vec![0u8; DISK_SIZE],
35            last_avail: 0,
36            used_idx: 0,
37        }
38    }
39}
40
41impl VirtioDevice for BlkDisk {
42    fn device_id(&self) -> u32 {
43        2
44    }
45
46    /// virtio-blk config: capacity at bytes 0-7 (little-endian u64 of sectors).
47    fn config_read(&self, offset: usize) -> u32 {
48        match offset {
49            0 => (NUM_SECTORS & 0xFFFF_FFFF) as u32,
50            4 => (NUM_SECTORS >> 32) as u32,
51            _ => 0,
52        }
53    }
54
55    fn notify(&mut self, q: usize, qcfg: &QueueCfg, vm_id: usize, vcpu_id: usize) {
56        if q != 0 {
57            return;
58        }
59        // Disjoint field borrows: data / last_avail / used_idx
60        let disk = self.data.as_mut_slice();
61        process_notify(
62            vm_id,
63            qcfg,
64            &mut self.last_avail,
65            &mut self.used_idx,
66            |bufs| handle_blk_request(disk, bufs, vm_id),
67        );
68        crate::vmm::inject_irq(vm_id, vcpu_id, DISK_SPI);
69    }
70}
71
72fn handle_blk_request(disk: &mut [u8], bufs: &[DescBuf], vm_id: usize) -> u32 {
73    if bufs.len() < 3 {
74        return 0;
75    }
76    let status_idx = bufs.len() - 1;
77
78    let mut hdr = [0u8; 16];
79    if crate::vmm::read_guest_memory(vm_id, bufs[0].gpa, &mut hdr) != 16 {
80        write_status(vm_id, bufs[status_idx].gpa, 1);
81        return 1;
82    }
83    let req_type = u32::from_le_bytes([hdr[0], hdr[1], hdr[2], hdr[3]]);
84    let sector = u64::from_le_bytes(hdr[8..16].try_into().unwrap_or([0u8; 8]));
85
86    let data_bufs = &bufs[1..status_idx];
87    let status = match req_type {
88        BLK_T_IN => blk_read(disk, sector, data_bufs, vm_id),
89        BLK_T_OUT => blk_write(disk, sector, data_bufs, vm_id),
90        BLK_T_FLUSH => 0u8,
91        _ => 2u8, // VIRTIO_BLK_S_UNSUPP
92    };
93    write_status(vm_id, bufs[status_idx].gpa, status);
94    1 // bytes placed in used ring (status byte)
95}

97/// READ: copy disk sectors into driver-writable guest buffers.
98fn blk_read(disk: &[u8], sector: u64, bufs: &[DescBuf], vm_id: usize) -> u8 {
99    let mut lba = sector;
100    for buf in bufs {
101        let off = (lba as usize).saturating_mul(SECTOR_SIZE);
102        if off >= disk.len() {
103            break;
104        }
105        let n = (buf.len as usize).min(disk.len() - off);
106        crate::vmm::write_guest_memory(vm_id, buf.gpa, &disk[off..off + n]);
107        lba += (n.div_ceil(SECTOR_SIZE)) as u64;
108    }
109    0
110}
111
112/// WRITE: copy driver-readable guest buffers into disk sectors.
113fn blk_write(disk: &mut [u8], sector: u64, bufs: &[DescBuf], vm_id: usize) -> u8 {
114    let mut lba = sector;
115    for buf in bufs {
116        let off = (lba as usize).saturating_mul(SECTOR_SIZE);
117        if off >= disk.len() {
118            break;
119        }
120        let n = (buf.len as usize).min(disk.len() - off);
121        let mut tmp = vec![0u8; n];
122        let got = crate::vmm::read_guest_memory(vm_id, buf.gpa, &mut tmp);
123        if got == 0 || got == usize::MAX {
124            break;
125        }
126        disk[off..off + got].copy_from_slice(&tmp[..got]);
127        lba += (got.div_ceil(SECTOR_SIZE)) as u64;
128    }
129    0
130}
131
132fn write_status(vm_id: usize, gpa: u64, status: u8) {
133    crate::vmm::write_guest_memory(vm_id, gpa, &[status]);
