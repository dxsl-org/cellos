1//! Split-virtqueue device-side ring processor.
2//!
3//! The VMM is on the device side: it reads the avail ring, walks desc chains,
4//! hands buffers to the device model, then writes the used ring.
5//! All guest memory accesses go via `crate::vmm` syscall wrappers — never raw
6//! pointer dereference from the cell (Law 4).
7
8extern crate alloc;
9use alloc::vec::Vec;
10
11use crate::virtio_mmio::QueueCfg;
12
13/// One segment from a virtqueue descriptor chain.
14pub struct DescBuf {
15    pub gpa: u64,
16    pub len: u32,
17    /// true = device writes into this buffer (VRING_DESC_F_WRITE); false = device reads.
18    pub writable: bool,
19}
20
21const FLAGS_NEXT: u16 = 1;
22const FLAGS_WRITE: u16 = 2;
23const MAX_CHAIN: usize = 64; // guard against infinite chains
24
25/// Process one QueueNotify: drain avail ring → walk desc chains → call `handle` → update used ring.
26///
27/// `last_avail_idx` and `used_idx` are per-queue device-side counters (NOT in guest memory);
28/// the caller (device model) owns them and passes them mutably across calls.
29pub fn process_notify<F>(
30    vm_id: usize,
31    qcfg: &QueueCfg,
32    last_avail_idx: &mut u16,
33    used_idx: &mut u16,
34    mut handle: F,
35) where
36    F: FnMut(&[DescBuf]) -> u32,
37{
38    let q_size = qcfg.num as usize;
39    if q_size == 0 {
40        return;
41    }
42
43    // Read avail.idx (u16 at avail_ring + 2).
44    let mut b2 = [0u8; 2];
45    if crate::vmm::read_guest_memory(vm_id, qcfg.avail_gpa + 2, &mut b2) != 2 {
46        return;
47    }
48    let avail_idx = u16::from_le_bytes(b2);
49
50    while *last_avail_idx != avail_idx {
51        // Read avail.ring[last_avail_idx % q_size] — the desc head index.
52        let ring_off = 4 + (*last_avail_idx as usize % q_size) * 2;
53        if crate::vmm::read_guest_memory(vm_id, qcfg.avail_gpa + ring_off as u64, &mut b2) != 2 {
54            break;
55        }
56        let head = u16::from_le_bytes(b2) as usize;
57        *last_avail_idx = last_avail_idx.wrapping_add(1);
58
59        // Walk descriptor chain starting at `head`.
60        let mut bufs: Vec<DescBuf> = Vec::new();
61        let mut cur = head;
62        for _ in 0..MAX_CHAIN {
63            let mut raw = [0u8; 16]; // VirtqDesc = 16 bytes
64            let desc_gpa = qcfg.desc_gpa + (cur as u64) * 16;
65            if crate::vmm::read_guest_memory(vm_id, desc_gpa, &mut raw) != 16 {
66                break;
67            }
68
69            let addr = u64::from_le_bytes([
70                raw[0], raw[1], raw[2], raw[3], raw[4], raw[5], raw[6], raw[7],
71            ]);
72            let len = u32::from_le_bytes([raw[8], raw[9], raw[10], raw[11]]);
73            let flags = u16::from_le_bytes([raw[12], raw[13]]);
74            let next = u16::from_le_bytes([raw[14], raw[15]]) as usize;
75
76            bufs.push(DescBuf {
77                gpa: addr,
78                len,
79                writable: flags & FLAGS_WRITE != 0,
80            });
81            if flags & FLAGS_NEXT == 0 {
82                break;
83            }
84            cur = next;
85        }
86
87        // Deliver buffers to device model; it returns how many bytes were produced.
88        let written = handle(&bufs);
89
90        // Write used ring entry { id: u32, len: u32 } at used.ring[used_idx % q_size].
91        let elem_off = 4 + (*used_idx as usize % q_size) * 8;
92        let mut elem = [0u8; 8];
93        elem[0..4].copy_from_slice(&(head as u32).to_le_bytes());
94        elem[4..8].copy_from_slice(&written.to_le_bytes());
95        crate::vmm::write_guest_memory(vm_id, qcfg.used_gpa + elem_off as u64, &elem);
96
97        // Advance used.idx with a store that the guest will see (TCG is SC).
98        *used_idx = used_idx.wrapping_add(1);
99        crate::vmm::write_guest_memory(vm_id, qcfg.used_gpa + 2, &used_idx.to_le_bytes());
100    }
101}
