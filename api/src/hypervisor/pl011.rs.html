1//! Emulated ARM PL011 UART for the guest console.
2//!
3//! The guest PL011 lives at IPA 0x09000000 (GPA in Stage-2 unmapped = traps here).
4//! Only the registers Linux actually touches during early console output are emulated;
5//! all others return a safe default.
6
7use ostd::io::print;
8
9/// PL011 base IPA in the guest address space.
10pub const PL011_BASE_IPA: u64 = 0x0900_0000;
11pub const PL011_SIZE: u64 = 0x1000; // 4 KiB MMIO window
12
13/// PL011 register offsets (byte addresses from base).
14///
15/// Registers below that are never matched in `write`/`read` (RSR, IBRD, FBRD,
16/// IFLS, DMACR) document the full PL011 map so the next person extending this
17/// emulation to baud/FIFO-level trapping doesn't have to re-derive the offsets.
18mod reg {
19    pub const UARTDR: u64 = 0x00; // Data register: TX byte on write
20    #[allow(dead_code)] // documents full register map; not trapped (RX status unsupported)
21    pub const UARTRSR: u64 = 0x04; // Receive status / error clear
22    pub const UARTFR: u64 = 0x18; // Flag register: read → TX empty/ready bits
23    #[allow(dead_code)] // documents full register map; baud rate not modeled (bytes forwarded synchronously)
24    pub const UARTIBRD: u64 = 0x24; // Integer baud rate
25    #[allow(dead_code)] // documents full register map; baud rate not modeled (bytes forwarded synchronously)
26    pub const UARTFBRD: u64 = 0x28; // Fractional baud rate
27    pub const UARTLCR: u64 = 0x2C; // Line control (8N1)
28    pub const UARTCR: u64 = 0x30; // Control: UARTEN|TXE|RXE
29    #[allow(dead_code)] // documents full register map; FIFO level select not modeled (no interrupt delivery yet)
30    pub const UARTIFLS: u64 = 0x34; // FIFO level select
31    pub const UARTIMSC: u64 = 0x38; // Interrupt mask
32    pub const UARTRIS: u64 = 0x3C; // Raw interrupt status
33    pub const UARTMIS: u64 = 0x40; // Masked interrupt status
34    pub const UARTICR: u64 = 0x44; // Interrupt clear
35    #[allow(dead_code)] // documents full register map; DMA not modeled
36    pub const UARTDMACR: u64 = 0x48; // DMA control
37}
38
39/// UARTFR bits.
40#[allow(dead_code)] // documents the real PL011 flag bit; TX-full backpressure not modeled (synchronous forwarding)
41const FR_TXFF: u64 = 1 << 5; // TX FIFO full
42const FR_RXFE: u64 = 1 << 4; // RX FIFO empty
43const FR_TXFE: u64 = 1 << 7; // TX FIFO empty (all data shifted out)
44#[allow(dead_code)] // documents the real PL011 flag bit; busy-state not modeled (synchronous forwarding)
45const FR_BUSY: u64 = 1 << 3; // UART busy
46
47/// Minimal PL011 state.
48pub struct Pl011 {
49    cr: u64,   // control register shadow
50    lcr: u64,  // line control register shadow
51    imsc: u64, // interrupt mask shadow
52}
53
54impl Pl011 {
55    pub const fn new() -> Self {
56        Self {
57            cr: 0x300,
58            lcr: 0,
59            imsc: 0,
60        }
61    }
62
63    /// Handle a guest MMIO write to `offset` (relative to PL011_BASE_IPA) with `val`.
64    pub fn write(&mut self, offset: u64, val: u64) {
65        match offset {
66            reg::UARTDR => {
67                // Forward TX byte to ViCell serial output.
68                let byte = (val & 0xFF) as u8;
69                let buf = [byte];
70                if let Ok(s) = core::str::from_utf8(&buf) {
71                    print(s);
72                }
73            }
74            reg::UARTCR => {
75                self.cr = val;
76            }
77            reg::UARTLCR => {
78                self.lcr = val;
79            }
80            reg::UARTIMSC => {
81                self.imsc = val;
82            }
83            reg::UARTICR => { /* clear interrupts — no interrupt delivery yet */ }
84            _ => { /* ignore: IBRD, FBRD, IFLS, DMACR etc. */ }
85        }
86    }
87
88    /// Handle a guest MMIO read from `offset`; returns the register value.
89    pub fn read(&self, offset: u64) -> u64 {
90        match offset {
91            reg::UARTFR => {
92                // TX always ready (we forward bytes synchronously), RX FIFO empty.
93                FR_TXFE | FR_RXFE
94            }
95            reg::UARTCR => self.cr,
96            reg::UARTLCR => self.lcr,
97            reg::UARTIMSC => self.imsc,
98            reg::UARTRIS => 0, // no raw interrupts
99            reg::UARTMIS => 0, // no masked interrupts
100            _ => 0,
101        }
102    }
103
104    /// True if `ipa` falls within this device's MMIO window.
105    pub fn owns(ipa: u64) -> bool {
106        (PL011_BASE_IPA..PL011_BASE_IPA + PL011_SIZE).contains(&ipa)
107    }
108}
