1//! Minimal DTB builder for the Alpine Linux guest.
2//!
3//! Emits the 7 mandatory nodes using `vm-fdt`. Missing any node causes a silent
4//! guest panic during Linux GIC/timer/console init (R4 risk #1, HIGHEST severity).
5//!
6//! Node list (from phase-05 Key Insights):
7//! 1. /memory@40000000  — guest RAM
8//! 2. /cpus/cpu@0       — cortex-a72, enable-method = "psci"
9//! 3. /psci             — method = "hvc", compatible = "arm,psci-1.0"
10//! 4. /intc             — GICv2 (arm,cortex-a15-gic), GICD@0x08000000 GICC@0x08010000
11//! 5. /timer            — armv8-timer PPIs 13/14/11/10, level-low
12//! 6. /chosen           — bootargs, initrd-start/end, stdout-path
13//! 7. /pl011@9000000    — arm,pl011 console
14
15extern crate alloc;
16use alloc::vec;
17use alloc::vec::Vec;
18use vm_fdt::{FdtWriter, FdtWriterResult};
19
20/// Boot arguments passed to the Linux kernel command line.
21pub const BOOTARGS: &str =
22    "console=hvc0 console=ttyAMA0 earlycon=pl011,0x9000000 rdinit=/bin/sh panic=1 loglevel=8";
23
24/// Build a minimal DTB for a single-CPU Alpine guest.
25///
26/// Parameters:
27/// - `ram_base` / `ram_size`: guest physical RAM region (usually 0x40000000, configurable MB)
28/// - `initrd_start` / `initrd_end`: guest physical addresses of the initramfs blob
29///
30/// Returns the raw DTB bytes.
31pub fn build_dtb(
32    ram_base: u64,
33    ram_size: u64,
34    initrd_start: u64,
35    initrd_end: u64,
36) -> FdtWriterResult<Vec<u8>> {
37    let mut fdt = FdtWriter::new()?;
38
39    let root = fdt.begin_node("")?;
40    fdt.property_u32("#address-cells", 2)?;
41    fdt.property_u32("#size-cells", 2)?;
42    fdt.property_string("compatible", "linux,dummy-virt")?;
43
44    // ── 1. /memory ───────────────────────────────────────────────────────────
45    let mem = fdt.begin_node(&alloc::format!("memory@{:x}", ram_base))?;
46    fdt.property_string("device_type", "memory")?;
47    // reg = <base_hi base_lo size_hi size_lo>
48    fdt.property_array_u32(
49        "reg",
50        &[
51            (ram_base >> 32) as u32,
52            (ram_base & 0xFFFF_FFFF) as u32,
53            (ram_size >> 32) as u32,
54            (ram_size & 0xFFFF_FFFF) as u32,
55        ],
56    )?;
57    fdt.end_node(mem)?;
58
59    // ── 2. /cpus ─────────────────────────────────────────────────────────────
60    let cpus = fdt.begin_node("cpus")?;
61    fdt.property_u32("#address-cells", 1)?;
62    fdt.property_u32("#size-cells", 0)?;
63
64    let cpu0 = fdt.begin_node("cpu@0")?;
65    fdt.property_string("device_type", "cpu")?;
66    fdt.property_string("compatible", "arm,cortex-a72")?;
67    fdt.property_string("enable-method", "psci")?;
68    fdt.property_u32("reg", 0)?;
69    fdt.end_node(cpu0)?;
70    fdt.end_node(cpus)?;
71
72    // ── 3. /psci ─────────────────────────────────────────────────────────────
73    let psci = fdt.begin_node("psci")?;
74    // arm,psci-1.0 compatible — must use "hvc" method (not "smc").
75    fdt.property_string_list(
76        "compatible",
77        vec![
78            alloc::string::String::from("arm,psci-1.0"),
79            alloc::string::String::from("arm,psci-0.2"),
80            alloc::string::String::from("arm,psci"),
81        ],
82    )?;
83    fdt.property_string("method", "hvc")?;
84    // PSCI 1.0 function IDs (for SMCCC 32-bit / hvc call convention).
85    fdt.property_u32("cpu_suspend", 0x8400_0001)?;
86    fdt.property_u32("cpu_off", 0x8400_0002)?;
87    fdt.property_u32("cpu_on", 0x8400_0003)?;
88    fdt.property_u32("migrate", 0x8400_0005)?;
89    fdt.end_node(psci)?;
90
91    // ── 4. /intc — GICv2 ─────────────────────────────────────────────────────
92    // reg: GICD base (64KiB) + GICC base (64KiB).
93    // arm,cortex-a15-gic is the compatible string for GICv2 used by QEMU virt.
94    let intc = fdt.begin_node("intc@8000000")?;
95    fdt.property_string_list(
96        "compatible",
97        vec![
98            alloc::string::String::from("arm,cortex-a15-gic"),
99            alloc::string::String::from("arm,gic-400"),
100        ],
101    )?;
102    fdt.property_null("interrupt-controller")?;
103    fdt.property_u32("#interrupt-cells", 3)?;
104    fdt.property_u32("#address-cells", 0)?;
105    fdt.property_array_u32(
106        "reg",
107        &[
108            0x0,
109            0x0800_0000,
110            0x0,
111            0x0001_0000, // GICD
112            0x0,
113            0x0801_0000,
114            0x0,
115            0x0001_0000, // GICC
116        ],
117    )?;
118    fdt.property_phandle(1)?; // phandle=1 for interrupt-parent references
119    fdt.end_node(intc)?;
120
121    // ── 5. /timer — armv8-timer ──────────────────────────────────────────────
122    // PPIs: secure EL1 (13), non-secure EL1 (14), virtual (11), hypervisor (10).
123    // All level-low (GIC_PPI | IRQ_TYPE_LEVEL_LOW = <1 N 8>).
124    let timer = fdt.begin_node("timer")?;
125    fdt.property_string("compatible", "arm,armv8-timer")?;
126    fdt.property_u32("interrupt-parent", 1)?; // phandle of /intc
127                                              // Each PPI = <GIC_PPI irq flags>; GIC_PPI=1, flags=0x8=IRQ_TYPE_LEVEL_LOW.
128    fdt.property_array_u32(
129        "interrupts",
130        &[
131            1, 13, 0x8, // secure EL1 PPI 13
132            1, 14, 0x8, // non-secure EL1 PPI 14
133            1, 11, 0x8, // virtual PPI 11
134            1, 10, 0x8, // hypervisor PPI 10
135        ],
136    )?;
137    fdt.property_null("always-on")?;
138    fdt.end_node(timer)?;
139
140    // ── 6. /chosen ───────────────────────────────────────────────────────────
141    let chosen = fdt.begin_node("chosen")?;
142    fdt.property_string("bootargs", BOOTARGS)?;
143    fdt.property_string("stdout-path", "pl011@9000000")?;
144    fdt.property_u64("linux,initrd-start", initrd_start)?;
145    fdt.property_u64("linux,initrd-end", initrd_end)?;
146    fdt.end_node(chosen)?;
147
148    // ── 7. /pl011@9000000 ────────────────────────────────────────────────────
149    let uart = fdt.begin_node("pl011@9000000")?;
150    fdt.property_string_list(
151        "compatible",
152        vec![
153            alloc::string::String::from("arm,pl011"),
154            alloc::string::String::from("arm,primecell"),
155        ],
156    )?;
157    fdt.property_u32("interrupt-parent", 1)?;
158    // UART0 SPI 1 level-high = <0 1 4> (GIC_SPI=0, irq=1, IRQ_TYPE_LEVEL_HIGH=4).
159    fdt.property_array_u32("interrupts", &[0, 1, 4])?;
160    fdt.property_array_u32("reg", &[0x0, 0x0900_0000, 0x0, 0x1000])?;
161    fdt.property_u32("clock-frequency", 0x16E360)?; // 1.5 MHz (QEMU default)
162    fdt.end_node(uart)?;
163
164    // ── 8. /virtio_mmio@a000000 — console (slot 0, SPI 16) ──────────────────
165    let vio = fdt.begin_node("virtio_mmio@a000000")?;
166    fdt.property_string("compatible", "virtio,mmio")?;
167    fdt.property_u32("interrupt-parent", 1)?;
168    // SPI 16, IRQ_TYPE_EDGE_RISING=1: <GIC_SPI=0, irq=16, flags=1>
169    fdt.property_array_u32("interrupts", &[0, 16, 1])?;
170    fdt.property_array_u32("reg", &[0x0, 0x0a00_0000, 0x0, 0x200])?;
171    fdt.end_node(vio)?;
172
173    // ── 9. /virtio_mmio@a000200 — block device (slot 1, SPI 17) ─────────────
174    let vio_blk = fdt.begin_node("virtio_mmio@a000200")?;
175    fdt.property_string("compatible", "virtio,mmio")?;
176    fdt.property_u32("interrupt-parent", 1)?;
177    // SPI 17, IRQ_TYPE_EDGE_RISING=1: <GIC_SPI=0, irq=17, flags=1>
178    fdt.property_array_u32("interrupts", &[0, 17, 1])?;
179    fdt.property_array_u32("reg", &[0x0, 0x0a00_0200, 0x0, 0x200])?;
180    fdt.end_node(vio_blk)?;
181
182    // ── 10. /virtio_mmio@a000400 — network device (slot 2, SPI 18) ───────────
183    let vio_net = fdt.begin_node("virtio_mmio@a000400")?;
184    fdt.property_string("compatible", "virtio,mmio")?;
185    fdt.property_u32("interrupt-parent", 1)?;
186    // SPI 18, IRQ_TYPE_EDGE_RISING=1: <GIC_SPI=0, irq=18, flags=1>
187    fdt.property_array_u32("interrupts", &[0, 18, 1])?;
188    fdt.property_array_u32("reg", &[0x0, 0x0a00_0400, 0x0, 0x200])?;
189    fdt.end_node(vio_net)?;
190
191    fdt.end_node(root)?;
192    fdt.finish()
193}
