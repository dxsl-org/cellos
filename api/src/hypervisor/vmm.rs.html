1//! Low-level VMM syscall wrappers (220-226) for the hypervisor service cell.
2//!
3//! These are ARM64-only at runtime (guarded by cpu_features::has_el2 at cell start).
4//! RISC-V stubs return a NotSupported sentinel so the code compiles on all targets.
5
6use api::hypervisor::ViVmExit;
7use api::syscall::ViSyscall;
8
9/// Scheduler tick budget for each RunVcpu call (~10ms in 10 MHz ticks = 100_000 ticks).
10pub const SCHED_TICK_BUDGET_NS: u64 = 10_000_000; // 10ms in nanoseconds
11
12/// Error sentinel returned by VMM syscalls on failure.
13const ERR: usize = usize::MAX;
14
15#[inline]
16unsafe fn syscall4(id: ViSyscall, a0: usize, a1: usize, a2: usize, a3: usize) -> usize {
17    let ret: usize;
18    #[cfg(target_arch = "aarch64")]
19    core::arch::asm!(
20        "svc #0",
21        inlateout("x0") id as usize => ret,
22        in("x1") a0, in("x2") a1, in("x3") a2, in("x4") a3,
23        options(nostack, preserves_flags),
24    );
25    #[cfg(not(target_arch = "aarch64"))]
26    {
27        let _ = (id, a0, a1, a2, a3);
28        ret = ERR;
29    }
30    ret
31}
32
33/// Allocate guest RAM + Stage-2 table; returns vm_id (> 0) or 0 on error.
34pub fn create_vm(guest_pages: usize) -> usize {
35    unsafe { syscall4(ViSyscall::CreateVm, guest_pages, 0, 0, 0) }
36}
37
38/// Create a vCPU with initial PC `entry_pc` in `vm_id`; returns vcpu_id or 0.
39pub fn create_vcpu(vm_id: usize, entry_pc: u64) -> usize {
40    unsafe { syscall4(ViSyscall::CreateVcpu, vm_id, entry_pc as usize, 0, 0) }
41}
42
43/// Map guest IPA range in `vm_id`; returns 0 on success.
44pub fn map_guest_memory(vm_id: usize, ipa: u64, size: usize, writable: bool) -> usize {
45    unsafe {
46        syscall4(
47            ViSyscall::MapGuestMemory,
48            vm_id,
49            ipa as usize,
50            size,
51            writable as usize,
52        )
53    }
54}
55
56/// Copy `src` bytes into guest RAM at `gpa`; returns bytes written or ERR.
57pub fn write_guest_memory(vm_id: usize, gpa: u64, src: &[u8]) -> usize {
58    unsafe {
59        syscall4(
60            ViSyscall::WriteGuestMemory,
61            vm_id,
62            gpa as usize,
63            src.as_ptr() as usize,
64            src.len(),
65        )
66    }
67}
68
69/// Copy `len` bytes from guest RAM at `gpa` into `dst`; returns bytes read or ERR.
70pub fn read_guest_memory(vm_id: usize, gpa: u64, dst: &mut [u8]) -> usize {
71    unsafe {
72        syscall4(
73            ViSyscall::ReadGuestMemory,
74            vm_id,
75            gpa as usize,
76            dst.as_mut_ptr() as usize,
77            dst.len(),
78        )
79    }
80}

82/// World-switch into `vcpu_id`; writes exit reason to `*exit`. Returns 0 or ERR.
83pub fn run_vcpu(vm_id: usize, vcpu_id: usize, exit: &mut ViVmExit) -> usize {
84    unsafe {
85        syscall4(
86            ViSyscall::RunVcpu,
87            vm_id,
88            vcpu_id,
89            SCHED_TICK_BUDGET_NS as usize,
90            exit as *mut ViVmExit as usize,
91        )
92    }
93}

95/// Read (write=false) or write (write=true) vCPU GP registers (32×u64 = x0..x30+pc).
96pub fn vcpu_regs(vm_id: usize, vcpu_id: usize, regs: &mut [u64; 32], write: bool) -> usize {
97    unsafe {
98        syscall4(
99            ViSyscall::VcpuRegs,
100            vm_id,
101            vcpu_id,
102            regs.as_mut_ptr() as usize,
103            write as usize,
104        )
105    }
106}

108/// Inject GICv2 virtual IRQ (0 ≤ intid ≤ 1019) into vCPU; returns 0 or ERR.
109pub fn inject_irq(vm_id: usize, vcpu_id: usize, intid: u32) -> usize {
110    unsafe { syscall4(ViSyscall::InjectIrq, vm_id, vcpu_id, intid as usize, 0) }
