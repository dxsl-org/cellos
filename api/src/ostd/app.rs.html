3//! ViCell App SDK — structured entry point for Cell applications.
4//!
5//! Instead of writing raw `sys_recv` dispatch loops, cells declare a handler and
6//! call `AppContext::run()`. The context manages the receive buffer, interprets
7//! the message envelope, and delivers typed [`AppEvent`]s.
8//!
9//! # Message envelope
10//!
11//! Messages sent through [`AppContext::send_msg`] carry a 2-byte header:
12//! - byte 0: [`APP_MSG_MAGIC`] (0xAC) — namespace guard, prevents raw-protocol collisions
13//! - byte 1: event type discriminant (0x00 = Message, 0xFF = Shutdown)
14//! - bytes 2..: payload (caller-defined)
15//!
16//! Messages that do not start with `APP_MSG_MAGIC` are delivered as raw
17//! [`AppEvent::RawMessage`] so legacy IPC senders remain compatible.
18//!
19//! # Example
20//! ```no_run
21//! use ostd::app::{AppContext, AppEvent};
22//!
23//! fn handler(ctx: &mut AppContext, ev: AppEvent) {
24//!     if let AppEvent::Message { sender_tid, .. } = ev {
25//!         ctx.send(sender_tid, b"\xAC\x00pong").ok();
26//!     }
27//! }
28//!
29//! AppContext::new().run(handler);
30//! ```

37/// Magic byte prefixing every App SDK envelope.
38///
39/// Prevents collisions between typed AppContext messages and raw `sys_send` clients.
40pub const APP_MSG_MAGIC: u8 = 0xAC;
41
42const RECV_BUF_SIZE: usize = 4096;
43
44/// Reason delivered with a structured [`AppEvent::ShutdownWith`] event.
45#[derive(Debug, Clone, Copy, PartialEq, Eq)]
46pub enum ShutdownReason {
47    /// Graceful shutdown requested by init / power management.
48    Requested,
49    /// Heartbeat deadline missed — kernel terminated the cell; supervisor will restart.
50    Watchdog,
51    /// A parent cell the cell subscribed to via `NotifyOnExit` has died.
52    ParentDied,
53}
54
55/// Typed event delivered to an [`AppContext`] handler.
56///
57/// AppEvent owns its payload data so the handler can store or forward it without
58/// lifetime constraints tied to the receive buffer.
59///
60/// New variants may be added in future SDK releases.  Always add a wildcard arm
61/// `_ => {}` to future-proof your match.
62#[non_exhaustive]
63#[derive(Debug)]
64pub enum AppEvent {
65    /// Fires exactly once before the first `sys_recv` when using
66    /// [`AppContext::run_with_lifecycle`] or [`CellRuntime::run`].
67    /// Use for startup work: config read, service registration, UI init.
68    Init,
69    /// An App SDK message (envelope magic verified).
70    Message {
71        /// TID of the sending cell.
72        sender_tid: usize,
73        /// Message payload (bytes after the 2-byte header), heap-owned.
74        data: Vec<u8>,
75    },
76    /// A raw IPC message not using the App SDK envelope (legacy / non-SDK senders).
77    RawMessage {
78        sender_tid: usize,
79        /// Full message bytes, heap-owned.
80        data: Vec<u8>,
81    },
82    /// A keyboard or mouse event delivered by the input service.
83    ///
84    /// Automatically decoded from the 65-byte `0x10`-opcode IPC message.
85    /// Cells must call [`AppContext::request_input_focus`] at startup to
86    /// receive these events.
87    Input(api::input::InputEvent),
88    /// The receive deadline elapsed (only from [`AppContext::run_with_timeout`]).
89    Timeout,
90    /// Kernel-requested graceful shutdown (legacy form — no reason byte in envelope).
91    Shutdown,
92    /// Structured shutdown with reason (emitted when the envelope carries a reason byte).
93    ShutdownWith {
94        /// Why the shutdown is happening.
95        reason: ShutdownReason,
96    },
97
98    // ── Hot-swap lifecycle events (Phase 20) ─────────────────────────────────
99    /// Kernel-sent signal to the OLD cell during Step 2 of a hot-swap sequence.
100    ///
101    /// The cell must serialize its state, stash it under `swap_id` as the key
102    /// via [`ostd::syscall::sys_state_stash`], then yield or exit.  The kernel
103    /// polls the stash; once the bytes appear the swap proceeds to Step 3.
104    ///
105    /// Default arm: cells that do not implement hot-swap state transfer may
106    /// ignore this event — the swap continues with an empty stash (no state restored).
107    Snapshot {
108        /// Monotonically increasing swap identifier assigned by the kernel orchestrator.
109        /// Must be used as the stash key so the replacement instance can recover it.
110        swap_id: u64,
111    },
112
113    /// Kernel-sent signal to the NEW cell during Step 4 of a hot-swap sequence.
114    ///
115    /// The cell must restore its state from the stash under `key` via
116    /// [`ostd::syscall::sys_state_restore`], then call
117    /// [`ostd::syscall::sys_hotswap_ready`] to signal completion.
118    ///
119    /// `key` is a null-terminated UTF-8 string encoding the decimal `swap_id`,
120    /// held in a fixed 64-byte buffer.  Parse with
121    /// `core::str::from_utf8(&key[..key.iter().position(|&b| b==0).unwrap_or(64)])`
122    /// and convert via `str::parse::<u64>()`.
123    Restore {
124        /// Fixed-size stash key (null-terminated, decimal swap_id).
125        key: [u8; 64],
126    },
127}

129/// Execution context for a Cell application.
130///
131/// Owns the receive buffer and drives the IPC loop.  Create once in `main()` and
132/// call [`run`][AppContext::run] — it never returns unless the handler panics.
133///
134/// Lazy service client accessors (`.vfs()`, `.net()`, `.input()`, `.cluster()`)
135/// initialize on first use and are re-used across handler invocations.
136pub struct AppContext {
137    recv_buf: [u8; RECV_BUF_SIZE],
138    vfs_client: Option<crate::clients::VfsClient>,
139    net_client: Option<crate::clients::NetClient>,
140    input_client: Option<crate::clients::InputClient>,
141    cluster_client: Option<crate::cluster::ClusterRef>,
142}
143
144impl AppContext {
145    /// Create a new context with a stack-allocated receive buffer.
146    pub fn new() -> Self {
147        Self {
148            recv_buf: [0u8; RECV_BUF_SIZE],
149            vfs_client: None,
150            net_client: None,
151            input_client: None,
152            cluster_client: None,
153        }
154    }
155
156    /// Run the IPC event loop, calling `handler` for every incoming event.
157    ///
158    /// Blocks indefinitely.  The handler receives `&mut AppContext` so it can call
159    /// [`send`][AppContext::send], [`lookup_service`][AppContext::lookup_service], etc.
160    pub fn run(&mut self, mut handler: impl FnMut(&mut AppContext, AppEvent)) -> ! {
161        loop {
162            let result = syscall::sys_recv(0, &mut self.recv_buf);
163            let sender_tid = match result {
164                syscall::SyscallResult::Ok(tid) => tid,
165                syscall::SyscallResult::Err(_) => {
166                    syscall::sys_yield();
167                    continue;
168                }
169            };
170            // Parse and own the event data before calling handler so there is no
171            // outstanding borrow on self.recv_buf when handler runs.
172            let event = Self::parse_event_owned(sender_tid, &self.recv_buf);
173            handler(self, event);
174        }
175    }
176
177    /// Run the IPC event loop with a per-iteration receive timeout.
178    ///
179    /// `timeout_ticks` is the maximum scheduler ticks to wait (one tick ≈ 10 ms).
180    /// When the deadline elapses the handler is called with [`AppEvent::Timeout`].
181    pub fn run_with_timeout(
182        &mut self,
183        timeout_ticks: u64,
184        mut handler: impl FnMut(&mut AppContext, AppEvent),
185    ) -> ! {
186        loop {
187            let result = syscall::sys_recv_timeout(0, &mut self.recv_buf, timeout_ticks);
188            let sender_tid = match result {
189                syscall::SyscallResult::Ok(0) => {
190                    handler(self, AppEvent::Timeout);
191                    continue;
192                }
193                syscall::SyscallResult::Ok(tid) => tid,
194                syscall::SyscallResult::Err(_) => {
195                    syscall::sys_yield();
196                    continue;
197                }
198            };
199            let event = Self::parse_event_owned(sender_tid, &self.recv_buf);
200            handler(self, event);
201        }
202    }
203
204    /// Send raw bytes `data` to another cell by TID.
205    pub fn send(&self, tid: usize, data: &[u8]) -> ViResult<()> {
206        match syscall::sys_send(tid, data) {
207            syscall::SyscallResult::Ok(_) => Ok(()),
208            syscall::SyscallResult::Err(_) => Err(ViError::IO),
209        }
210    }
211
212    /// Send an App SDK envelope message to `tid`.
213    ///
214    /// Prepends `[APP_MSG_MAGIC, 0x00]` so the receiver's `AppContext` decodes it
215    /// as [`AppEvent::Message`].
216    pub fn send_msg(&self, tid: usize, payload: &[u8]) -> ViResult<()> {
217        let mut buf = [0u8; RECV_BUF_SIZE];
218        if payload.len() + 2 > RECV_BUF_SIZE {
219            return Err(ViError::InvalidArgument);
220        }
221        buf[0] = APP_MSG_MAGIC;
222        buf[1] = 0x00;
223        buf[2..2 + payload.len()].copy_from_slice(payload);
224        self.send(tid, &buf[..2 + payload.len()])
225    }
226
227    /// Resolve the live provider tid of a well-known service.
228    pub fn lookup_service(&self, service_id: u16) -> Option<usize> {
229        syscall::sys_lookup_service(service_id)
230    }
231
232    /// Register this cell as the keyboard/mouse focus recipient.
233    ///
234    /// Delegates to [`ostd::input::request_focus`].  Call once at startup
235    /// after any compositor surface is ready.  Returns `true` when the input
236    /// service grants focus; `false` on boot race (retry with a yield).
237    pub fn request_input_focus(&self) -> bool {
238        crate::input::request_focus()
239    }
240
241    /// Arm (or re-arm) the kernel watchdog heartbeat for this cell.
242    ///
243    /// The cell must call a syscall (any syscall) within `ticks` scheduler ticks
244    /// (1 tick ≈ 10 ms) or the kernel terminates and restarts it.  Pass `0` to
245    /// disable the heartbeat.  Called automatically by [`CellRuntime::run`].
246    pub fn arm_heartbeat(&self, ticks: u64) {
247        syscall::sys_heartbeat(ticks);
248    }
249
250    /// Run the IPC event loop, firing [`AppEvent::Init`] exactly once before
251    /// the first `sys_recv`, then calling `handler` for every subsequent event.
252    ///
253    /// Prefer this over [`run`][Self::run] for cells that need startup work.
254    pub fn run_with_lifecycle(&mut self, mut handler: impl FnMut(&mut AppContext, AppEvent)) -> ! {
255        handler(self, AppEvent::Init);
256        self.run(handler)
257    }

283    /// Lazy accessor for the cluster net-broker client.

285    /// Returns `None` if no net-broker is running (single-node deployment).

294    /// Parse `buf` into an owned `AppEvent`, copying the payload into a `Vec`.
295    ///
296    /// Owning the data eliminates the lifetime coupling between the event and the
297    /// receive buffer, so the caller can pass `&mut self` to the handler freely.
298    fn parse_event_owned(sender_tid: usize, buf: &[u8]) -> AppEvent {
299        if buf.len() >= 2 && buf[0] == APP_MSG_MAGIC {
300            match buf[1] {
301                0xFF => {
302                    // Structured shutdown: byte 2 carries the reason (if present).
303                    if buf.len() >= 3 {
304                        let reason = Self::parse_shutdown_reason(buf);
305                        AppEvent::ShutdownWith { reason }
306                    } else {
307                        AppEvent::Shutdown
308                    }
309                }
310                // Hot-swap Step 2: kernel asks this (old) cell to serialize state.
311                // Envelope: [0xAC, 0xF0, swap_id_le8 (8 bytes)]
312                0xF0 if buf.len() >= 10 => {
313                    let mut id_bytes = [0u8; 8];
314                    id_bytes.copy_from_slice(&buf[2..10]);
315                    let swap_id = u64::from_le_bytes(id_bytes);
316                    AppEvent::Snapshot { swap_id }
317                }
318                // Hot-swap Step 4: kernel asks this (new) cell to restore state.
319                // Envelope: [0xAC, 0xF1, key (64 bytes)]
320                0xF1 if buf.len() >= 66 => {
321                    let mut key = [0u8; 64];
322                    key.copy_from_slice(&buf[2..66]);
323                    AppEvent::Restore { key }
324                }
325                _ => AppEvent::Message {
326                    sender_tid,
327                    data: buf[2..].to_vec(),
328                },
329            }
330        } else if buf.len() >= 2 && buf[0] == api::input::INPUT_EVENT_OPCODE {
331            if let Some(ev) = crate::input::parse_frame(buf) {
332                return AppEvent::Input(ev);
333            }
334            AppEvent::RawMessage {
335                sender_tid,
336                data: buf.to_vec(),
337            }
338        } else {
339            AppEvent::RawMessage {
340                sender_tid,
341                data: buf.to_vec(),
342            }
343        }
344    }
345
346    /// Decode shutdown reason byte from a `[0xAC, 0xFF, reason]` envelope.
347    ///
348    /// Missing or unknown reason bytes default to [`ShutdownReason::Requested`].
349    pub(crate) fn parse_shutdown_reason(buf: &[u8]) -> ShutdownReason {
350        match buf.get(2) {
351            Some(1) => ShutdownReason::Watchdog,
352            Some(2) => ShutdownReason::ParentDied,
353            _ => ShutdownReason::Requested,
354        }
355    }
