3//! Input service client — keyboard / mouse focus registration.
4
5use super::vierr_from_code;
6use crate::service::InputRef;
7use crate::{ViError, ViResult};
8use api::ipc::{InputRequest, InputResponse, IPC_BUF_SIZE};
9
10/// Ergonomic client for the input service.
11///
12/// Wraps [`InputRef`] and provides focus registration helpers.
13pub struct InputClient {
14    svc: InputRef,
15}
16
17impl InputClient {
18    /// Create a new unresolved client. Resolution is lazy (first call).
19    pub fn new() -> Self {
20        Self {
21            svc: InputRef::new(),
22        }
23    }
24
25    /// Register the calling cell as the keyboard/mouse focus recipient.
26    ///
27    /// Returns `true` on success. May return `false` on a boot race (the input
28    /// service may not be registered yet) — retry after a `sys_yield`.
29    pub fn request_focus(&mut self) -> bool {
30        // Use the ostd::input helper which handles the TID lookup internally.
31        crate::input::request_focus()
32    }
33
34    /// Query which cell currently has focus. Returns the TID (0 = none).
35    pub fn get_focus(&mut self) -> ViResult<u32> {
36        let req = InputRequest::GetFocus;
37        let mut resp_buf = [0u8; IPC_BUF_SIZE];
38        match self
39            .svc
40            .call::<InputRequest, InputResponse>(&req, &mut resp_buf)?
41        {
42            InputResponse::Focus(tid) => Ok(tid),
43            InputResponse::Err(code) => Err(vierr_from_code(code)),
44            _ => Err(ViError::IO),
45        }
46    }
47
48    /// Unregister the calling cell from receiving input events.
49    pub fn clear_focus(&mut self, cell_tid: u32) -> ViResult<()> {
50        let req = InputRequest::ClearFocus { cell_tid };
51        let mut resp_buf = [0u8; IPC_BUF_SIZE];
52        match self
53            .svc
54            .call::<InputRequest, InputResponse>(&req, &mut resp_buf)?
55        {
56            InputResponse::Ok => Ok(()),
57            InputResponse::Err(code) => Err(vierr_from_code(code)),
58            _ => Err(ViError::IO),
59        }
60    }
61}
62
63impl Default for InputClient {
64    fn default() -> Self {
65        Self::new()
66    }
67}
