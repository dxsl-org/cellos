3//! `TlsStream` — an [`embedded_io`] adapter over the raw `ostd::tls` IPC.
4//!
5//! Mirrors [`TcpStream`][super::net::TcpStream]: it turns the net cell's
6//! request/reply TLS opcodes ([`tls_connect`]/[`tls_write`]/[`tls_read`]/
7//! [`tls_close`]) into a blocking `Read + Write` stream so a generic
8//! [`HttpClient`][crate::http::HttpClient] can drive HTTPS exactly the same way
9//! it drives plaintext HTTP over `TcpStream`.
10//!
11//! # Blocking / IPC contract
12//! - [`connect`][TlsStream::connect] resolves the NET service TID, then performs
13//!   the TCP + TLS handshake synchronously inside the net cell. It returns only
14//!   after the handshake completes (or the net cell reports failure).
15//! - [`Write::write`] loops over the 503-byte-per-call cap of [`tls_write`]
16//!   until every byte of `buf` is accepted (write-all semantics), because a
17//!   single TLS record write may be partially accepted by the net cell.
18//! - [`Read::read`] spin-yields while [`tls_read`] returns 0. A 0-read over TLS
19//!   is ambiguous: it means EITHER "no application data buffered yet" OR
20//!   "peer closed" (net `handlers.rs:462`) — there is no EOF signal. Body
21//!   completion is therefore decided by HTTP framing in
22//!   [`BodyReader`][crate::http::BodyReader], NOT by a 0-read here. To stop a
23//!   genuinely dead connection from hanging forever, the spin is bounded by a
24//!   retry budget; exhausting it surfaces as [`ViError::WouldBlock`] (mapped to
25//!   `embedded_io::ErrorKind::Other`) rather than an unbounded loop.
26//!
27//! # Security posture
28//! `TlsStream` inherits the net cell's current TLS verifier — today that is
29//! embedded-tls `UnsecureProvider` with **no certificate verification**.
30//! `HttpClient<TlsStream>` is therefore only as trustworthy as the net cell;
31//! hardening cert verification is a separate net-cell workstream. Do not treat
32//! this as authenticated HTTPS.
33
34extern crate alloc;

36use crate::service;
37use crate::tls::{tls_close, tls_connect, tls_read, tls_write};
38use crate::{ViError, ViResult};

40/// Max bytes accepted by a single [`tls_write`] IPC call (mirrors the cap in
41/// `tls.rs`: 512-byte IPC buffer minus the 9-byte `[op][cap]` header).
42const MAX_TLS_WRITE: usize = 503;
43
44/// Number of consecutive 0-byte reads tolerated before [`Read::read`] gives up.
45///
46/// Each miss yields the scheduler, so this is a "no progress for N scheduler
47/// turns" budget, not a wall-clock timeout (ostd has no per-read timer here).
48/// Sized generously so a slow-but-live LLM response never trips it, while a
49/// mid-body peer drop still terminates instead of hanging. A caller that wants
50/// a stricter bound layers its own timeout above the stream.
51const READ_RETRY_BUDGET: u32 = 100_000;
52
53/// A TLS 1.3 connection handle implementing [`embedded_io::Read`] +
54/// [`embedded_io::Write`].
55///
56/// The underlying connection (and its TCP socket) is closed when the handle is
57/// dropped (RAII — Law 8).
58pub struct TlsStream {
59    net_tid: usize,
60    cap_id: u64,
61}
62
63impl TlsStream {
64    /// Open a TLS 1.3 connection to `addr:port` with SNI `hostname`.
65    ///
66    /// Resolves the NET service via [`service::lookup`] and performs the
67    /// handshake synchronously. Returns `ViError::NotFound` if the net service
68    /// is unavailable, or `ViError::IO` if the handshake fails.
69    pub fn connect(addr: [u8; 4], port: u16, hostname: &str) -> ViResult<Self> {
70        let net_tid = service::lookup(service::service::NET).ok_or(ViError::NotFound)?;
71        let cap_id = tls_connect(net_tid, addr, port, hostname);
72        if cap_id == 0 {
73            return Err(ViError::IO);
74        }
75        Ok(Self { net_tid, cap_id })
76    }
77}
78
79impl Drop for TlsStream {
80    fn drop(&mut self) {
81        tls_close(self.net_tid, self.cap_id);
82    }
83}
84
85impl embedded_io::ErrorType for TlsStream {
86    type Error = crate::io::OstdError;
87}

89impl embedded_io::Read for TlsStream {
90    /// Block (with cooperative yields) until at least one byte is available.
91    ///
92    /// A 0-read from the net cell is treated as "not ready yet" and retried, up
93    /// to [`READ_RETRY_BUDGET`] times. The HTTP layer never relies on a 0-read
94    /// to mean EOF — see the module doc.
95    fn read(&mut self, buf: &mut [u8]) -> Result<usize, crate::io::OstdError> {
96        if buf.is_empty() {
97            return Ok(0);
98        }
99        let mut misses: u32 = 0;
100        loop {
101            let n = tls_read(self.net_tid, self.cap_id, buf);
102            if n > 0 {
103                return Ok(n);
104            }
105            misses += 1;
106            if misses >= READ_RETRY_BUDGET {
107                // Dead connection or stalled peer — surface as WouldBlock rather
108                // than spinning forever (EOF-less TLS guard, plan risk #2).
109                return Err(crate::io::OstdError(ViError::WouldBlock));
110            }
111            crate::task::yield_now();
112        }
113    }
114}
115
116impl embedded_io::Write for TlsStream {
117    /// Write-all: loop over the 503-byte [`tls_write`] cap until every byte of
118    /// `buf` has been accepted by the net cell.
119    ///
120    /// Returns `Ok(buf.len())` on success. A stalled send (the net cell accepts
121    /// 0 bytes) surfaces as `ViError::WouldBlock` after the retry budget,
122    /// rather than looping forever.
123    fn write(&mut self, buf: &[u8]) -> Result<usize, crate::io::OstdError> {
124        if buf.is_empty() {
125            return Ok(0);
126        }
127        let mut sent = 0usize;
128        let mut stalls: u32 = 0;
129        while sent < buf.len() {
130            let chunk_end = (sent + MAX_TLS_WRITE).min(buf.len());
131            let n = tls_write(self.net_tid, self.cap_id, &buf[sent..chunk_end]);
132            if n == 0 {
133                stalls += 1;
134                if stalls >= READ_RETRY_BUDGET {
135                    return Err(crate::io::OstdError(ViError::WouldBlock));
136                }
137                crate::task::yield_now();
138                continue;
139            }
140            stalls = 0;
141            sent += n;
142        }
143        Ok(sent)
144    }
145
146    fn flush(&mut self) -> Result<(), crate::io::OstdError> {
147        Ok(()) // tls_write commits synchronously through the net cell
148    }
149}
