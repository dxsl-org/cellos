4/// Root `_start` against `--gc-sections`.
5///
6/// `_start` lives in `.text.boot`, which cell linker scripts pull without
7/// `KEEP`. Nothing in Rust code *references* the symbol — it is only named by
8/// the linker script's `ENTRY(_start)`, which is NOT a GC root — so it was
9/// collected out of every cell, and the ELF entry silently fell through to
10/// `main`. This `#[used]` pointer creates a real relocation to `_start` from an
11/// object that is always linked (this module holds the `#[panic_handler]`),
12/// keeping it against GC. Belt-and-suspenders alongside `ENTRY(_start)`.
13#[used]
14static _KEEP_START: unsafe extern "C" fn() -> ! = _start;

16/// Cell entry point — sets up gp/sp and jumps into the ostd runtime.
17///
18/// # Safety
19/// Never call from Rust: the kernel loader jumps here exactly once on a fresh
20/// stack; the naked asm assumes registers per the cell spawn ABI.
21#[no_mangle]
22#[unsafe(naked)]
23#[link_section = ".text.boot"]
24pub unsafe extern "C" fn _start() -> ! {
25    #[cfg(target_arch = "riscv64")]
26    core::arch::naked_asm!(
27        ".option push",
28        ".option norelax",
29        "la gp, __global_pointer$",
30        ".option pop",
31        "andi sp, sp, -16",
32        // Use callee-saved s0/s1 as iterators — a0/a1 are caller-saved and
33        // would be clobbered by any constructor called via jalr.
34        // `lla` (load LOCAL address) is strictly PC-relative (auipc+addi); `la`
35        // for a linker-defined global would go GOT-indirect, which is fragile
36        // under -pie/pic (the GOT must be relocated first). PC-relative is
37        // correct here — the init-array is inside the cell's own image.
38        "lla  s0, __init_array_start",
39        "lla  s1, __init_array_end",
40        "1:",
41        "beq  s0, s1, 2f",
42        "ld   t0, 0(s0)",
43        "jalr t0",
44        "addi s0, s0, 8",
45        "j    1b",
46        "2:",
47        "call main",
48        "li a7, 60", // ViSyscall::Exit
49        "li a0, 0",  // exit code = 0 in a0 (ViCell ABI: syscall nr in a7, arg in a0)
50        "ecall",
51        "1: j 1b"
52    );
53    // ViCell ARM64 ABI: x0=syscall_nr, x1=a0 (exit code).
54    // Stack is kernel-aligned on entry; skip re-alignment to avoid clobbering sp.
55    #[cfg(target_arch = "aarch64")]
56    core::arch::naked_asm!(
57        // adrp+add (:lo12:) forms the address PC-relatively. `ldr x,=sym` would
58        // stash the address in a literal pool and emit an R_AARCH64_ABS64 that
59        // rust-lld REJECTS under -pie ("cannot be used against symbol; recompile
60        // with -fPIC") — which is why cells previously entered at bare `main`,
61        // skipping this crt0 and the post-main Exit ecall (CLI cells crashed
62        // scause=0xc sepc=0 on return). See docs/specs/17 §8 / cell-build.
63        "adrp x19, __init_array_start",
64        "add  x19, x19, :lo12:__init_array_start",
65        "adrp x20, __init_array_end",
66        "add  x20, x20, :lo12:__init_array_end",
67        "1:",
68        "cmp  x19, x20",
69        "b.eq 2f",
70        "ldr  x21, [x19], #8",
71        "blr  x21",
72        "b    1b",
73        "2:",
74        "bl   main",
75        "mov  x0, #60", // ViSyscall::Exit
76        "mov  x1, #0",  // exit code = 0
77        "svc  #0",
78        "1: b 1b"
79    );
80    // ViCell x86_64 ABI: RAX=syscall_nr, RDI=a0.
81    // The kernel-supplied stack is 16-byte-aligned at entry; AND to guarantee it
82    // before the CALL so the SysV AMD64 ABI requirement is always satisfied.
83    // sys_exit(0): ViSyscall::Exit = 60, exit code in RDI = 0.
84    #[cfg(target_arch = "x86_64")]
85    core::arch::naked_asm!(
86        "and rsp, -16",
87        "lea rbx, [rip + __init_array_start]",
88        "lea r12, [rip + __init_array_end]",
89        "1:",
90        "cmp rbx, r12",
91        "je 2f",
92        "call [rbx]",
93        "add rbx, 8",
94        "jmp 1b",
95        "2:",
96        "call main",
97        "mov rax, 60",  // ViSyscall::Exit
98        "xor rdi, rdi", // exit code = 0
99        "syscall",
100        "2: jmp 2b",
101    );
102}

104// User applications must define `fn main() -> !` or `fn main()`.
105// Since we don't have a standardized `main` signature yet in ostd macro,
106// we will assume the app defines `no_mangle pub extern "C" fn main()`.
107extern "C" {
108    fn main();
109}
110
111#[no_mangle]
112pub extern "C" fn generic_main() -> ! {
113    unsafe {
114        main();
115    }
116    // If main returns, we exit
117    crate::syscall::sys_exit(0);
118}

120#[panic_handler]
121fn panic(info: &PanicInfo) -> ! {
122    // Log panic
123    // We don't have a proper writer yet, so just stringify manually?
124    // Or just sys_log("PANIC!");
125    let _ = sys_log("PANIC: Application crashed!\n");
126    if let Some(location) = info.location() {
127        // simple formatting
128        let _ = sys_log("Location: ");
129        let _ = sys_log(location.file());
130    }
131
132    // Exit
133    crate::syscall::sys_exit(1);
134}
