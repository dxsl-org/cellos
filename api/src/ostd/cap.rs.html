2//! Runtime capability delegation handles.
3//!
4//! A `CapHandle` represents a revocable capability delegation from a supervisor
5//! to a child cell.  The supervisor (e.g. `init`) constructs a `CapHandle` after
6//! spawning a child and noting which caps were granted.  Calling [`CapHandle::revoke`]
7//! issues `sys_cap_revoke`, stripping those caps from the live target in-kernel.
8//!
9//! The target cell continues running but any subsequent syscall that requires a
10//! revoked cap is denied with `PermissionDenied`.  For system cells (`block_io` /
11//! `network` holders) revocation is blocked — use `sys_hotswap` instead.
12
13use crate::syscall::{sys_cap_revoke, SyscallError};
14use api::syscall::cap_mask;
15
16/// A handle representing a revocable capability delegation to another cell.
17///
18/// # Monotonic downgrade invariant
19/// `CapHandle` does not grant new capabilities; it only tracks what was already
20/// granted at spawn time.  A cell can never hand a child a `CapHandle` whose
21/// `cap_mask` exceeds the caps the supervisor itself holds.
22///
23/// # Revocation semantics
24/// Revocation is best-effort and immediate: the kernel clears the TCB field
25/// atomically under the scheduler lock.  The target cell may have already
26/// consumed the cap (e.g. opened an MMIO range via `sys_request_mmio`); those
27/// live effects are not undone.  Revocation prevents future use only.
28///
29/// For full teardown (revoke + kill + respawn) combine with `sys_force_exit`
30/// and a subsequent `sys_spawn_from_path`.
31#[derive(Debug, Clone, Copy, PartialEq, Eq)]
32pub struct CapHandle {
33    target_tid: usize,
34    cap_mask: u32,
35}
36
37impl CapHandle {
38    /// Create a handle representing the given `cap_mask` granted to `target_tid`.
39    ///
40    /// No syscall is issued; this only records the association so [`revoke`]
41    /// knows what to strip.
42    ///
43    /// [`revoke`]: Self::revoke
44    pub const fn new(target_tid: usize, cap_mask: u32) -> Self {
45        Self {
46            target_tid,
47            cap_mask,
48        }
49    }
50
51    /// Revoke all capabilities this handle represents from the live target cell.
52    ///
53    /// Equivalent to calling `sys_cap_revoke(target_tid, cap_mask)`.
54    ///
55    /// # Errors
56    /// See [`sys_cap_revoke`] — `PermissionDenied` if caller lacks `SpawnCap`,
57    /// target is a system cell, or `InvalidCommand` if the target no longer exists.
58    pub fn revoke(&self) -> Result<(), SyscallError> {
59        sys_cap_revoke(self.target_tid, self.cap_mask)
60    }
61
62    /// Revoke only the specified subset of capabilities (must be a subset of
63    /// `self.cap_mask`; extra bits are silently ignored by the kernel).
64    pub fn revoke_partial(&self, mask: u32) -> Result<(), SyscallError> {
65        sys_cap_revoke(self.target_tid, self.cap_mask & mask)
66    }
67
68    /// Task ID of the cell this handle controls.
69    pub const fn target_tid(&self) -> usize {
70        self.target_tid
71    }
72
73    /// The full capability mask this handle can revoke.
74    pub const fn cap_mask(&self) -> u32 {
75        self.cap_mask
76    }
77
78    /// Return `true` if this handle controls the network capability.
79    pub const fn has_network(&self) -> bool {
80        self.cap_mask & cap_mask::NETWORK != 0
81    }
82
83    /// Return `true` if this handle controls the spawn capability.
84    pub const fn has_spawn(&self) -> bool {
85        self.cap_mask & cap_mask::SPAWN != 0
86    }
87}
