3//! `HttpClient<T>` — drives an HTTP/1.1 request/response exchange over any
4//! blocking [`embedded_io::Read`] + [`embedded_io::Write`] transport.
5//!
6//! Works identically over [`TcpStream`][crate::clients::TcpStream] (plaintext
7//! HTTP) and [`TlsStream`][crate::clients::TlsStream] (HTTPS) — that generic
8//! transport is the whole point of the embedded_io adapters.
9//!
10//! # Exchange contract (what completes before what)
11//! 1. [`send`][HttpClient::send] writes the full request bytes via `write_all`
12//!    (the transport handles its own chunking).
13//! 2. It then reads response bytes into a growing buffer, calling
14//!    [`parse_response_headers`] after each read until it returns `Ok` (headers
15//!    complete). A `NeedMoreData` means "keep accumulating".
16//! 3. A [`BodyReader`] is constructed seeded with the post-header bytes already
17//!    in the buffer (`buf[body_offset..]`) — CRITICAL because the first chunk or
18//!    a short Content-Length body often arrives in the SAME packet as the
19//!    headers. The caller then drains the body via `reader.read(transport, ..)`.
20
21extern crate alloc;
22
23use alloc::vec::Vec;
24use embedded_io::{Read, Write};
25use http_core::request::RequestBuilder;
26use http_core::response::parse_response_headers;
27use http_core::{BodyReader, HttpError, ParsedHeaders};
28
29/// Initial response accumulation buffer capacity (one IPC-sized read).
30const INITIAL_RECV: usize = 4096;
31
32/// Hard cap on accumulated *header* bytes before giving up, guarding against a
33/// hostile server that never terminates the header block. The body is NOT
34/// buffered here (that is `BodyReader`'s streaming job) — only headers.
35const MAX_HEADER_BYTES: usize = 64 * 1024;
36
37/// A generic blocking HTTP/1.1 client over a `Read + Write` transport.
38pub struct HttpClient<T> {
39    transport: T,
40}
41
42impl<T: Read + Write> HttpClient<T> {
43    /// Wrap an already-connected transport.
44    pub fn new(transport: T) -> Self {
45        Self { transport }
46    }
47
48    /// Consume the client and return the underlying transport (e.g. to close it).
49    pub fn into_inner(self) -> T {
50        self.transport
51    }
52
53    /// Send a pre-built request and parse the response headers.
54    ///
55    /// Returns the parsed headers plus a [`BodyReader`] seeded with any body
56    /// bytes that arrived alongside the headers. Drain the body by repeatedly
57    /// calling [`BodyReader::read`] with [`transport`][HttpClient::transport].
58    ///
59    /// # Errors
60    /// - Transport write/read failure → `HttpError::Other`.
61    /// - Headers never complete within [`MAX_HEADER_BYTES`] → `HttpError::Other`.
62    /// - Malformed status/headers → propagated from [`parse_response_headers`].
63    pub fn send(&mut self, req: &[u8]) -> Result<(ParsedHeaders, BodyReader), HttpError> {
64        self.transport
65            .write_all(req)
66            .map_err(|_| HttpError::Other(alloc::string::String::from("transport write failed")))?;
67        self.transport
68            .flush()
69            .map_err(|_| HttpError::Other(alloc::string::String::from("transport flush failed")))?;
70
71        let mut buf: Vec<u8> = Vec::with_capacity(INITIAL_RECV);
72        let mut tmp = [0u8; INITIAL_RECV];
73
74        loop {
75            // Try to parse what we have. The first iteration may already hold a
76            // complete header block if the request and response raced; but with
77            // an empty buf parse returns NeedMoreData, so read first below.
78            match parse_response_headers(&buf) {
79                Ok(headers) => {
80                    let body = &buf[headers.body_offset..];
81                    let reader = BodyReader::new(headers.framing, headers.content_length, body);
82                    return Ok((headers, reader));
83                }
84                Err(HttpError::NeedMoreData) => { /* fall through to read more */ }
85                Err(e) => return Err(e),
86            }
87
88            if buf.len() >= MAX_HEADER_BYTES {
89                return Err(HttpError::Other(alloc::string::String::from(
90                    "response headers exceeded limit",
91                )));
92            }
93
94            let n = self.transport.read(&mut tmp).map_err(|_| {
95                HttpError::Other(alloc::string::String::from("transport read failed"))
96            })?;
97            if n == 0 {
98                // Transport reported EOF before headers completed. For TLS this
99                // is also "no data yet"; TlsStream::read already spins+bounds,
100                // so a 0 here from TcpStream means a real FIN — headers can never
101                // complete.
102                return Err(HttpError::Other(alloc::string::String::from(
103                    "connection closed before headers completed",
104                )));
105            }
106            buf.extend_from_slice(&tmp[..n]);
107        }
108    }
109
110    /// Convenience POST built on [`RequestBuilder`].
111    ///
112    /// Emits `Content-Type: <content_type>` as the sole extra header; the
113    /// builder auto-adds `Host`, `Content-Length`, and `Connection: close`.
114    pub fn post(
115        &mut self,
116        host: &str,
117        path: &str,
118        content_type: &str,
119        body: &[u8],
120    ) -> Result<(ParsedHeaders, BodyReader), HttpError> {
121        let extra = [("Content-Type", content_type)];
122        let req = RequestBuilder::new("POST", path, host, &extra, Some(body)).build();
123        self.send(&req)
124    }
125}
