3//! Filesystem utilities for ViCell cells.
4//!
5//! `File` is backed by a kernel capability (`CapId`) obtained via `OpenCap`.
6//! Single-owner: moving a `File` transfers the capability.  Dropping without
7//! calling `close()` issues an implicit close (which revokes the capability)
8//! and, in debug builds, emits a warning about the implicit close.
9
10use crate::syscall;
11use alloc::string::String;
12use alloc::vec::Vec;
13use api::ipc::{VfsRequest, VfsResponse};
14use types::*;

16/// Iterator over directory entries returned by the kernel.
17pub struct ReadDir {
18    /// fd from legacy `sys_open` — kept for directory listing (caps are file-only for now).
19    fd: usize,
20}
21
22impl Iterator for ReadDir {
23    type Item = DirEntry;
24
25    fn next(&mut self) -> Option<Self::Item> {
26        let mut entry = DirEntry::default();
27        // SAFETY: entry is a valid stack-allocated DirEntry; pointer is valid for the call.
28        let ptr = &mut entry as *mut _ as *mut u8;
29        let size = core::mem::size_of::<DirEntry>();
30        let slice = unsafe { core::slice::from_raw_parts_mut(ptr, size) };
31        match syscall::sys_read_dir(self.fd, slice) {
32            Ok(bytes) if bytes == size => Some(entry),
33            _ => None,
34        }
35    }
36}
37
38impl Drop for ReadDir {
39    fn drop(&mut self) {
40        syscall::sys_close(self.fd);
41    }
42}
43
44/// Open directory for reading.
45pub fn read_dir(path: &str) -> ViResult<ReadDir> {
46    let fd = syscall::sys_open(path).map_err(|_| ViError::NotFound)?;
47    Ok(ReadDir { fd })
48}

52/// An open file backed by a kernel capability (`CapId`).
53///
54/// Moving `File` transfers ownership of the underlying capability.  Dropping
55/// calls `close()` implicitly; in debug builds a warning is emitted when this
56/// happens without an explicit `close()` call (handle-leak detection).
57///
58/// `path` and `vfs_tid` are cached at `open()` time to support `write_all` via
59/// `VfsRequest::Append` IPC. Reads still use the faster cap-based kernel path.
60pub struct File {
61    cap_id: u64,
62    /// Set to `true` by `close()` to suppress the drop warning.
63    closed: bool,
64    /// Owned copy of the path, needed for VFS Append IPC.
65    path: String,
66    /// Cached VFS service tid; 0 means VFS was unavailable at open time.
67    vfs_tid: usize,
68}
69
70impl File {
71    /// Open a file at `path`.
72    ///
73    /// Caches the VFS tid at open time for use by `write_all`. If the VFS
74    /// service is temporarily unavailable, `write_all` will return `NotFound`
75    /// until the file is re-opened after VFS restarts.
76    ///
77    /// # Errors
78    /// Returns `ViError::NotFound` if the path does not exist in the kernel FS.
79    pub fn open(path: &str) -> ViResult<Self> {
80        let cap_id = syscall::sys_open_cap(path).map_err(|_| ViError::NotFound)?;
81        let vfs_tid = crate::service::lookup(crate::service::service::VFS).unwrap_or(0);
82        Ok(Self {
83            cap_id,
84            closed: false,
85            path: String::from(path),
86            vfs_tid,
87        })
88    }
89
90    /// Read all bytes until EOF into `buf`.
91    pub fn read_to_end(&mut self, buf: &mut Vec<u8>) -> ViResult<usize> {
92        let mut temp = [0u8; 512];
93        let mut total = 0;
94        loop {
95            match syscall::sys_read_cap(self.cap_id, &mut temp) {
96                Ok(0) => break,
97                Ok(n) => {
98                    buf.extend_from_slice(&temp[..n]);
99                    total += n;
100                }
101                Err(_) => return Err(ViError::IO),
102            }
103        }
104        Ok(total)
105    }
106
107    /// Read up to `buf.len()` bytes from the file.
108    ///
109    /// Returns the number of bytes actually read (0 = EOF).
110    pub fn read(&mut self, buf: &mut [u8]) -> ViResult<usize> {
111        syscall::sys_read_cap(self.cap_id, buf).map_err(|_| ViError::IO)
112    }
113
114    /// Read the entire file into a `String`.  Returns `Err(IO)` if content is not valid UTF-8.
115    pub fn read_to_string(&mut self) -> ViResult<alloc::string::String> {
116        let mut bytes = alloc::vec::Vec::new();
117        self.read_to_end(&mut bytes)?;
118        alloc::string::String::from_utf8(bytes).map_err(|_| ViError::IO)
119    }
120
121    /// Append `buf` to the file via VFS IPC.
122    ///
123    /// **Append semantics:** each call extends the file at its current end.
124    /// Multiple `write_all` calls accumulate — correct for `embedded_io::Write`.
125    /// For overwrite / truncate semantics, that is future work.
126    ///
127    /// Large buffers are chunked into ≤400-byte pieces to fit the 512-byte
128    /// `vfs_call` send buffer (path + content + postcard framing ≤ 512 bytes).
129    /// Not atomic across chunks — a quota failure mid-write leaves a partial append.
130    ///
131    /// # Errors
132    /// - `NotFound` if VFS is unavailable (re-open after VFS restarts to refresh tid).
133    /// - `InvalidInput` if `path.len() > 96` (postcard frame would overflow send buffer).
134    /// - `IO` on VFS error response (permission denied, quota exceeded, backend failure).
135    pub fn write_all(&mut self, buf: &[u8]) -> ViResult<()> {
136        if self.vfs_tid == 0 {
137            return Err(ViError::NotFound);
138        }
139        if self.path.len() > 96 {
140            // postcard: discriminant(1) + path_len(2) + path(≤96) + content_len(2) + content(≤400) = ≤501 < 512
141            return Err(ViError::InvalidInput);
142        }
143        for chunk in buf.chunks(CHUNK_CONTENT) {
144            let req = VfsRequest::Append {
145                path: &self.path,
146                content: chunk,
147            };
148            let mut resp_buf = [0u8; 512];
149            match vfs_call(self.vfs_tid, &req, &mut resp_buf)? {
150                VfsResponse::Ok => {}
151                _ => return Err(ViError::IO),
152            }
153        }
154        Ok(())
155    }
156
157    /// Explicitly close the file and revoke its capability.
158    pub fn close(mut self) -> ViResult<()> {
159        self.closed = true;
160        syscall::sys_close_cap(self.cap_id);
161        Ok(())
162    }
163
164    /// Return the file's size in bytes without moving the cursor.
165    ///
166    /// # Errors
167    /// Returns `ViError::IO` if the cap is invalid or the query fails.
168    pub fn size(&self) -> ViResult<u64> {
169        syscall::sys_stat_cap(self.cap_id).map_err(|_| ViError::IO)
170    }
171
172    /// Truncate the file to exactly `len` bytes.
173    ///
174    /// Returns `ViError::IO` if `len > current_size` or the backend does not
175    /// support truncation.  Use `write` / `embedded_io::Write` to extend a file.
176    pub fn truncate(&self, len: u64) -> ViResult<()> {
177        syscall::sys_truncate_cap(self.cap_id, len).map_err(|_| ViError::IO)
178    }
179
180    /// Return the raw capability ID (for passing to kernel APIs).
181    pub fn cap_id(&self) -> u64 {
182        self.cap_id
183    }
184}
185
186impl Drop for File {
187    fn drop(&mut self) {
188        if !self.closed {
189            // Revoke the kernel capability so it doesn't leak after the File is gone.
190            // This is the normal Rust drop path (error propagation, end-of-scope, etc.).
191            // Calling `File::close()` first is preferred so errors can be observed,
192            // but this implicit close is always safe.
193            syscall::sys_close_cap(self.cap_id);
194        }
195    }
196}

200/// Max content bytes per `VfsRequest::Append` IPC call.
201///
202/// Bound by `vfs_call`'s 512-byte send buffer: discriminant(1) + path_len(1,
203/// since ≤96 < 128 → 1-byte postcard varint) + path(≤96) + content_len(2,
204/// since 400 ≥ 128 → 2-byte varint) + content(≤400) + slack(12) = 512.
205/// Changing this constant requires re-verifying that budget.
206const CHUNK_CONTENT: usize = 400;
207
208impl embedded_io::ErrorType for File {
209    type Error = crate::io::OstdError;
210}

212impl embedded_io::Read for File {
213    fn read(&mut self, buf: &mut [u8]) -> Result<usize, crate::io::OstdError> {
214        // Explicit path to avoid ambiguity with the trait method of the same name.
215        // Inherent methods win in Rust method resolution, but this is clearer.
216        File::read(self, buf).map_err(crate::io::OstdError)
217    }
218}

220impl embedded_io::Seek for File {
221    fn seek(&mut self, pos: embedded_io::SeekFrom) -> Result<u64, crate::io::OstdError> {
222        let (offset, whence) = match pos {
223            embedded_io::SeekFrom::Start(n) => (n as i64, 0u8),
224            embedded_io::SeekFrom::Current(n) => (n, 1u8),
225            embedded_io::SeekFrom::End(n) => (n, 2u8),
226        };
227        crate::syscall::sys_seek_cap(self.cap_id, offset, whence)
228            .map_err(|_| crate::io::OstdError(ViError::IO))
229    }
230}

232impl embedded_io::Write for File {
233    /// Write `buf` into the file at the current cursor position via `WriteCap` syscall.
234    ///
235    /// Uses the kernel cap path (symmetric with `ReadCap`) rather than VFS IPC.
236    /// Read-only filesystems (BootFS) will return `Err(IO)` from the kernel.
237    fn write(&mut self, buf: &[u8]) -> Result<usize, crate::io::OstdError> {
238        crate::syscall::sys_write_cap(self.cap_id, buf)
239            .map_err(|_| crate::io::OstdError(ViError::IO))
240    }
241
242    fn flush(&mut self) -> Result<(), crate::io::OstdError> {
243        crate::syscall::sys_sync_cap(self.cap_id).map_err(|_| crate::io::OstdError(ViError::IO))
244    }
245}

249/// Blocking IPC call to the VFS service: encode `req`, send, receive, decode.
250///
251/// Uses a stack-allocated 512-byte buffer for both directions.
252fn vfs_call<'r>(
253    vfs_tid: usize,
254    req: &VfsRequest<'_>,
255    resp_buf: &'r mut [u8; 512],
256) -> ViResult<VfsResponse<'r>> {
257    let mut send_buf = [0u8; 512];
258    let n = api::ipc::encode(req, &mut send_buf)
259        .map(|s| s.len())
260        .map_err(|_| ViError::IO)?;
261    syscall::sys_send(vfs_tid, &send_buf[..n]);
262    syscall::sys_recv(0, resp_buf);
263    api::ipc::decode::<VfsResponse>(resp_buf).map_err(|_| ViError::IO)
264}

266/// Read the ENTIRE file at `path` into a freshly-allocated Grant via the VFS mount
267/// table (so it reaches the `/bin` cell-store overlay AND VIFS1 bootstrap cells).
268/// Returns `(grant_id, len)`; the caller OWNS the grant and MUST `sys_grant_free`
269/// it after use. Backs `sys_spawn_from_path`'s VFS routing — reads a cell ELF for
270/// `sys_spawn_from_elf` so the kernel loader needs no disk access post-boot.
271///
272/// Uses masked `service_call_typed` (recv pinned to `vfs_tid`, spec 17 §2) so a
273/// spawning cell that holds input focus can't decode a queued keystroke as the reply.
274pub fn read_full_via_grant(path: &str, vfs_tid: usize) -> ViResult<(usize, usize)> {
275    // 1. Stat for the file size (bounds the grant).
276    let mut sb = [0u8; 512];
277    let mut rb = [0u8; 64];
278    let size = match crate::ipc::service_call_typed::<_, VfsResponse>(
279        vfs_tid,
280        &VfsRequest::Stat(path),
281        &mut sb,
282        &mut rb,
283    ) {
284        Ok(VfsResponse::Stat { size, is_dir }) if !is_dir && size > 0 => size as usize,
285        _ => return Err(ViError::NotFound),
286    };
287    // 2. Allocate a grant sized to the file; share RW so VFS can fill it.
288    let grant_id = syscall::sys_grant_alloc(size).ok_or(ViError::OutOfMemory)?;
289    syscall::sys_grant_share(grant_id, vfs_tid, 2 /* ReadWrite */);
290    // 3. One-shot full-file read into the grant (bounded by `max = size`).
291    let mut sb2 = [0u8; 512];
292    let mut rb2 = [0u8; 64];
293    let bytes = match crate::ipc::service_call_typed::<_, VfsResponse>(
294        vfs_tid,
295        &VfsRequest::ReadFileGrant {
296            path,
297            grant: grant_id,
298            max: size,
299        },
300        &mut sb2,
301        &mut rb2,
302    ) {
303        Ok(VfsResponse::GrantDone { bytes }) if bytes > 0 => bytes,
304        _ => {
305            syscall::sys_grant_free(grant_id);
306            return Err(ViError::IO);
307        }
308    };
309    Ok((grant_id, bytes))
310}

312/// Read up to `buf.len()` bytes from a file cap using the optimal I/O path.
313///
314/// - `buf.len() < 4096`: kernel ReadCap path (no Grant overhead)
315/// - `buf.len() >= 4096`: zero-copy Grant path (one VFS round-trip per 4096 bytes)
316///
317/// # F14 contract
318/// The Grant is freed only AFTER `GrantDone` is received from VFS, ensuring
319/// VFS has finished reading the buffer before the caller reclaims the frames.
320///
321/// # Errors
322/// Returns `ViError::IO` on any transport or permission failure.
323pub fn read_all(cap_id: u64, buf: &mut [u8], vfs_tid: usize) -> ViResult<usize> {
324    if buf.len() < 4096 {
325        syscall::sys_read_cap(cap_id, buf).map_err(|_| ViError::IO)
326    } else {
327        grant_read(cap_id, buf, vfs_tid)
328    }
329}

331/// Write `data` to a file cap using the optimal I/O path.
332///
333/// - `data.len() < 4096`: kernel WriteGrant IPC path (no Grant overhead; caller
334///   uses existing `VfsRequest::Write` via IPC — stub, returns 0 for now)
335/// - `data.len() >= 4096`: zero-copy Grant path
336///
337/// # F14 contract
338/// The caller waits for `GrantDone` before freeing the grant, so VFS finishes
339/// writing to disk before the frames are returned to the allocator.
340pub fn write_all(cap_id: u64, data: &[u8], vfs_tid: usize) -> ViResult<usize> {
341    if data.len() < 4096 {
342        // Small writes: caller uses existing VfsRequest::Write IPC directly.
343        // This wrapper covers the large-file case only; return 0 to signal fallback.
344        let _ = (cap_id, vfs_tid);
345        Ok(0)
346    } else {
347        grant_write(cap_id, data, vfs_tid)
348    }
349}

351fn grant_read(cap_id: u64, buf: &mut [u8], vfs_tid: usize) -> ViResult<usize> {
352    let size = buf.len().min(4096);
353    let grant_id = syscall::sys_grant_alloc(size).ok_or(ViError::OutOfMemory)?;
354    // Share RW with VFS so it can fill the grant buffer.
355    syscall::sys_grant_share(grant_id, vfs_tid, 2 /* ReadWrite */);
356
357    // Control message fits in 512B IPC buffer.
358    let req = VfsRequest::ReadGrant {
359        cap: cap_id,
360        offset: 0,
361        size,
362        grant: grant_id,
363    };
364    let mut resp_buf = [0u8; 512];
365    let resp = vfs_call(vfs_tid, &req, &mut resp_buf).inspect_err(|_e| {
366        syscall::sys_grant_free(grant_id);
367    })?;
368
369    let bytes = match resp {
370        // F14: GrantDone arrives only AFTER VFS has filled the grant buffer.
371        VfsResponse::GrantDone { bytes } => bytes,
372        _ => {
373            syscall::sys_grant_free(grant_id);
374            return Err(ViError::IO);
375        }
376    };
377
378    // SAFETY: grant was allocated with `size` bytes; VFS filled `bytes` of it.
379    let ptr = syscall::sys_grant_slice(grant_id).ok_or_else(|| {
380        syscall::sys_grant_free(grant_id);
381        ViError::IO
382    })?;
383    let src = unsafe { core::slice::from_raw_parts(ptr as *const u8, bytes) };
384    buf[..bytes].copy_from_slice(src);
385
386    // F14: safe to free — GrantDone already received above.
387    syscall::sys_grant_free(grant_id);
388    Ok(bytes)
389}

391fn grant_write(cap_id: u64, data: &[u8], vfs_tid: usize) -> ViResult<usize> {
392    let bytes = data.len().min(4096);
393    let grant_id = syscall::sys_grant_alloc(bytes).ok_or(ViError::OutOfMemory)?;
394
395    // Fill grant buffer BEFORE sharing — we own it exclusively here.
396    // SAFETY: grant was allocated for `bytes`; ptr is valid for that range.
397    let ptr = syscall::sys_grant_slice(grant_id).ok_or_else(|| {
398        syscall::sys_grant_free(grant_id);
399        ViError::IO
400    })?;
401    unsafe { core::ptr::copy_nonoverlapping(data.as_ptr(), ptr, bytes) };
402
403    // Share WriteOnly (VFS reads, can't modify).
404    syscall::sys_grant_share(grant_id, vfs_tid, 1 /* WriteOnly */);
405
406    let req = VfsRequest::WriteGrant {
407        cap: cap_id,
408        offset: 0,
409        grant: grant_id,
410        bytes,
411    };
412    let mut resp_buf = [0u8; 512];
413    // ipc_call blocks until VFS replies — F14 guarantees VFS drained the grant.
414    let resp = vfs_call(vfs_tid, &req, &mut resp_buf).inspect_err(|_e| {
415        syscall::sys_grant_free(grant_id);
416    })?;
417
418    syscall::sys_grant_free(grant_id);
419    match resp {
420        VfsResponse::GrantDone { bytes: written } => Ok(written),
421        _ => Err(ViError::IO),
422    }
