1//! DMA buffer helpers for Tier-1 Driver Cells.
2//!
3//! Driver Cells allocate physically contiguous memory for DMA rings via `sys_grant_alloc`
4//! (which uses `allocate_contiguous` internally), then authorise the IOMMU via
5//! `sys_grant_dma`. This module wraps both into a safe interface.
6//!
7//! # Usage
8//! ```ignore
9//! let buf = DmaBuf::alloc(4).expect("OOM");      // 4 pages (16 KiB)
10//! buf.authorize(0x0300).expect("IOMMU deny");     // BDF 0:3:0
11//! let phys = buf.phys();                         // program into NVMe/e1000 regs
12//! ```
13
14use crate::syscall::{sys_grant_alloc, sys_grant_dma, sys_grant_free, SyscallError};
15
16/// A physically-contiguous, page-aligned DMA buffer backed by a Grant region.
17///
18/// In SAS the virtual address equals the physical address (identity mapping),
19/// so [`phys`](DmaBuf::phys) and [`virt`](DmaBuf::virt) return the same value.
20///
21/// Drop does NOT free the buffer — call [`DmaBuf::free`] explicitly.
22/// Driver Cells typically hold DMA buffers for their entire lifetime.
23pub struct DmaBuf {
24    grant_id: usize, // physical base == virtual base in SAS
25    n_pages: usize,
26}
27
28impl DmaBuf {
29    /// Allocate `n_pages` contiguous 4-KiB pages suitable for DMA.
30    pub fn alloc(n_pages: usize) -> Option<Self> {
31        let grant_id = sys_grant_alloc(n_pages * 4096)?;
32        Some(Self { grant_id, n_pages })
33    }
34
35    /// Physical base address — program this into DMA descriptor registers.
36    #[inline]
37    pub fn phys(&self) -> usize {
38        self.grant_id
39    }
40
41    /// Virtual address — identical to `phys()` in SAS.
42    #[inline]
43    pub fn virt(&self) -> *mut u8 {
44        self.grant_id as *mut u8
45    }
46
47    /// Total size in bytes.
48    #[inline]
49    pub fn size(&self) -> usize {
50        self.n_pages * 4096
51    }
52
53    /// Authorise DMA for the PCIe device at `bdf` covering this buffer.
54    ///
55    /// Must be called before issuing any DMA operation.  The IOMMU maps
56    /// `[phys, phys + size)` for this BDF in the calling cell's domain.
57    /// Returns `Ok(iova)` on success (`iova == phys` in SAS).
58    pub fn authorize(&self, bdf: u32) -> Result<u64, SyscallError> {
59        sys_grant_dma(bdf, self.grant_id as u64, self.size())
60    }
61
62    /// Release the Grant region and return frames to the kernel.
63    pub fn free(self) {
64        sys_grant_free(self.grant_id);
65    }
66}
