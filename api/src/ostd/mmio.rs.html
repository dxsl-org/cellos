1//! Safe MMIO accessor for Driver Cells.
2//!
3//! Driver Cells use `#![forbid(unsafe_code)]` and cannot directly deref raw
4//! hardware addresses.  `MmioRegion` hides the `unsafe` inside this trusted
5//! library and enforces bounds checking on every access.
6//!
7//! A Cell cannot construct `MmioRegion` directly — the constructor is
8//! `pub(crate)`.  Cells obtain one by calling `request_region()`, which
9//! goes through the kernel Resource Registry syscall (Phase 02).
10
11use crate::syscall;
12use types::{ViError, ViResult};
13
14/// Request exclusive access to the MMIO range `[base, base+len)` from the kernel.
15///
16/// On success, returns an `MmioRegion` the Driver Cell can use for volatile
17/// register reads and writes.  The kernel:
18/// 1. Checks `[base, base+len)` against the per-arch allowlist.
19/// 2. Verifies no other Cell currently owns the range.
20/// 3. Checks the caller's ELF manifest declared the matching peripheral cap.
21///
22/// # Errors
23/// - `PermissionDenied` — range not in allowlist or manifest cap missing.
24/// - `AlreadyExists` — another Cell owns an overlapping range.
25/// - `InvalidInput` — arithmetic overflow in `base + len`.
26pub fn request_region(base: usize, len: usize) -> ViResult<MmioRegion> {
27    let ret = syscall::sys_request_mmio(base, len);
28    if ret == 0 {
29        // SAFETY: kernel approved the range; it is valid device MMIO.
30        Ok(MmioRegion::new(base, len))
31    } else {
32        Err(match ret {
33            1 => ViError::PermissionDenied,
34            2 => ViError::AlreadyExists,
35            _ => ViError::InvalidInput,
36        })
37    }
38}

40/// A bounds-checked, volatile read/write accessor for a device MMIO window.
41///
42/// Fields are private; Cell code (a separate Rust crate) cannot forge an
43/// `MmioRegion`.  Obtain one via [`request_region`].
44pub struct MmioRegion {
45    base: usize,
46    len: usize,
47}
48
49impl MmioRegion {
50    /// Construct an `MmioRegion` for `[base, base+len)`.
51    ///
52    /// `pub(crate)` — only ostd internals (i.e. `request_region`) call this
53    /// after the kernel has confirmed exclusive access for the calling Cell.
54    pub(crate) fn new(base: usize, len: usize) -> Self {
55        Self { base, len }
56    }
57
58    /// Read a `T`-sized value at `offset` bytes from the region base.
59    ///
60    /// Returns `Err(InvalidInput)` if `offset + size_of::<T>()` would exceed
61    /// `self.len` or if the arithmetic overflows.
62    pub fn read<T: Copy>(&self, offset: usize) -> ViResult<T> {
63        let size = core::mem::size_of::<T>();
64        let end = offset.checked_add(size).ok_or(ViError::InvalidInput)?;
65        if end > self.len {
66            return Err(ViError::InvalidInput);
67        }
68        let ptr = (self.base + offset) as *const T;
69        // SAFETY: bounds-check above ensures [ptr, ptr+size) ⊆ [base, base+len).
70        // The Region was granted by the kernel Resource Registry, guaranteeing
71        // the address range is valid device MMIO.  `read_volatile` prevents the
72        // compiler from caching or eliding the hardware read.
73        Ok(unsafe { core::ptr::read_volatile(ptr) })
74    }
75
76    /// Write `val` of type `T` at `offset` bytes from the region base.
77    ///
78    /// Returns `Err(InvalidInput)` on out-of-bounds.
79    pub fn write<T: Copy>(&self, offset: usize, val: T) -> ViResult<()> {
80        let size = core::mem::size_of::<T>();
81        let end = offset.checked_add(size).ok_or(ViError::InvalidInput)?;
82        if end > self.len {
83            return Err(ViError::InvalidInput);
84        }
85        let ptr = (self.base + offset) as *mut T;
86        // SAFETY: same contract as `read` above.
87        unsafe { core::ptr::write_volatile(ptr, val) };
88        Ok(())
89    }
90
91    /// Read a 32-bit device register at `offset` (most common register width).
92    #[inline]
93    pub fn read_u32(&self, offset: usize) -> ViResult<u32> {
94        self.read::<u32>(offset)
95    }
96
97    /// Write a 32-bit device register at `offset`.
98    #[inline]
99    pub fn write_u32(&self, offset: usize, val: u32) -> ViResult<()> {
100        self.write::<u32>(offset, val)
101    }
102
103    /// Read an 8-bit device register at `offset`.
104    #[inline]
105    pub fn read_u8(&self, offset: usize) -> ViResult<u8> {
106        self.read::<u8>(offset)
107    }
108
109    /// Write an 8-bit device register at `offset`.
110    #[inline]
111    pub fn write_u8(&self, offset: usize, val: u8) -> ViResult<()> {
112        self.write::<u8>(offset, val)
113    }
114
115    /// Base address of this region (physical/MMIO address).
116    #[inline]
117    pub fn base(&self) -> usize {
118        self.base
119    }
120
121    /// Length of this region in bytes.
122    #[inline]
123    pub fn len(&self) -> usize {
124        self.len
125    }
126
127    /// Returns `true` if the region has zero length.
128    #[inline]
129    pub fn is_empty(&self) -> bool {
130        self.len == 0
131    }
132}

134#[cfg(test)]
135mod tests {
136    use super::*;
137
138    /// Helper: back an MmioRegion with a stack buffer for testing.
139    fn make_region(buf: &mut [u8]) -> MmioRegion {
140        MmioRegion::new(buf.as_mut_ptr() as usize, buf.len())
141    }
142
143    #[test]
144    fn rejects_oob_read() {
145        let mut buf = [0u8; 4];
146        let r = make_region(&mut buf);
147        // offset=0, size=4 → end=4 == len=4 → OK
148        assert!(r.read_u32(0).is_ok());
149        // offset=1, size=4 → end=5 > len=4 → Err
150        assert_eq!(r.read_u32(1), Err(ViError::InvalidInput));
151        // offset=4, size=4 → end=8 > len=4 → Err
152        assert_eq!(r.read_u32(4), Err(ViError::InvalidInput));
153    }
154
155    #[test]
156    fn rejects_oob_write() {
157        let mut buf = [0u8; 4];
158        let r = make_region(&mut buf);
159        assert!(r.write_u32(0, 0x1234_5678).is_ok());
160        assert_eq!(r.write_u32(1, 0), Err(ViError::InvalidInput));
161    }
162
163    #[test]
164    fn round_trip_u32() {
165        let mut buf = [0u8; 16];
166        let r = make_region(&mut buf);
167        r.write_u32(0, 0xDEAD_BEEF).unwrap();
168        assert_eq!(r.read_u32(0).unwrap(), 0xDEAD_BEEF);
169        r.write_u32(4, 0x1234_5678).unwrap();
170        assert_eq!(r.read_u32(4).unwrap(), 0x1234_5678);
171        // independent offsets do not clobber each other
172        assert_eq!(r.read_u32(0).unwrap(), 0xDEAD_BEEF);
173    }
174
175    #[test]
176    fn round_trip_u8() {
177        let mut buf = [0u8; 4];
178        let r = make_region(&mut buf);
179        r.write_u8(0, 0xAB).unwrap();
180        r.write_u8(3, 0xCD).unwrap();
181        assert_eq!(r.read_u8(0).unwrap(), 0xAB);
182        assert_eq!(r.read_u8(3).unwrap(), 0xCD);
183    }
184
185    #[test]
186    fn overflow_offset_rejected() {
187        let mut buf = [0u8; 4];
188        let r = make_region(&mut buf);
189        // offset near usize::MAX + size_of::<u32>()=4 must not overflow
190        assert_eq!(r.read_u32(usize::MAX - 2), Err(ViError::InvalidInput));
191        assert_eq!(r.write_u32(usize::MAX, 0), Err(ViError::InvalidInput));
192    }
193}
