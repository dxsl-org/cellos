1//! App-side display helpers: `ViSurface` for Grant-backed compositor surfaces.
2//!
3//! ## Usage
4//! ```
5//! let comp_tid = wait_for_compositor();
6//! let mut surf = ViSurface::create(comp_tid, 640, 480, PixelFormat::Bgra8888)?;
7//! let px = surf.pixels_mut();
8//! // draw into px ...
9//! surf.damage_all();
10//! ```

25/// Block until the compositor service is registered and return its TID.
26pub fn wait_for_compositor() -> usize {
27    loop {
28        if let Some(tid) = sys_lookup_service(service::COMPOSITOR) {
29            return tid;
30        }
31        crate::task::yield_now();
32    }
33}

37/// A compositor surface backed by a Grant buffer the app cell owns directly.
38///
39/// The app writes pixels into `pixels_mut()` and calls `damage()` / `damage_all()`
40/// to tell the compositor which regions need to be re-blended.  No pixel data
41/// crosses an IPC boundary — only a 24-byte `DamageNotify` is sent per dirty region.
42///
43/// ## Lifecycle
44/// `ViSurface::create` → write pixels → `damage` → … → `drop` (auto-cleans up).
45///
46/// `ViSurface` is `!Send`: the Grant pointer must stay on the cell's task.
47pub struct ViSurface {
48    comp_tid: usize,
49    cap: u32,
50    reg_id: usize,
51    ptr: *mut u8,
52    width: u32,
53    height: u32,
54    fmt: PixelFormat,
55    /// Makes `ViSurface` !Send on stable Rust — the raw pointer must stay on its origin task.
56    _not_send: core::marker::PhantomData<*mut ()>,
57}
58
59impl ViSurface {
60    /// Create a new surface of `(width × height)` pixels, attaching it to the
61    /// running compositor at `comp_tid`.
62    ///
63    /// Allocates a persistent Grant buffer, shares it read-only with the compositor,
64    /// and sends `CREATE_SURFACE` + `ATTACH_GRANT` IPC.
65    ///
66    /// # Errors
67    /// - `OutOfMemory` if `sys_grant_register` fails.
68    /// - `IO` if the compositor rejects `ATTACH_GRANT` (e.g. too many surfaces).
69    pub fn create(comp_tid: usize, width: u32, height: u32, fmt: PixelFormat) -> ViResult<Self> {
70        let size = (width * height * fmt.bpp()) as usize;
71
72        // 1. Allocate a persistent physical Grant buffer (lives until we call unregister).
73        let reg_id = sys_grant_register(size).ok_or(ViError::OutOfMemory)?;
74
75        // 2. Share read-only with compositor so it can read our pixels.
76        sys_grant_share(reg_id, comp_tid, 0 /* ReadOnly */);
77
78        // 3. Get our own write pointer into the Grant.
79        let ptr = sys_grant_slice(reg_id).ok_or_else(|| {
80            sys_grant_unregister(reg_id);
81            ViError::IO
82        })?;
83
84        // 4. Ask compositor to create a surface slot → get cap.
85        let cap = ipc_create_surface(comp_tid, width, height).inspect_err(|_e| {
86            sys_grant_unregister(reg_id);
87        })?;
88
89        // 5. Tell compositor to attach our Grant to that slot.
90        ipc_attach_grant(comp_tid, cap, reg_id, width, height, fmt).inspect_err(|_e| {
91            let _ = ipc_destroy_surface(comp_tid, cap);
92            sys_grant_unregister(reg_id);
93        })?;
94
95        Ok(Self {
96            comp_tid,
97            cap,
98            reg_id,
99            ptr,
100            width,
101            height,
102            fmt,
103            _not_send: core::marker::PhantomData,
104        })
105    }
106
107    /// Direct mutable access to the pixel buffer.
108    ///
109    /// The app writes directly here; the compositor reads it via a read-only Grant.
110    /// After writing, call `damage()` or `damage_all()` to trigger a repaint.
111    pub fn pixels_mut(&mut self) -> &mut [u8] {
112        let len = (self.width * self.height * self.fmt.bpp()) as usize;
113        // SAFETY: ptr is our own registered Grant buffer (sys_grant_register).
114        // We hold &mut self so no other code can call pixels_mut concurrently.
115        // The compositor holds ReadOnly access — the kernel blocks any compositor write.
116        unsafe { core::slice::from_raw_parts_mut(self.ptr, len) }
117    }
118
119    /// Stride in bytes (width × bytes-per-pixel).
120    pub fn stride(&self) -> usize {
121        self.width as usize * self.fmt.bpp() as usize
122    }
123
124    /// Surface width in pixels.
125    pub fn width(&self) -> u32 {
126        self.width
127    }
128
129    /// Surface height in pixels.
130    pub fn height(&self) -> u32 {
131        self.height
132    }
133
134    /// Signal a dirty region to the compositor (fire-and-forget, 24-byte IPC).
135    ///
136    /// The compositor will re-blend this rect on the next render tick.
137    pub fn damage(&self, rect: Rect) {
138        let msg = DamageNotify {
139            opcode: compositor_ops::DAMAGE_NOTIFY,
140            _pad: [0; 3],
141            cap: self.cap,
142            rect,
143        };
144        sys_send(self.comp_tid, &msg.encode());
145    }
146
147    /// Signal the entire surface as dirty.
148    pub fn damage_all(&self) {
149        self.damage(Rect {
150            x: 0,
151            y: 0,
152            w: self.width,
153            h: self.height,
154        });
155    }
156
157    /// Move this surface to a new screen position.
158    pub fn move_to(&self, x: i32, y: i32) {
159        let mut buf = [0u8; 13];
160        buf[0] = compositor_ops::MOVE_SURFACE;
161        buf[1..9].copy_from_slice(&(self.cap as u64).to_le_bytes());
162        buf[9..13].copy_from_slice(&x.to_le_bytes());
163        // y needs 4 more bytes — extend to 17
164        let mut buf17 = [0u8; 17];
165        buf17[..13].copy_from_slice(&buf);
166        buf17[13..17].copy_from_slice(&y.to_le_bytes());
167        sys_send(self.comp_tid, &buf17);
168    }
169
170    /// Raise this surface to the top of the z-order.
171    pub fn raise(&self) {
172        let mut buf = [0u8; 9];
173        buf[0] = compositor_ops::RAISE_SURFACE;
174        buf[1..9].copy_from_slice(&(self.cap as u64).to_le_bytes());
175        sys_send(self.comp_tid, &buf);
176    }
177
178    /// Explicitly destroy the surface (also called by `Drop`).
179    pub fn destroy(self) {
180        drop(self);
181    }
182}
183
184impl Drop for ViSurface {
185    fn drop(&mut self) {
186        // 1. Detach grant — compositor stops reading from our buffer.
187        let mut detach = [0u8; 9];
188        detach[0] = compositor_ops::DETACH_GRANT;
189        detach[1..9].copy_from_slice(&(self.cap as u64).to_le_bytes());
190        sys_send(self.comp_tid, &detach);
191        // Drain reply (compositor sends [0x01] on success).
192        let mut resp = [0u8; 8];
193        let _ = sys_recv(0, &mut resp);
194
195        // 2. Destroy surface slot in compositor.
196        let _ = ipc_destroy_surface(self.comp_tid, self.cap);
197
198        // 3. Release physical Grant pages.
199        sys_grant_unregister(self.reg_id);
200    }
201}

203// ─── Private IPC helpers ──────────────────────────────────────────────────────

205/// Send `CREATE_SURFACE` and return the cap (u32).
206fn ipc_create_surface(comp_tid: usize, w: u32, h: u32) -> ViResult<u32> {
207    let mut req = [0u8; 9];
208    req[0] = compositor_ops::CREATE_SURFACE;
209    req[1..5].copy_from_slice(&w.to_le_bytes());
210    req[5..9].copy_from_slice(&h.to_le_bytes());
211    sys_send(comp_tid, &req);
212
213    let mut resp = [0u8; 8];
214    match sys_recv(0, &mut resp) {
215        SyscallResult::Ok(_) => {
216            let cap = u32::from_le_bytes([resp[0], resp[1], resp[2], resp[3]]);
217            if cap == 0 {
218                Err(ViError::IO)
219            } else {
220                Ok(cap)
221            }
222        }
223        _ => Err(ViError::IO),
224    }
225}

227/// Send `ATTACH_GRANT` and verify the compositor accepted it.
228fn ipc_attach_grant(
229    comp_tid: usize,
230    cap: u32,
231    reg_id: usize,
232    w: u32,
233    h: u32,
234    fmt: PixelFormat,
235) -> ViResult<()> {
236    let ag = AttachGrant {
237        opcode: compositor_ops::ATTACH_GRANT,
238        fmt: fmt as u8,
239        _pad: [0; 2],
240        cap,
241        reg_id: reg_id as u64,
242        width: w,
243        height: h,
244    };
245    sys_send(comp_tid, &ag.encode());
246
247    let mut resp = [0u8; 8];
248    match sys_recv(0, &mut resp) {
249        SyscallResult::Ok(_) if resp[0] == 0x01 => Ok(()),
250        _ => Err(ViError::IO),
251    }
252}
253
254/// Send `DESTROY_SURFACE` (best-effort; errors silently ignored on Drop path).
255fn ipc_destroy_surface(comp_tid: usize, cap: u32) -> ViResult<()> {
256    let mut req = [0u8; 9];
257    req[0] = compositor_ops::DESTROY_SURFACE;
258    req[1..9].copy_from_slice(&(cap as u64).to_le_bytes());
259    sys_send(comp_tid, &req);
260
261    let mut resp = [0u8; 8];
262    match sys_recv(0, &mut resp) {
263        SyscallResult::Ok(_) => Ok(()),
264        _ => Err(ViError::IO),
265    }
