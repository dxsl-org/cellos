1//! Typed request/reply IPC helpers — the compliant path for talking to a
2//! service cell (Spec 17 — Cell IPC Wire Contract).
3//!
4//! Prefer [`service_call`] / [`service_call_typed`] over a hand-rolled
5//! `sys_send` + `sys_recv(0)`: they recv **masked to the service tid** (Spec 17
6//! §2), so a queued input key event can never be mistaken for the reply, and
7//! they surface every failure as a typed [`IpcError`] instead of a silent empty
8//! result (Spec 17 §7).
9
10#![allow(unsafe_code)]
11
12use crate::syscall::{sys_recv, sys_send, sys_try_recv, SyscallResult};
13use core::future::Future;
14use core::pin::Pin;
15use core::task::{Context, Poll};
16use serde::{Deserialize, Serialize};

18/// Why a [`service_call`] did not complete. Never silently swallowed.
19#[derive(Debug, Clone, Copy, PartialEq, Eq)]
20pub enum IpcError {
21    /// The request did not fit `send_buf` (postcard encode failed).
22    Encode,
23    /// `sys_send` to the service failed (service gone / bad tid).
24    Send,
25    /// `sys_recv` returned an error.
26    Recv,
27    /// A message arrived, but from a cell other than the service — never
28    /// treated as the reply (Spec 17 §7 silent-wrong-sender guard).
29    WrongSender,
30    /// The reply bytes did not decode into the expected type.
31    Decode,
32}
33
34/// One request/reply exchange with `service_tid`, recv **masked** to it.
35///
36/// `send_buf` encodes the request; `recv_buf` receives the reply and backs the
37/// returned slice (caller-owned so the borrow outlives the call). The reply is
38/// accepted only if it came from `service_tid` — a message from any other
39/// sender (e.g. a queued input event, Spec 17 §2) is an [`IpcError::WrongSender`],
40/// not a decode of the wrong bytes.
41pub fn service_call<'r, Req: Serialize>(
42    service_tid: usize,
43    req: &Req,
44    send_buf: &mut [u8],
45    recv_buf: &'r mut [u8],
46) -> Result<&'r [u8], IpcError> {
47    let encoded = api::ipc::encode(req, send_buf).map_err(|_| IpcError::Encode)?;
48    if let SyscallResult::Err(_) = sys_send(service_tid, encoded) {
49        return Err(IpcError::Send);
50    }
51    // MASKED recv — Spec 17 §2. Only the service's reply, never a keystroke.
52    match sys_recv(service_tid, recv_buf) {
53        SyscallResult::Ok(sender) if sender == service_tid => {
54            let len = recv_buf.len();
55            Ok(&recv_buf[..len])
56        }
57        SyscallResult::Ok(_) => Err(IpcError::WrongSender),
58        SyscallResult::Err(_) => Err(IpcError::Recv),
59    }
60}
61
62/// [`service_call`] that decodes the reply into `Resp`.
63///
64/// `Resp` borrows `recv_buf` for types with `&str`/`&[u8]` fields — consume it
65/// before reusing the buffer.
66pub fn service_call_typed<'r, Req, Resp>(
67    service_tid: usize,
68    req: &Req,
69    send_buf: &mut [u8],
70    recv_buf: &'r mut [u8],
71) -> Result<Resp, IpcError>
72where
73    Req: Serialize,
74    Resp: Deserialize<'r>,
75{
76    let raw = service_call(service_tid, req, send_buf, recv_buf)?;
77    api::ipc::decode::<Resp>(raw).map_err(|_| IpcError::Decode)
78}

82/// Future that waits for a message to arrive. Returns the sender id.
83pub struct AsyncRecv<'a> {
84    pub mask: usize,
85    pub buf: &'a mut [u8],
86}
87
88impl<'a> Future for AsyncRecv<'a> {
89    type Output = SyscallResult;
90
91    fn poll(mut self: Pin<&mut Self>, _cx: &mut Context<'_>) -> Poll<Self::Output> {
92        match sys_try_recv(self.mask, self.buf) {
93            // No message yet — the naive executor yields and polls again.
94            SyscallResult::Ok(0) => Poll::Pending,
95            SyscallResult::Ok(id) => Poll::Ready(SyscallResult::Ok(id)),
96            err => Poll::Ready(err),
97        }
98    }
99}
100
101/// Await a message on `mask` (0 = wildcard). Prefer a service tid for
102/// request/reply — see Spec 17 §2.
103pub fn recv_async(mask: usize, buf: &mut [u8]) -> AsyncRecv<'_> {
104    AsyncRecv { mask, buf }
