1//! Cell heap allocator — a real freeing allocator (linked-list free list).
2//!
3//! Replaces the former bump allocator whose `dealloc` was a no-op: that leaked
4//! EVERY allocation, so any long-running cell (shell, services) inexorably
5//! exhausted its 4 MiB arena and then store-faulted on the first failed
6//! allocation — a guaranteed slow death, the opposite of "never-die". This wraps
7//! `linked_list_allocator` (the same crate the kernel heap uses) so transient
8//! `String`/`Vec` allocations are actually reclaimed and a cell can run forever.
9//!
10//! ## Why `static mut Heap` and not `LockedHeap`
11//! Every ViCell cell is single-task / single-hart, so allocator entries never
12//! overlap and no lock is needed. Crucially, `LockedHeap` embeds a spinlock whose
13//! atomic write-back FAULTS in a cell: the const-initialised allocator static lands
14//! in the ELF's RELRO (read-only) segment, and an `amoor.w` to a read-only page
15//! traps (scause=0xf). A zero-sized `CellAllocator` plus mutable state in `.bss`
16//! (`static mut`) sidesteps that entirely — there is no writable allocator static
17//! to be placed read-only.
18
19use core::alloc::{GlobalAlloc, Layout};
20use core::ptr::{addr_of_mut, null_mut, NonNull};
21use linked_list_allocator::Heap;
22
23/// Per-cell heap size. 8 MiB gives the compositor room for its two full-screen
24/// framebuffer Vecs (pixels + staging, ~3 MiB each at 1024×768) while leaving
25/// headroom for all other cells' dynamic allocations.
26const HEAP_SIZE: usize = 8 * 1024 * 1024;
27
28/// Backing arena for the cell heap. 16-byte aligned so the allocator can satisfy
29/// the largest natural alignment without wasting the first bytes.
30#[repr(align(16))]
31#[allow(dead_code)] // reason: backing bytes are accessed via addr_of_mut!(HEAP_MEM) raw-pointer cast, not field syntax
32struct HeapArena([u8; HEAP_SIZE]);
33static mut HEAP_MEM: HeapArena = HeapArena([0; HEAP_SIZE]);
34
35/// Free-list allocator state. `static mut` (writable `.bss`), serialized solely by
36/// the single-task execution model — see the module note on why no lock is used.
37static mut HEAP: Heap = Heap::empty();
38static mut HEAP_INIT: bool = false;
39
40/// Zero-sized handle; all real state lives in the `static mut`s above so this
41/// `#[global_allocator]` static carries no writable data that could land read-only.
42struct CellAllocator;
43
44// SAFETY: single-task cells never enter the allocator concurrently, so the
45// unsynchronised access to the `static mut` heap state is race-free. Allocation
46// correctness is delegated to the well-tested `linked_list_allocator::Heap`.
47unsafe impl GlobalAlloc for CellAllocator {
48    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
49        let heap = &mut *addr_of_mut!(HEAP);
50        if !HEAP_INIT {
51            // SAFETY: HEAP_MEM is a 'static, 16-byte-aligned arena owned solely by the
52            // heap; init runs exactly once (single-task → no race on HEAP_INIT).
53            heap.init(addr_of_mut!(HEAP_MEM) as *mut u8, HEAP_SIZE);
54            HEAP_INIT = true;
55        }
56        heap.allocate_first_fit(layout)
57            .map_or(null_mut(), |p| p.as_ptr())
58    }
59
60    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
61        if let Some(p) = NonNull::new(ptr) {
62            let heap = &mut *addr_of_mut!(HEAP);
63            heap.deallocate(p, layout);
64        }
65    }
66}
67
68#[global_allocator]
69static ALLOCATOR: CellAllocator = CellAllocator;
70
71/// Allocation-failure handler. We are genuinely out of heap (4 MiB of LIVE
72/// objects), so in-process recovery is hopeless. Rather than hang — which would
73/// leave a paralyzed-but-alive cell the supervisor cannot detect — log via the
74/// raw syscall (no allocation) and exit abnormally so the supervisor restarts the
75/// cell with a fresh heap. That is the "never-die" response to OOM.
76#[alloc_error_handler]
77fn alloc_error(_layout: Layout) -> ! {
78    let _ = crate::syscall::sys_log("OOM: cell heap exhausted — exiting for restart\n");
79    crate::syscall::sys_exit(0xEE); // non-zero = abnormal → supervisor restarts it
80}
