1//! Typed linear handles for kernel Grant regions.
2//!
3//! A [`GrantHandle<T>`] wraps a raw grant ID (physical base address in SAS).
4//! It is `!Copy + !Clone` — Rust affine ownership enforces that each grant
5//! has exactly one live owner at a time. Dropping the handle automatically
6//! calls `sys_grant_free`, returning the frames to the kernel.
7//!
8//! This implements the Singularity "exchange heap" lesson: ownership transfer
9//! of shared memory is explicit and compile-time verifiable, not a runtime
10//! convention that can be accidentally violated.
11//!
12//! # Typical flow
13//!
14//! ```rust,no_run
15//! // Allocate a 4-KiB region typed as raw bytes.
16//! let mut handle = GrantHandle::<u8>::alloc(4096).expect("OOM");
17//! // ... fill with data ...
18//! let (id, len) = handle.into_raw();          // hand off to another Cell via IPC
19//! // The grant is NOT freed here — receiver calls GrantHandle::from_raw(id, len).
20//! ```
21
22use core::marker::PhantomData;
23
24use crate::syscall::{sys_grant_alloc, sys_grant_free, sys_grant_slice};
25
26/// A typed, linear handle to a kernel-managed grant region.
27///
28/// `T` is a logical element type — the kernel manages raw byte pages and has
29/// no knowledge of `T`. The region holds `len / size_of::<T>()` elements.
30///
31/// # Linear type invariant
32/// `!Copy + !Clone`. At most one `GrantHandle<T>` exists per `grant_id` in
33/// this cell's address space. This is enforced by Rust's affine type system
34/// (move semantics), not by hardware.
35///
36/// # Drop
37/// `Drop` calls [`sys_grant_free`], releasing the frames back to the kernel.
38/// Use [`into_raw`](GrantHandle::into_raw) to transfer ownership without freeing.
39pub struct GrantHandle<T> {
40    id: usize,
41    len: usize,
42    _type: PhantomData<*mut T>,
43}
44
45// SAFETY: In SAS all cells share the address space. GrantHandle is Send because
46// only one cell can own the handle at a time (move semantics prevent aliasing).
47unsafe impl<T: Send> Send for GrantHandle<T> {}
48
49impl<T> GrantHandle<T> {
50    /// Allocate a new grant region holding `count` elements of type `T`.
51    ///
52    /// The kernel allocates page-aligned contiguous frames; the actual region
53    /// may be slightly larger than `count * size_of::<T>()` due to alignment.
54    ///
55    /// Returns `None` on OOM or if `count * size_of::<T>()` overflows.
56    pub fn alloc(count: usize) -> Option<Self> {
57        let byte_len = count.checked_mul(core::mem::size_of::<T>())?;
58        let byte_len = byte_len.max(1); // zero-size grants are not useful
59        let id = sys_grant_alloc(byte_len)?;
60        Some(Self {
61            id,
62            len: byte_len,
63            _type: PhantomData,
64        })
65    }
66
67    /// Wrap a raw grant ID received from another Cell via IPC.
68    ///
69    /// # Safety
70    /// `id` must be a valid grant owned by this cell (e.g. via `sys_grant_share`
71    /// and the receiver's confirmation). `len` must equal the byte length returned
72    /// by the original `sys_grant_alloc`. Calling this twice with the same `id`
73    /// creates two owners and will double-free on drop — undefined behaviour.
74    pub unsafe fn from_raw(id: usize, len: usize) -> Self {
75        Self {
76            id,
77            len,
78            _type: PhantomData,
79        }
80    }
81
82    /// Consume the handle, returning `(grant_id, byte_len)` **without** freeing.
83    ///
84    /// Use when passing grant ownership to another Cell via IPC. The receiver
85    /// must call [`GrantHandle::from_raw`] to re-wrap it.
86    #[inline]
87    pub fn into_raw(self) -> (usize, usize) {
88        let id = self.id;
89        let len = self.len;
90        // Prevent Drop from calling sys_grant_free — ownership is transferred.
91        core::mem::forget(self);
92        (id, len)
93    }
94
95    /// Raw kernel grant ID (physical base address in SAS == virtual address).
96    #[inline]
97    pub fn id(&self) -> usize {
98        self.id
99    }
100
101    /// Byte length of the grant region.
102    #[inline]
103    pub fn len(&self) -> usize {
104        self.len
105    }
106
107    /// True when the grant region has zero length (never the case for a
108    /// successful `sys_grant_alloc`, but clippy requires `is_empty` beside `len`).
109    #[inline]
110    pub fn is_empty(&self) -> bool {
111        self.len == 0
112    }
113
114    /// Get an exclusive byte slice over the entire grant region.
115    ///
116    /// # Safety
117    /// No other live reference to this grant must exist. The handle must still
118    /// be valid (not freed). The caller must ensure the data at `[0, len)` is
119    /// initialised before reading.
120    pub unsafe fn as_bytes_mut(&mut self) -> &mut [u8] {
121        let ptr = sys_grant_slice(self.id)
122            .expect("GrantHandle::as_bytes_mut: grant not found or permission denied");
123        core::slice::from_raw_parts_mut(ptr, self.len)
124    }
125}
126
127impl<T: Copy> GrantHandle<T> {
128    /// Get an exclusive typed slice over the grant region.
129    ///
130    /// # Safety
131    /// No other live reference must exist. All bytes must be initialised as
132    /// valid `T` values (bitwise validity).
133    pub unsafe fn as_slice_mut(&mut self) -> &mut [T] {
134        let ptr = sys_grant_slice(self.id)
135            .expect("GrantHandle::as_slice_mut: grant not found or permission denied");
136        let count = self.len / core::mem::size_of::<T>();
137        core::slice::from_raw_parts_mut(ptr as *mut T, count)
138    }
139}
140
141impl<T> Drop for GrantHandle<T> {
142    fn drop(&mut self) {
143        sys_grant_free(self.id);
144    }
145}
