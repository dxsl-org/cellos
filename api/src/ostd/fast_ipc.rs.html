1//! Fast-IPC handler table for direct Cell-to-service calls.
2//!
3//! Bypasses the ecall trap for high-frequency VFS operations (~3 cycles vs
4//! ~100 cycles for a full syscall round-trip in a Single Address Space OS).
5//!
6//! ## Safety invariant
7//! The handler function pointer is written once at VFS startup (before any
8//! Cell requests it) and never changed thereafter.  On single-hart QEMU,
9//! there is no concurrent modification risk.  `Ordering::Release` on write
10//! and `Ordering::Acquire` on read ensure the handler body is visible to
11//! the caller after the pointer is stored.
12//!
13//! ## Preemption
14//! `call_vfs` disables S-mode interrupts for the duration of the handler call.
15//! The VFS FAT16 driver holds a spinlock; if the timer ISR preempted the
16//! handler mid-spinlock and switched to another Cell that also called VFS,
17//! the spinlock would deadlock.  Interrupt-disable makes the fast-path call
18//! behave as an atomic critical section w.r.t. the scheduler.
19
20use api::fast_ipc::{TrustedHandle, VfsCell};
21use api::ipc::{VfsRequest, IPC_BUF_SIZE};
22use core::sync::atomic::{AtomicPtr, AtomicUsize, Ordering};
23
24/// Signature of a registered VFS fast-IPC handler.
25///
26/// Reads the request and writes the response into `out`.
27/// Returns the number of bytes written into `out`.
28pub type VfsFastHandler = unsafe fn(req: &VfsRequest<'_>, out: &mut [u8; IPC_BUF_SIZE]) -> usize;
29
30static VFS_HANDLER_PTR: AtomicPtr<()> = AtomicPtr::new(core::ptr::null_mut());
31/// Raw CellId of the cell that registered the VFS handler; 0 = unregistered.
32/// Used to clear the handler pointer when the VFS cell crashes.
33static VFS_HANDLER_CELL: AtomicUsize = AtomicUsize::new(0);
34
35/// Register the VFS fast-IPC handler.
36///
37/// Called once by the VFS service at startup.  The pointer is published with
38/// `Release` ordering so subsequent `Acquire` loads in `call_vfs` observe the
39/// complete handler function.
40pub fn register_vfs(handler: VfsFastHandler) {
41    VFS_HANDLER_PTR.store(
42        // fn pointer → *mut () for atomic storage; recovered with the same
43        // type in call_vfs.
44        handler as *mut (),
45        Ordering::Release,
46    );
47}
48
49/// Record which cell (by CellId raw value) owns the registered VFS handler.
50///
51/// Call this immediately after `register_vfs` so the kernel can clear the
52/// handler pointer if the owning cell crashes.
53pub fn set_vfs_handler_cell(cell_id_raw: usize) {
54    VFS_HANDLER_CELL.store(cell_id_raw, Ordering::Relaxed);
55}
56
57/// Clear the VFS handler pointer if `cell_id_raw` is the registered owner.
58///
59/// Called by the kernel fault-isolation path when a cell is terminated, so
60/// that a future `call_vfs` does not jump into stale/replaced code.
61pub fn clear_vfs_if_cell(cell_id_raw: usize) {
62    if VFS_HANDLER_CELL.load(Ordering::Relaxed) == cell_id_raw && cell_id_raw != 0 {
63        VFS_HANDLER_PTR.store(core::ptr::null_mut(), Ordering::Release);
64        VFS_HANDLER_CELL.store(0, Ordering::Relaxed);
65    }
66}
67
68/// `extern "Rust"` shim so the kernel (which does not depend on ostd) can
69/// call `set_vfs_handler_cell` via link-time symbol resolution.
70#[no_mangle]
71pub extern "Rust" fn vi_set_fast_ipc_vfs_cell(cell_id: usize) {
72    set_vfs_handler_cell(cell_id);
73}
74
75/// `extern "Rust"` shim so the kernel can call `clear_vfs_if_cell`.
76#[no_mangle]
77pub extern "Rust" fn vi_clear_fast_ipc_vfs_cell(cell_id: usize) {
78    clear_vfs_if_cell(cell_id);
79}

81/// RAII guard: disables S-mode interrupt enable (SIE) on construction, restores on drop.
82///
83/// Ensures SIE is always restored even if the fast-IPC handler panics (drop glue
84/// runs before the panic handler, so the restore is guaranteed).
85struct SieGuard(bool);
86
87impl SieGuard {
88    /// Disable SIE and return a guard.  Noop on non-riscv64.
89    ///
90    /// # Safety
91    /// Must be called from S-mode.
92    #[inline]
93    unsafe fn disable() -> Self {
94        #[cfg(target_arch = "riscv64")]
95        {
96            let v: usize;
97            // SAFETY: csrrci reads-and-clears sstatus.SIE (bit 1) atomically.
98            core::arch::asm!("csrrci {}, sstatus, 0x2", out(reg) v);
99            Self(v & 0x2 != 0)
100        }
101        #[cfg(not(target_arch = "riscv64"))]
102        Self(false)
103    }
104}
105
106impl Drop for SieGuard {
107    fn drop(&mut self) {
108        if self.0 {
109            // SAFETY: restoring SIE to the state saved in disable(); S-mode only.
110            #[cfg(target_arch = "riscv64")]
111            unsafe {
112                core::arch::asm!("csrsi sstatus, 0x2");
113            }
114        }
115    }
116}

118/// Call the registered VFS handler directly, bypassing the ecall trap.
119///
120/// Returns the number of bytes written into `out`, or 0 if no handler is
121/// registered (caller should fall back to the `sys_send` / `sys_recv` path).
122///
123/// # Note (PIE limitation)
124/// For non-PIE cells (current default), each cell ELF has its own copy of
125/// `VFS_HANDLER_PTR` (statically linked from `libs/ostd`).  VFS writes to its
126/// copy; client cells read from theirs (null) → always fallback to ecall.
127/// The fast path activates once cells are compiled as PIE and the kernel loader
128/// patches JUMP_SLOT relocations to `kernel::fast_ipc::call_vfs` (G2 work).
129///
130/// # Safety
131/// The caller must own `out` exclusively for the duration of this call.
132/// `_handle: TrustedHandle<VfsCell>` documents that the caller has been
133/// granted fast-path access; it does not enforce this at runtime.
134pub unsafe fn call_vfs(
135    _handle: TrustedHandle<VfsCell>,
136    req: &VfsRequest<'_>,
137    out: &mut [u8; IPC_BUF_SIZE],
138) -> usize {
139    let ptr = VFS_HANDLER_PTR.load(Ordering::Acquire);
140    if ptr.is_null() {
141        return 0; // VFS not yet registered — caller falls back to ecall path
142    }
143    // SAFETY: ptr was stored by register_vfs from a valid VfsFastHandler.
144    let handler: VfsFastHandler = core::mem::transmute(ptr);
145
146    // Disable S-mode interrupts for the handler's duration. VFS's FAT32 driver
147    // holds a spinlock; timer-preemption mid-handler to another VFS caller would
148    // deadlock on it. SieGuard restores on drop — safe even on handler panic.
149    // SAFETY: called from S-mode; SieGuard::disable is safe here.
150    let _sie = SieGuard::disable();
151
152    handler(req, out)
153}
