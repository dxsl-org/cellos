3//! `ostd::http` — feature-gated HTTP/1.1 client surface (`feature = "http"`).
4//!
5//! Re-exports the pure protocol logic from [`http_core`] (host-testable) and
6//! adds the ostd-only transport glue: [`HttpClient`] (generic over any
7//! `embedded_io::Read + Write`) and [`TlsStream`] (HTTPS adapter over the net
8//! cell's raw TLS IPC).
9//!
10//! ```no_run
11//! use ostd::http::{HttpClient, TlsStream};
12//! use ostd::json;
13//!
14//! let tls = TlsStream::connect([10, 0, 2, 2], 8443, "mock")?;
15//! let mut client = HttpClient::new(tls);
16//! let (headers, mut body) = client.post(
17//!     "mock", "/v1/chat/completions", "application/json", br#"{"q":"hi"}"#,
18//! )?;
19//! // drain `body` via body.read(&mut transport, &mut out) ...
20//! # Ok::<(), http_core::HttpError>(())
21//! ```
22
23#[path = "http/client.rs"]
24mod client;
25
26pub use client::HttpClient;
27
28/// HTTPS transport adapter (TLS over the net cell). Re-exported here so callers
29/// only need `use ostd::http::{HttpClient, TlsStream}`.
30pub use crate::clients::TlsStream;
31
32// ── Re-exports from the pure, host-tested protocol crate ───────────────────────
33pub use http_core::request::RequestBuilder;
34pub use http_core::response::parse_response_headers;
35pub use http_core::{BodyReader, Framing, HttpError, ParsedHeaders};
