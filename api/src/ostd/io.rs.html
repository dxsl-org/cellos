12/// Opaque I/O error wrapping a [`ViError`] for [`embedded_io`] trait impls.
13///
14/// `ViError` lives in `libs/types`; implementing a foreign trait on a foreign type
15/// violates the orphan rule. This newtype lives in ostd and bridges the two.
16#[derive(Debug)]
17pub struct OstdError(pub ViError);
18
19impl core::fmt::Display for OstdError {
20    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
21        write!(f, "{:?}", self.0)
22    }
23}
24
25impl core::error::Error for OstdError {}

27impl embedded_io::Error for OstdError {
28    fn kind(&self) -> embedded_io::ErrorKind {
29        match self.0 {
30            ViError::NotFound => embedded_io::ErrorKind::NotFound,
31            ViError::PermissionDenied => embedded_io::ErrorKind::PermissionDenied,
32            ViError::OutOfMemory => embedded_io::ErrorKind::OutOfMemory,
33            ViError::WouldBlock => embedded_io::ErrorKind::Other,
34            _ => embedded_io::ErrorKind::Other,
35        }
36    }
37}

50pub fn print_usize(n: usize) {
51    let mut buf = [0u8; 20];
52    let mut i = 0;
53    let mut num = n;
54
55    if num == 0 {
56        print("0");
57        return;
58    }
59
60    while num > 0 {
61        buf[i] = (num % 10) as u8 + b'0';
62        num /= 10;
63        i += 1;
64    }
65
66    // Reverse
67    let mut start = 0;
68    let mut end = i - 1;
69    while start < end {
70        buf.swap(start, end);
71        start += 1;
72        end -= 1;
73    }
74
75    if let Ok(s) = core::str::from_utf8(&buf[..i]) {
76        print(s);
77    }
78}

80pub struct Stdin;

82impl Stdin {
83    /// Read one line from stdin, echoing characters as they arrive.
84    ///
85    /// When the input service is online (registered via `sys_lookup_service`),
86    /// reads via the input-service IPC path so that the caller participates in
87    /// the focus model. Callers must have called `ostd::input::request_focus()`
88    /// at least once before this returns meaningful data; without focus the input
89    /// service will not dispatch key events to this cell.
90    ///
91    /// Falls back to the kernel `sys_read(fd=0)` path when the input service is
92    /// not yet registered (early boot, or input cell not present).
93    pub fn read_line(&self, buf: &mut String) -> ViResult<usize> {
94        if sys_lookup_service(service::INPUT).is_some() {
95            // Input-service path: receive InputEvent frames dispatched to this cell.
96            // Uses a per-iteration timeout only to detect input-service restarts —
97            // we re-request focus only when the TID changes, NOT on every timeout.
98            // Re-requesting focus on every timeout created a ~10ms dead zone where
99            // shell was in sys_send(SetFocus) and sys_try_send drops from dispatcher
100            // would cause burst keystrokes to be silently lost (G18 fix regression).
101            const TIMEOUT_TICKS: u64 = 20; // ~200ms between TID-change checks
102            let mut bytes_read: usize = 0;
103            let mut focused_on_tid: usize = 0; // 0 = no focus yet
104                                               // Re-check the input service TID on each iteration — it may have
105                                               // restarted. If it disappears entirely, fall through to sys_read.
106            while let Some(input_tid) = sys_lookup_service(service::INPUT) {
107                // Request focus only on startup or when input service restarted (TID changed).
108                if focused_on_tid != input_tid {
109                    if !crate::input::request_focus() {
110                        // Input service up but SetFocus failed — service may be
111                        // initialising.  Try again next iteration.
112                        continue;
113                    }
114                    focused_on_tid = input_tid;
115                }
116
117                let mut frame = [0u8; IPC_BUF_SIZE];
118                match sys_recv_timeout(0, &mut frame, TIMEOUT_TICKS) {
119                    SyscallResult::Ok(0) => {
120                        // Timeout — loop back to check if TID changed (restart detection).
121                        // Do NOT reset focused_on_tid here — that would trigger a SetFocus
122                        // sys_send dead zone on every 200ms timeout, causing dropped chars.
123                    }
124                    SyscallResult::Ok(sender) if sender == input_tid => {
125                        if frame[0] != INPUT_EVENT_OPCODE {
126                            continue;
127                        }
128                        let Some(ev) = decode_event(&frame[1..]) else {
129                            continue;
130                        };
131                        if let InputEvent::Key(k) = ev {
132                            if k.state != KeyState::Pressed {
133                                continue;
134                            }
135                            match k.keysym {
136                                KeySym::Return => {
137                                    print("\n");
138                                    buf.push('\n');
139                                    return Ok(bytes_read + 1);
140                                }
141                                KeySym::Backspace if !buf.is_empty() => {
142                                    print("\x08 \x08");
143                                    buf.pop();
144                                    bytes_read = bytes_read.saturating_sub(1);
145                                }
146                                KeySym::Printable if k.character > 0 => {
147                                    if let Some(ch) = char::from_u32(k.character) {
148                                        let mut tmp = [0u8; 4];
149                                        let s = ch.encode_utf8(&mut tmp);
150                                        print(s);
151                                        buf.push(ch);
152                                        bytes_read += 1;
153                                    }
154                                }
155                                _ => {}
156                            }
157                        }
158                    }
159                    SyscallResult::Ok(_other) => {
160                        // Unexpected sender — discard and keep waiting.
161                    }
162                    _ => return Err(ViError::IO),
163                }
164            }
165        }
166        {
167            // Fallback: input service not online; use direct kernel UART buffer.
168            let mut bytes_read: usize = 0;
169            loop {
170                let mut c = [0u8; 1];
171                if let Ok(n) = syscall::sys_read(0, &mut c) {
172                    if n > 0 {
173                        let ch = c[0] as char;
174                        if ch == '\r' || ch == '\n' {
175                            print("\n");
176                            buf.push('\n');
177                            return Ok(bytes_read + 1);
178                        }
179                        if c[0] == 8 || c[0] == 127 {
180                            if !buf.is_empty() {
181                                print("\x08 \x08");
182                                buf.pop();
183                                bytes_read = bytes_read.saturating_sub(1);
184                            }
185                            continue;
186                        }
187                        let mut tmp = [0u8; 4];
188                        let s = ch.encode_utf8(&mut tmp);
189                        print(s);
190                        buf.push(ch);
191                        bytes_read += 1;
192                    }
193                } else {
194                    return Err(ViError::IO);
195                }
196            }
197        }
198    }
199}

217/// Handle to the standard output stream.
218pub struct Stdout;

220pub fn stdout() -> Stdout {
221    Stdout
222}

224impl embedded_io::ErrorType for Stdout {
225    type Error = OstdError;
226}
227
228impl embedded_io::Write for Stdout {
229    fn write(&mut self, buf: &[u8]) -> Result<usize, OstdError> {
230        match core::str::from_utf8(buf) {
231            Ok(s) => {
232                syscall::sys_log(s); // always succeeds; return value is ignored
233                Ok(buf.len())
234            }
235            Err(_) => Err(OstdError(ViError::InvalidInput)),
236        }
237    }
238
239    fn flush(&mut self) -> Result<(), OstdError> {
240        Ok(()) // sys_log is synchronous; no buffering to flush
241    }
