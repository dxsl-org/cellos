1//! Shell command parser — tokenizes a line and builds an `Ast`.
2//!
3//! Supported syntax (v1.0):
4//!   - Simple command: `ls /bin`
5//!   - Pipeline: `cat /etc/hosts | grep 127`
6//!   - Output redirect: `echo hello > /tmp/a.txt`
7//!   - Input redirect: `cat < /tmp/a.txt`
8//!   - Append redirect: `echo hi >> /tmp/log.txt`
9//!   - Background: `sleep 10 &`
10//!   - Sequence: `echo a ; echo b`
11//!
12//! Intentionally simple: no subshells, no quoting beyond `"..."`, no globs.
13
14extern crate alloc;
15
16use alloc::string::String;
17use alloc::vec::Vec;

19/// One redirect target.
20#[derive(Debug, Clone, PartialEq, Eq)]
21pub enum Redirect {
22    /// `> path`
23    StdoutTo(String),
24    /// `>> path`
25    StdoutAppend(String),
26    /// `< path`
27    StdinFrom(String),
28    /// `2> path`
29    StderrTo(String),
30}
31
32/// A single command with its arguments and any redirects.
33#[derive(Debug, Clone)]
34pub struct Cmd {
35    /// `argv[0]` and arguments.
36    pub argv: Vec<String>,
37    /// Redirects attached to this command.
38    pub redirects: Vec<Redirect>,
39}
40
41impl Cmd {
42    fn new() -> Self {
43        Cmd {
44            argv: Vec::new(),
45            redirects: Vec::new(),
46        }
47    }
48
49    /// True if the command has no name (empty line or whitespace-only).
50    pub fn is_empty(&self) -> bool {
51        self.argv.is_empty()
52    }
53}

55/// Top-level abstract syntax tree for one shell line.
56#[derive(Debug, Clone)]
57pub enum Ast {
58    /// Empty input.
59    Empty,
60    /// A single simple command.
61    Simple(Cmd),
62    /// `cmd1 | cmd2 | …` — pipeline of commands.
63    Pipeline(Vec<Cmd>),
64    /// `cmd &` — run in background.
65    Background(Cmd),
66    /// `cmd1 ; cmd2` — sequential execution.
67    Sequence(Vec<Ast>),
68    /// `cmd1 && cmd2` — run cmd2 only if cmd1 exits 0 (success).
69    And(alloc::boxed::Box<Ast>, alloc::boxed::Box<Ast>),
70    /// `cmd1 || cmd2` — run cmd2 only if cmd1 exits non-zero (failure).
71    Or(alloc::boxed::Box<Ast>, alloc::boxed::Box<Ast>),
72    /// `while COND; do BODY; done` — loop while COND exits 0.
73    While {
74        cond: alloc::boxed::Box<Ast>,
75        body: alloc::boxed::Box<Ast>,
76    },
77    /// `name() { body; }` — define a shell function.
78    ///
79    /// The body is stored as a reconstructed string; the executor registers it
80    /// in the function table so later invocations of `name` run the body.
81    FuncDef {
82        name: alloc::string::String,
83        body: alloc::string::String,
84    },
85    /// `case EXPR in pattern) BODY ;; … esac` — pattern-match dispatch.
86    ///
87    /// `expr` is the unexpanded word after `case`; expansion happens at
88    /// execute time.  Patterns are exact strings or `*` (wildcard).
89    /// Note: the whole statement must be on a single line or in a single
90    /// `source`-script line; mixing with `;`-sequences on the same line is
91    /// not supported (the outer `;` split would fragment the `;;` markers).
92    Case {
93        expr: alloc::string::String,
94        arms: alloc::vec::Vec<(alloc::string::String, alloc::boxed::Box<Ast>)>,
95    },
96    /// `for VAR in word1 word2 …; do BODY; done` — iterate over a word list.
97    ///
98    /// Sets `$VAR` to each word in order, runs BODY, then advances. `$VAR`
99    /// expansion in BODY uses the same static var store as `VAR=value`.
100    For {
101        var: alloc::string::String,
102        words: alloc::vec::Vec<alloc::string::String>,
103        body: alloc::boxed::Box<Ast>,
104    },
105    /// `if COND; then BODY; fi` — conditional execution.
106    ///
107    /// `cond` exit-code 0 → run `then_b`; non-zero → run `else_b` if present.
108    If {
109        cond: alloc::boxed::Box<Ast>,
110        then_b: alloc::boxed::Box<Ast>,
111        else_b: Option<alloc::boxed::Box<Ast>>,
112    },
113}

117/// Raw token before AST construction.
118#[derive(Debug, Clone, PartialEq, Eq)]
119enum Tok {
120    Word(String),
121    Pipe,           // |
122    RedirectOut,    // >
123    RedirectAppend, // >>
124    RedirectIn,     // <
125    RedirectErr,    // 2>
126    Ampersand,      // &   (background marker — single &)
127    And,            // &&  (short-circuit AND)
128    Or,             // ||  (short-circuit OR)
129    Semicolon,      // ;
130    LBrace,         // {  (function body open)
131    RBrace,         // }  (function body close)
132    // ── Conditional keywords ─────────────────────────────────────────────────
133    // These variants are NEVER emitted by the tokenizer — `if`/`then`/`else`/`fi`
134    // always remain as Word tokens.  parse_if_stmt detects them by string
135    // comparison so they never silently disappear from external command arguments
136    // (e.g. `lua -e "if x then ... end"` must reach Lua intact).
137    #[allow(dead_code)] // reserved — kept for exhaustive match arms in parse_cmd
138    If,
139    #[allow(dead_code)] // reserved
140    Then,
141    #[allow(dead_code)] // reserved
142    Else,
143    #[allow(dead_code)] // reserved
144    Fi,
145}

147/// Tokenize a shell input line.
148///
149/// Handles:
150/// - Whitespace separation
151/// - Simple `"..."` double-quoted strings (no escape sequences)
152/// - Single-character operators: `|`, `<`, `>`, `&`, `;`
153/// - Two-character operators: `>>`, `2>`
154fn tokenize(line: &str) -> Vec<Tok> {
155    let mut tokens = Vec::new();
156    let mut chars = line.chars().peekable();
157    let mut current = String::new();
158
159    macro_rules! flush {
160        () => {
161            if !current.is_empty() {
162                tokens.push(Tok::Word(current.clone()));
163                current.clear();
164            }
165        };
166    }
167
168    while let Some(c) = chars.next() {
169        match c {
170            ' ' | '\t' => {
171                flush!();
172            }
173            '"' => {
174                // Consume until closing '"'.
175                loop {
176                    match chars.next() {
177                        Some('"') | None => break,
178                        Some(ch) => current.push(ch),
179                    }
180                }
181            }
182            '|' => {
183                flush!();
184                if chars.peek() == Some(&'|') {
185                    chars.next();
186                    tokens.push(Tok::Or);
187                } else {
188                    tokens.push(Tok::Pipe);
189                }
190            }
191            '&' => {
192                flush!();
193                if chars.peek() == Some(&'&') {
194                    chars.next();
195                    tokens.push(Tok::And);
196                } else {
197                    tokens.push(Tok::Ampersand);
198                }
199            }
200            ';' => {
201                flush!();
202                tokens.push(Tok::Semicolon);
203            }
204            '{' => {
205                flush!();
206                tokens.push(Tok::LBrace);
207            }
208            '}' => {
209                flush!();
210                tokens.push(Tok::RBrace);
211            }
212            '<' => {
213                flush!();
214                tokens.push(Tok::RedirectIn);
215            }
216            '>' => {
217                flush!();
218                if chars.peek() == Some(&'>') {
219                    chars.next();
220                    tokens.push(Tok::RedirectAppend);
221                } else {
222                    tokens.push(Tok::RedirectOut);
223                }
224            }
225            '2' if chars.peek() == Some(&'>') => {
226                // "2>" — only if current buffer is empty (i.e. not part of a word).
227                if current.is_empty() {
228                    chars.next();
229                    tokens.push(Tok::RedirectErr);
230                } else {
231                    current.push(c);
232                }
233            }
234            // `$(...)` — command substitution.  Consume everything up to the
235            // matching `)` into the current word so spaces inside don't split
236            // the token.  Nested `$( $() )` increments depth correctly.
237            '$' if chars.peek() == Some(&'(') => {
238                chars.next(); // consume '('
239                current.push('$');
240                current.push('(');
241                let mut depth = 1usize;
242                loop {
243                    match chars.next() {
244                        None => break, // unclosed: pass through
245                        Some('(') => {
246                            depth += 1;
247                            current.push('(');
248                        }
249                        Some(')') => {
250                            depth -= 1;
251                            current.push(')');
252                            if depth == 0 {
253                                break;
254                            }
255                        }
256                        Some(ch) => {
257                            current.push(ch);
258                        }
259                    }
260                }
261            }
262            other => {
263                current.push(other);
264            }
265        }
266    }
267    flush!();
268    // All tokens remain as their natural type — no keyword conversion.
269    // The if-statement parser detects `if`/`then`/`else`/`fi` by string
270    // comparison on Word tokens so they are never eaten from command arguments.
271    tokens
272}

276/// Parse a shell line into an `Ast`.
277///
278/// Returns `Ast::Empty` for blank input.
279pub fn parse(line: &str) -> Ast {
280    let tokens = tokenize(line.trim());
281    if tokens.is_empty() {
282        return Ast::Empty;
283    }
284
285    // `if...then...fi` must be parsed BEFORE semicolon splitting, because the
286    // semicolons inside an if-statement are structural (not sequence separators).
287    // Keywords remain as Word tokens (not converted) so they survive in external
288    // command argument strings (e.g. `lua -e "if x then ... end"`).
289    // `name() { body; }` — function definition.
290    // name() is a single Word token ending with "()" (no spaces inside).
291    if let Some(Tok::Word(w)) = tokens.first() {
292        if w.ends_with("()")
293            && tokens.get(1) == Some(&Tok::LBrace)
294            && tokens.last() == Some(&Tok::RBrace)
295        {
296            let name = String::from(&w[..w.len() - 2]);
297            let body = tokens_to_string(&tokens[2..tokens.len() - 1]);
298            return Ast::FuncDef { name, body };
299        }
300    }
301
302    if tokens.first() == Some(&Tok::Word("if".into())) {
303        return parse_if_stmt(&tokens);
304    }
305    if tokens.first() == Some(&Tok::Word("while".into())) {
306        return parse_while_stmt(&tokens);
307    }
308    if tokens.first() == Some(&Tok::Word("for".into())) {
309        return parse_for_stmt(&tokens);
310    }
311    // `case` must be parsed before semicolon splitting because `;;` arm
312    // separators are also Semicolon tokens and would fragment the statement.
313    if tokens.first() == Some(&Tok::Word("case".into())) {
314        return parse_case_stmt(&tokens);
315    }
316
317    // Split on `;` into sub-sequences first.
318    let segments: Vec<&[Tok]> = split_on(&tokens, |t| t == &Tok::Semicolon);
319    if segments.len() > 1 {
320        let seq: Vec<Ast> = segments.iter().map(|seg| parse_pipeline(seg)).collect();
321        return Ast::Sequence(seq);
322    }
323
324    parse_pipeline(&tokens)
325}

327/// Parse a token sub-slice that may contain `;`-separated commands.
328///
329/// Equivalent to the main `parse()` body but operating on a pre-tokenized
330/// slice — used by `parse_if_stmt` to parse condition and body sections.
331fn parse_tokens(tokens: &[Tok]) -> Ast {
332    // Strip leading/trailing semicolons that linger from the structural split.
333    let start = tokens
334        .iter()
335        .position(|t| t != &Tok::Semicolon)
336        .unwrap_or(tokens.len());
337    let end = tokens
338        .iter()
339        .rposition(|t| t != &Tok::Semicolon)
340        .map(|i| i + 1)
341        .unwrap_or(0);
342    let tokens = &tokens[start..end];
343    if tokens.is_empty() {
344        return Ast::Empty;
345    }
346    let segments: Vec<&[Tok]> = split_on(tokens, |t| t == &Tok::Semicolon);
347    if segments.len() > 1 {
348        let seq: Vec<Ast> = segments.iter().map(|seg| parse_pipeline(seg)).collect();
349        return Ast::Sequence(seq);
350    }
351    parse_pipeline(tokens)
352}

354/// Parse `if COND; then BODY; fi` or `if COND; then BODY; else BODY; fi`.
355///
356/// Handles any number of semicolons around the keywords — the structure is
357/// determined by the `Then`, `Else`, and `Fi` token positions.
358/// Serialize a token slice back to a space-separated shell line.
359///
360/// Used by function definition to store the body as a re-parseable string.
361fn tokens_to_string(tokens: &[Tok]) -> String {
362    let parts: alloc::vec::Vec<&str> = tokens
363        .iter()
364        .map(|t| match t {
365            Tok::Word(w) => w.as_str(),
366            Tok::Pipe => "|",
367            Tok::And => "&&",
368            Tok::Or => "||",
369            Tok::Semicolon => ";",
370            Tok::Ampersand => "&",
371            Tok::RedirectOut => ">",
372            Tok::RedirectAppend => ">>",
373            Tok::RedirectIn => "<",
374            Tok::RedirectErr => "2>",
375            Tok::LBrace => "{",
376            Tok::RBrace => "}",
377            _ => "",
378        })
379        .collect();
380    parts.join(" ")
381}

383/// Helper: returns true when a token is the keyword word `w`.
384fn is_kw(tok: &Tok, w: &str) -> bool {
385    tok == &Tok::Word(w.into())
386}

388/// Parse `while COND; do BODY; done`.
389///
390/// Keywords stay as `Word` tokens (no Tok variants) so `while`/`do`/`done`
391/// survive intact when used as external command arguments.  Malformed input
392/// (missing `do` or `done`) calls `parse_tokens` — NOT `parse()` — to avoid
393/// re-dispatching on the leading `while` and recursing infinitely.
394/// Parse `for VAR in word1 word2 …; do BODY; done`.
395///
396/// Keywords stay as `Word` tokens (same Phase N/O rule) so `for`/`in`/`do`/`done`
397/// survive as external command arguments.  Malformed input falls back to
398/// `parse_tokens` (not `parse()`) to prevent infinite recursion.
399fn parse_for_stmt(tokens: &[Tok]) -> Ast {
400    // tokens[1] = variable name; tokens[2] should be Word("in").
401    let var = match tokens.get(1) {
402        Some(Tok::Word(w)) if w != "in" => w.clone(),
403        _ => return parse_tokens(tokens),
404    };
405    let in_pos = match tokens.iter().position(|t| is_kw(t, "in")) {
406        Some(p) => p,
407        None => return parse_tokens(tokens),
408    };
409    let do_pos = tokens.iter().position(|t| is_kw(t, "do"));
410    let done_pos = tokens.iter().rposition(|t| is_kw(t, "done"));
411    let (dp, np) = match (do_pos, done_pos) {
412        (Some(d), Some(n)) if n > d => (d, n),
413        _ => return parse_tokens(tokens),
414    };
415    // Word list: tokens between `in` and `do`, stripping Semicolons.
416    let words: alloc::vec::Vec<alloc::string::String> = tokens[in_pos + 1..dp]
417        .iter()
418        .filter_map(|t| {
419            if let Tok::Word(w) = t {
420                Some(w.clone())
421            } else {
422                None
423            }
424        })
425        .collect();
426    let body = parse_tokens(&tokens[dp + 1..np]);
427    Ast::For {
428        var,
429        words,
430        body: alloc::boxed::Box::new(body),
431    }
432}

434/// Parse `case EXPR in pattern1) BODY ;; pattern2) BODY ;; esac`.
435///
436/// `;;` is two consecutive `Tok::Semicolon` tokens.  Patterns are the first
437/// token of each arm with a trailing `)` stripped.  `*` is a catch-all.
438/// Malformed input (missing `in` or `esac`) falls back to `parse_tokens`.
439fn parse_case_stmt(tokens: &[Tok]) -> Ast {
440    let in_pos = tokens.iter().position(|t| is_kw(t, "in"));
441    let esac_pos = tokens.iter().rposition(|t| is_kw(t, "esac"));
442    let (ip, ep) = match (in_pos, esac_pos) {
443        (Some(i), Some(e)) if e > i => (i, e),
444        _ => return parse_tokens(tokens),
445    };
446
447    // Expression: single word between `case` and `in`.
448    let expr = match tokens.get(1) {
449        Some(Tok::Word(w)) => w.clone(),
450        _ => String::new(),
451    };
452
453    // Arms: tokens[ip+1..ep] split on `;;` (two consecutive Semicolons).
454    let arm_tokens = &tokens[ip + 1..ep];
455    let mut arms: alloc::vec::Vec<(String, alloc::boxed::Box<Ast>)> = alloc::vec::Vec::new();
456    let mut start = 0;
457    let mut i = 0;
458    while i < arm_tokens.len() {
459        if i + 1 < arm_tokens.len()
460            && arm_tokens[i] == Tok::Semicolon
461            && arm_tokens[i + 1] == Tok::Semicolon
462        {
463            push_arm(&arm_tokens[start..i], &mut arms);
464            i += 2;
465            start = i;
466        } else {
467            i += 1;
468        }
469    }
470    push_arm(&arm_tokens[start..], &mut arms); // last arm (no trailing ;;)
471
472    Ast::Case { expr, arms }
473}
474
475fn push_arm(tokens: &[Tok], arms: &mut alloc::vec::Vec<(String, alloc::boxed::Box<Ast>)>) {
476    // Strip leading/trailing Semicolons left from the split.
477    let start = tokens
478        .iter()
479        .position(|t| t != &Tok::Semicolon)
480        .unwrap_or(tokens.len());
481    let tokens = &tokens[start..];
482    if let Some(Tok::Word(pat)) = tokens.first() {
483        let pattern = String::from(pat.trim_end_matches(')'));
484        let body = parse_tokens(&tokens[1..]);
485        arms.push((pattern, alloc::boxed::Box::new(body)));
486    }
487}

489fn parse_while_stmt(tokens: &[Tok]) -> Ast {
490    let do_pos = tokens.iter().position(|t| is_kw(t, "do"));
491    let done_pos = tokens.iter().rposition(|t| is_kw(t, "done"));
492    let (dp, np) = match (do_pos, done_pos) {
493        (Some(d), Some(n)) if n > d => (d, n),
494        _ => return parse_tokens(tokens), // malformed: fall back without infinite recursion
495    };
496    let cond = parse_tokens(&tokens[1..dp]);
497    let body = parse_tokens(&tokens[dp + 1..np]);
498    Ast::While {
499        cond: alloc::boxed::Box::new(cond),
500        body: alloc::boxed::Box::new(body),
501    }
502}

504fn parse_if_stmt(tokens: &[Tok]) -> Ast {
505    // Locate structural keywords after the leading `if` Word.
506    // Keywords are plain Word tokens — never converted — so they survive intact
507    // in external command argument strings.
508    let then_pos = tokens
509        .iter()
510        .position(|t| is_kw(t, "then"))
511        .unwrap_or(tokens.len());
512    let else_pos = tokens.iter().position(|t| is_kw(t, "else"));
513    let fi_pos = tokens
514        .iter()
515        .rposition(|t| is_kw(t, "fi"))
516        .unwrap_or(tokens.len());
517
518    // Condition: tokens[1..then_pos]   (skip leading `If`)
519    let cond_slice = &tokens[1..then_pos];
520    let cond = parse_tokens(cond_slice);
521
522    // Then body: tokens[then_pos+1..else_or_fi]
523    let then_end = else_pos.unwrap_or(fi_pos);
524    let then_slice = &tokens[then_pos + 1..then_end];
525    let then_b = parse_tokens(then_slice);
526
527    // Else body (optional): tokens[else_pos+1..fi_pos]
528    let else_b = else_pos.map(|ep| {
529        let slice = &tokens[ep + 1..fi_pos];
530        alloc::boxed::Box::new(parse_tokens(slice))
531    });
532
533    Ast::If {
534        cond: alloc::boxed::Box::new(cond),
535        then_b: alloc::boxed::Box::new(then_b),
536        else_b,
537    }
538}

540fn parse_pipeline(tokens: &[Tok]) -> Ast {
541    // `&&` / `||` have lower precedence than pipelines — check first.
542    // Split on the FIRST occurrence; right side is parsed recursively so
543    // `A && B && C` builds `And(A, And(B, C))` with left-to-right evaluation.
544    if let Some(pos) = tokens.iter().position(|t| t == &Tok::And || t == &Tok::Or) {
545        let left = parse_pipeline(&tokens[..pos]);
546        let right = parse_pipeline(&tokens[pos + 1..]);
547        return match &tokens[pos] {
548            Tok::And => Ast::And(alloc::boxed::Box::new(left), alloc::boxed::Box::new(right)),
549            Tok::Or => Ast::Or(alloc::boxed::Box::new(left), alloc::boxed::Box::new(right)),
550            _ => unreachable!(),
551        };
552    }
553
554    let pipe_segs: Vec<&[Tok]> = split_on(tokens, |t| t == &Tok::Pipe);
555
556    let cmds: Vec<Cmd> = pipe_segs
557        .iter()
558        .map(|seg| {
559            // Ignore the per-segment `bg` flag here — the trailing `&` check on
560            // `tokens.last()` below is the authoritative background detector.
561            // Filtering out `bg=true` segments caused single-command background
562            // jobs (`httpd 9091 /path &`) to be parsed as Ast::Empty.
563            let (cmd, _bg) = parse_cmd(seg);
564            cmd
565        })
566        .filter(|c| !c.is_empty())
567        .collect();
568
569    // Check for trailing `&` (background marker).
570    let background = tokens.last() == Some(&Tok::Ampersand);
571
572    if background && cmds.len() == 1 {
573        return Ast::Background(cmds.into_iter().next().unwrap_or_else(Cmd::new));
574    }
575    match cmds.len() {
576        0 => Ast::Empty,
577        1 => Ast::Simple(cmds.into_iter().next().unwrap_or_else(Cmd::new)),
578        _ => Ast::Pipeline(cmds),
579    }
580}

582/// Parse one command segment (no `|`, `;`, or `&` except trailing `&`).
583/// Returns (Cmd, is_background).
584fn parse_cmd(tokens: &[Tok]) -> (Cmd, bool) {
585    let mut cmd = Cmd::new();
586    let mut background = false;
587    let mut iter = tokens.iter().peekable();
588
589    while let Some(tok) = iter.next() {
590        match tok {
591            Tok::Word(w) => cmd.argv.push(w.clone()),
592            Tok::Ampersand => background = true,
593            Tok::RedirectOut => {
594                if let Some(Tok::Word(path)) = iter.next() {
595                    cmd.redirects.push(Redirect::StdoutTo(path.clone()));
596                }
597            }
598            Tok::RedirectAppend => {
599                if let Some(Tok::Word(path)) = iter.next() {
600                    cmd.redirects.push(Redirect::StdoutAppend(path.clone()));
601                }
602            }
603            Tok::RedirectIn => {
604                if let Some(Tok::Word(path)) = iter.next() {
605                    cmd.redirects.push(Redirect::StdinFrom(path.clone()));
606                }
607            }
608            Tok::RedirectErr => {
609                if let Some(Tok::Word(path)) = iter.next() {
610                    cmd.redirects.push(Redirect::StderrTo(path.clone()));
611                }
612            }
613            _ => {}
614        }
615    }
616    (cmd, background)
617}

619/// Split a token slice on positions where `pred` returns true.
620fn split_on<F>(tokens: &[Tok], pred: F) -> Vec<&[Tok]>
621where
622    F: Fn(&Tok) -> bool,
623{
624    let mut result = Vec::new();
625    let mut start = 0;
626    for (i, tok) in tokens.iter().enumerate() {
627        if pred(tok) {
628            result.push(&tokens[start..i]);
629            start = i + 1;
630        }
631    }
632    result.push(&tokens[start..]);
633    result
634}
