1//! Shell job table — tracks background and foreground jobs.
2
3extern crate alloc;
4use alloc::{string::String, vec::Vec};
5
6#[derive(Debug, Clone, Copy, PartialEq, Eq)]
7pub enum JobState {
8    Running,
9    Done,
10}
11
12/// One job entry.
13struct Job {
14    id: usize,
15    state: JobState,
16    name: String,
17}
18
19/// Active job registry.
20#[derive(Default)]
21pub struct Jobs {
22    jobs: Vec<Job>,
23    next_id: usize,
24}
25
26impl Jobs {
27    pub fn new() -> Self {
28        Self {
29            jobs: Vec::new(),
30            next_id: 1,
31        }
32    }
33
34    /// Add a new running job and return its job ID.
35    pub fn add(&mut self, name: &str) -> usize {
36        let id = self.next_id;
37        self.next_id += 1;
38        self.jobs.push(Job {
39            id,
40            state: JobState::Running,
41            name: String::from(name),
42        });
43        id
44    }
45
46    /// Update the state of a job by ID.
47    pub fn set_state(&mut self, id: usize, state: JobState) {
48        if let Some(j) = self.jobs.iter_mut().find(|j| j.id == id) {
49            j.state = state;
50        }
51    }
52
53    /// Remove all completed jobs.
54    pub fn reap_done(&mut self) {
55        self.jobs.retain(|j| j.state != JobState::Done);
56    }
57
58    /// Iterate (id, state, name) for all jobs.
59    pub fn list(&self) -> impl Iterator<Item = (usize, JobState, &str)> {
60        self.jobs.iter().map(|j| (j.id, j.state, j.name.as_str()))
61    }
62}
