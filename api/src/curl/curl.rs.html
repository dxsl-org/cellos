12/// Maximum accumulated response size (stack-allocated; avoids the 4 MB alloc BSS).
13const RESP_BUF: usize = 4096;

15api::declare_syscalls![Send, Recv, Log, StateRestore, LookupService];

17#[no_mangle]
18pub fn main() {
19    // ── Parse argv ───────────────────────────────────────────────────────────
20    let mut arg_buf = [0u8; 128];
21    let arg_len = sys_spawn_args(&mut arg_buf);
22    if arg_len == 0 {
23        println("Usage: curl http://IP[:PORT][/path]");
24        return;
25    }
26    let url = match core::str::from_utf8(&arg_buf[..arg_len]) {
27        Ok(s) => s.trim(),
28        Err(_) => {
29            println("curl: bad args");
30            return;
31        }
32    };
33
34    // ── Parse URL ────────────────────────────────────────────────────────────
35    let (host, port, path) = match parse_url(url) {
36        Some(t) => t,
37        None => {
38            println("curl: invalid URL — expected http://IP[:PORT][/path]");
39            return;
40        }
41    };
42    let addr = match resolve_host(host) {
43        Some(a) => a,
44        None => {
45            println("curl: invalid host");
46            return;
47        }
48    };
49
50    // ── Guard: HTTP request must fit in the request_data stack buffer ─────────
51    let overhead =
52        b"GET ".len() + b" HTTP/1.0\r\nHost: ".len() + b"\r\nConnection: close\r\n\r\n".len();
53    if overhead + path.len() + host.len() > 500 {
54        println("curl: URL too long");
55        return;
56    }
57
58    // ── Resolve net service endpoint ──────────────────────────────────────────
59    let net_ep = match sys_lookup_service(service::NET) {
60        Some(ep) => ep,
61        None => {
62            println("curl: no net service");
63            return;
64        }
65    };
66
67    // ── TcpConnect (atomic create + connect) ─────────────────────────────────
68    let mut req_buf = [0u8; IPC_BUF_SIZE];
69    let len = api::ipc::encode(&NetRequest::TcpConnect { addr, port }, &mut req_buf)
70        .map(|b| b.len())
71        .unwrap_or(0);
72    sys_send(net_ep, &req_buf[..len]);
73    let mut resp_buf = [0u8; IPC_BUF_SIZE];
74    let cap_id = match sys_recv(0, &mut resp_buf) {
75        SyscallResult::Ok(_) => match api::ipc::decode::<NetResponse>(&resp_buf) {
76            Ok(NetResponse::CapId(c)) => c,
77            _ => {
78                println("curl: connect failed");
79                return;
80            }
81        },
82        _ => {
83            println("curl: TcpConnect syscall failed");
84            return;
85        }
86    };
87
88    // ── Build HTTP GET request into a stack buffer ────────────────────────────
89    let mut request_data = [0u8; 512];
90    let mut pos = 0usize;
91    pos = wb(&mut request_data, pos, b"GET ");
92    pos = wb(&mut request_data, pos, path.as_bytes());
93    pos = wb(&mut request_data, pos, b" HTTP/1.0\r\nHost: ");
94    pos = wb(&mut request_data, pos, host.as_bytes());
95    pos = wb(&mut request_data, pos, b"\r\nConnection: close\r\n\r\n");
96    let request_len = pos;
97
98    // ── Send HTTP request via TcpSend with retry ──────────────────────────────
99    let mut sent_bytes = 0usize;
100    for _ in 0..500 {
101        if sent_bytes >= request_len {
102            break;
103        }
104        let rem = &request_data[sent_bytes..request_len];
105        let chunk = rem.len().min(480);
106        let mut send_buf = [0u8; IPC_BUF_SIZE];
107        let send_len = api::ipc::encode(
108            &NetRequest::TcpSend {
109                cap_id,
110                data: &rem[..chunk],
111            },
112            &mut send_buf,
113        )
114        .map(|b| b.len())
115        .unwrap_or(0);
116        sys_send(net_ep, &send_buf[..send_len]);
117        let mut cnt_buf = [0u8; IPC_BUF_SIZE];
118        match sys_recv(0, &mut cnt_buf) {
119            SyscallResult::Ok(_) => match api::ipc::decode::<NetResponse>(&cnt_buf) {
120                Ok(NetResponse::Data(b)) if b.len() >= 4 => {
121                    let mut arr = [0u8; 4];
122                    arr.copy_from_slice(&b[0..4]);
123                    let n = u32::from_le_bytes(arr) as usize;
124                    sent_bytes += n;
125                    if n == 0 {
126                        sys_yield();
127                    }
128                }
129                _ => break,
130            },
131            _ => break,
132        }
133    }
134
135    // ── Accumulate HTTP response via TcpRecv ──────────────────────────────────
136    let mut response = [0u8; RESP_BUF];
137    let mut resp_len = 0usize;
138
139    // Pre-encode the TcpRecv request once and reuse it across all iterations.
140    let mut recv_req_buf = [0u8; IPC_BUF_SIZE];
141    let recv_req_len = api::ipc::encode(
142        &NetRequest::TcpRecv {
143            cap_id,
144            buf_len: 500,
145        },
146        &mut recv_req_buf,
147    )
148    .map(|b| b.len())
149    .unwrap_or(0);
150
151    'recv: for _ in 0..500 {
152        sys_send(net_ep, &recv_req_buf[..recv_req_len]);
153        let mut data_buf = [0u8; IPC_BUF_SIZE];
154        match sys_recv(0, &mut data_buf) {
155            SyscallResult::Ok(_) => {
156                match api::ipc::decode::<NetResponse>(&data_buf) {
157                    Ok(NetResponse::Data(b)) if !b.is_empty() => {
158                        let n = b.len().min(RESP_BUF - resp_len);
159                        response[resp_len..resp_len + n].copy_from_slice(&b[..n]);
160                        resp_len += n;
161                        if resp_len >= RESP_BUF {
162                            break 'recv;
163                        }
164                    }
165                    Ok(NetResponse::Data(_)) => {
166                        // 0 bytes: check TCP state to distinguish "nothing yet" from FIN.
167                        let st = query_state(cap_id, net_ep);
168                        if st == 0x06 || st == 0x00 {
169                            // CloseWait / Closed — drain one final recv before exiting.
170                            sys_send(net_ep, &recv_req_buf[..recv_req_len]);
171                            let mut fb = [0u8; IPC_BUF_SIZE];
172                            if let SyscallResult::Ok(_) = sys_recv(0, &mut fb) {
173                                if let Ok(NetResponse::Data(b)) =
174                                    api::ipc::decode::<NetResponse>(&fb)
175                                {
176                                    let n = b.len().min(RESP_BUF - resp_len);
177                                    response[resp_len..resp_len + n].copy_from_slice(&b[..n]);
178                                    resp_len += n;
179                                }
180                            }
181                            break 'recv;
182                        }
183                        sys_yield();
184                    }
185                    _ => break,
186                }
187            }
188            _ => break,
189        }
190    }
191
192    // ── Print status line + body ─────────────────────────────────────────────
193    let resp = &response[..resp_len];
194    if let Some(header_end) = resp.windows(4).position(|w| w == b"\r\n\r\n") {
195        let status_end = resp.iter().position(|&b| b == b'\r').unwrap_or(header_end);
196        if let Ok(status) = core::str::from_utf8(&resp[..status_end]) {
197            println(status);
198        }
199        let body = &resp[header_end + 4..];
200        if let Ok(s) = core::str::from_utf8(body) {
201            print(s);
202        }
203    } else if resp_len > 0 {
204        if let Ok(s) = core::str::from_utf8(resp) {
205            print(s);
206        }
207    } else {
208        println("curl: empty response");
209    }
210
211    close_socket(cap_id, net_ep);
212}

214/// Send `NetRequest::SocketState` and return the 1-byte smoltcp state code.
215fn query_state(cap_id: u32, net_ep: usize) -> u8 {
216    let mut req_buf = [0u8; IPC_BUF_SIZE];
217    let len = api::ipc::encode(&NetRequest::SocketState { cap_id }, &mut req_buf)
218        .map(|b| b.len())
219        .unwrap_or(0);
220    sys_send(net_ep, &req_buf[..len]);
221    let mut resp_buf = [0u8; IPC_BUF_SIZE];
222    match sys_recv(0, &mut resp_buf) {
223        SyscallResult::Ok(_) => match api::ipc::decode::<NetResponse>(&resp_buf) {
224            Ok(NetResponse::State(s)) => s,
225            _ => 0x00,
226        },
227        _ => 0x00,
228    }
229}

231/// Send `NetRequest::TcpClose` and drain the acknowledgement.
232fn close_socket(cap_id: u32, net_ep: usize) {
233    let mut req_buf = [0u8; IPC_BUF_SIZE];
234    let len = api::ipc::encode(&NetRequest::TcpClose { cap_id }, &mut req_buf)
235        .map(|b| b.len())
236        .unwrap_or(0);
237    sys_send(net_ep, &req_buf[..len]);
238    let mut resp_buf = [0u8; IPC_BUF_SIZE];
239    let _ = sys_recv(0, &mut resp_buf);
240}

242/// Append `src` into `buf` at `pos`; returns new pos.
243fn wb(buf: &mut [u8], pos: usize, src: &[u8]) -> usize {
244    buf[pos..pos + src.len()].copy_from_slice(src);
245    pos + src.len()
246}

248fn parse_url(s: &str) -> Option<(&str, u16, &str)> {
249    let rest = s.strip_prefix("http://")?;
250    if rest.is_empty() {
251        return None;
252    }
253    let (host_port, path) = match rest.find('/') {
254        Some(i) => (&rest[..i], &rest[i..]),
255        None => (rest, "/"),
256    };
257    let (host, port) = match host_port.rfind(':') {
258        Some(i) => (&host_port[..i], parse_u16(&host_port[i + 1..])?),
259        None => (host_port, 80u16),
260    };
261    if host.is_empty() {
262        return None;
263    }
264    Some((host, port, path))
265}

267fn resolve_host(s: &str) -> Option<[u8; 4]> {
268    match s {
269        "gateway" | "host" => Some([10, 0, 2, 2]),
270        "dns" => Some([10, 0, 2, 3]),
271        "localhost" => Some([127, 0, 0, 1]),
272        _ => parse_ipv4(s),
273    }
274}

276fn parse_ipv4(s: &str) -> Option<[u8; 4]> {
277    let mut it = s.splitn(5, '.');
278    let a = parse_octet(it.next()?)?;
279    let b = parse_octet(it.next()?)?;
280    let c = parse_octet(it.next()?)?;
281    let d = parse_octet(it.next()?)?;
282    if it.next().is_some() {
283        return None;
284    }
285    Some([a, b, c, d])
286}

288fn parse_octet(s: &str) -> Option<u8> {
289    let mut n: u16 = 0;
290    if s.is_empty() {
291        return None;
292    }
293    for ch in s.bytes() {
294        if !ch.is_ascii_digit() {
295            return None;
296        }
297        n = n * 10 + (ch - b'0') as u16;
298        if n > 255 {
299            return None;
300        }
301    }
302    Some(n as u8)
303}

305fn parse_u16(s: &str) -> Option<u16> {
306    let mut n: u32 = 0;
307    if s.is_empty() {
308        return None;
309    }
310    for ch in s.bytes() {
311        if !ch.is_ascii_digit() {
312            return None;
313        }
314        n = n * 10 + (ch - b'0') as u32;
315        if n > 65535 {
316            return None;
317        }
318    }
319    Some(n as u16)
