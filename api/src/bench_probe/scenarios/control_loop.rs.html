1//! control_loop_jitter — periodic deadline adherence under load.
2//!
3//! A RealTime cell wakes every `PERIOD_TICKS` (via `recv_timeout`, which blocks
4//! until timeout since nothing sends to it), measures the actual elapsed period,
5//! and records the per-cycle error (|actual − period|) plus a deadline-miss when
6//! the cycle overruns by more than `SLACK_TICKS`. Mirrors a fixed-rate control
7//! loop (PID / software PWM) and proves "control-loop meets deadline" (G1 #3).
8//!
9//! Period is 50 ms (5 scheduler ticks) so jitter is meaningful above the 10 ms
10//! tick quantum. Runs under background load cells spawned by the orchestrator.
11//!
12//! ⚠️ Runtime numbers require the bench cell embedded at `/bin/bench` (phase-05).
13
14extern crate alloc;
15use crate::framework::rt_report::RtReport;
16use alloc::vec::Vec;
17use ostd::syscall::{sys_exit, sys_get_time, sys_recv, sys_recv_timeout, sys_send, SyscallResult};
18
19/// Target loop period: 50 ms @ 10 MHz mtime (5 scheduler ticks).
20const PERIOD_TICKS: u64 = 50 * 10_000;
21/// Allowed overrun before a cycle counts as a deadline miss: 5 ms.
22const SLACK_TICKS: u64 = 5 * 10_000;
23/// Number of measured periods.
24const CL_ITERS: u32 = 200;
25
26/// RealTime probe role: learn the orchestrator's tid from its start ping, run
27/// the periodic loop, print the jitter/deadline report, then signal done + exit.
28pub fn run_control_loop() -> ! {
29    let mut buf = [0u8; 8];
30    // Block for the orchestrator's start ping so we can reply "done" to it later.
31    let orch = loop {
32        match sys_recv(0, &mut buf) {
33            SyscallResult::Ok(s) if s > 0 => break s,
34            _ => ostd::task::yield_now(),
35        }
36    };
37
38    let mut errors: Vec<u64> = Vec::with_capacity(CL_ITERS as usize);
39    let mut miss = 0u32;
40    let mut prev = sys_get_time();
41    for _ in 0..CL_ITERS {
42        // Sleep ~one period: recv_timeout returns on timeout (no sender) and
43        // is a real block — the scheduler is free to run load cells meanwhile.
44        let _ = sys_recv_timeout(0, &mut buf, PERIOD_TICKS);
45        let now = sys_get_time();
46        let actual = now.saturating_sub(prev);
47        let err = actual.abs_diff(PERIOD_TICKS);
48        errors.push(err);
49        if actual > PERIOD_TICKS + SLACK_TICKS {
50            miss += 1;
51        }
52        prev = now;
53    }
54
55    let r = RtReport::build("control_loop_jitter", &mut errors, miss);
56    r.print();
57    r.print_json();
58    if r.deadline_miss == 0 {
59        ostd::io::println("[rt] control_loop PASS");
60    } else {
61        ostd::io::println("[rt] control_loop FAIL (deadline misses)");
62    }
63
64    let _ = sys_send(orch, &[1u8]); // signal orchestrator we are done
65    sys_exit(0);
66}
