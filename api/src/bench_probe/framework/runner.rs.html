1//! Benchmark runner: warmup loop + measurement loop + percentile computation.
2
3extern crate alloc;
4use super::{report, timer};
5use alloc::vec::Vec;
6use api::benchmark::{BenchReport, ViBenchmark};
7
8/// Default warmup iterations (discarded; exist to heat up caches and QEMU JIT).
9pub const DEFAULT_WARMUP: u32 = 100;
10/// Default measurement iterations per scenario.
11pub const DEFAULT_ITERS: u32 = 1_000;
12
13/// Run a benchmark through warmup + measurement and return a `BenchReport`.
14///
15/// # Arguments
16/// * `bench` — mutable reference to the scenario implementing `ViBenchmark`
17/// * `warmup` — number of iterations whose results are discarded
18/// * `iters` — number of iterations to measure
19pub fn run<B: ViBenchmark>(bench: &mut B, warmup: u32, iters: u32) -> BenchReport {
20    // ── Warmup ────────────────────────────────────────────────────────────────
21    bench.setup().ok();
22    for _ in 0..warmup {
23        let _ = bench.run_once();
24    }
25
26    // ── Measurement ──────────────────────────────────────────────────────────
27    let mut samples: Vec<u64> = Vec::with_capacity(iters as usize);
28    for _ in 0..iters {
29        let t0 = timer::read_ticks();
30        let _ = bench.run_once();
31        let t1 = timer::read_ticks();
32        samples.push(t1.saturating_sub(t0));
33    }
34
35    bench.teardown().ok();
36    report::build_report(bench.name(), &mut samples)
37}
38
39/// Run a benchmark with the default warmup and iteration counts.
40pub fn run_default<B: ViBenchmark>(bench: &mut B) -> BenchReport {
41    run(bench, DEFAULT_WARMUP, DEFAULT_ITERS)
42}
