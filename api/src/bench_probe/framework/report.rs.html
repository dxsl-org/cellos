1//! Benchmark report: statistics computation and JSON/text emission.
2
3extern crate alloc;
4use super::timer::ticks_to_ns;
5use alloc::vec::Vec;
6use api::benchmark::BenchReport;
7use ostd::io::println;
8
9/// Compute a `BenchReport` from a raw (unsorted) tick-delta sample buffer.
10///
11/// Sorts `samples` in-place, converts each tick delta to nanoseconds, then
12/// builds percentile stats.
13pub fn build_report(name: &'static str, samples: &mut [u64]) -> BenchReport {
14    samples.sort_unstable();
15    let ns: Vec<u64> = samples.iter().map(|&t| ticks_to_ns(t)).collect();
16    BenchReport::from_sorted(name, &ns)
17}
18
19/// Print a human-readable summary of a `BenchReport` to the serial console.
20pub fn print_report(r: &BenchReport) {
21    println(&format_report(r));
22}
23
24/// Print a machine-readable JSON line (for CI parsing).
25pub fn print_json(r: &BenchReport) {
26    let mut buf = [0u8; 256];
27    let len = r.write_json(&mut buf);
28    if len > 0 {
29        if let Ok(s) = core::str::from_utf8(&buf[..len]) {
30            println(s);
31        }
32    }
33}
34
35/// Format a single-line human-readable report string.
36fn format_report(r: &BenchReport) -> alloc::string::String {
37    use alloc::format;
38    format!(
39        "[bench] {:20} n={:>6}  min={:>6}ns  p50={:>6}ns  p99={:>6}ns  max={:>6}ns",
40        r.name, r.n, r.min, r.p50, r.p99, r.max
41    )
42}
