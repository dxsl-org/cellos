23pub fn send_response(cap: u32, net_ep: usize, status: u16, content_type: &str, body: &[u8]) {
24    let status_text = match status {
25        200 => "OK",
26        204 => "No Content",
27        400 => "Bad Request",
28        404 => "Not Found",
29        500 => "Internal Server Error",
30        _ => "OK",
31    };
32    let header = format!(
33        "HTTP/1.1 {} {}\r\nContent-Type: {}\r\nContent-Length: {}\r\nConnection: close\r\n\r\n",
34        status,
35        status_text,
36        content_type,
37        body.len()
38    );
39    net_ipc::tcp_send_all(cap, net_ep, header.as_bytes());
40    net_ipc::tcp_send_all(cap, net_ep, body);
41}
42
43fn send_json(cap: u32, net_ep: usize, status: u16, json: &str) {
44    let body = json.as_bytes();
45    let status_text = if status == 200 { "OK" } else { "Not Found" };
46    let header = format!(
47        "HTTP/1.1 {} {}\r\nContent-Type: application/json\r\nContent-Length: {}\r\nAccess-Control-Allow-Origin: *\r\nConnection: close\r\n\r\n",
48        status, status_text, body.len()
49    );
50    net_ipc::tcp_send_all(cap, net_ep, header.as_bytes());
51    net_ipc::tcp_send_all(cap, net_ep, body);
52}

132pub fn serve_file(cap: u32, net_ep: usize, vfs_ep: usize, path: &str) {
133    let content_type = mime_from_ext(path);
134    let data = net_ipc::vfs_read_file(path, vfs_ep);
135    if data.is_empty() {
136        not_found(cap, net_ep);
137    } else {
138        send_response(cap, net_ep, 200, content_type, &data);
139    }
140}
141
142fn mime_from_ext(path: &str) -> &'static str {
143    match path.rsplit('.').next() {
144        Some("html") | Some("htm") => "text/html; charset=utf-8",
145        Some("css") => "text/css",
146        Some("js") => "application/javascript",
147        Some("json") => "application/json",
148        Some("png") => "image/png",
149        Some("jpg") | Some("jpeg") => "image/jpeg",
150        Some("svg") => "image/svg+xml",
151        Some("ico") => "image/x-icon",
152        Some("txt") => "text/plain",
153        _ => "application/octet-stream",
154    }
155}

167pub fn api_cells(cap: u32, net_ep: usize) {
168    // Well-known service IDs probed via service registry
169    use ostd::service::{lookup, service};
170    let mut entries = Vec::new();
171    let known: &[(&str, u16)] = &[
172        ("net", service::NET),
173        ("vfs", service::VFS),
174        ("input", service::INPUT),
175        ("compositor", service::COMPOSITOR),
176        ("config", service::CONFIG),
177    ];
178    for (name, id) in known {
179        if let Some(tid) = lookup(*id) {
180            entries.push(format!(
181                r#"{{"name":"{}","tid":{},"state":"Running"}}"#,
182                name, tid
183            ));
184        }
185    }
186    let list = entries.join(",");
187    let json = format!(r#"{{"cells":[{}]}}"#, list);
188    send_json(cap, net_ep, 200, &json);
189}

191pub fn api_files(cap: u32, net_ep: usize, vfs_ep: usize, path: &str) {
192    let raw = net_ipc::vfs_list_dir(path, vfs_ep);
193    let listing = core::str::from_utf8(&raw).unwrap_or("");
194    let entries: Vec<String> = listing
195        .lines()
196        .filter(|l| !l.is_empty())
197        .map(|name| format!(r#"{{"name":"{}"}}"#, name))
198        .collect();
199    let list = entries.join(",");
200    let json = format!(r#"{{"path":"{}","entries":[{}]}}"#, path, list);
201    send_json(cap, net_ep, 200, &json);
202}
