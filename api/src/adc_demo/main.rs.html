4extern crate alloc;
5
6use api::declare_manifest;
7use driver_adc_sim::{SimAdc, SimChannel};
8use hal_adc::ViAdc;
9use ostd::io::println;
10
11// No MMIO required — simulation only. No capability flags needed.
12declare_manifest!(
13    block_io = false,
14    network = false,
15    spawn = false,
16    gpio = false,
17    uart = false
18);
19
20const SUPPLY_MV: u32 = 3300; // 3.3 V reference
21
22#[no_mangle]
23pub fn main() {
24    println("[adc-demo] ADC simulation demo (3 channels, 12-bit, 3.3V ref)");
25
26    let mut adc = SimAdc::new();
27
28    // Channel 0: slow temperature-like ramp (5 s cycle), with noise.
29    adc.configure(0, SimChannel::new(5_000, 3));
30    // Channel 1: medium ramp (2 s cycle), clean.
31    adc.configure(1, SimChannel::new(2_000, 0));
32    // Channel 2: fast ramp (800 step cycle), with noise.
33    adc.configure(2, SimChannel::new(800, 2));
34
35    for i in 0..5u32 {
36        adc.step();
37
38        let r0 = adc.read_raw(0).unwrap_or(0);
39        let r1 = adc.read_raw(1).unwrap_or(0);
40        let r2 = adc.read_raw(2).unwrap_or(0);
41
42        let mv0 = adc.to_millivolts(r0, SUPPLY_MV);
43        let mv1 = adc.to_millivolts(r1, SUPPLY_MV);
44        let mv2 = adc.to_millivolts(r2, SUPPLY_MV);
45
46        let msg = alloc::format!(
47            "[adc-demo] step={} ch0={}mV ch1={}mV ch2={}mV (raw {}/{}/{})",
48            i,
49            mv0,
50            mv1,
51            mv2,
52            r0,
53            r1,
54            r2
55        );
56        println(&msg);
57    }
58
59    println("[adc-demo] done");
60}
