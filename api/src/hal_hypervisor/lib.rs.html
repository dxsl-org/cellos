4/// VM exit reason returned by `ViHypervisor::run_vcpu`.
5///
6/// Variants are stable across P01-P03; the binary layout (not `#[repr(C)]`) is
7/// intentionally left non-ABI-frozen until P04 per Law 1 — callers must match
8/// the VMM crate version at compile time.
9#[derive(Debug, Clone, Copy)]
10pub enum ViVmExit {
11    MmioRead {
12        ipa: u64,
13        size: u8,
14        reg: u8,
15    },
16    MmioWrite {
17        ipa: u64,
18        size: u8,
19        val: u64,
20    },
21    Hvc {
22        imm: u16,
23        regs: [u64; 8],
24    },
25    Wfi,
26    /// System-register access (EC=0x18) — timer register emulation (P05+).
27    SysReg {
28        op0: u8,
29        op1: u8,
30        crn: u8,
31        crm: u8,
32        op2: u8,
33        rt: u8,
34        is_write: bool,
35    },
36    Preempted,
37    Shutdown,
38    Unknown {
39        ec: u32,
40        iss: u32,
41    },
42}

44/// Hypervisor trait — implemented by the architecture-specific VMM.
45///
46/// # Law 1 note
47/// This trait and its associated types constitute public API.  Changing
48/// method signatures requires 2× user confirmation once P04 freezes the ABI.
49pub trait ViHypervisor {
50    type Vm;
51    type Vcpu;
52    type Stage2Table;
53
54    fn create_vm(&self) -> ViResult<Self::Vm>;
55    fn create_vcpu(&self, vm: &mut Self::Vm) -> ViResult<Self::Vcpu>;
56    fn map_guest(
57        &self,
58        table: &mut Self::Stage2Table,
59        ipa: u64,
60        hpa: u64,
61        pages: usize,
62        writable: bool,
63    ) -> ViResult<()>;
64    fn run_vcpu(&self, vcpu: &mut Self::Vcpu) -> ViResult<ViVmExit>;
65    fn inject_irq(&self, vcpu: &mut Self::Vcpu, intid: u32) -> ViResult<()>;
66}
