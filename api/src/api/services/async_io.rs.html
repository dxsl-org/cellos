6//! Async I/O traits for non-blocking operations.
7//!
8//! These traits provide async variants of blocking I/O operations,
9//! enabling high-performance concurrent I/O for network servers and
10//! other latency-sensitive applications.
11
12use crate::*;
13use alloc::boxed::Box;
14use core::future::Future;
15use core::pin::Pin;
16use types::ViResult; // Fix import // Fix import
17
18/// Type alias for boxed futures.
19pub type BoxFuture<'a, T> = Pin<Box<dyn Future<Output = T> + Send + 'a>>;
20
21/// Async filesystem interface.
22///
23/// Provides non-blocking file operations for concurrent I/O.
24pub trait ViAsyncFileSystem: Send + Sync {
25    /// Open a file asynchronously.
26    fn open_async(
27        &mut self,
28        path: &str,
29        mode: OpenMode,
30    ) -> BoxFuture<'_, ViResult<Box<dyn ViAsyncFile>>>;
31
32    /// Create a directory asynchronously.
33    fn mkdir_async(&mut self, path: &str) -> BoxFuture<'_, ViResult<()>>;
34
35    /// Remove a file or directory asynchronously.
36    fn remove_async(&mut self, path: &str) -> BoxFuture<'_, ViResult<()>>;
37}
38
39/// Async file interface.
40///
41/// Provides non-blocking read/write operations.
42pub trait ViAsyncFile: Send + Sync {
43    /// Read data asynchronously.
44    fn read_async<'a>(&'a mut self, buf: &'a mut [u8]) -> BoxFuture<'a, ViResult<usize>>;
45
46    /// Write data asynchronously.
47    fn write_async<'a>(&'a mut self, buf: &'a [u8]) -> BoxFuture<'a, ViResult<usize>>;
48
49    /// Seek asynchronously.
50    fn seek_async(&mut self, pos: SeekFrom) -> BoxFuture<'_, ViResult<u64>>;
51}
52
53/// Async TCP stack interface.
54pub trait ViAsyncTcpStack: Send + Sync {
55    /// Connect asynchronously.
56    fn connect_async(&self, addr: IpEndpoint)
57        -> BoxFuture<'_, ViResult<Box<dyn ViAsyncTcpStream>>>;
58
59    /// Listen asynchronously.
60    fn listen_async(&self, port: u16) -> BoxFuture<'_, ViResult<Box<dyn ViAsyncTcpListener>>>;
61}
62
63/// Async TCP stream interface.
64pub trait ViAsyncTcpStream: Send + Sync {
65    /// Read asynchronously.
66    fn read_async<'a>(&'a mut self, buf: &'a mut [u8]) -> BoxFuture<'a, ViResult<usize>>;
67
68    /// Write asynchronously.
69    fn write_async<'a>(&'a mut self, buf: &'a [u8]) -> BoxFuture<'a, ViResult<usize>>;
70
71    /// Close asynchronously.
72    fn close_async(&mut self) -> BoxFuture<'_, ViResult<()>>;
73}
74
75/// Async TCP listener interface.
76pub trait ViAsyncTcpListener: Send + Sync {
77    /// Accept asynchronously.
78    fn accept_async(&self) -> BoxFuture<'_, ViResult<Box<dyn ViAsyncTcpStream>>>;
79}
80
81/// Async block device interface.
82pub trait ViAsyncBlockDevice: Send + Sync {
83    /// Read sector asynchronously.
84    fn read_sector_async<'a>(
85        &'a self,
86        sector: u64,
87        buf: &'a mut [u8],
88    ) -> BoxFuture<'a, ViResult<()>>;
89
90    /// Write sector asynchronously.
91    fn write_sector_async<'a>(&'a self, sector: u64, buf: &'a [u8]) -> BoxFuture<'a, ViResult<()>>;
92
93    /// Flush asynchronously.
94    fn flush_async(&self) -> BoxFuture<'_, ViResult<()>>;
95
96    /// Get sector count (synchronous, metadata only).
97    fn sector_count(&self) -> u64;
98
99    /// Get sector size (synchronous, metadata only).
100    fn sector_size(&self) -> usize;
101}
