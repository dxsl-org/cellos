2//! Cluster-membership types and the [`declare_cluster!`] macro.
3//!
4//! A Cell announces cluster participation by embedding a `__ViCell_cluster`
5//! ELF section (emitted by [`declare_cluster!`]).  The kernel loader reads this
6//! section at spawn time and writes the parsed mode and cluster ID into the TCB.
7//!
8//! ## ClusterId is NOT a credential
9//!
10//! [`ClusterId`] is a **routing / dedup identifier** (FNV-1a-64 of the cluster
11//! name).  It MUST NOT be used for any access decision.  The net-broker's Noise
12//! KKpsk0 handshake (PSK possession) is the sole authentication mechanism.
13//! A `ClusterId` mismatch is a routing reject logged *distinctly* from an auth
14//! failure — never collapse the two, and a routing reject must not leak whether
15//! the PSK would have matched.
16//!
17//! ## RT cells are barred from cluster participation
18//!
19//! A cell spawned with `TaskPriority::RealTime` AND a non-`Isolated`
20//! `__ViCell_cluster` section is rejected at `SpawnPinned` time (the only
21//! place both the priority argument and the loaded `cluster_mode` are
22//! simultaneously available).  RT cells must not enter the cluster.
23
24/// Cluster participation mode for a Cell.
25///
26/// Encoded as a single `u8` at offset 0 of the [`ClusterSection`].
27/// `Isolated` (0) is the default for any Cell that omits the section,
28/// preserving full backwards compatibility.
29#[derive(Debug, Clone, Copy, PartialEq, Eq)]
30#[repr(u8)]
31pub enum ClusterMode {
32    /// Not a cluster participant (default). Full SAS isolation.
33    Isolated = 0,
34    /// Publicly reachable by any cluster peer.
35    Public = 1,
36    /// Private to the named cluster — reachable only by peers sharing the same K1 PSK.
37    Private = 2,
38}
39
40/// Cluster routing identifier — FNV-1a-64 hash of the cluster name string.
41///
42/// ⚠️ **Routing only, NOT a credential.**  A `ClusterId` mismatch is a routing
43/// reject (logged before the Noise handshake) and MUST be logged distinctly from
44/// an authentication failure.  Possession of the correct K1 PSK (proved by
45/// completing the Noise KKpsk0 handshake) is the sole authentication mechanism.
46/// Two different names may theoretically produce the same hash — that costs a
47/// rejected handshake at worst; it is not a security boundary.
48#[derive(Debug, Clone, Copy, PartialEq, Eq)]
49#[repr(transparent)]
50pub struct ClusterId(pub u64);
51
52impl ClusterId {
53    /// Compute the [`ClusterId`] for `name` at **const-evaluation** time.
54    ///
55    /// Algorithm: FNV-1a-64 (offset basis = 14695981039346656037, prime = 1099511628211).
56    /// Stable and deterministic across architectures.
57    pub const fn from_name(name: &str) -> ClusterId {
58        const FNV_OFFSET: u64 = 14_695_981_039_346_656_037;
59        const FNV_PRIME: u64 = 1_099_511_628_211;
60        let bytes = name.as_bytes();
61        let mut hash = FNV_OFFSET;
62        let mut i = 0;
63        while i < bytes.len() {
64            hash ^= bytes[i] as u64;
65            hash = hash.wrapping_mul(FNV_PRIME);
66            i += 1;
67        }
68        ClusterId(hash)
69    }
70
71    /// The zero cluster id — corresponds to `ClusterMode::Isolated`.
72    pub const NONE: ClusterId = ClusterId(0);
73}
74
75/// Fixed 16-byte `#[repr(C)]` layout stored in the `__ViCell_cluster` ELF section.
76///
77/// Written by [`declare_cluster!`] at compile time; read by the kernel loader
78/// at spawn time.  Layout is little-endian on all supported architectures.
79///
80/// ```text
81/// offset  0 :  u8   mode       (0=Isolated  1=Public  2=Private)
82/// offset  1 :  u8   _pad[7]    (zero-fill)
83/// offset  8 :  u64  cluster_id (LE, FNV-1a-64 of the cluster name)
84/// total: 16 bytes
85/// ```
86///
87/// The loader tolerates a section larger than 16 bytes (forward-compat) and
88/// ignores trailing bytes.
89#[repr(C)]
90pub struct ClusterSection {
91    pub mode: u8,
92    pub _pad: [u8; 7],
93    /// Little-endian cluster routing id.  See [`ClusterId`] docs on routing vs. auth.
94    pub cluster_id: u64,
95}
96
97impl ClusterSection {
98    /// Build a `ClusterSection` for `mode` and cluster `name` at **const-eval** time.
99    pub const fn new(mode: ClusterMode, name: &str) -> ClusterSection {
100        let id = ClusterId::from_name(name);
101        ClusterSection {
102            mode: mode as u8,
103            _pad: [0u8; 7],
104            cluster_id: id.0,
105        }
106    }
107}

109// ── CellNetId ─────────────────────────────────────────────────────────────────
110
111/// Stable per-machine network identity — 32-byte public key.
112///
113/// G1: derived from the X25519 Noise static keypair (no separate Ed25519 keygen).
114/// G2: proper Ed25519 public key with signing capability.
115///
116/// Bound into the Noise prologue so both machines must agree on each other's
117/// NodeId during the handshake — prevents routing spoofing (FATAL-1 fix).
118#[derive(Debug, Clone, Copy, PartialEq, Eq)]
119#[repr(transparent)]
120pub struct CellNetId(pub [u8; 32]);
121
122impl CellNetId {
123    pub const ZERO: Self = Self([0u8; 32]);
124    pub const fn from_bytes(b: [u8; 32]) -> Self {
125        Self(b)
126    }
127    pub fn as_bytes(&self) -> &[u8; 32] {
128        &self.0
129    }
130}

132// ── PeerTicket ────────────────────────────────────────────────────────────────
133
134/// Peer address book — how to connect to one remote Cellos machine.
135///
136/// Fixed-size, IPv4-only. No hostname fields (DNS resolver absent in G1).
137/// Max encoded size: 57 bytes (fits comfortably in IPC_BUF_SIZE).
138#[derive(Debug, Clone, Copy)]
139pub struct PeerTicket {
140    pub node_id: CellNetId,
141    pub relay_ip: [u8; 4],
142    pub relay_port: u16,
143    /// Direct IPv4:port addrs (LAN address + STUN reflexive, up to 3).
144    pub addrs: [([u8; 4], u16); 3],
145    pub addrs_len: u8,
146}
147
148impl PeerTicket {
149    pub const ENCODED_MAX: usize = 32 + 4 + 2 + 1 + 3 * 6; // = 57
150
151    pub fn encode(&self, out: &mut [u8]) -> usize {
152        assert!(out.len() >= Self::ENCODED_MAX);
153        let mut p = 0;
154        out[p..p + 32].copy_from_slice(&self.node_id.0);
155        p += 32;
156        out[p..p + 4].copy_from_slice(&self.relay_ip);
157        p += 4;
158        out[p..p + 2].copy_from_slice(&self.relay_port.to_le_bytes());
159        p += 2;
160        out[p] = self.addrs_len;
161        p += 1;
162        for i in 0..self.addrs_len as usize {
163            out[p..p + 4].copy_from_slice(&self.addrs[i].0);
164            p += 4;
165            out[p..p + 2].copy_from_slice(&self.addrs[i].1.to_le_bytes());
166            p += 2;
167        }
168        p
169    }
170
171    pub fn decode(data: &[u8]) -> Option<Self> {
172        if data.len() < 39 {
173            return None;
174        }
175        let mut node_id = [0u8; 32];
176        node_id.copy_from_slice(&data[0..32]);
177        let mut relay_ip = [0u8; 4];
178        relay_ip.copy_from_slice(&data[32..36]);
179        let relay_port = u16::from_le_bytes([data[36], data[37]]);
180        let addrs_len = (data[38] as usize).min(3) as u8;
181        let mut addrs = [([0u8; 4], 0u16); 3];
182        let mut p = 39;
183        for addr in addrs.iter_mut().take(addrs_len as usize) {
184            if p + 6 > data.len() {
185                break;
186            }
187            let mut ip = [0u8; 4];
188            ip.copy_from_slice(&data[p..p + 4]);
189            let port = u16::from_le_bytes([data[p + 4], data[p + 5]]);
190            *addr = (ip, port);
191            p += 6;
192        }
193        Some(Self {
194            node_id: CellNetId(node_id),
195            relay_ip,
196            relay_port,
197            addrs,
198            addrs_len,
199        })
200    }
201}

203/// Emit the `__ViCell_cluster` ELF section declaring this Cell's cluster mode.
204///
205/// The section is read by the kernel loader at spawn time.  Cells that omit
206/// this macro default to `ClusterMode::Isolated` and are unaffected by cluster
207/// operations.
208///
209/// # Example
210///
211/// ```ignore
212/// api::declare_cluster!(mode = Private, name = "robots");
213/// ```
214///
215/// `mode` must be one of `Isolated`, `Public`, or `Private`.
216/// `name` is the cluster name string used to derive the [`ClusterId`].
217#[macro_export]
218macro_rules! declare_cluster {
219    (mode = $mode:ident, name = $name:literal) => {
220        #[used]
221        #[link_section = "__ViCell_cluster"]
222        pub static VICELL_CLUSTER: $crate::cluster::ClusterSection =
223            $crate::cluster::ClusterSection::new($crate::cluster::ClusterMode::$mode, $name);
224    };
225}
