45// ---------------------------------------------------------------------------
46// Helper: extract fd from a FILE*, handling both our FILE structs and
47// foreign picolibc FILE* (which has _flags at offset 0, not fd).
48// ---------------------------------------------------------------------------
49unsafe fn fd_of(stream: *mut FILE) -> i32 {
50    if stream.is_null() {
51        return 1;
52    }
53    if core::ptr::eq(stream, core::ptr::addr_of!(STDOUT_FILE) as *const _) {
54        return 1;
55    }
56    if core::ptr::eq(stream, core::ptr::addr_of!(STDERR_FILE) as *const _) {
57        return 2;
58    }
59    if core::ptr::eq(stream, core::ptr::addr_of!(STDIN_FILE) as *const _) {
60        return 0;
61    }
62    let raw = (*stream).fd;
63    if raw > 0 {
64        raw
65    } else {
66        2
67    }
68}

74/// Formats `fmt` with `args` into `buf`, writing at most `size` bytes (including
75/// the NUL terminator).
76///
77/// # Safety
78/// `buf` must be null (with `size == 0`) or valid for writes of at least `size`
79/// bytes; `fmt` must point to a valid NUL-terminated C string; `args` must be a
80/// `VaList` whose remaining arguments match the conversion specifiers in `fmt`.
81#[no_mangle]
82pub unsafe extern "C" fn vsnprintf(
83    buf: *mut c_char,
84    size: usize,
85    fmt: *const c_char,
86    args: VaList<'_>,
87) -> c_int {
88    vsnprintf_core(buf, size, fmt, args) as c_int
89}
90
91/// Formats `fmt` with `args` into `buf` with no length limit (C `sprintf` semantics).
92///
93/// # Safety
94/// `buf` must be valid for writes large enough to hold the entire formatted
95/// output plus a NUL terminator — the caller is responsible for sizing it, as
96/// this function performs no bounds checking beyond an internal cap; `fmt` must
97/// point to a valid NUL-terminated C string; `args` must match `fmt`'s specifiers.
98#[no_mangle]
99pub unsafe extern "C" fn vsprintf(buf: *mut c_char, fmt: *const c_char, args: VaList<'_>) -> c_int {
100    vsnprintf_core(buf, usize::MAX / 2, fmt, args) as c_int
101}

103/// Formats `fmt` with `args` and writes the result to stdout (fd 1).
104///
105/// # Safety
106/// `fmt` must point to a valid NUL-terminated C string; `args` must be a
107/// `VaList` whose remaining arguments match the conversion specifiers in `fmt`.
108#[no_mangle]
109pub unsafe extern "C" fn vprintf(fmt: *const c_char, args: VaList<'_>) -> c_int {
110    let mut tmp = [0u8; 1024];
111    let n = vsnprintf_core(tmp.as_mut_ptr(), tmp.len(), fmt, args);
112    _write(1, tmp.as_ptr() as *const c_void, n.min(tmp.len()));
113    n as c_int
114}

116/// Formats `fmt` with `args` and writes the result to `stream`.
117///
118/// # Safety
119/// `stream` must be null or a valid `FILE*` (one of `stdout`/`stderr`/`stdin`
120/// or a pointer previously returned by [`fopen`]), since it is dereferenced by
121/// [`fd_of`]; `fmt` must point to a valid NUL-terminated C string; `args` must
122/// match `fmt`'s specifiers.
123#[no_mangle]
124pub unsafe extern "C" fn vfprintf(
125    stream: *mut FILE,
126    fmt: *const c_char,
127    args: VaList<'_>,
128) -> c_int {
129    let fd = fd_of(stream);
130    let mut tmp = [0u8; 1024];
131    let n = vsnprintf_core(tmp.as_mut_ptr(), tmp.len(), fmt, args);
132    _write(fd, tmp.as_ptr() as *const c_void, n.min(tmp.len()));
133    n as c_int
134}

196/// Opens a file. Returns a heap-allocated FILE* on success, NULL on failure.
197///
198/// # Safety
199/// `path` must point to a valid NUL-terminated C string; `mode` must be null
200/// or point to a valid NUL-terminated C string (only its first byte is read).
201#[no_mangle]
202pub unsafe extern "C" fn fopen(path: *const c_char, mode: *const c_char) -> *mut FILE {
203    let flags = if !mode.is_null() && *mode == b'w' as c_char {
204        0x201
205    } else {
206        0
207    };
208    let fd = _open(path, flags, 0o644);
209    if fd < 0 {
210        return core::ptr::null_mut();
211    }
212    let f = super::alloc::malloc_impl(core::mem::size_of::<FILE>()) as *mut FILE;
213    if f.is_null() {
214        _close(fd);
215        return core::ptr::null_mut();
216    }
217    (*f).fd = fd;
218    (*f)._pad = [0; 28];
219    f
220}

222/// Closes `stream`, freeing it if it was heap-allocated by [`fopen`].
223///
224/// # Safety
225/// `stream` must be null or a valid `FILE*` (one of `stdout`/`stderr`/`stdin`
226/// or a pointer previously returned by [`fopen`] and not yet closed) — it is
227/// dereferenced by [`fd_of`] and, if heap-allocated, passed to `free_impl`.
228#[no_mangle]
229pub unsafe extern "C" fn fclose(stream: *mut FILE) -> c_int {
230    if stream.is_null() {
231        return -1;
232    }
233    let fd = fd_of(stream);
234    // Only free heap-allocated FILE* (opened via fopen); never free static streams.
235    if fd > 2 {
236        super::alloc::free_impl(stream as *mut c_void);
237    }
238    _close(fd)
239}

241/// Reads up to `size * nmemb` bytes from `stream` into `ptr`, looping until
242/// the full amount is read or the underlying `_read` returns EOF/error.
243///
244/// # Safety
245/// `ptr` must be null or valid for writes of at least `size * nmemb` bytes;
246/// `stream` must be null or a valid `FILE*` (dereferenced by [`fd_of`]).
247#[no_mangle]
248pub unsafe extern "C" fn fread(
249    ptr: *mut c_void,
250    size: usize,
251    nmemb: usize,
252    stream: *mut FILE,
253) -> usize {
254    if stream.is_null() || ptr.is_null() {
255        return 0;
256    }
257    let total = size.saturating_mul(nmemb);
258    let fd = fd_of(stream);
259    // Loop to match C standard fread semantics: read until all bytes received
260    // or EOF/error. A single _read may return fewer bytes than requested.
261    let mut done = 0usize;
262    while done < total {
263        let n = _read(fd, (ptr as *mut u8).add(done) as *mut c_void, total - done);
264        if n <= 0 {
265            break;
266        }
267        done += n as usize;
268    }
269    done / size.max(1)
270}

272/// Writes `size * nmemb` bytes from `ptr` to `stream`.
273///
274/// # Safety
275/// `ptr` must be null or valid for reads of at least `size * nmemb` bytes;
276/// `stream` must be null or a valid `FILE*` (dereferenced by [`fd_of`]).
277#[no_mangle]
278pub unsafe extern "C" fn fwrite(
279    ptr: *const c_void,
280    size: usize,
281    nmemb: usize,
282    stream: *mut FILE,
283) -> usize {
284    if stream.is_null() || ptr.is_null() {
285        return 0;
286    }
287    let total = size.saturating_mul(nmemb);
288    let n = _write(fd_of(stream), ptr, total);
289    if n <= 0 {
290        0
291    } else {
292        n as usize / size.max(1)
293    }
294}

310/// Writes the single byte `c` to `stream`.
311///
312/// # Safety
313/// `stream` must be null or a valid `FILE*` (dereferenced by [`fd_of`]).
314#[no_mangle]
315pub unsafe extern "C" fn fputc(c: c_int, stream: *mut FILE) -> c_int {
316    let b = [c as u8];
317    if _write(fd_of(stream), b.as_ptr() as *const c_void, 1) == 1 {
318        c
319    } else {
320        -1
321    }
322}

324/// Reads a single byte from `stream`, returning it as an unsigned `c_int` or
325/// `-1` on EOF/error.
326///
327/// # Safety
328/// `stream` must be null or a valid `FILE*` (dereferenced by [`fd_of`]).
329#[no_mangle]
330pub unsafe extern "C" fn fgetc(stream: *mut FILE) -> c_int {
331    let mut b = 0u8;
332    // For reads, fd_of returns 2 for unknown streams, but 0 (stdin) for stdin.
333    let fd = if stream.is_null() { 0 } else { fd_of(stream) };
334    if _read(fd, &mut b as *mut u8 as *mut c_void, 1) == 1 {
335        b as c_int
336    } else {
337        -1
338    }
339}

341/// Reads up to `n - 1` bytes from `stream` into `buf` and NUL-terminates the result.
342///
343/// # Safety
344/// `buf` must be valid for writes of at least `n` bytes when `n > 0`; `stream`
345/// must be null or a valid `FILE*` (dereferenced by [`fd_of`]).
346#[no_mangle]
347pub unsafe extern "C" fn fgets(buf: *mut c_char, n: c_int, stream: *mut FILE) -> *mut c_char {
348    if buf.is_null() || n <= 0 {
349        return core::ptr::null_mut();
350    }
351    let fd = if stream.is_null() { 0 } else { fd_of(stream) };
352    let r = _read(fd, buf as *mut c_void, (n - 1) as usize);
353    if r <= 0 {
354        return core::ptr::null_mut();
355    }
356    *buf.add(r as usize) = 0;
357    buf
358}

390/// fseek: seek fd to `offset` from `whence` (SEEK_SET=0, SEEK_CUR=1, SEEK_END=2).
391/// We implement this ourselves so our simple FILE* (fd at offset 0) is used
392/// correctly — picolibc's fseek reads _file from a different offset and would
393/// call lseek(0, ...) instead of the real fd.
394/// Seeks `stream` to `offset` from `whence` (SEEK_SET=0, SEEK_CUR=1, SEEK_END=2).
395///
396/// # Safety
397/// `stream` must be null or a valid `FILE*` (dereferenced by [`fd_of`]).
398#[no_mangle]
399pub unsafe extern "C" fn fseek(stream: *mut FILE, offset: c_long, whence: c_int) -> c_int {
400    if stream.is_null() {
401        return -1;
402    }
403    let fd = fd_of(stream);
404    let r = super::sysio::_lseek(fd, offset, whence);
405    if r < 0 {
406        -1
407    } else {
408        0
409    }
410}

412/// Returns the current seek position of `stream`.
413///
414/// # Safety
415/// `stream` must be null or a valid `FILE*` (dereferenced by [`fd_of`]).
416#[no_mangle]
417pub unsafe extern "C" fn ftell(stream: *mut FILE) -> c_long {
418    if stream.is_null() {
419        return -1;
420    }
421    super::sysio::_lseek(fd_of(stream), 0, 1) // SEEK_CUR
422}
423
424/// Resets `stream`'s seek position to the beginning.
425///
426/// # Safety
427/// `stream` must be null or a valid `FILE*` (dereferenced by [`fd_of`]).
428#[no_mangle]
429pub unsafe extern "C" fn rewind(stream: *mut FILE) {
430    if stream.is_null() {
431        return;
432    }
433    super::sysio::_lseek(fd_of(stream), 0, 0); // SEEK_SET
434}
