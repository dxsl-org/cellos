10/// Fill `buf` with `buflen` cryptographically random bytes via VirtIO-RNG.
11///
12/// Maps to `sys_get_random(214)`. Returns 0 on success, −1 if the device is
13/// absent or `buflen` exceeds the 256-byte POSIX limit.
14///
15/// # Safety
16/// `buf` must be either null or point to at least `buflen` writable bytes
17/// for the duration of the call.
18#[no_mangle]
19pub unsafe extern "C" fn getentropy(buf: *mut c_void, buflen: usize) -> i32 {
20    if buf.is_null() || buflen > 256 {
21        return -1;
22    }
23    let mut written = 0usize;
24    let dst = buf as *mut u8;
25    while written < buflen {
26        let chunk = (buflen - written).min(64);
27        let ret = raw_syscall(ViSyscall::GetRandom, dst.add(written) as usize, chunk, 0, 0);
28        if ret <= 0 {
29            return -1;
30        }
31        written += ret as usize;
32    }
33    0
34}

36/// arc4random_buf — fills `buf` with `n` random bytes; no return value.
37///
38/// # Safety
39/// `buf` must be either null or point to at least `n` writable bytes for
40/// the duration of the call (only up to 256 of them are actually written,
41/// per `getentropy`'s POSIX limit).
42#[no_mangle]
43pub unsafe extern "C" fn arc4random_buf(buf: *mut c_void, n: usize) {
44    getentropy(buf, n.min(256));
45}
