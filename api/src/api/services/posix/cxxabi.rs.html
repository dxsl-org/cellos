2//! Minimal C++ ABI stubs for freestanding C++ (no exceptions, no RTTI).
3//!
4//! SAFETY: These stubs ONLY support -fno-exceptions -fno-rtti compiled C++ code.
5//! Do NOT attempt to use exceptions, RTTI, or STL containers — they require
6//! full libcxxabi/libstdc++ which is SAS-unsafe. Use Tier 3b Linux VM instead.
7
8#![allow(unsafe_code)]
9
10use super::sysio::raw_syscall;
11use crate::syscall::ViSyscall;
12
13/// Called when a pure virtual function is invoked (programming error).
14/// Terminates the cell immediately — equivalent to a Rust panic.
15///
16/// # Safety
17/// Must only be reached via a compiler-generated vtable thunk for an
18/// unoverridden pure virtual slot; never call directly. Never returns.
19#[no_mangle]
20pub unsafe extern "C" fn __cxa_pure_virtual() -> ! {
21    raw_syscall(
22        ViSyscall::Log,
23        b"FATAL: pure virtual call\n".as_ptr() as usize,
24        25,
25        0,
26        0,
27    );
28    raw_syscall(ViSyscall::Exit, 134, 0, 0, 0); // 134 = 128 + SIGABRT(6)
29    loop {
30        core::hint::spin_loop();
31    }
32}

34/// Thread-safe static local init guard — single-threaded stub.
35/// C++ emits __cxa_guard_acquire/release around function-local statics.
36/// In ViCell's single-threaded cells, a simple flag suffices.
37///
38/// # Safety
39/// `guard` must be a valid, properly aligned pointer to a `u64` owned by the
40/// enclosing function-local static's compiler-generated guard variable, and
41/// must not be concurrently accessed (single-threaded cells only).
42#[no_mangle]
43pub unsafe extern "C" fn __cxa_guard_acquire(guard: *mut u64) -> i32 {
44    if *guard == 0 {
45        1
46    } else {
47        0
48    } // 1 = needs init, 0 = already done
49}
50
51/// Marks the function-local static guarded by `guard` as initialized.
52///
53/// # Safety
54/// `guard` must be the same pointer previously passed to a matching
55/// `__cxa_guard_acquire` call and must still be valid and properly aligned.
56#[no_mangle]
57pub unsafe extern "C" fn __cxa_guard_release(guard: *mut u64) {
58    *guard = 1; // mark initialized
59}
60
61/// Signals that function-local static initialization failed, leaving the
62/// guard clear so the next execution retries initialization.
63///
64/// # Safety
65/// `_guard` must be the same pointer previously passed to a matching
66/// `__cxa_guard_acquire` call and must still be valid and properly aligned.
67#[no_mangle]
68pub unsafe extern "C" fn __cxa_guard_abort(_guard: *mut u64) {
69    // Init failed — leave guard at 0 so next attempt retries
70}

72/// abort() — terminates cell immediately. No cleanup, no atexit handlers.
73/// This is the correct behavior for SAS: kernel reclaims all resources.
74///
75/// # Safety
76/// Callable from any context; never returns. Caller-held resources are
77/// reclaimed by the kernel on cell exit, not by this function.
78#[no_mangle]
79pub unsafe extern "C" fn abort() -> ! {
80    raw_syscall(ViSyscall::Exit, 134, 0, 0, 0);
81    loop {
82        core::hint::spin_loop();
83    }
84}
