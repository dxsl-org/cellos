8use core::alloc::Layout;
9use core::ffi::c_void;
10use core::ptr;

12#[repr(C)]
13struct AllocHeader {
14    size: usize,
15    magic: usize,
16}
17
18const HEADER_MAGIC: usize = 0xDEAD_C0DE;
19const HEADER_SIZE: usize = core::mem::size_of::<AllocHeader>();
20const ALIGN: usize = 16;
21
22/// Allocates a block of at least `size` bytes, 16-byte aligned.
23///
24/// # Safety
25/// The returned pointer must eventually be released via `free` or `realloc`
26/// from this same allocator (never via the global Rust allocator or a
27/// different `malloc` implementation), and must not be freed more than once.
28#[no_mangle]
29pub unsafe extern "C" fn malloc(size: usize) -> *mut c_void {
30    malloc_impl(size)
31}
32
33/// Releases a block previously allocated by `malloc`, `calloc`, or `realloc`.
34///
35/// # Safety
36/// `ptr` must be null or a pointer previously returned by this allocator's
37/// `malloc`/`calloc`/`realloc` family, must not already have been freed, and
38/// must not be accessed by the caller after this call returns.
39#[no_mangle]
40pub unsafe extern "C" fn free(ptr: *mut c_void) {
41    free_impl(ptr)
42}
43
44/// Resizes a previously allocated block, preserving its contents up to
45/// `min(old_size, size)` bytes; a `size` of 0 behaves like `free`.
46///
47/// # Safety
48/// `ptr` must be null or a pointer previously returned by this allocator's
49/// `malloc`/`calloc`/`realloc` family and not already freed. On success the
50/// caller must stop using `ptr` and use only the returned pointer.
51#[no_mangle]
52pub unsafe extern "C" fn realloc(ptr: *mut c_void, size: usize) -> *mut c_void {
53    realloc_impl(ptr, size)
54}
55
56/// Allocates zero-initialized storage for `nmemb` elements of `size` bytes each.
57///
58/// # Safety
59/// The returned pointer must eventually be released via `free` or `realloc`
60/// from this same allocator, and must not be freed more than once. Overflow
61/// of `nmemb * size` is detected internally and yields a null pointer.
62#[no_mangle]
63pub unsafe extern "C" fn calloc(nmemb: usize, size: usize) -> *mut c_void {
64    let total_size = match nmemb.checked_mul(size) {
65        Some(s) => s,
66        None => return ptr::null_mut(),
67    };
68    let p = malloc(total_size);
69    if !p.is_null() {
70        core::ptr::write_bytes(p as *mut u8, 0, total_size);
71    }
72    p
73}
74
75pub(super) unsafe fn malloc_impl(size: usize) -> *mut c_void {
76    let total = match size.checked_add(HEADER_SIZE) {
77        Some(s) => s,
78        None => return ptr::null_mut(),
79    };
80    let layout = match Layout::from_size_align(total, ALIGN) {
81        Ok(l) => l,
82        Err(_) => return ptr::null_mut(),
83    };
84    let raw = alloc::alloc::alloc(layout);
85    if raw.is_null() {
86        return ptr::null_mut();
87    }
88    let header = raw as *mut AllocHeader;
89    (*header).size = size;
90    (*header).magic = HEADER_MAGIC;
91    raw.add(HEADER_SIZE) as *mut c_void
92}
93
94pub(super) unsafe fn free_impl(ptr: *mut c_void) {
95    if ptr.is_null() {
96        return;
97    }
98    let raw = (ptr as *mut u8).sub(HEADER_SIZE);
99    let header = raw as *mut AllocHeader;
100    if (*header).magic != HEADER_MAGIC {
101        return;
102    }
103    let total = (*header).size + HEADER_SIZE;
104    let layout = Layout::from_size_align_unchecked(total, ALIGN);
105    alloc::alloc::dealloc(raw, layout);
106}
107
108unsafe fn realloc_impl(ptr: *mut c_void, new_size: usize) -> *mut c_void {
109    if ptr.is_null() {
110        return malloc_impl(new_size);
111    }
112    if new_size == 0 {
113        free_impl(ptr);
114        return ptr::null_mut();
115    }
116    let raw = (ptr as *mut u8).sub(HEADER_SIZE);
117    let header = raw as *mut AllocHeader;
118    if (*header).magic != HEADER_MAGIC {
119        return ptr::null_mut();
120    }
121    let old_size = (*header).size;
122    let old_layout = Layout::from_size_align_unchecked(old_size + HEADER_SIZE, ALIGN);
123    let total_new = match new_size.checked_add(HEADER_SIZE) {
124        Some(s) => s,
125        None => return ptr::null_mut(),
126    };
127    let new_raw = alloc::alloc::realloc(raw, old_layout, total_new);
128    if new_raw.is_null() {
129        return ptr::null_mut();
130    }
131    let new_header = new_raw as *mut AllocHeader;
132    (*new_header).size = new_size;
133    (*new_header).magic = HEADER_MAGIC;
134    new_raw.add(HEADER_SIZE) as *mut c_void
135}

141/// C++ `operator new(size_t)` — same allocation semantics as `malloc`.
142///
143/// # Safety
144/// The returned pointer must eventually be released via `_ZdlPv`/`_ZdlPvm`
145/// (`operator delete`), never via `free` or a mismatched deallocator, and
146/// must not be freed more than once.
147#[no_mangle]
148pub unsafe extern "C" fn _Znwm(size: usize) -> *mut c_void {
149    malloc_impl(size)
150}
151
152/// C++ `operator delete(void*)` — same semantics as `free`.
153///
154/// # Safety
155/// `ptr` must be null or a pointer previously returned by `_Znwm`, must not
156/// already have been freed, and must not be accessed after this call.
157#[no_mangle]
158pub unsafe extern "C" fn _ZdlPv(ptr: *mut c_void) {
159    free_impl(ptr)
160}
161
162/// C++ sized `operator delete(void*, size_t)`; the size is compiler-provided
163/// bookkeeping and is unused here since the allocator header already tracks it.
164///
165/// # Safety
166/// `ptr` must be null or a pointer previously returned by `_Znwm`, must not
167/// already have been freed, and `_size` must match the size originally
168/// requested from `_Znwm`.
169#[no_mangle]
170pub unsafe extern "C" fn _ZdlPvm(ptr: *mut c_void, _size: usize) {
171    free_impl(ptr)
172}
