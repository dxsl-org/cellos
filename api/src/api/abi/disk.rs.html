1//! Canonical MBR partition layout for `disk_v3.img` — the contract shared by
2//! the kernel (`loader/disk_layout.rs`), cells (VFS block stream, littlefs
3//! backend) and the image builders (`tools/write-mbr.py`, `gen_disk.ps1`,
4//! `tools/mkfat32_inplace.py` — the Python tools carry a must-match copy).
5//!
6//! The on-disk MBR at LBA 0 is the runtime source of truth; the kernel parses
7//! and cross-checks it at boot (`disk_layout::verify_mbr`, warn-only). These
8//! constants stay authoritative so a blank or legacy image still boots.
9//!
10//! ```text
11//! P1  0x0C  FAT32 interop volume      (VFS /data today, /mnt/sd after P04)
12//! P2  0x7F  cell bootstrap table      (kernel early loader ONLY — never granted)
13//! P3  0x7D  kernel heap snapshot      (Phase 29 — kernel ONLY, never granted)
14//! P4  0x7E  littlefs /data            (Milestone 2.5 Phase 04)
15//! ```
16
17/// P1: first absolute LBA of the FAT32 volume (MBR-standard 1 MiB alignment).
18pub const PART_FAT32_BASE_LBA: u64 = 2_048;
19/// P1: FAT32 volume size in sectors.
20pub const PART_FAT32_SECTORS: u64 = 524_288;
21
22/// P2: first LBA of the cell bootstrap table (header + entries + ELF blobs).
23pub const PART_CELLTBL_BASE_LBA: u64 = 526_336;
24/// P2 size in sectors (16 MiB).
25pub const PART_CELLTBL_SECTORS: u64 = 33_664;
26
27/// P3: kernel heap snapshot region (raw LBA, no filesystem).
28pub const PART_SNAPSHOT_BASE_LBA: u64 = 560_000;
29/// P3 size in sectors (~117 MB — covers the full 64 MB kernel heap + headers).
30pub const PART_SNAPSHOT_SECTORS: u64 = 240_000;
31
32/// P4: littlefs persistent /data store.
33pub const PART_LFS_BASE_LBA: u64 = 800_000;
34/// P4 size in sectors (64 MB).
35pub const PART_LFS_SECTORS: u64 = 131_072;
36
37/// P5: RedoxFS native filesystem — VFS `/srv` (G1+, spec §09b).
38///
39/// Follows immediately after P4.  Pre-formatted offline with `redoxfs-mkfs`;
40/// opened at runtime with `FileSystem::open()` (no_std, read-write).
41/// Partition-capability gating (`MANIFEST_FLAG_PART_SRV`) requires expanding
42/// `flags` to `u16` in a future manifest version; until then, `/srv` access is
43/// controlled exclusively by the VFS `AccessTable`.
44pub const PART_SRV_BASE_LBA: u64 = 931_072; // LBA 800_000 + 131_072
45/// P5 size in sectors (64 MB RedoxFS volume).
46pub const PART_SRV_SECTORS: u64 = 131_072;
47
48/// P6: FAT cell-store — non-bootstrap cell ELFs served under VFS `/bin`
49/// (G2 loader redesign). Follows P5; constant-addressed (no MBR slot, like P5).
50/// Read-only at runtime: the VFS `/bin` overlay reads it after a VIFS1 miss.
51/// gen_disk formats + populates it with the signed P2-only cell ELFs.
52pub const PART_CELLSTORE_BASE_LBA: u64 = 1_062_144; // LBA 931_072 + 131_072
53/// P6 size in sectors (32 MB — ample for the P2-only cell set).
54pub const PART_CELLSTORE_SECTORS: u64 = 65_536;
