1//! `declare_manifest!` — embeds a [`super::manifest::CellManifest`] into the
2//! current Cell's `__ViCell_manifest` ELF section.  Split out of `manifest.rs`
3//! to keep that file under the 200-LOC law.
4
5/// Embed a capability manifest into the current Cell's ELF binary.
6///
7/// Places a fixed 16-byte `CellManifest` (v2) into the `__ViCell_manifest` ELF
8/// section.  The cell linker script must `KEEP` that section or `--gc-sections`
9/// will silently drop it in release/LTO builds.
10///
11/// The back-compat forms default `tier = TIER_LEGACY` (v1 behaviour preserved).
12/// Add `tier = <TIER_*>` to opt a cell into an explicit isolation domain (e.g. a
13/// Tier-1b C/FFI cell requesting `tier = 2` for PKU key 2).
14#[macro_export]
15macro_rules! declare_manifest {
16    // Full form + explicit tier (v2 opt-in).
17    (block_io = $bio:literal, network = $net:literal, spawn = $spawn:literal, gpio = $gpio:literal, uart = $uart:literal, hypervisor = $hv:literal, part_data = $pd:literal, part_lfs = $pl:literal, can = $can:literal, adc = $adc:literal, tier = $tier:expr) => {
18        #[used]
19        #[link_section = "__ViCell_manifest"]
20        pub static VICELL_MANIFEST: $crate::manifest::CellManifest =
21            $crate::manifest::CellManifest::with_all(
22                $bio, $net, $spawn, $gpio, $uart, $hv, $pd, $pl, $can, $adc, $tier,
23            );
24    };
25    // Full form — block-I/O partition range grants (tier defaults to LEGACY).
26    (block_io = $bio:literal, network = $net:literal, spawn = $spawn:literal, gpio = $gpio:literal, uart = $uart:literal, hypervisor = $hv:literal, part_data = $pd:literal, part_lfs = $pl:literal) => {
27        #[used]
28        #[link_section = "__ViCell_manifest"]
29        pub static VICELL_MANIFEST: $crate::manifest::CellManifest =
30            $crate::manifest::CellManifest::with_parts(
31                $bio, $net, $spawn, $gpio, $uart, $hv, $pd, $pl,
32            );
33    };
34    // Convenience form: block_io with partition scopes, no gpio/uart/hypervisor.
35    (block_io = $bio:literal, network = $net:literal, spawn = $spawn:literal, part_data = $pd:literal, part_lfs = $pl:literal) => {
36        $crate::declare_manifest!(
37            block_io = $bio,
38            network = $net,
39            spawn = $spawn,
40            gpio = false,
41            uart = false,
42            hypervisor = false,
43            part_data = $pd,
44            part_lfs = $pl
45        );
46    };
47    // 3-cap form + explicit tier (v2 opt-in), no gpio/uart/hypervisor/parts/can/adc.
48    // The pairing `app_entry!` uses for a plain Rust-cell app that wants an
49    // explicit isolation domain (e.g. a Tier-1b C/FFI cell requesting tier=2).
50    (block_io = $bio:literal, network = $net:literal, spawn = $spawn:literal, tier = $tier:expr) => {
51        $crate::declare_manifest!(
52            block_io = $bio,
53            network = $net,
54            spawn = $spawn,
55            gpio = false,
56            uart = false,
57            hypervisor = false,
58            part_data = false,
59            part_lfs = false,
60            can = false,
61            adc = false,
62            tier = $tier
63        );
64    };
65    // 6-param form — includes hypervisor flag.
66    (block_io = $bio:literal, network = $net:literal, spawn = $spawn:literal, gpio = $gpio:literal, uart = $uart:literal, hypervisor = $hv:literal) => {
67        #[used]
68        #[link_section = "__ViCell_manifest"]
69        pub static VICELL_MANIFEST: $crate::manifest::CellManifest =
70            $crate::manifest::CellManifest::new($bio, $net, $spawn, $gpio, $uart, $hv);
71    };
72    // 5-param form (no hypervisor) — hypervisor defaults to false.
73    (block_io = $bio:literal, network = $net:literal, spawn = $spawn:literal, gpio = $gpio:literal, uart = $uart:literal) => {
74        $crate::declare_manifest!(
75            block_io = $bio,
76            network = $net,
77            spawn = $spawn,
78            gpio = $gpio,
79            uart = $uart,
80            hypervisor = false
81        );
82    };
83    // 3-param back-compat form (no gpio/uart/hypervisor) — all default to false.
84    (block_io = $bio:literal, network = $net:literal, spawn = $spawn:literal) => {
85        $crate::declare_manifest!(
86            block_io = $bio,
87            network = $net,
88            spawn = $spawn,
89            gpio = false,
90            uart = false,
91            hypervisor = false
92        );
93    };
94}
