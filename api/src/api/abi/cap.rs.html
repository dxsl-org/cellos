1//! Capability types for ViCell.
2//!
3//! A capability is an opaque, unforgeable token assigned by the kernel.
4//! Holding a `CapId` grants the owner a bounded set of operations on a
5//! specific resource (file, socket, surface, …).  The kernel validates
6//! ownership on every use — callers cannot forge or guess valid IDs.
7
8/// Opaque kernel-assigned capability identifier.
9///
10/// Values are assigned by the kernel cap registry and are unique per session.
11/// `CapId(0)` is the null / invalid sentinel — the registry never assigns it.
12#[repr(transparent)]
13#[derive(Copy, Clone, Eq, PartialEq, Ord, PartialOrd, Hash, Debug, Default)]
14pub struct CapId(pub u64);
15
16impl CapId {
17    /// The invalid/null capability sentinel.
18    pub const INVALID: Self = Self(0);
19
20    /// Returns `true` if this is the null sentinel.
21    pub fn is_invalid(self) -> bool {
22        self.0 == 0
23    }
24}
25
26/// Permissions encoded in a capability.
27///
28/// Packed as a `u32` for compact IPC transfer.
29#[derive(Copy, Clone, Eq, PartialEq, Debug, Default)]
30pub struct CapPerms(pub u32);
31
32impl CapPerms {
33    pub const READ: Self = Self(1 << 0);
34    pub const WRITE: Self = Self(1 << 1);
35    pub const SEEK: Self = Self(1 << 2);
36
37    /// Read + Seek (standard file-read access).
38    pub const FILE_READ: Self = Self(Self::READ.0 | Self::SEEK.0);
39    /// Read + Write + Seek (read-write file access).
40    pub const FILE_RW: Self = Self(Self::READ.0 | Self::WRITE.0 | Self::SEEK.0);
41
42    pub fn has(self, perm: Self) -> bool {
43        (self.0 & perm.0) == perm.0
44    }
45}
