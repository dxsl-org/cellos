1//! Constants for [`super::manifest::CellManifest`] — magic/version, isolation
2//! tiers, and capability flag bits.  Split out of `manifest.rs` to keep that file
3//! under the 200-LOC law; re-exported from `manifest.rs` so callers see one
4//! surface (`api::manifest::MANIFEST_FLAG_*` etc. keep working unchanged).
5
6/// Magic value identifying a valid manifest (`0x5649_4345`, "VICE" as a u32).
7pub const MANIFEST_MAGIC: u32 = 0x5649_4345;
8
9/// Current manifest layout version.  Bump on any field addition or reorder.
10pub const MANIFEST_VERSION: u8 = 2;
11
12/// The v1 layout version (8-byte record).  Recognised by `from_bytes` for upcast.
13pub const MANIFEST_VERSION_V1: u8 = 1;
14
15// ─── Isolation tiers (x86 PKU protection domain request) ─────────────────────
16// A tier is a FLOOR request, the inverse of a capability: a higher number means
17// MORE isolation / LESS authority.  A cell may always raise its own tier
18// (self-restriction); it may NOT lower it below the floor the loader derives from
19// its capabilities (see `kernel/src/loader.rs`).  This is why declaring a tier is
20// safe without a ceiling check for the raise direction, and gated for the lower.
21
22/// Trusted-core domain (PKU key 0 — no fencing).  Only reachable if the cell's
23/// caps already authorise it (the loader floors non-privileged cells above this).
24pub const TIER_TRUSTED_CORE: u8 = 0;
25/// Standard Rust cell (PKU key 1).  The default authority floor for a plain cell.
26pub const TIER_STANDARD: u8 = 1;
27/// Tier-1b C/FFI cell (PKU key 2).  Fences untrusted FFI code from the cell's Rust
28/// data — the key the v1 `TODO(pku-ffi)` could not reach.
29pub const TIER_TIER1B_FFI: u8 = 2;
30/// Untrusted domain (maps to the most-isolated available key).
31pub const TIER_UNTRUSTED: u8 = 3;
32/// Sentinel meaning "no explicit tier requested — apply the caller's floor
33/// policy (the legacy `is_trusted` heuristic)."  Set automatically on upcast from
34/// a v1 manifest (which had no tier field), AND baked in by the tier-less
35/// constructors (`CellManifest::new`/`with_parts`, used by `declare_manifest!`'s
36/// back-compat macro forms) into a native v2 record — so it is a valid value in
37/// BOTH v1-upcast and native-v2 manifests, never an error on its own.
38pub const TIER_LEGACY: u8 = 0xFF;
39
40// ─── Capability flags (u16 in v2; low 8 bits are bit-identical to v1) ─────────
41
42/// Raw block-device access (BlkRead/BlkWrite/BlkFlush).  Grants `BlockIoCap`.
43pub const MANIFEST_FLAG_BLOCK_IO: u16 = 1 << 0;
44/// Network transmit/receive (NetTx/NetRx).  Grants `NetworkCap`.
45pub const MANIFEST_FLAG_NETWORK: u16 = 1 << 1;
46/// Cell spawning and hot-swap (SpawnFromPath/SpawnPinned/HotSwap).  Grants `SpawnCap`.
47pub const MANIFEST_FLAG_SPAWN: u16 = 1 << 2;
48/// GPIO pin control (ViGpio driver cell).  MMIO range via `sys_request_mmio`.
49pub const MANIFEST_FLAG_GPIO: u16 = 1 << 3;
50/// UART serial access (ViUart driver cell).  MMIO range via `sys_request_mmio`.
51pub const MANIFEST_FLAG_UART: u16 = 1 << 4;
52/// RISC-V H-extension (hypervisor) CSR access for VMM cells.  Grants `HypervisorCap`
53/// only when the CPU also reports H-ext at boot.
54pub const MANIFEST_FLAG_HYPERVISOR: u16 = 1 << 5;
55/// Block-I/O sector range grant: MBR partition P1 (FAT32, `api::disk`).
56pub const MANIFEST_FLAG_PART_DATA: u16 = 1 << 6;
57/// Block-I/O sector range grant: MBR partition P4 (littlefs, `api::disk`).
58pub const MANIFEST_FLAG_PART_LFS: u16 = 1 << 7;
59/// CAN bus controller MMIO (v2 — freed by the u16 widening).  Grants the CAN
60/// device class via `sys_request_mmio`.
61pub const MANIFEST_FLAG_CAN: u16 = 1 << 8;
62/// ADC controller MMIO (v2).  Grants the ADC device class via `sys_request_mmio`.
63pub const MANIFEST_FLAG_ADC: u16 = 1 << 9;
64
65/// Bitmask of all defined flags.  `from_bytes` rejects manifests setting any bit
66/// outside this mask (a stale/forward binary is treated as malformed → legacy path
67/// grants, the fail-safe direction).
68pub const MANIFEST_FLAGS_MASK: u16 = MANIFEST_FLAG_BLOCK_IO
69    | MANIFEST_FLAG_NETWORK
70    | MANIFEST_FLAG_SPAWN
71    | MANIFEST_FLAG_GPIO
72    | MANIFEST_FLAG_UART
73    | MANIFEST_FLAG_HYPERVISOR
74    | MANIFEST_FLAG_PART_DATA
75    | MANIFEST_FLAG_PART_LFS
76    | MANIFEST_FLAG_CAN
77    | MANIFEST_FLAG_ADC;
