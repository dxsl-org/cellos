1//! Cell capability manifest embedded in the `__ViCell_manifest` ELF section.
2//!
3//! A fixed **16-byte** `#[repr(C)]` record (v2) declaring which privileged
4//! capabilities a Cell requests, its isolation **tier**, and a reserved hook for
5//! future per-cell cap arguments.  The kernel reads it at spawn time (see
6//! `kernel/src/loader.rs::spawn_from_path`) to grant capability tokens, choose the
7//! x86 PKU protection domain, and reject user Cells that over-declare privilege.
8//!
9//! Binary layout (16 bytes, little-endian):
10//! ```text
11//!   offset  0–3 : magic        u32  = MANIFEST_MAGIC (0x5649_4345)
12//!   offset  4   : version      u8   = MANIFEST_VERSION (2)
13//!   offset  5   : tier         u8   = TIER_* (isolation floor request; TIER_LEGACY
14//!                                     on upcast from v1 → keep the is_trusted heuristic)
15//!   offset  6–7 : flags        u16  = bitwise-OR of MANIFEST_FLAG_*
16//!   offset  8–11: cap_args_off u32  = RESERVED (0) — future offset into a
17//!                                     __ViCell_cap_args section (do not repurpose)
18//!   offset 12–15: reserved     u32  = 0
19//! ```
20//!
21//! ## v1 compatibility
22//! v1 was an 8-byte record `{magic, version=1, flags:u8, _pad:[u8;2]}`.  A v2
23//! kernel reads a v1 manifest via `from_bytes` (zero-extends `flags`, sets
24//! `tier = TIER_LEGACY` so the loader keeps v1's `is_trusted`→PKU-key behaviour
25//! byte-for-byte).  A v1 kernel reading a v2 manifest sees `version != 1` and
26//! rejects it (fail-closed → legacy path grants).  `TIER_LEGACY` is ALSO a valid
27//! tier value in a native v2 record (not just a v1-upcast artifact) — it is what
28//! the tier-less constructors (`CellManifest::new`/`with_parts`) bake in by
29//! default, meaning "no explicit tier requested." ABI-stable — see Law 1.
30

37/// Fixed-layout capability manifest (v2).  ABI-stable — see Law 1.
38///
39/// Always 16 bytes due to `#[repr(C)]` and explicit reserved fields.  Version the
40/// struct via `MANIFEST_VERSION` before adding fields.
41#[repr(C)]
42#[derive(Copy, Clone, Debug)]
43pub struct CellManifest {
44    /// Must equal `MANIFEST_MAGIC`; `from_bytes` rejects any other value.
45    pub magic: u32,
46    /// `MANIFEST_VERSION` (2) for a native v2 manifest; a v1 upcast keeps this at 2
47    /// (the value in `tier` records that it came from v1).
48    pub version: u8,
49    /// Isolation tier request (`TIER_*`); `TIER_LEGACY` when upcast from v1.
50    pub tier: u8,
51    /// Bitwise-OR of `MANIFEST_FLAG_*` constants (u16 in v2).
52    pub flags: u16,
53    /// RESERVED — must be 0.  Future offset into a `__ViCell_cap_args` section for
54    /// parameterized capabilities; kept here so filling it is an additive section
55    /// parse, not a third ABI confirmation.
56    pub cap_args_off: u32,
57    /// RESERVED — must be 0.
58    pub reserved: u32,
59}

61impl CellManifest {
62    /// Construct a manifest from capability bits (tier defaults to `TIER_LEGACY`,
63    /// preserving v1's `is_trusted`→PKU-key behaviour for cells that do not opt in).
64    ///
65    /// Evaluates at compile time; safe to use as a `static` initializer.
66    pub const fn new(
67        block_io: bool,
68        network: bool,
69        spawn: bool,
70        gpio: bool,
71        uart: bool,
72        hypervisor: bool,
73    ) -> Self {
74        Self::with_all(
75            block_io,
76            network,
77            spawn,
78            gpio,
79            uart,
80            hypervisor,
81            false,
82            false,
83            false,
84            false,
85            TIER_LEGACY,
86        )
87    }
88
89    /// Construct a manifest including block-I/O partition range grants.
90    #[allow(clippy::too_many_arguments)]
91    pub const fn with_parts(
92        block_io: bool,
93        network: bool,
94        spawn: bool,
95        gpio: bool,
96        uart: bool,
97        hypervisor: bool,
98        part_data: bool,
99        part_lfs: bool,
100    ) -> Self {
101        Self::with_all(
102            block_io,
103            network,
104            spawn,
105            gpio,
106            uart,
107            hypervisor,
108            part_data,
109            part_lfs,
110            false,
111            false,
112            TIER_LEGACY,
113        )
114    }
115
116    /// Full constructor — all flags + tier.  The macro is the public face.
117    #[allow(clippy::too_many_arguments)]
118    pub const fn with_all(
119        block_io: bool,
120        network: bool,
121        spawn: bool,
122        gpio: bool,
123        uart: bool,
124        hypervisor: bool,
125        part_data: bool,
126        part_lfs: bool,
127        can: bool,
128        adc: bool,
129        tier: u8,
130    ) -> Self {
131        Self {
132            magic: MANIFEST_MAGIC,
133            version: MANIFEST_VERSION,
134            tier,
135            flags: (block_io as u16 * MANIFEST_FLAG_BLOCK_IO)
136                | (network as u16 * MANIFEST_FLAG_NETWORK)
137                | (spawn as u16 * MANIFEST_FLAG_SPAWN)
138                | (gpio as u16 * MANIFEST_FLAG_GPIO)
139                | (uart as u16 * MANIFEST_FLAG_UART)
140                | (hypervisor as u16 * MANIFEST_FLAG_HYPERVISOR)
141                | (part_data as u16 * MANIFEST_FLAG_PART_DATA)
142                | (part_lfs as u16 * MANIFEST_FLAG_PART_LFS)
143                | (can as u16 * MANIFEST_FLAG_CAN)
144                | (adc as u16 * MANIFEST_FLAG_ADC),
145            cap_args_off: 0,
146            reserved: 0,
147        }
148    }

150    // `from_bytes` (the ELF-section parser) lives in the sibling `manifest_parse`

153    /// Returns `true` if the cell declared raw block-device access.
154    pub fn has_block_io(&self) -> bool {
155        self.flags & MANIFEST_FLAG_BLOCK_IO != 0
156    }
157    /// Returns `true` if the cell declared network transmit/receive.
158    pub fn has_network(&self) -> bool {
159        self.flags & MANIFEST_FLAG_NETWORK != 0
160    }
161    /// Returns `true` if the cell declared cell-spawning and hot-swap.
162    pub fn has_spawn(&self) -> bool {
163        self.flags & MANIFEST_FLAG_SPAWN != 0
164    }
165    /// Returns `true` if the cell declared GPIO pin-control access.
166    pub fn has_gpio(&self) -> bool {
167        self.flags & MANIFEST_FLAG_GPIO != 0
168    }
169    /// Returns `true` if the cell declared UART serial access.
170    pub fn has_uart(&self) -> bool {
171        self.flags & MANIFEST_FLAG_UART != 0
172    }
173    /// Returns `true` if the cell declared H-extension hypervisor CSR access.
174    pub fn has_hypervisor(&self) -> bool {
175        self.flags & MANIFEST_FLAG_HYPERVISOR != 0
176    }
177    /// Returns `true` if the cell's block I/O is granted the P1 (FAT32) range.
178    pub fn has_part_data(&self) -> bool {
179        self.flags & MANIFEST_FLAG_PART_DATA != 0
180    }
181    /// Returns `true` if the cell's block I/O is granted the P4 (littlefs) range.
182    pub fn has_part_lfs(&self) -> bool {
183        self.flags & MANIFEST_FLAG_PART_LFS != 0
184    }
185    /// Returns `true` if the cell declared CAN controller MMIO access (v2).
186    pub fn has_can(&self) -> bool {
187        self.flags & MANIFEST_FLAG_CAN != 0
188    }
189    /// Returns `true` if the cell declared ADC controller MMIO access (v2).
190    pub fn has_adc(&self) -> bool {
191        self.flags & MANIFEST_FLAG_ADC != 0
192    }
193
194    /// The declared isolation tier (`TIER_*`, or `TIER_LEGACY` if upcast from v1).
195    pub fn tier(&self) -> u8 {
196        self.tier
197    }
198
199    /// Returns `true` if any privileged capability bit is set.  Used by
200    /// `spawn_from_path` to reject over-declaring user Cells (non-`/bin/` paths).
201    pub fn declares_any_privilege(&self) -> bool {
202        self.flags != 0
203    }
204}
