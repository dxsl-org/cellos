3/// System Call Identifiers (The Contract)
4///
5/// These IDs must match between Kernel and User (libs/ostd).
6#[derive(Debug, Clone, Copy, PartialEq, Eq)]
7#[repr(usize)]
8pub enum ViSyscall {
9    // === IPC (0-9) ===
10    Send = 0,
11    Recv = 1,
12    Call = 2,
13    Reply = 3,
14    /// Non-blocking send: delivers immediately if target is in `Recv`, otherwise
15    /// drops the message and returns without blocking the caller.
16    /// Prevents deadlock when the focused cell is busy (e.g. reading UART).
17    TrySend = 4,
18    /// Non-blocking receive: returns 0 immediately when no message is queued
19    /// (vs `Recv`, which parks the task until a message arrives).
20    TryRecv = 7,
21
22    // === Process Management (10-49) ===
23    Exit = 60, // Linux compat usually, but we define our own space
24    /// Force-terminate another task by TID.  Non-blocking: caller continues.
25    /// Requires `SpawnCap` on caller; rejected on system cells (block_io/network_cap).
26    /// ABI: a0 = target_tid → 0 on success, usize::MAX on error.
27    ForceExit = 61,
28    Spawn = 5,
29    Exec = 6,          // Deprecated/Legacy
30    SpawnFromMem = 10, // New Spawn from Memory (Struct based)
31    /// Spawn a cell by reading its ELF from a VFS path.
32    /// ABI: a0 = path_ptr, a1 = path_len; returns cell id or error code.
33    SpawnFromPath = 12,
34    /// Open a file by path, returning a kernel capability ID.
35    /// ABI: a0 = path_ptr, a1 = path_len → CapId (> 0) on success.
36    OpenCap = 13,
37    /// Read bytes from a cap-backed file.
38    /// ABI: a0 = cap_id, a1 = buf_ptr, a2 = buf_len → bytes_read.
39    ReadCap = 14,
40    /// Revoke a capability (close).
41    /// ABI: a0 = cap_id → 0 on success.
42    CloseCap = 15,
43    /// Seek a cap-backed file cursor.
44    /// ABI: a0 = cap_id, a1 = offset (i64 bits), a2 = whence (0=Start,1=Current,2=End)
45    ///      → new absolute position on success.
46    /// Requires the same allowlist bit as ReadCap (bit 16).
47    SeekCap = 228,
48    /// Write bytes into a cap-backed file at the current cursor position.
49    /// ABI: a0 = cap_id, a1 = buf_ptr, a2 = buf_len → bytes_written.
50    /// Requires WriteCap allowlist bit (bit 45).
51    WriteCap = 229,
52    /// Query the size of a cap-backed file in bytes.
53    /// ABI: a0 = cap_id → file_size on success. Does NOT move the cursor.
54    /// Shares the ReadCap allowlist bit (bit 16).
55    StatCap = 230,
56    /// Truncate a cap-backed file to exactly `len` bytes.
57    /// ABI: a0 = cap_id, a1 = len → 0 on success.
58    /// Returns an error if `len > current_size`; use `WriteCap` to extend.
59    /// Requires TruncateCap allowlist bit (bit 46).
60    TruncateCap = 231,
61    /// Flush all dirty pages for a cap-backed file to the underlying block device (fsync).
62    /// ABI: a0 = cap_id → 0 on success.
63    /// No-op on write-through filesystems (RamDisk). Hooks into device flush on NVMe (G2).
64    /// Shares the WriteCap allowlist bit (bit 45).
65    SyncCap = 232,
66    /// Map a PCIe device's DMA range into the IOMMU for the calling Cell.
67    /// ABI: a0 = bdf: u32, a1 = phys: u64, a2 = size: usize → IOVA (== phys for SAS) on success.
68    /// Caller must own the BDF via Resource Registry; range must be within Cell memory quota.
69    GrantDma = 233,
70    /// Block the calling Driver Cell until hardware IRQ `irq_num` fires.
71    /// The kernel ISR sets an atomic flag (no lock); the scheduler sweep transitions the task
72    /// to Ready. Lost-wakeup safe: an IRQ that fires between the Cell's queue-check and this
73    /// call is not missed (take_pending before park). Requires PcieDriverCap or PlatformCap.
74    /// ABI: a0 = irq_num (u8), a1 = mmio_base (VirtIO MMIO slot base for InterruptACK, or 0).
75    WaitIrq = 234,
76    /// Register a PCIe device BAR in the kernel ECAM BAR table (Platform Cell only).
77    /// Called by the Platform Cell after scanning ECAM config space; populates the allowlist
78    /// that Driver Cells consult via `sys_request_mmio`. Singleton PlatformCap required.
79    /// ABI: a0 = bdf (u32), a1 = base (usize), a2 = len (usize) → 0 on success.
80    RegisterPcieBar = 235,
81    /// Register a discovered PCI device with class/BAR info (Platform Cell only).
82    ///
83    /// Called by the Platform Cell after ECAM scan; populates kernel `PCI_DEVICES` so
84    /// `sys_find_pcie_device` queries work without a kernel-side ECAM scan.
85    /// `bdf`       = (bus<<16)|(dev<<8)|fun
86    /// `cls`       = (class<<16)|(subclass<<8)|prog_if
87    /// `bar0_base` = physical base of BAR0 (or 0 if absent)
88    /// `bar0_size` = probed BAR0 size in bytes (or 0)
89    /// Requires singleton PlatformCap (allowlist bit 53).
90    RegisterPciDevice = 236,
91    /// Drain up to `max` bytes from the kernel user-log ring buffer into `buf`.
92    ///
93    /// Returns the number of bytes actually copied (may be 0 if no new output).
94    /// The ring buffer is written by `sys_log` (cell stdout) and `print_user_log`
95    /// (kernel's USER: prefixed UART path) — gives the fb-console cell a live
96    /// mirror of everything that appears on the serial console.
97    ///
98    /// ABI: a0 = buf_ptr, a1 = max → bytes_copied.
99    /// Gated by ReadLogCap (allowlist bit 54) — declare with `ReadLog` in manifest.
100    ReadLog = 237,
101    /// Spawn a cell from ELF bytes supplied in a caller-owned Grant region.
102    ///
103    /// The userspace spawner (init/supervisor/shell) reads a cell's ELF from VFS
104    /// into a Grant, then calls this. The kernel applies the IDENTICAL spawn trust
105    /// gate as `SpawnFromPath` (Ed25519 signature, manifest-privilege, capability
106    /// intersection, policy, syscall allowlist, cluster, quota, measurement) over
107    /// the grant bytes — trust is over the bytes, so the source is irrelevant.
108    /// Lets the kernel loader stop reading a filesystem for post-boot spawns
109    /// (G2 loader redesign). Requires SpawnCap (same as SpawnFromPath).
110    ///
111    /// ABI: a0 = grant_id, a1 = len (bytes), a2 = path_hint_ptr, a3 = path_hint_len
112    ///      → tid (>0) on success. `path_hint` is advisory (drives the /bin/
113    ///      privilege check, policy lookup, measurement label); a caller lying
114    ///      about it can only LOSE privilege, never gain it.
115    SpawnFromElf = 238,
116    /// Spawn a cell pinned to a specific hardware core.
117    /// ABI: a0 = path_ptr, a1 = path_len, a2 = priority: u8, a3 = core_id: usize.
118    /// On single-core systems core_id must be 0; any other value returns NotSupported.
119    SpawnPinned = 16,
120    /// Serialize all allocated physical frames to the snapshot sector range.
121    /// Triggers a warm-boot-capable snapshot.  Returns frame count on success.
122    Snapshot = 420,
123    Wait = 8,      // Wait for task
124    Yield = 104,   // Linux sched_yield is 24, but we use 104 in current code
125    SetTimer = 35, // Added SetTimer
126    ShmAlloc = 20, // Allocate Shared Memory
127    ShmMap = 21,   // Map Shared Memory to Task
128
129    // === Logging (50-59) ===
130    Log = 11, // Current implementation uses 11
131
132    // === Filesystem (100-199) ===
133    Open = 101,
134    Read = 102,
135    Close = 103,
136    ReadDir = 105,
137    Write = 109,
138    Seek = 106,
139    FileOp = 107, // Rename, Remove
140
141    // === Time/System (120-129) ===
142    GetTime = 120,
143
144    // === GPU / Display (Phase 16) ===
145    /// Flush a pixel rectangle from a cell-provided buffer to the GPU framebuffer.
146    /// ABI: a0 = data_ptr, a1 = data_len, a2 = x, a3 = y (w+h embedded in len)
147    GpuFlush = 300,
148    /// Set or move the VirtIO GPU hardware cursor.
149    /// ABI: a0 = op (0=set sprite, 1=move), a1 = data_ptr (op=0: 64×64 BGRA sprite),
150    ///      a2 = xy ((x<<16)|y), a3 = hot ((hot_x<<16)|hot_y, op=0 only).
151    GpuCursor = 301,
152    /// Query the VirtIO GPU's current scanout resolution.
153    /// ABI: no args → returns (width as u32) << 32 | (height as u32).
154    /// Falls back to (1280, 800) if GPU not yet initialized.
155    GpuGetResolution = 302,
156
157    // === Network ===
158    /// Transmit one Ethernet frame through the kernel VirtIO NIC.
159    /// ABI: a0 = frame_ptr, a1 = frame_len → 1 on success, 0 on failure.
160    NetTx = 310,
161    /// Receive one pending Ethernet frame from the kernel VirtIO NIC.
162    /// ABI: a0 = buf_ptr, a1 = buf_len → bytes written (0 if none ready).
163    NetRx = 311,
164
165    // === Advanced IPC (Phase 20) ===
166    /// Recv with a deadline in monotonic ticks.
167    /// ABI: a0 = mask, a1 = buf_ptr, a2 = buf_len, a3 = timeout_ticks → sender_id or ViError::Timeout.
168    RecvTimeout = 201,
169    /// Gather-send: one IPC message assembled from up to 8 non-contiguous buffers.
170    /// ABI: a0 = target, a1 = iovec_ptr ([ptr:usize, len:usize] × count), a2 = iovec_count.
171    SendGather = 202,
172    /// Scatter-receive: one IPC message written into up to 8 non-contiguous buffers.
173    /// ABI: a0 = mask, a1 = iovec_ptr, a2 = iovec_count → sender_id.
174    RecvScatter = 203,
175    /// Register the caller as a watcher of `watched_tid`. When that task exits OR
176    /// faults, the kernel delivers a death notification to the caller's pending
177    /// `Recv` (`sender_id` = the dead tid, payload = the exit/fault reason), so a
178    /// supervisor can wait-any across many children with a single recv loop.
179    /// Requires `SpawnCap`. ABI: a0 = watched_tid → 0 on success, usize::MAX on error.
180    NotifyOnExit = 204,
181    /// Register `tid` as the current provider of well-known `service_id`. SpawnCap-gated:
182    /// the supervisor owns the service namespace (prevents a cell from hijacking, say, the
183    /// VFS endpoint). On respawn the supervisor re-registers the new tid, so clients
184    /// reconnect transparently. ABI: a0 = service_id (u16), a1 = tid → 0 / usize::MAX.
185    RegisterService = 205,
186    /// Resolve a well-known `service_id` to its current provider tid. Open to all cells so
187    /// any client can reconnect after a service restart. ABI: a0 = service_id (u16) →
188    /// provider tid (> 0), or 0 if no live provider is currently registered.
189    LookupService = 206,
190    /// Liveness heartbeat: the caller asserts it is alive and will call again within
191    /// `interval_ticks` 10 ms scheduler ticks. If it misses that deadline the kernel
192    /// terminates it as HUNG — catching a silent hang (deadlock / stuck loop) that the
193    /// CPU-monopoly watchdog cannot see — so the supervisor restarts it. `interval_ticks
194    /// = 0` disables the heartbeat. Open to all cells. ABI: a0 = interval_ticks → 0.
195    Heartbeat = 207,
196
197    // === Zero-Copy Grant (Storage 2.0, Phase 01) ===
198    /// Allocate a kernel-managed Grant region of up to 4096 pages (16 MB).
199    /// ABI: a0 = size (rounded up to pages, max 16777216) → base_paddr (> 0) on success, 0 on OOM.
200    /// Requires GrantCap (allowlist bit 39).
201    GrantAlloc = 208,
202    /// Share Grant access with `target_task`.
203    /// ABI: a0 = grant_id, a1 = target_task_id, a2 = GrantPerm (0=RO, 1=WO, 2=RW) → 0 on success.
204    /// Requires GrantCap (bit 39).
205    GrantShare = 209,
206    /// Return the raw pointer to a Grant region the caller has access to.
207    /// ABI: a0 = grant_id → ptr (usize, same as grant_id for identity-mapped SAS) on success.
208    /// Returns usize::MAX on permission denied or not found.
209    GrantSlice = 210,
210    /// Release a Grant region: unmaps its pages and frees its frames.
211    /// ABI: a0 = grant_id → 0 on success.
212    GrantFree = 211,
213    /// Start an async sector read into a Grant buffer (poll-based stub; Phase 04 for true async).
214    /// ABI: a0 = sector, a1 = grant_id → async_id (1 = immediately complete) on success.
215    /// Requires BlockIoCap (bit 36) — same gate as raw block-I/O opcodes 500/501.
216    BlkReadAsync = 212,
217    /// Request exclusive MMIO ownership for a peripheral Driver Cell.
218    /// ABI: a0 = base (physical MMIO address), a1 = len → 0 on success.
219    /// Requires the corresponding manifest flag (GPIO or UART) to be set.
220    /// The kernel checks the Resource Registry allowlist; rejects unknown ranges.
221    RequestMmio = 213,
222    /// Fill caller buffer with VirtIO-RNG entropy (true hardware randomness).
223    /// Required for TLS key generation — mtime-seeded PRNG is cryptographically broken.
224    /// ABI: a0 = buf_ptr, a1 = len (max 64 per call, one VirtIO descriptor) → bytes written.
225    /// Returns 0 if no VirtIO-RNG device is present.
226    GetRandom = 214,
227    /// Allocate a persistent, pre-pinned Grant buffer for the cell's lifetime.
228    /// Unlike GrantAlloc, the buffer is not freed until GrantUnregister or cell exit.
229    /// ABI: a0 = size → reg_id (physical base, > 0) on success, 0 on OOM.
230    /// Requires GrantCap (allowlist bit 39).
231    GrantRegister = 215,
232    /// Release a registered buffer allocated via GrantRegister.
233    /// ABI: a0 = reg_id → 0 on success.
234    /// Requires GrantCap (allowlist bit 39).
235    GrantUnregister = 216,
236    /// Block until one or more bits in `mask` fire, or `timeout_ticks` 10ms ticks elapse.
237    /// `timeout_ticks = 0` means block indefinitely.
238    /// Returns the fired event bits (>0) on wake, or 0 on timeout.
239    /// Bit 0 = NET_RX (NIC frame available). See `api::syscall::events`.
240    /// ABI: a0 = mask (u32), a1 = timeout_ticks (u64 lo, a2 = hi) → fired_mask (usize).
241    /// Allowlist-gated (bit 42). Requires `WaitForEvent` in `declare_syscalls!`.
242    WaitForEvent = 217,
243
244    /// Play raw PCM frames on the VirtIO sound device's output stream (blocking).
245    /// PCM format is fixed: signed 16-bit LE, 2 channels, 44100 Hz (frame = 4 bytes).
246    /// The kernel configures + starts the output stream on first use.
247    /// ABI: a0 = pcm_ptr, a1 = pcm_len (bytes) → frames written (0 on no device/error).
248    /// Allowlist-gated (bit 47). Requires `AudioPlay` in `declare_syscalls!`.
249    AudioPlay = 218,

251    /// Revoke one or more capabilities from a live running cell (runtime revocation).
252    ///
253    /// The caller must hold `SpawnCap`.  System cells (`block_io_cap` / `network_cap`
254    /// holders) are protected — revoking I/O caps from a running service mid-flight
255    /// would corrupt driver state; use `HotSwap` to replace them safely.
256    ///
257    /// `cap_mask` bit layout (see `cap_mask` module constants):
258    ///   bit 0 → block_io · bit 1 → network · bit 2 → spawn · bit 3 → hypervisor
259    ///   bits  8-15 → mmio_devices sub-mask (matches resource_registry::DEV_*)
260    ///   bits 16-23 → block_regions sub-mask (partition access bits)
261    ///
262    /// ABI: a0 = target_tid, a1 = cap_mask (u32) → 0 on success.
263    /// Returns `PermissionDenied` if caller lacks SpawnCap or target is a system cell.
264    /// Always permitted past the syscall allowlist (SpawnCap is the authority gate).
265    CapRevoke = 219,

268    // All six syscalls require `hypervisor = true` in the cell manifest AND the

270    // without HypervisorCap calling any of these receives `PermissionDenied`.
271    /// Allocate guest RAM and a Stage-2 table; returns opaque `vm_id`.
272    /// ABI: a0 = guest_pages → vm_id (>0) on success, 0 on OOM / unsupported.
273    CreateVm = 220,

276    CreateVcpu = 221,

292    /// ABI: a0 = vm_id, a1 = gpa, a2 = src_ptr, a3 = len → bytes_written.
293    WriteGuestMemory = 226,
294    /// Copy `len` bytes from guest physical RAM at `gpa` into caller's `dst_ptr`.
295    /// Kernel bounds-checks both sides: `dst_ptr + len` within caller's address space

297    /// ABI: a0 = vm_id, a1 = gpa, a2 = dst_ptr, a3 = len → bytes_read.
298    ReadGuestMemory = 227,
299
300    // === Hot-swap (Phase 20) ===
301    /// Live-replace a running Cell without message loss.
302    /// ABI: a0 = cell_id, a1 = path_ptr, a2 = path_len → new_task_id or error.

323    FreezeCell = 413,

353/// Bit constants for the `cap_mask` argument of `ViSyscall::CapRevoke`.
354///
355/// The 32-bit mask encodes which capability classes to strip from the target cell.
356/// Boolean caps occupy the low 4 bits; parameterized caps use two 8-bit sub-fields.
357pub mod cap_mask {
358    /// Revoke raw block-device access (BlkRead/BlkWrite/BlkFlush).
359    pub const BLOCK_IO: u32 = 1 << 0;
360    /// Revoke network transmit/receive (NetTx/NetRx).
361    pub const NETWORK: u32 = 1 << 1;
362    /// Revoke cell-spawning authority (SpawnFromPath/HotSwap).
363    pub const SPAWN: u32 = 1 << 2;
364    /// Revoke RISC-V H-extension / ARM64 EL2 hypervisor access.
365    pub const HYPERVISOR: u32 = 1 << 3;
366
367    /// Bit offset of the `mmio_devices` sub-field within the mask.
368    pub const MMIO_SHIFT: u32 = 8;
369    /// Mask selecting the `mmio_devices` 8-bit sub-field (bits 8-15).
370    /// Each bit within this byte corresponds to a `resource_registry::DEV_*` flag.
371    pub const MMIO_MASK: u32 = 0xFF << MMIO_SHIFT;
372
373    /// Bit offset of the `block_regions` sub-field within the mask.
374    pub const BLKREGION_SHIFT: u32 = 16;
375    /// Mask selecting the `block_regions` 8-bit sub-field (bits 16-23).
376    pub const BLKREGION_MASK: u32 = 0xFF << BLKREGION_SHIFT;
377
378    /// Revoke all capability classes at once.
379    pub const ALL: u32 = BLOCK_IO | NETWORK | SPAWN | HYPERVISOR | MMIO_MASK | BLKREGION_MASK;
380}

382/// Compact bitset of permitted syscalls, stored as a `u64`.
383///
384/// Used to build the allowlist embedded in the ELF `__ViCell_syscalls` section
385/// via [`declare_syscalls!`].  All methods are `const` so the value can be
386/// computed at compile time.
387///
388/// ```ignore
389/// const MY_ALLOWED: SyscallSet = SyscallSet::EMPTY
390///     .with(ViSyscall::Send)
391///     .with(ViSyscall::Recv)
392///     .with(ViSyscall::Log);
393/// ```
394#[derive(Copy, Clone, Debug, PartialEq, Eq)]
395pub struct SyscallSet(pub u64);
396
397impl SyscallSet {
398    /// Empty set — all syscalls denied (not useful on its own; combine with `with`).
399    pub const EMPTY: Self = Self(0);
400    /// Permit-all sentinel — the default for cells without a `__ViCell_syscalls` section.
401    pub const ALL: Self = Self(u64::MAX);
402
403    /// Return a new `SyscallSet` with `syscall` added.
404    ///
405    /// If `syscall` has no allowlist bit (always-permitted syscalls like `Yield`
406    /// or `Exit`) the set is returned unchanged.
407    pub const fn with(self, syscall: ViSyscall) -> Self {
408        match syscall.allowlist_bit() {
409            Some(bit) => Self(self.0 | (1u64 << bit)),
410            None => self,
411        }
412    }
413
414    /// Raw bit-mask value to embed in `__ViCell_syscalls`.
415    pub const fn bits(self) -> u64 {
416        self.0
417    }
418
419    /// Returns `true` if `syscall` is permitted by this set.
420    ///
421    /// Always-permitted syscalls (`Yield`, `Exit`, …) return `true` regardless
422    /// of the stored bits.
423    pub const fn permits(self, syscall: ViSyscall) -> bool {
424        match syscall.allowlist_bit() {
425            Some(bit) => (self.0 >> bit) & 1 == 1,
426            None => true,
427        }
428    }
429}

431/// Embed a syscall allowlist into the current Cell's ELF binary.
432///
433/// Places a `u64` bitset into the `__ViCell_syscalls` ELF section.  The kernel
434/// reads it at spawn time and enforces it on every syscall — any call whose bit
435/// is not set is rejected with `PermissionDenied` before the handler runs.
436///
437/// Cells that do **not** call this macro default to permit-all (backwards
438/// compatible).  Cells that call it are restricted to exactly the listed
439/// syscalls plus always-permitted ones (`Yield`, `Exit`, `ForceExit`, …).
440///
441/// # Usage
442/// ```ignore
443/// api::declare_syscalls![Send, Recv, TryRecv, Log, Heartbeat, LookupService];
444/// ```
445#[macro_export]
446macro_rules! declare_syscalls {
447    ($($syscall:ident),* $(,)?) => {
448        #[used]
449        #[link_section = "__ViCell_syscalls"]
450        pub static VICELL_SYSCALLS: u64 = const {
451            let mut mask: u64 = 0u64;
452            $(
453                if let Some(bit) = $crate::syscall::ViSyscall::$syscall.allowlist_bit() {
454                    mask |= 1u64 << bit;
455                }
456            )*
457            mask
458        };
459    };
460}

462impl ViSyscall {
463    /// Stable bit index (0-38) for the per-Cell syscall allowlist stored in
464    /// `Task::syscall_allowlist`.
465    ///
466    /// Bit indices are independent of raw opcode values so they remain stable
467    /// even if opcodes are renumbered.  Returns `None` for syscalls that are
468    /// always permitted (Yield, Exit) — filtering them would prevent a Cell
469    /// from ever yielding or cleanly shutting down.
470    ///
471    /// Bit 36 is reserved for raw block-I/O syscalls (opcodes 500-503) which
472    /// bypass `ViSyscall::from()` and must be checked separately at dispatch.
473    pub const fn allowlist_bit(self) -> Option<u8> {
474        match self {
475            Self::Send => Some(0),
476            Self::TrySend => Some(0), // same allowlist bit as Send (subset of Send)
477            Self::Recv => Some(1),
478            Self::TryRecv => Some(2),
479            Self::Reply => Some(3),
480            Self::Call => Some(4),
481            Self::Spawn => Some(5),
482            Self::SpawnFromMem => Some(6),
483            Self::SpawnFromPath => Some(7),
484            // SpawnFromElf shares the SpawnCap allowlist bit (7): same authority as
485            // SpawnFromPath (both require SpawnCap, enforced again in the handler).
486            Self::SpawnFromElf => Some(7),
487            Self::SpawnPinned => Some(8),
488            Self::Wait => Some(9),
489            Self::Log => Some(10),
490            Self::SetTimer => Some(11),
491            Self::ShmAlloc => Some(12),
492            Self::ShmMap => Some(13),
493            Self::GetProcs => Some(14),
494            Self::OpenCap => Some(15),
495            Self::ReadCap | Self::SeekCap | Self::StatCap => Some(16),
496            Self::CloseCap => Some(17),
497            Self::WriteCap | Self::SyncCap => Some(45),
498            Self::TruncateCap => Some(46),
499            Self::Open => Some(18),
500            Self::Read => Some(19),
501            Self::Write => Some(20),
502            Self::Close => Some(21),
503            Self::ReadDir => Some(22),
504            Self::Seek => Some(23),
505            Self::FileOp => Some(24),
506            Self::GetTime => Some(25),
507            Self::GpuFlush => Some(26),
508            Self::GpuCursor => Some(43),
509            Self::GpuGetResolution => Some(44),
510            Self::NetTx => Some(27),
511            Self::NetRx => Some(28),
512            Self::RecvTimeout => Some(29),
513            Self::SendGather => Some(30),
514            Self::RecvScatter => Some(31),
515            Self::HotSwap | Self::HotSwapReady => Some(32),
516            Self::StateStash => Some(33),
517            Self::StateRestore => Some(34),
518            // StateStashClear: same allowlist bit as StateRestore (both are stash housekeeping).
519            Self::StateStashClear => Some(34),
520            Self::Exec => Some(35),
521            // Snapshot: privileged warm-boot operation; reuses HotSwap bit (SpawnCap required).
522            Self::Snapshot => Some(32),
523            // LookupService is an open syscall (any client resolves a service endpoint).
524            Self::LookupService => Some(37),
525            // Heartbeat is an open syscall (any cell asserts its own liveness).
526            Self::Heartbeat => Some(38),
527            // GrantCap (bit 39): cells that need zero-copy large-file I/O via Grant API.
528            Self::GrantAlloc
529            | Self::GrantShare
530            | Self::GrantSlice
531            | Self::GrantFree
532            | Self::GrantRegister
533            | Self::GrantUnregister => Some(39),
534            // BlkReadAsync reuses BlockIoCap (bit 36) — same authority as raw block I/O.
535            Self::BlkReadAsync => Some(36),
536            // RequestMmio: gated by GPIO (bit 40) or UART (bit 41) manifest flags.
537            // The kernel re-checks the manifest and allowlist at dispatch; here we
538            // assign bit 40 as the generic "peripheral MMIO" allowlist bit.
539            Self::RequestMmio => Some(40),
540            // GetRandom: any cell that needs entropy (TLS, crypto) declares this.
541            // Bit 41 was previously allocated as a UART MMIO alias; repurposing for entropy
542            // is safe because UART MMIO is enforced at dispatch by resource-registry check,
543            // not by this bit alone.
544            Self::GetRandom => Some(41),
545            // WaitForEvent: IRQ-driven sleep (net RX waker, Phase 04 SMP).
546            Self::WaitForEvent => Some(42),
547            // AudioPlay: VirtIO sound output (bit 47 — next free after TruncateCap 46).
548            Self::AudioPlay => Some(47),
549            // HypervisorCap (bit 44): all 6 VMM syscalls share one bit.
550            // Bit 43 is already assigned to GpuCursor; 44 is the next free slot.
551            Self::CreateVm
552            | Self::CreateVcpu
553            | Self::MapGuestMemory
554            | Self::RunVcpu
555            | Self::VcpuRegs
556            | Self::InjectIrq
557            | Self::WriteGuestMemory
558            | Self::ReadGuestMemory => Some(44),
559            // GrantDma (bit 48): PCIe DMA authorization. Cells that need to drive
560            // hardware DMA (Driver Cells) declare this capability in their manifest.
561            Self::GrantDma => Some(48),
562            // SupervisorOp (bit 49): freeze/resume/kill/query a live Cell.
563            // Gated by SupervisorCap which is ONLY granted via direct kernel TCB write.
564            Self::FreezeCell | Self::ResumeCell | Self::KillCell | Self::QueryHotswapReady => {
565                Some(49)
566            }
567            // DriverRegistration (bit 50): announce as the active block/NIC driver,
568            // and discover PCIe device BARs. All gated by PcieDriverCap.
569            Self::RegisterBlockDriver | Self::RegisterNicDriver | Self::FindPcieDevice => Some(50),
570            // WaitIrq (bit 51): block a Driver Cell until a hardware IRQ fires.
571            // Gated by PcieDriverCap or PlatformCap — only Driver Cells may wait on IRQs.
572            Self::WaitIrq => Some(51),
573            // RegisterPcieBar (bit 52): Platform Cell announces a discovered BAR.
574            // Singleton PlatformCap is the gate — only one trusted Platform Cell may call this.
575            Self::RegisterPcieBar => Some(52),
576            // RegisterPciDevice (bit 53): Platform Cell announces a PCI device with class info.
577            // Populates kernel PCI_DEVICES so find_class() queries work without kernel ECAM scan.
578            Self::RegisterPciDevice => Some(53),
579            // ReadLog (bit 54): read the kernel user-log ring buffer.
580            // Gated so random cells cannot snoop on other cells' stdout.
581            // Declare with `ReadLog` in the cell manifest to enable.
582            Self::ReadLog => Some(54),
583            // Yield, Exit, and ForceExit are always permitted — a Cell must be able
584            // to yield the CPU, exit cleanly, and force-terminate unresponsive tasks
585            // regardless of its allowlist.  SpawnCap is the authority gate for ForceExit.
586            // NotifyOnExit, RegisterService, and CapRevoke are privileged (SpawnCap-gated),
587            // so they are always permitted past the allowlist — SpawnCap is the gate at dispatch.
588            Self::Yield
589            | Self::Exit
590            | Self::ForceExit
591            | Self::NotifyOnExit
592            | Self::RegisterService
593            | Self::CapRevoke
594            | Self::Unknown => None,
595        }
596    }
597}

700/// Well-known service IDs for the kernel Service Registry (`RegisterService` /
701/// `LookupService`). Stable ABI — values must not change once shipped. `0` is reserved
702/// to mean "no provider" (the value `LookupService` returns when nothing is registered).
703///
704/// A client resolves the live provider tid of a service by `LookupService(service::VFS)`
705/// instead of hard-coding a tid, so it transparently reconnects when the supervisor
706/// respawns that service under a new tid.
707pub mod service {
708    /// Virtual filesystem service (`/bin/vfs`).
709    pub const VFS: u16 = 1;
710    /// Network stack service (`/bin/net`).
711    pub const NET: u16 = 2;
712    /// Input/event service (`/bin/input`).
713    pub const INPUT: u16 = 3;
714    /// Configuration store service (`/bin/config`).
715    pub const CONFIG: u16 = 4;
716    /// Display compositor service (`/bin/compositor`).
717    pub const COMPOSITOR: u16 = 5;
718    /// Hot-swap demo cell — used by the hotswap integration demo only.
719    /// Not a production service; registered at demo spawn time.
720    pub const HOTSWAP_DEMO: u16 = 7;
721    /// Cluster net-broker — the userspace trust anchor for all cross-machine
722    /// communication (Noise KKpsk0 p2p, XChaCha20-Poly1305 gossip, RemoteServiceProxy).
723    /// Supervised by init; respawning re-registers under a new tid, so clients
724    /// must resolve via `sys_lookup_service(service::NET_BROKER)`.
725    pub const NET_BROKER: u16 = 8;
726    /// NVMe / block device driver cell. Clients (VFS) call `LookupService(BLOCK_DRIVER)`
727    /// to find the provider tid and send IPC directly instead of using `BlkReadAsync`.
728    pub const BLOCK_DRIVER: u16 = 9;
729    /// NIC driver cell. The net cell resolves this instead of using kernel `NetTx`/`NetRx`
730    /// once a Driver Cell has registered.
731    pub const NIC_DRIVER: u16 = 10;
732    /// Supervisor Cell — hotswap/snapshot orchestration. Trusted by init; holds SupervisorCap.
733    /// Send `HotswapRequest` / `SnapshotRequest` IPC to trigger live cell replacement.
734    pub const SUPERVISOR: u16 = 11;
735    /// VirtIO GPU Driver Cell — owns the VirtIO GPU MMIO and framebuffer DMA.
736    /// Kernel `GpuFlush`/`GpuCursor` syscalls forward to this Cell when registered.
737    pub const GPU_DRIVER: u16 = 12;
738}
