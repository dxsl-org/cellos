2//! ViMte — cross-arch Memory Tagging trait.
3//!
4//! ⚠️  MTE is HARDENING, NOT a security boundary. It detects use-after-free
5//! and buffer overflow probabilistically (1/16 tag collision chance).
6//! TikTag (2024) demonstrates speculative side-channel bypass. Do NOT treat
7//! MTE as an isolation primitive; it complements, not replaces, Rust safety.
8//!
9//! ## Current implementations
10//! - `aarch64`: MTE2 (ARMv8.5-A), synchronous or async tag-fault mode.
11//! - All other targets: `NoMte` stub (compile-time no-op).
12//!
13//! A second implementation (e.g., RISC-V Zimt, draft spec v0.2) would justify
14//! the trait abstraction at the hardware interface layer; the stub keeps
15//! non-aarch64 builds clean in the meantime.
16
17/// Memory Tagging hardware abstraction.
18///
19/// Implementors must be zero-sized and stateless; tag operations act directly
20/// on hardware registers and Normal-Tagged memory granules.
21pub trait ViMte {
22    /// Returns `true` when MTE2 (full checking) hardware is available.
23    ///
24    /// Checks `ID_AA64PFR1_EL1.MTE` ≥ 2 on AArch64.
25    fn is_available() -> bool;
26
27    /// Tag all 16-byte granules in `ptr..ptr+len` with `color` (4-bit, 0–15).
28    ///
29    /// # Safety
30    /// - `ptr` must be 16-byte aligned.
31    /// - `len` must be a non-zero multiple of 16.
32    /// - The entire range `ptr..ptr+len` must be valid, allocated Normal-Tagged
33    ///   memory owned exclusively by the caller.
34    unsafe fn tag_region(ptr: *mut u8, len: usize, color: u8);
35
36    /// Read the 4-bit allocation tag for the 16-byte granule containing `ptr`.
37    ///
38    /// # Safety
39    /// - `ptr` must point into a valid Normal-Tagged memory region.
40    unsafe fn get_tag(ptr: *const u8) -> u8;
41
42    /// Switch tag-check mode for EL0 and EL1.
43    ///
44    /// `true`  → synchronous fault on mismatch (default, lowest latency for
45    ///            deterministic error detection).
46    /// `false` → asynchronous fault (lower overhead; fault delivered later).
47    fn set_check_mode(synchronous: bool);
48}
49
50/// No-op stub for non-AArch64 targets.
51///
52/// Allows code that is generic over `ViMte` to compile everywhere without
53/// `#[cfg(target_arch = "aarch64")]` at every call site.
54#[cfg(not(target_arch = "aarch64"))]
55pub struct NoMte;
56
57#[cfg(not(target_arch = "aarch64"))]
58impl ViMte for NoMte {
59    fn is_available() -> bool {
60        false
61    }
62
63    unsafe fn tag_region(_ptr: *mut u8, _len: usize, _color: u8) {}
64
65    unsafe fn get_tag(_ptr: *const u8) -> u8 {
66        0
67    }
68
69    fn set_check_mode(_synchronous: bool) {}
70}
