2//! ConnectionManager — multi-path peer connection with relay fallback.
3//!
4//! Connection strategy per peer:
5//!   1. Try direct TCP (LAN address from ticket.addrs[0])       — 2s timeout
6//!   2. Try direct TCP (STUN reflexive, ticket.addrs[1])        — 2s timeout
7//!   3. Fall back: route through relay (ticket.relay_ip:port)
8//!
9//! The caller (dispatch loop) does not need to know which path is active.
10//! All paths end in a Noise KKpsk0 session in the ConnectionPool.
11
12// reason: implements the multi-path (direct TCP / STUN / relay-fallback)
13// peer connection manager for the net-broker robot-swarm feature; not yet
14// constructed from main.rs — the dispatch loop currently only polls
15// `relay_client.is_connected()` directly and has no routing/connection
16// wiring (P06/P08 TODOs).
17#![allow(dead_code)]
18

30const HEARTBEAT_MS: u64 = 500;
31const CONNECT_TIMEOUT_MS: u32 = 2000;
32
33/// Manages peer connections: direct TCP preferred, relay fallback.
34pub struct ConnectionManager<'a> {
35    pool: &'a mut ConnectionPool,
36    relay: &'a mut RelayClient,
37    identity: &'a BrokerIdentity,
38}
39
40impl<'a> ConnectionManager<'a> {
41    pub fn new(
42        pool: &'a mut ConnectionPool,
43        relay: &'a mut RelayClient,
44        identity: &'a BrokerIdentity,
45    ) -> Self {
46        Self {
47            pool,
48            relay,
49            identity,
50        }
51    }
52
53    /// Ensure a Noise session exists for `peer`. Returns the pool slot index.
54    ///
55    /// Tries direct TCP paths first (up to 2 addrs), then relay.
56    /// `psk` is the K1 cluster PSK. `rng` is the broker's PRNG.
57    pub fn ensure_connected(
58        &mut self,
59        net: &mut NetRef,
60        peer: &PeerTicket,
61        psk: &[u8; 32],
62        my_static: &StaticKeypair,
63        rng: &mut BrokerRng,
64        cluster_id: u64,
65    ) -> ViResult<usize> {
66        // Already connected?
67        if let Some(slot) = self.find_session(&peer.node_id) {
68            return Ok(slot);
69        }
70
71        // Try direct TCP paths.
72        for i in 0..peer.addrs_len as usize {
73            let (ip, port) = peer.addrs[i];
74            if ip == [0, 0, 0, 0] {
75                continue;
76            }
77            match self.try_direct_connect(net, peer, ip, port, psk, my_static, rng, cluster_id) {
78                Ok(slot) => return Ok(slot),
79                Err(_) => continue,
80            }
81        }
82
83        // Relay path: connect relay if not already connected.
84        if !self.relay.is_connected() {
85            self.relay.connect(net)?;
86        }
87
88        // For relay path we don't have a Noise session yet — the relay just forwards
89        // raw frames. The Noise handshake happens over the relay channel.
90        // Return a sentinel slot that indicates relay-only mode.
91        // TODO: implement relay-mediated Noise handshake (requires signaling peer to initiate).
92        Err(ViError::NotFound)
93    }
94
95    /// Find an existing session by node_id. Returns pool slot or None.
96    pub fn find_session(&self, node_id: &CellNetId) -> Option<usize> {
97        // ConnectionPool doesn't expose node_id directly — iterate via cluster_id proxy.
98        // For now, node_id equality is checked externally by routing.rs.
99        // This will be wired properly when routing.rs holds node_id→slot mapping.
100        let _ = node_id;
101        None
102    }
103
104    #[allow(clippy::too_many_arguments)] // reason: Noise handshake needs the full key/identity set; a params struct is planned with the routing wiring
105    fn try_direct_connect(
106        &mut self,
107        net: &mut NetRef,
108        peer: &PeerTicket,
109        addr: [u8; 4],
110        port: u16,
111        psk: &[u8; 32],
112        my_static: &StaticKeypair,
113        rng: &mut BrokerRng,
114        cluster_id: u64,
115    ) -> ViResult<usize> {
116        let mut resp = [0u8; api::ipc::IPC_BUF_SIZE];
117
118        sys_heartbeat(HEARTBEAT_MS);
119        let cap = match net
120            .call::<NetRequest, NetResponse>(&NetRequest::TcpConnect { addr, port }, &mut resp)
121            .map_err(|_| ViError::IO)?
122        {
123            NetResponse::CapId(id) => id,
124            _ => return Err(ViError::IO),
125        };
126
127        // Build Noise session (we are initiator on direct connect).
128        let mut session = NoiseSession::new(
129            rng,
130            psk,
131            my_static,
132            peer.node_id.0,
133            cluster_id,
134            &self.identity.node_id,
135            &peer.node_id,
136            cap,
137            true, // initiator
138        )?;
139
140        session.do_handshake(net)?;
141        let slot = self.pool.insert(session);
142        Ok(slot)
143    }
144}

146/// Look up the reflexive address hint from identity (if STUN has run).
147pub fn reflexive_or_direct(identity: &BrokerIdentity, peer: &PeerTicket) -> Option<([u8; 4], u16)> {
148    // Prefer peer's first direct addr; fall back to reflexive.
149    if peer.addrs_len > 0 && peer.addrs[0].0 != [0, 0, 0, 0] {
150        Some(peer.addrs[0])
151    } else {
152        identity.reflexive_addr
153    }
