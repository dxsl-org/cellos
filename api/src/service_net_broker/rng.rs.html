1use ostd::syscall::sys_get_random;
2/// BrokerRng — VirtIO-RNG-backed ChaCha20 PRNG for Noise ephemeral keygen.
3///
4/// Mirrors `cells/services/net/src/tls/rng.rs`. Panic policy: if the kernel
5/// returns 0 bytes from GetRandom after 64 attempts the device is absent —
6/// the broker must not start the crypto transport without real entropy (the
7/// silent xorshift32 fallback in kernel/src/task/syscall.rs:2493 makes Noise
8/// ephemerals predictable if we proceed).
9use rand_chacha::ChaCha20Rng;
10use rand_core::{CryptoRng, RngCore, SeedableRng};
11
12/// ChaCha20 PRNG seeded from VirtIO-RNG hardware entropy.
13///
14/// Implements `clatter::traits::Rng` (= `RngCore + CryptoRng + Default + Clone`).
15/// `Default` calls `new_seeded()` so clatter's internal ephemeral keygen is
16/// also fail-closed — if it ever calls `RNG::default()`, it panics correctly.
17#[derive(Clone)]
18pub struct BrokerRng(ChaCha20Rng);
19
20impl BrokerRng {
21    /// Seed from VirtIO-RNG. Loops until 32 bytes are filled.
22    ///
23    /// # Panics
24    /// Panics if VirtIO-RNG is absent (returns 0 bytes after 64 attempts).
25    pub fn new_seeded() -> Self {
26        let mut seed = [0u8; 32];
27        let mut filled = 0usize;
28        let mut attempts = 0u32;
29        while filled < 32 {
30            let n = sys_get_random(&mut seed[filled..]);
31            if n > 0 {
32                filled += n;
33            } else {
34                attempts += 1;
35                assert!(
36                    attempts < 64,
37                    "[net-broker] VirtIO-RNG absent — cannot generate Noise ephemerals. \
38                     Add -object rng-random,id=rng0 -device virtio-rng-device,rng=rng0 to QEMU."
39                );
40                for _ in 0..1000 {
41                    core::hint::spin_loop();
42                }
43            }
44        }
45        Self(ChaCha20Rng::from_seed(seed))
46    }
47}
48
49/// Fail-closed Default: new_seeded() panics if VirtIO-RNG is absent.
50/// Required by clatter's Rng bound; ensures ANY internal ephemeral keygen
51/// in clatter is also fail-closed, not just explicit calls.
52impl Default for BrokerRng {
53    fn default() -> Self {
54        Self::new_seeded()
55    }
56}
57
58impl RngCore for BrokerRng {
59    fn next_u32(&mut self) -> u32 {
60        self.0.next_u32()
61    }
62    fn next_u64(&mut self) -> u64 {
63        self.0.next_u64()
64    }
65    fn fill_bytes(&mut self, dest: &mut [u8]) {
66        self.0.fill_bytes(dest)
67    }
68    fn try_fill_bytes(&mut self, dest: &mut [u8]) -> Result<(), rand_core::Error> {
69        self.0.try_fill_bytes(dest)
70    }
71}
