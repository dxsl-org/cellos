2//! Cellos relay client — raw TCP relay for internet connectivity.
3//!
4//! Implements a simple relay protocol over raw TCP (no WebSocket).
5//! Payload on the relay wire is Noise-encrypted end-to-end; the relay only
6//! sees NodeIds and byte counts.
7//!
8//! ## Frame format (all fields big-endian)
9//! ```text
10//! [4B length (u32 BE)][1B frame_type][data]
11//! ```
12//!
13//! Frame types:
14//!   CLIENT_REGISTER (0x01): node_id(32)        — register with relay
15//!   SERVER_ACK      (0x02): status(1)           — 0x00 = ok
16//!   SEND_PACKET     (0x08): dest_node_id(32) + payload(N)
17//!   RECV_PACKET     (0x09): src_node_id(32)  + payload(N)
18//!   PING            (0x0b): data(8)
19//!   PONG            (0x0c): data(8)
20//!
21//! ## Blocking contract
22//! `recv_one` BLOCKS until a frame arrives or the connection closes.
23//! Re-arm sys_heartbeat before calling to prevent watchdog expiry.
24//! Call `recv_one` only when the dispatch loop has a free slot — not on every
25//! iteration. See `main.rs` for the recommended polling cadence.
26
27// reason: this module implements the raw-TCP relay fallback protocol for the
28// net-broker robot-swarm feature (internet relay when peers aren't LAN-reachable).
29// `main.rs` imports and calls only `RelayClient::new` + `is_connected` (main.rs:71,
30// 122, 132) as a liveness stub; the frame types, `recv_one`, and send/register paths
31// are unused because inbound relay frames are not yet wired into dispatch
32// (main.rs:131 TODO). Partially connected, not fully wired.
33#![allow(dead_code)]
34
35use api::cluster::CellNetId;
36use api::ipc::{NetRequest, NetResponse};
37use ostd::service::NetRef;
38use ostd::syscall::sys_heartbeat;
39use ostd::{ViError, ViResult};
40
41const HEARTBEAT_MS: u64 = 500;
42
43const FT_CLIENT_REGISTER: u8 = 0x01;
44const FT_SERVER_ACK: u8 = 0x02;
45const FT_SEND_PACKET: u8 = 0x08;
46const FT_RECV_PACKET: u8 = 0x09;
47const FT_PING: u8 = 0x0b;
48const FT_PONG: u8 = 0x0c;
49
50const MAX_FRAME: usize = api::ipc::IPC_BUF_SIZE - 32; // leave room for NodeId header
51
52/// Cellos relay client. Holds one persistent TCP connection to a relay server.
53pub struct RelayClient {
54    pub node_id: CellNetId,
55    relay_ip: [u8; 4],
56    relay_port: u16,
57    tcp_cap: Option<u32>,
58}
59
60impl RelayClient {
61    pub fn new(node_id: CellNetId, relay_ip: [u8; 4], relay_port: u16) -> Self {
62        Self {
63            node_id,
64            relay_ip,
65            relay_port,
66            tcp_cap: None,
67        }
68    }
69
70    pub fn is_connected(&self) -> bool {
71        self.tcp_cap.is_some()
72    }
73
74    /// Connect to the relay and register our NodeId.
75    pub fn connect(&mut self, net: &mut NetRef) -> ViResult<()> {
76        if self.tcp_cap.is_some() {
77            return Ok(());
78        }
79
80        let mut resp = [0u8; api::ipc::IPC_BUF_SIZE];
81        sys_heartbeat(HEARTBEAT_MS);
82        let cap = match net
83            .call::<NetRequest, NetResponse>(
84                &NetRequest::TcpConnect {
85                    addr: self.relay_ip,
86                    port: self.relay_port,
87                },
88                &mut resp,
89            )
90            .map_err(|_| ViError::IO)?
91        {
92            NetResponse::CapId(id) => id,
93            _ => return Err(ViError::IO),
94        };
95
96        // CLIENT_REGISTER: [4B len=33][0x01][node_id(32)]
97        let mut reg = [0u8; 37];
98        reg[0..4].copy_from_slice(&33u32.to_be_bytes());
99        reg[4] = FT_CLIENT_REGISTER;
100        reg[5..37].copy_from_slice(&self.node_id.0);
101        sys_heartbeat(HEARTBEAT_MS);
102        net.call::<NetRequest, NetResponse>(
103            &NetRequest::TcpSend {
104                cap_id: cap,
105                data: &reg,
106            },
107            &mut resp,
108        )
109        .map_err(|_| ViError::IO)?;
110
111        // Read SERVER_ACK frame (2 bytes body: type + status).
112        let mut frame_buf = [0u8; api::ipc::IPC_BUF_SIZE];
113        let n = recv_frame_into(net, cap, &mut frame_buf)?;
114        if n < 2 || frame_buf[0] != FT_SERVER_ACK || frame_buf[1] != 0x00 {
115            let _ = net
116                .call::<NetRequest, NetResponse>(&NetRequest::TcpClose { cap_id: cap }, &mut resp);
117            return Err(ViError::IO);
118        }
119
120        self.tcp_cap = Some(cap);
121        Ok(())
122    }
123
124    /// Send a Noise-encrypted payload to `dest` via the relay.
125    pub fn send(&mut self, net: &mut NetRef, dest: &CellNetId, payload: &[u8]) -> ViResult<()> {
126        let cap = self.tcp_cap.ok_or(ViError::IO)?;
127        if payload.len() > MAX_FRAME {
128            return Err(ViError::InvalidArgument);
129        }
130
131        // Header: [4B len = 1 + 32 + payload_len][0x08][dest(32)]
132        let data_len: u32 = (1 + 32 + payload.len()) as u32;
133        let mut hdr = [0u8; 37];
134        hdr[0..4].copy_from_slice(&data_len.to_be_bytes());
135        hdr[4] = FT_SEND_PACKET;
136        hdr[5..37].copy_from_slice(&dest.0);
137
138        let mut resp = [0u8; api::ipc::IPC_BUF_SIZE];
139        sys_heartbeat(HEARTBEAT_MS);
140        net.call::<NetRequest, NetResponse>(
141            &NetRequest::TcpSend {
142                cap_id: cap,
143                data: &hdr,
144            },
145            &mut resp,
146        )
147        .map_err(|_| ViError::IO)?;
148        net.call::<NetRequest, NetResponse>(
149            &NetRequest::TcpSend {
150                cap_id: cap,
151                data: payload,
152            },
153            &mut resp,
154        )
155        .map_err(|_| ViError::IO)?;
156        Ok(())
157    }
158
159    /// Receive one relay frame (BLOCKS until data arrives or connection closes).
160    ///
161    /// Re-arm heartbeat before calling. Returns:
162    /// - `Ok(Some((src_node_id, payload_end)))` — RECV_PACKET received; payload is
163    ///   stored in `buf[33..payload_end]`.
164    /// - `Ok(None)` — PING (handled internally) or unknown frame type.
165    /// - `Err(_)` — connection lost; caller should mark relay as disconnected.
166    pub fn recv_one(
167        &mut self,
168        net: &mut NetRef,
169        buf: &mut [u8; api::ipc::IPC_BUF_SIZE],
170    ) -> ViResult<Option<(CellNetId, usize)>> {
171        let cap = self.tcp_cap.ok_or(ViError::IO)?;
172        sys_heartbeat(HEARTBEAT_MS);
173        let n = match recv_frame_into(net, cap, buf) {
174            Ok(n) => n,
175            Err(e) => {
176                self.tcp_cap = None;
177                return Err(e);
178            }
179        };
180        if n == 0 {
181            self.tcp_cap = None;
182            return Err(ViError::IO);
183        }
184
185        match buf[0] {
186            FT_RECV_PACKET if n >= 33 => {
187                let mut src = [0u8; 32];
188                src.copy_from_slice(&buf[1..33]);
189                Ok(Some((CellNetId::from_bytes(src), n)))
190            }
191            FT_PING if n >= 9 => {
192                let mut ping_data = [0u8; 8];
193                ping_data.copy_from_slice(&buf[1..9]);
194                self.send_pong(net, &ping_data);
195                Ok(None)
196            }
197            _ => Ok(None),
198        }
199    }
200
201    /// Check TCP connection state (0x03 = Established).
202    pub fn check_connected(&mut self, net: &mut NetRef) -> bool {
203        let cap = match self.tcp_cap {
204            Some(c) => c,
205            None => return false,
206        };
207        let mut resp = [0u8; api::ipc::IPC_BUF_SIZE];
208        matches!(
209            net.call::<NetRequest, NetResponse>(
210                &NetRequest::SocketState { cap_id: cap },
211                &mut resp
212            ),
213            Ok(NetResponse::State(0x03))
214        )
215    }
216
217    pub fn disconnect(&mut self, net: &mut NetRef) {
218        if let Some(cap) = self.tcp_cap.take() {
219            let mut resp = [0u8; api::ipc::IPC_BUF_SIZE];
220            let _ = net
221                .call::<NetRequest, NetResponse>(&NetRequest::TcpClose { cap_id: cap }, &mut resp);
222        }
223    }
224
225    fn send_pong(&self, net: &mut NetRef, data: &[u8; 8]) {
226        if let Some(cap) = self.tcp_cap {
227            let mut frame = [0u8; 13];
228            frame[0..4].copy_from_slice(&9u32.to_be_bytes()); // len = 1 + 8
229            frame[4] = FT_PONG;
230            frame[5..13].copy_from_slice(data);
231            let mut resp = [0u8; api::ipc::IPC_BUF_SIZE];
232            let _ = net.call::<NetRequest, NetResponse>(
233                &NetRequest::TcpSend {
234                    cap_id: cap,
235                    data: &frame,
236                },
237                &mut resp,
238            );
239        }
240    }
241}

245/// Read exactly `n` bytes from TCP into `dest[..n]`, looping on short reads.
246///
247/// Each TcpRecv call may return fewer bytes than requested (TCP segmentation).
248/// This function retries until all `n` bytes are received or an error occurs.
249fn tcp_read_exact(net: &mut NetRef, cap_id: u32, dest: &mut [u8], n: usize) -> ViResult<()> {
250    let mut received = 0;
251    while received < n {
252        sys_heartbeat(HEARTBEAT_MS);
253        let want = (n - received) as u32;
254        let mut tmp = [0u8; api::ipc::IPC_BUF_SIZE];
255        let chunk = match net
256            .call::<NetRequest, NetResponse>(
257                &NetRequest::TcpRecv {
258                    cap_id,
259                    buf_len: want,
260                },
261                &mut tmp,
262            )
263            .map_err(|_| ViError::IO)?
264        {
265            NetResponse::Data(d) => d,
266            _ => return Err(ViError::IO),
267        };
268        if chunk.is_empty() {
269            return Err(ViError::IO);
270        } // connection closed
271        let len = chunk.len().min(n - received);
272        dest[received..received + len].copy_from_slice(&chunk[..len]);
273        received += len;
274    }
275    Ok(())
276}

278/// Read one length-prefixed relay frame into `out`.
279/// Returns the number of bytes written (= frame body length, including type byte).
280/// `out[0]` = frame type; `out[1..]` = frame data.
281fn recv_frame_into(
282    net: &mut NetRef,
283    cap_id: u32,
284    out: &mut [u8; api::ipc::IPC_BUF_SIZE],
285) -> ViResult<usize> {
286    // Read 4-byte big-endian length prefix.
287    let mut hdr = [0u8; 4];
288    tcp_read_exact(net, cap_id, &mut hdr, 4)?;
289    let frame_len = u32::from_be_bytes(hdr) as usize;
290    if frame_len == 0 || frame_len > MAX_FRAME {
291        return Err(ViError::IO);
292    }
293
294    // Read frame body (type + data) directly into `out`.
295    tcp_read_exact(net, cap_id, &mut out[..frame_len], frame_len)?;
296    Ok(frame_len)
297}
