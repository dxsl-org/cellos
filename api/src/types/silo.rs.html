2// Host-side protocol types for the Tier 3a Security Silo.
3//
4// The silo is a bare-metal AArch64 guest that holds a P-256 private key in
5// Stage-2 fenced memory. The host communicates through a shared mailbox page
6// mapped at IPA 0x4000_3000.  All constants here must stay byte-for-byte
7// consistent with the guest-side `mailbox.rs`.
8
9/// IPA of the 4 KiB mailbox page pre-mapped by the VMM before guest boot.
10pub const MAILBOX_IPA: u64 = 0x4000_3000;
11
12// ── HVC function IDs (SMCCC-style vendor range 0xC600_0080..0xC600_00FF) ──
13
14/// Guest signals: key initialisation complete; response data = uncompressed pub key.
15pub const HVC_SILO_READY: u64 = 0xC600_0080;
16/// Guest signals: operation finished; response data in mailbox.
17pub const HVC_SILO_DONE: u64 = 0xC600_0081;
18/// Guest signals: unrecoverable error; mailbox data[0] = error code.
19pub const HVC_SILO_FAULT: u64 = 0xC600_0082;
20
21// ── Mailbox protocol ──────────────────────────────────────────────────────
22
23/// Commands the host can send to the silo guest.
24#[repr(u8)]
25#[derive(Debug, Clone, Copy, PartialEq, Eq)]
26pub enum SiloCmd {
27    /// Key initialisation: `data[0..32]` = 32-byte entropy seed.
28    Init = 0,
29    /// Sign a pre-hashed digest: `data[0..32]` = SHA-256 digest.
30    Sign = 1,
31    /// Static-static ECDH: `data[0..65]` = peer uncompressed public key.
32    Ecdh = 2,
33    /// Return own public key; no input data required.
34    GetPub = 3,
35}
36
37/// Response codes the silo guest writes back to the mailbox.
38#[repr(u8)]
39#[derive(Debug, Clone, Copy, PartialEq, Eq)]
40pub enum SiloRespCode {
41    /// Init succeeded: `data[0..65]` = uncompressed public key.
42    Ready = 0,
43    /// Sign succeeded: `data[0]` = DER length, `data[1..=len]` = DER signature.
44    Signature = 1,
45    /// ECDH succeeded: `data[0..32]` = raw shared secret.
46    Secret = 2,
47    /// GetPub succeeded: `data[0..65]` = uncompressed public key.
48    PubKey = 3,
49    /// Operation failed: `data[0]` = error code.
50    Fault = 0xFF,
51}
52
53/// 4 KiB shared-memory page at `MAILBOX_IPA`.
54///
55/// The host writes `seq`, `cmd`, and `data`; then resumes the guest.
56/// The guest reads the snapshot, processes it, writes `resp` + `data`, then
57/// fires an HVC to signal completion.  The host must never touch the page
58/// between guest resume and the HVC callback.
59#[repr(C)]
60#[derive(Clone, Copy)]
61pub struct MailboxPage {
62    /// Sequence number: host increments before each request; guest echoes it.
63    pub seq: u32,
64    /// Command discriminant — cast from `SiloCmd`.
65    pub cmd: u8,
66    /// Response code discriminant — written by guest, cast from `SiloRespCode`.
67    pub resp: u8,
68    pub _pad: [u8; 2],
69    /// Payload: input data (host→guest) / output data (guest→host).
70    pub data: [u8; 4088],
71}
72
73const _MAILBOX_SIZE_CHECK: () = assert!(core::mem::size_of::<MailboxPage>() == 4096);
74
75// ── IPC wire format (host cell ↔ silo service cell) ──────────────────────
76
77/// IPC request from a caller cell to the silo service cell (128 bytes total).
78///
79/// Fits in a single ViCell IPC message (≤ 4096 bytes; this is intentionally
80/// small so it routes through the fast-path without a grant).
81#[repr(C)]
82#[derive(Clone, Copy)]
83pub struct SiloRequest {
84    /// Operation code — cast from `SiloCmd`.
85    pub opcode: u8,
86    pub _pad: [u8; 31],
87    /// Payload: up to 65 bytes for an uncompressed P-256 public key (ECDH).
88    pub data: [u8; 96],
89}
90
91/// IPC response from the silo service cell to the caller (128 bytes total).
92#[repr(C)]
93#[derive(Clone, Copy)]
94pub struct SiloResponse {
95    /// `0` = success, `0xFF` = fault.
96    pub status: u8,
97    /// Number of valid bytes in `data`.
98    pub len: u8,
99    pub _pad: [u8; 2],
100    /// Result bytes: DER signature (≤ 72 B), shared secret (32 B), or pub key (65 B).
101    pub data: [u8; 124],
102}
103
104const _SILO_REQ_SIZE_CHECK: () = assert!(core::mem::size_of::<SiloRequest>() == 128);
105const _SILO_RESP_SIZE_CHECK: () = assert!(core::mem::size_of::<SiloResponse>() == 128);
106
107/// Service registry ID for the silo service cell.
108pub const SILO_SERVICE_ID: u16 = 6;
