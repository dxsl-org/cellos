8/// Read the spawn-args stash into `buf` and return the trimmed content.
9fn spawn_args(buf: &mut [u8]) -> &str {
10    let n = syscall::sys_spawn_args(buf);
11    core::str::from_utf8(&buf[..n]).unwrap_or("").trim()
12}
13
14/// kill <tid> — send cooperative shutdown signal or force-exit a task.
15///
16/// Cooperative (state=Waiting): sends 0xFF via sys_send; the task's recv loop
17/// checks for this sentinel and exits cleanly.
18/// Force (state=Ready/Running): calls sys_force_exit; the kernel terminates the
19/// task immediately.  System Cells with SpawnCap may reject force-exit.
20#[no_mangle]
21pub fn main() {
22    let mut arg_buf = [0u8; 64];
23    let arg = spawn_args(&mut arg_buf);
24    let tid_str = arg.split_whitespace().next().unwrap_or("");
25
26    if tid_str.is_empty() {
27        io::println("usage: kill <tid>");
28        syscall::sys_exit(1);
29    }
30
31    let tid: usize = match tid_str.parse() {
32        Ok(n) => n,
33        Err(_) => {
34            io::print("kill: invalid tid: ");
35            io::println(tid_str);
36            syscall::sys_exit(1);
37        }
38    };
39
40    // Read process table to determine the target task's state.
41    let mut buf = [ProcessInfo::default(); 32];
42    let count = syscall::sys_get_procs(&mut buf).unwrap_or(0);
43
44    let state = (0..count).find(|&i| buf[i].id == tid).map(|i| buf[i].state);
45
46    match state {
47        Some(2) => {
48            // Task is blocked in a recv call — send cooperative shutdown sentinel.
49            let _ = syscall::sys_send(tid, &[0xFF_u8]);
50            io::print("kill: sent shutdown signal to ");
51            io::print_usize(tid);
52            io::println("");
53        }
54        Some(0) | Some(1) => {
55            // Task is Ready or Running — force terminate.
56            let _ = syscall::sys_force_exit(tid);
57            io::print("kill: force-exited ");
58            io::print_usize(tid);
59            io::println("");
60        }
61        Some(3) => {
62            io::print("kill: task ");
63            io::print_usize(tid);
64            io::println(" is already terminated");
65        }
66        _ => {
67            io::print("kill: task not found: ");
68            io::print_usize(tid);
69            io::println("");
70            syscall::sys_exit(1);
71        }
72    }
73
74    syscall::sys_exit(0);
75}
