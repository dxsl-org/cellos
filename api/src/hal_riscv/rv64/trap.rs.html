5/// Trap frame saved on stack during exception/interrupt.
6/// Must match the layout in trap.S exactly!
7#[derive(Debug, Clone, Copy, Default)]
8#[repr(C)]
9pub struct ViTrapFrame {
10    pub regs: [usize; 32], // x0-x31 (x0 always 0 but slot exists)
11    pub sstatus: usize,
12    pub sepc: usize,
13    pub stval: usize,
14    pub scause: usize,
15}
16
17impl ViTrapFrame {
18    pub fn new() -> Self {
19        Self::default()
20    }
21}

29/// Initialize trap handling by setting stvec
30pub fn init() {
31    unsafe {
32        let trap_entry = __trap_entry as *const () as usize;
33        // Set stvec to direct mode (all traps go to __trap_entry)
34        core::arch::asm!("csrw stvec, {}", in(reg) trap_entry);
35        // Initialize sscratch to 0 (indicates S-mode context)
36        core::arch::asm!("csrw sscratch, zero");
37    }
38}

40/// No-op on RISC-V since the nested-safe sscratch protocol (bug #7 fix).
41///
42/// INVARIANT: sscratch == 0 for the entire time a hart runs in S-mode; the
43/// trap-exit path loads the task's kernel-stack top (frame base + frame size)
44/// right before `sret`. The scheduler used to call this mid-S-mode with the
45/// next task's SAVED sp — a mid-stack pointer — so any nested trap (timer IRQ
46/// after a context with SIE=1 was restored) swapped that pointer in as "the
47/// kernel stack" and sprayed a trap frame over live memory.
48///
49/// x86 (TSS RSP0 / syscall MSR) and ARM (TPIDR) still need their versions —
50/// the multi-arch call site in `task::yield_cpu` stays.
51pub fn set_kernel_stack(_kernel_stack_top: usize) {}

53pub fn enable_interrupts() {
54    unsafe {
55        #[cfg(target_arch = "riscv64")]
56        core::arch::asm!("csrsi sstatus, 0x2"); // SIE
57    }
58}

60/// Rust trap handler called from assembly (vi_trap_handler)
61/// Uses borrowed &mut ViTrapFrame per Luật 8
62/// This function handles all traps: syscalls, interrupts, exceptions
63#[no_mangle]
64pub extern "C" fn vi_trap_handler(frame: &mut ViTrapFrame) {
65    let scause = frame.scause;
66    let is_interrupt = (scause >> 63) != 0;
67    let code = scause & 0x7FFF_FFFF_FFFF_FFFF;
68
69    if is_interrupt {
70        // Handle interrupts
71        match code {
72            1 => {
73                // S-mode software interrupt — zero-latency RT preemption.
74                // Cleared here before yield so it does not re-fire immediately.
75                // SAFETY: csrci on sip.SSIP is permitted from S-mode (priv spec §4.1.3).
76                unsafe { core::arch::asm!("csrci sip, 0x2") };
77                // Reuse the timer tick path: just run the scheduler.
78                unsafe {
79                    vi_timer_tick();
80                }
81            }
82            5 => {
83                // S-mode timer interrupt — preemption point.
84                // SAFETY: vi_timer_tick is defined in kernel::task and linked
85                // via extern "Rust".  It increments the tick counter, rearmed
86                // the timer, and calls yield_cpu() to preempt if needed.
87                unsafe {
88                    vi_timer_tick();
89                }
90            }
91            9 => {
92                // S-mode external interrupt (PLIC)
93                // Claim first, dispatch handler, complete AFTER handler per PLIC spec.
94                if let Some(irq) = plic_claim() {
95                    if (1..=8).contains(&irq) {
96                        // SAFETY: vi_handle_virtio_irq is defined in kernel/src/task/drivers/virtio_blk.rs
97                        // and linked via extern "Rust". The irq argument is a valid PLIC claim value (1-8).
98                        unsafe {
99                            vi_handle_virtio_irq(irq);
100                        }
101                    } else if irq == 10 {
102                        // SAFETY: vi_handle_uart_irq is defined in the kernel and linked via extern "Rust".
103                        unsafe {
104                            vi_handle_uart_irq();
105                        }
106                    }
107                    // PLIC complete must come AFTER the device handler has run.
108                    plic_complete(irq);
109                }
110            }
111            _ => {
112                // Unknown interrupt - log but don't panic
113                // log::warn!("Unknown interrupt: {}", code);
114            }
115        }
116    } else {
117        // Handle exceptions
118        match code {
119            8 => {
120                // Environment call from U-mode (syscall)
121                vi_handle_syscall(frame);
122                // Advance PC past ecall instruction (4 bytes)
123                frame.sepc += 4;
124            }
125            9 => {
126                // Environment call from S-mode (should not happen normally)
127                frame.sepc += 4;
128            }
129            _ => {
130                // Illegal instruction, page faults, or other unhandled exception.
131                //
132                // Distinguish U-mode Cell faults from S-mode kernel faults using SPP
133                // (sstatus bit 8): SPP=0 means the trap came from U-mode (a Cell).
134                // Checking only `cell_id != 0` is insufficient — if the kernel faults
135                // while servicing a Cell's syscall, CURRENT_CELL_ID is still non-zero
136                // but the CPU is in S-mode.  Misclassifying that as a Cell fault
137                // silently kills the Cell and hides the kernel bug.
138                //
139                // SAFETY: vi_current_cell_id and vi_terminate_on_fault are defined
140                // in kernel::task and linked via extern "Rust".
141                let from_user = (frame.sstatus & 0x100) == 0; // SPP bit: 0=U-mode
142                let cell_id = unsafe { vi_current_cell_id() };
143                if from_user && cell_id != 0 {
144                    // Genuine U-mode Cell fault — terminate the Cell, let kernel continue.
145                    unsafe {
146                        vi_terminate_on_fault(code, frame.sepc, frame.stval);
147                    }
148                    // vi_terminate_on_fault calls yield_cpu() which switches away.
149                    // We should not reach here, but return safely if we do.
150                } else {
151                    // True kernel fault (S-mode) or U-mode fault without a registered Cell.
152                    panic!(
153                        "ViCell: Kernel exception: scause={} sepc={:#x} stval={:#x} sstatus={:#x}",
154                        code, frame.sepc, frame.stval, frame.sstatus
155                    );
156                }
157            }
158        }
159    }
160}

162/// Claim the highest-priority pending IRQ from PLIC (S-mode context 1).
163/// Returns the IRQ number, or None if no interrupt is pending.
164/// The caller MUST call `plic_complete(irq)` after the device handler returns.
165fn plic_claim() -> Option<u32> {
166    use crate::common::plic::PLIC;
167    let irq = PLIC.claim(1);
168    if irq != 0 {
169        Some(irq)
170    } else {
171        None
172    }
173}
174
175/// Notify PLIC that IRQ handling is complete (S-mode context 1).
176/// Must be called AFTER the device handler has finished.
177fn plic_complete(irq: u32) {
178    use crate::common::plic::PLIC;
179    PLIC.complete(1, irq);
180}

182/// Handle syscall from userspace (Vi prefix per Luật 6)
183fn vi_handle_syscall(frame: &mut ViTrapFrame) {
184    extern "Rust" {
185        fn ViCell_syscall_dispatch(frame: &mut ViTrapFrame);
