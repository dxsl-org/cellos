1/// Boot module - Assembly entry point and early initialization
2///
3/// This module handles the transition from bootloader to Rust code
4use core::arch::global_asm;
5
6// Secondary-hart entry point.
7// Hart 0's `_start` runs self-relocation + BSS clear and MUST NOT be re-entered.
8// Secondary harts start here — bare entry, no relocation, no BSS clear.
9// a0 = hart_id (set by OpenSBI HSM per SBI spec §9.1.1)
10// a1 = opaque = kernel stack top (our convention, passed in sbi_hart_start)
11global_asm!(
12    r#"
13    .section .text
14    .global _secondary_entry
15    .align 2
16_secondary_entry:
17    .option push
18    .option norelax
19    lla gp, __global_pointer$   # restore GP (PC-relative, GOT-independent)
20    .option pop
21    mv  sp, a1          # kernel stack = stack_top from SBI opaque
22    mv  tp, zero        # Phase 01 placeholder; Phase 02 sets real HartLocal ptr
23    call smp_hart_entry # fn(hart_id: usize) -> !   (a0 = hart_id from SBI)
24.secondary_halt:
25    wfi
26    j .secondary_halt
27    "#
28);
29
30// Assembly boot code
31global_asm!(
32    r#"
33    .section .text.boot
34    .global _start
35_start:
36    # 1. Disable interrupts
37    csrw sie, zero
38    csrw sip, zero
39
40    # CRITICAL: every address below uses `lla` (load-local-address), which
41    # always expands to PC-relative `auipc`+`addi`. Plain `la` under PIC
42    # codegen (relocation-model=pic) expands to a GOT-indirect load
43    # (`auipc`+`ld [GOT]`), and the GOT slots are exactly the un-relocated
44    # NULL pointers we are about to fix — a chicken-and-egg that left gp/sp/t*
45    # all zero and faulted before the first UART byte. `lla` is GOT-independent
46    # and correct at any load bias.
47    .option push
48    .option norelax
49
50    # 2. Initialize global pointer (gp) — PC-relative, GOT-independent.
51    lla gp, __global_pointer$
52
53    # 3. Initialize thread pointer (tp) - Clear it for now
54    mv tp, zero
55
56    # 4. Set up stack pointer
57    lla sp, __stack_top
58
59    # 5. Self-apply R_RISCV_RELATIVE relocations.
60    #    The kernel is linked -pie (ET_DYN). Under Limine the bootloader applies
61    #    these; under direct QEMU `-kernel` there is no loader, so we apply them
62    #    ourselves before any global/GOT slot is dereferenced (the first println!
63    #    reads a .data/.got pointer that is otherwise an un-relocated NULL).
64    #
65    #    Load bias (slide) = runtime(_start) - link_origin(0x80200000).
66    #    Each Elf64_Rela is 24 bytes: r_offset(8) r_info(8) r_addend(8).
67    #    For R_RISCV_RELATIVE: *(slide + r_offset) = slide + r_addend.
68    #    Uses only t0-t6, preserves a0/a1 (hartid/dtb) for kmain. At slide=0
69    #    (direct -kernel) this writes the identity-correct absolute addresses;
70    #    under Limine (slide!=0) it is also correct.
71    lla  t0, _start            # t0 = runtime address of _start
72    li   t1, 0x80200000        # t1 = link-time origin
73    sub  t2, t0, t1            # t2 = slide (load bias); 0 for direct -kernel
74    lla  t3, __rela_dyn_start  # t3 = &rela[0]
75    lla  t4, __rela_dyn_end    # t4 = end
766:
77    bgeu t3, t4, 7f            # done when t3 >= end
78    ld   t5, 8(t3)             # t5 = r_info
79    li   t6, 3                 # R_RISCV_RELATIVE == 3 (full r_info, sym idx 0)
80    bne  t5, t6, 8f            # skip non-RELATIVE entries (none expected)
81    ld   t5, 0(t3)             # t5 = r_offset (link-relative)
82    add  t5, t5, t2            # t5 = slot runtime addr = slide + r_offset
83    ld   t6, 16(t3)            # t6 = r_addend (link-relative target)
84    add  t6, t6, t2            # t6 = slide + r_addend
85    sd   t6, 0(t5)             # *slot = relocated value
868:
87    addi t3, t3, 24            # advance to next Elf64_Rela
88    j    6b
897:
90
91    # 6. Clear BSS section (PC-relative bounds).
92    lla t0, __bss_start
93    lla t1, __bss_end
941:
95    bgeu t0, t1, 2f
96    sd zero, 0(t0)
97    addi t0, t0, 8
98    j 1b
992:
100    .option pop
101
102    # Jump to Rust entry point (defined in kernel)
103    call kmain
104    
105    # If kmain returns, halt
1063:
107    wfi
108    j 3b
109    "#
110);
