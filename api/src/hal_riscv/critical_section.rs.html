1//! critical-section implementation for RISC-V (S-mode, both rv32 and rv64).
2//!
3//! Enabled via the `critical-section-impl` feature on `hal-riscv`.
4//! Only one impl may be linked per binary — the kernel enables this feature;
5//! Cell crates must NOT enable it.
6//!
7//! Mechanism: `csrrci` atomically reads and clears sstatus.SIE (bit 1).
8//! `release` restores the full saved sstatus, re-enabling interrupts only if
9//! they were enabled before `acquire`.
10//!
11//! # SMP limitation
12//!
13//! This impl masks interrupts on the **local hart only**. ViCell supports SMP
14//! (MAX_HARTS=2, hart 1 = RT scheduler). Code protected by
15//! `critical_section::with` is therefore NOT mutually exclusive against
16//! concurrent execution on hart 1.
17//!
18//! **Safe to use:** `heapless::spsc::Queue` (single-producer, single-consumer),
19//! any data only ever accessed from one hart, and interrupt-handler ↔ task
20//! synchronisation on the same hart.
21//!
22//! **Unsafe to use:** `heapless::mpmc::Queue` or any structure that requires
23//! exclusion across all harts simultaneously. Use a `Spinlock` for those.
24
25use critical_section::{set_impl, Impl, RawRestoreState};
26
27struct ViRiscvCs;
28set_impl!(ViRiscvCs);
29
30unsafe impl Impl for ViRiscvCs {
31    unsafe fn acquire() -> RawRestoreState {
32        let sstatus: usize;
33        // SAFETY: csrrci is an atomic read-modify-write on sstatus from S-mode.
34        // Returns the old sstatus so release() can restore it exactly.
35        // Bit 1 = SIE (Supervisor Interrupt Enable).
36        core::arch::asm!(
37            "csrrci {}, sstatus, 0x2",
38            out(reg) sstatus,
39            // nostack: no stack ops. nomem intentionally absent: the asm acts as a
40            // compiler barrier so memory ops do not migrate across acquire/release.
41            options(nostack),
42        );
43        sstatus
44    }
45
46    unsafe fn release(sstatus: RawRestoreState) {
47        // SAFETY: Restoring sstatus to the saved value from acquire().
48        // If SIE was set before acquire, this re-enables interrupts.
49        core::arch::asm!(
50            "csrw sstatus, {}",
51            in(reg) sstatus,
52            options(nostack),
53        );
54    }
55}
