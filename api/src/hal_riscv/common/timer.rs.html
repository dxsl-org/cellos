1//! Timer Interface (S-mode)
2//!
3//! Provides access to the RISC-V timer via SBI calls and the `time` CSR.
4//! Direct CLINT access is NOT available in S-mode — use SBI for `mtimecmp`.
5
6/// Ticks per 10 ms at the assumed 10 MHz `mtime` clock on QEMU virt.
7///
8/// Used to set the preemptive timeslice duration.  If the actual clock
9/// differs (detectable via DTB), callers should adjust accordingly.
10pub const TICKS_PER_10MS: u64 = 100_000;
11
12/// Read the current machine time (via 'time' CSR).
13///
14/// On RV32 the `time` CSR is only 32 bits wide; the upper 32 bits are in
15/// `timeh`.  We use a carry-safe retry loop to avoid a torn 64-bit read
16/// across a `timeh` overflow boundary.
17pub fn read_mtime() -> u64 {
18    #[cfg(target_arch = "riscv64")]
19    {
20        let time: u64;
21        // SAFETY: reads the read-only `time` CSR (0xC01), no side effects.
22        unsafe {
23            core::arch::asm!("csrr {0}, time", out(reg) time, options(nomem, nostack));
24        }
25        time
26    }
27    #[cfg(target_arch = "riscv32")]
28    {
29        // Retry if timeh changes between the two reads (i.e., the lower half
30        // rolled over between the timeh read and the time read).
31        loop {
32            let hi1: u32;
33            let lo: u32;
34            let hi2: u32;
35            // SAFETY: `timeh`/`time` are read-only user-mode CSRs; safe from S-mode.
36            unsafe {
37                core::arch::asm!("csrr {}, timeh", out(reg) hi1, options(nomem, nostack));
38                core::arch::asm!("csrr {}, time",  out(reg) lo,  options(nomem, nostack));
39                core::arch::asm!("csrr {}, timeh", out(reg) hi2, options(nomem, nostack));
40            }
41            if hi1 == hi2 {
42                return ((hi1 as u64) << 32) | (lo as u64);
43            }
44        }
45    }
46    #[cfg(not(any(target_arch = "riscv64", target_arch = "riscv32")))]
47    {
48        0
49    }
50}
51
52/// Get time in milliseconds (assuming 10MHz clock)
53pub fn time_ms() -> u64 {
54    read_mtime() / 10_000
55}
56
57/// Set a timer interrupt to fire after `ms` milliseconds
58pub fn set_timer_ms(ms: u64) {
59    let current = read_mtime();
60    let target = current + (ms * 10_000);
61
62    // Use SBI call to set timer in M-mode
63    crate::common::sbi::set_timer(target);
64}
