2//!
3//! Allows S-mode kernel to make requests to M-mode OpenSBI firmware.

7// SBI Extension IDs
8const SBI_EID_TIMER: usize = 0x54494D45;
9const SBI_EID_LEGACY_SET_TIMER: usize = 0x00;
10const SBI_EID_DBCN: usize = 0x4442434E;
11const SBI_FID_DBCN_WRITE: usize = 0;
12
13#[inline(always)]
14fn sbi_call(eid: usize, fid: usize, arg0: usize, arg1: usize, arg2: usize) -> (usize, usize) {
15    let mut error;
16    let mut value;
17    unsafe {
18        core::arch::asm!(
19            "ecall",
20            in("a7") eid,
21            in("a6") fid,
22            in("a0") arg0,
23            in("a1") arg1,
24            in("a2") arg2,
25            lateout("a0") error,
26            lateout("a1") value,
27            options(nostack)
28        );
29    }
30    (error, value)
31}

33/// Write a character to debug console (DBCN)
34pub fn console_putchar(c: u8) -> Result<(), usize> {
35    let byte = c;
36    let ptr = &byte as *const u8 as usize;
37    // console_write(num_bytes=1, base_addr_lo=ptr, base_addr_hi=0)
38    let (error, _value) = sbi_call(SBI_EID_DBCN, SBI_FID_DBCN_WRITE, 1, ptr, 0);
39
40    if error == 0 {
41        Ok(())
42    } else {
43        Err(error)
44    }
45}

47const SBI_FID_DBCN_READ: usize = 1;
48
49/// Read a character from debug console (DBCN)
50pub fn console_getchar() -> isize {
51    let mut byte: u8 = 0;
52    let ptr = &mut byte as *mut u8 as usize;
53    let (error, value) = sbi_call(SBI_EID_DBCN, SBI_FID_DBCN_READ, 1, ptr, 0);
54
55    if error == 0 && value == 1 {
56        byte as isize
57    } else {
58        -1
59    }
60}

62const SBI_FID_SET_TIMER: usize = 0;
63
64/// Set the next timer interrupt deadline.
65///
66/// On RV32 the SBI spec (v1.0 §5.1) requires passing `stime_value_lo` in
67/// `a0` and `stime_value_hi` in `a1` because `usize` is 32-bit — the 64-bit
68/// value cannot fit in a single argument register.
69pub fn set_timer(stime_value: u64) {
70    #[cfg(target_arch = "riscv64")]
71    sbi_call(SBI_EID_TIMER, SBI_FID_SET_TIMER, stime_value as usize, 0, 0);
72    #[cfg(target_arch = "riscv32")]
73    {
74        let lo = stime_value as usize; // bits 31:0
75        let hi = (stime_value >> 32) as usize; // bits 63:32
76        sbi_call(SBI_EID_TIMER, SBI_FID_SET_TIMER, lo, hi, 0);
77    }
78}

80// SBI System Reset (SRST) extension — used by the panic handler to reboot a
81// dead kernel instead of freezing (a robot must come back up, not brick).
82const SBI_EID_SRST: usize = 0x53525354;
83const SBI_FID_SYSTEM_RESET: usize = 0;
84/// SRST reset_type: graceful power-off.
85pub const SBI_RESET_SHUTDOWN: usize = 0x0;
86/// SRST reset_type: full re-init (re-runs firmware + kernel from scratch).
87pub const SBI_RESET_COLD_REBOOT: usize = 0x1;
88
89/// Request a system reset via SBI SRST.
90///
91/// On success the machine resets and this never returns; it returns only if the
92/// firmware does not implement SRST, letting the caller fall back to a halt loop.
93pub fn system_reset(reset_type: usize, reset_reason: usize) {
94    sbi_call(
95        SBI_EID_SRST,
96        SBI_FID_SYSTEM_RESET,
97        reset_type,
98        reset_reason,
99        0,
100    );
101}

103// --- SBI Hart State Management (HSM) extension  SBI spec v2.0 §9 ---
104const SBI_EID_HSM: usize = 0x48534D;
105const SBI_FID_HART_START: usize = 0;
106const SBI_FID_HART_GET_STATUS: usize = 2;
107
108/// Request OpenSBI to start a hart that is in STOPPED state.
109///
110/// `start_addr` is the PHYSICAL address of the secondary's first instruction
111/// (must be 4-byte aligned; hart starts bare — SATP=0).  `opaque` is passed
112/// in the secondary's `a1` register on entry (SBI HSM §9.1.1) and is
113/// conventionally the kernel stack top for that hart.
114///
115/// # Safety
116/// `start_addr` must be a valid, aligned physical address reachable without paging.
117pub fn sbi_hart_start(hart_id: usize, start_addr: usize, opaque: usize) -> Result<(), usize> {
118    let (error, _) = sbi_call(SBI_EID_HSM, SBI_FID_HART_START, hart_id, start_addr, opaque);
119    if error == 0 {
120        Ok(())
121    } else {
122        Err(error)
123    }
124}
125
126/// Query the current HSM state of a hart.
127///
128/// Returns: 0 = Started, 1 = Stopped, 2 = Start Pending, 3 = Stop Pending,
129/// 4 = Suspended, 5 = Suspend Pending, 6 = Resume Pending.
130pub fn sbi_hart_get_status(hart_id: usize) -> Result<usize, usize> {
131    let (error, value) = sbi_call(SBI_EID_HSM, SBI_FID_HART_GET_STATUS, hart_id, 0, 0);
132    if error == 0 {
133        Ok(value)
134    } else {
135        Err(error)
136    }
137}

139// --- SBI IPI (Send Inter-Processor Interrupt) extension  SBI spec v2.0 §4 ---
140const SBI_EID_IPI: usize = 0x735049;
141const SBI_FID_SEND_IPI: usize = 0;
142
143/// Send a software IPI to all harts selected by `hart_mask`.
144///
145/// Bit N in `hart_mask` targets hart `hart_mask_base + N`.  Pass
146/// `hart_mask_base = 0` for absolute hart IDs (e.g. `1 << 1` to ping hart 1).
147/// The IPI sets SSIP on each target; the target's trap handler sees scause=1.
148pub fn sbi_send_ipi(hart_mask: usize, hart_mask_base: usize) -> Result<(), usize> {
149    let (error, _) = sbi_call(SBI_EID_IPI, SBI_FID_SEND_IPI, hart_mask, hart_mask_base, 0);
150    if error == 0 {
151        Ok(())
152    } else {
153        Err(error)
154    }
