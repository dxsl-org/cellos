1//! RISC-V PMP (Physical Memory Protection) region definitions.
2//!
3//! # Architecture Constraint
4//!
5//! PMP CSRs (`pmpaddr*`, `pmpcfg*`) are **M-mode-only**.  Writing them from
6//! S-mode raises an Illegal Instruction trap.  ViCell runs in S-mode under
7//! OpenSBI, so this module **cannot write PMP registers at runtime**.
8//!
9//! This file documents the intended PMP layout for use by a future custom
10//! M-mode firmware shim (Phase 32+).  The `KERNEL_PMP_REGIONS` slice encodes
11//! the desired configuration that the shim should apply before `mret` into
12//! S-mode.
13//!
14//! # QEMU virt PMP budget
15//! - 16 total entries; OpenSBI claims 2 (entry 0 = firmware deny, entry 1 = all-access)
16//! - 14 free entries for ViCell (entries 2–15)
17//! - Entries 2–6 used for kernel + MMIO protection (see KERNEL_PMP_REGIONS)
18//! - Entries 7–15 reserved for per-Cell isolation (Phase 32)
19//!
20//! # NAPOT encoding
21//! `pmpaddr = (base >> 2) | (size/8 - 1)`
22//! Requirements: `base` aligned to `size`; `size` must be a power of two ≥ 8.
23
24/// Permission bits for PMP config entries (`pmpcfg` byte per region).
25pub mod perm {
26    /// Read permission.
27    pub const R: u8 = 0b001;
28    /// Write permission.
29    pub const W: u8 = 0b010;
30    /// Execute permission.
31    pub const X: u8 = 0b100;
32    /// Read + Write.
33    pub const RW: u8 = R | W;
34    /// Read + Execute.
35    pub const RX: u8 = R | X;
36    /// Read + Write + Execute.
37    pub const RWX: u8 = R | W | X;
38    /// Addressing mode: NAPOT (naturally aligned power-of-two).
39    pub const A_NAPOT: u8 = 0b11 << 3;
40    /// Lock bit: entry enforced on M-mode too; cannot be modified until reset.
41    /// Under Smepmp with MML=1, locked entries become M-mode-only rules.
42    pub const L: u8 = 1 << 7;
43}
44
45/// Compute the NAPOT `pmpaddr` value for a region.
46///
47/// `base` must be aligned to `size`; `size` must be a power of two ≥ 8.
48pub const fn napot_addr(base: usize, size: usize) -> usize {
49    (base >> 2) | (size / 8 - 1)
50}
51
52/// Descriptor for a single PMP region (consumed by M-mode firmware table).
53#[derive(Clone, Copy, Debug)]
54pub struct PmpRegion {
55    /// Physical base address (NAPOT-aligned to `size`).
56    pub base: usize,
57    /// Region size in bytes (must be a power of two ≥ 8).
58    pub size: usize,
59    /// Permission byte: R/W/X flags + addressing mode + optional lock.
60    pub perms: u8,
61}
62
63/// Intended kernel protection regions for entries 2–6.
64///
65/// Entries 0–1 are owned by OpenSBI.  These descriptors should be written
66/// to PMP entries 2–6 by the M-mode firmware shim before entering S-mode.
67///
68/// Physical addresses are for the QEMU virt machine with ViCell's default
69/// linker layout (`ORIGIN = 0x80200000`).  Adjust if the layout changes.
70pub const KERNEL_PMP_REGIONS: &[PmpRegion] = &[
71    // Kernel .text + .rodata: read + execute, locked.
72    // Prevents Cells from writing to kernel code via a wild physical pointer.
73    PmpRegion {
74        base: 0x8020_0000,
75        size: 4 * 1024 * 1024, // 4 MiB — covers typical kernel binary
76        perms: perm::RX | perm::A_NAPOT | perm::L,
77    },
78    // Kernel .data + .bss + stack: read + write, locked.
79    // W^X: data region is not executable.
80    PmpRegion {
81        base: 0x8060_0000,
82        size: 4 * 1024 * 1024,
83        perms: perm::RW | perm::A_NAPOT | perm::L,
84    },
85    // VirtIO MMIO range (UART, VirtIO block/net/keyboard/gpu).
86    PmpRegion {
87        base: 0x1000_0000,
88        size: 65536, // 64 KiB — covers UART + 8 VirtIO MMIO slots
89        perms: perm::RW | perm::A_NAPOT,
90    },
91    // PLIC (Platform-Level Interrupt Controller).
92    PmpRegion {
93        base: 0x0C00_0000,
94        size: 16 * 1024 * 1024, // 16 MiB
95        perms: perm::RW | perm::A_NAPOT,
96    },
97    // CLINT (Core-Local Interrupt: mtime, msip).
98    PmpRegion {
99        base: 0x0200_0000,
100        size: 65536,
101        perms: perm::RW | perm::A_NAPOT,
102    },
103];
