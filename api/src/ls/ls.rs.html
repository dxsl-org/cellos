7/// Read the spawn-args stash into `buf` and return the trimmed content.
8fn spawn_args(buf: &mut [u8]) -> &str {
9    let n = syscall::sys_spawn_args(buf);
10    core::str::from_utf8(&buf[..n]).unwrap_or("").trim()
11}
12
13/// ls [path] — list kernel FS directory entries, one per line.
14#[no_mangle]
15pub fn main() {
16    let mut arg_buf = [0u8; 256];
17    let arg = spawn_args(&mut arg_buf);
18    // Take the first whitespace-separated token as the path; default to "/".
19    let path = arg.split_whitespace().next().unwrap_or("/");
20    let path = if path.is_empty() { "/" } else { path };
21
22    match fs::read_dir(path) {
23        Ok(dir) => {
24            for entry in dir {
25                let name = core::str::from_utf8(&entry.name)
26                    .unwrap_or("?")
27                    .trim_matches('\0');
28                if !name.is_empty() {
29                    io::println(name);
30                }
31            }
32        }
33        Err(_) => {
34            io::print("ls: ");
35            io::print(path);
36            io::println(": no such directory");
37            syscall::sys_exit(1);
38        }
39    }
40    syscall::sys_exit(0);
41}
