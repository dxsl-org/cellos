1//! Hypha `core` — the agent brain (P3: fs + sys + spawn tools).
2//!
3//! Spawns `llm-gateway`, `tool-fs`, `tool-sys`, and `tool-spawn`, then loops:
4//! read a line from stdin (UART), keep the conversation in heap, run an agentic
5//! turn (LLM + optional tool sub-loop), print the final reply.
6//!
7//! Tool routing by name prefix:
8//! - fs tools  (read_file / write_file / list_dir) → tool-fs
9//! - sys tools (list_cells / sys_info / lookup_service) → tool-sys
10//! - spawn tools (spawn_cell / kill_cell) → tool-spawn
11
12#![no_std]
13#![no_main]

40const GATEWAY_PATH: &str = "/bin/llm-gateway";
41const TOOL_FS_PATH: &str = "/bin/tool-fs";
42const TOOL_SYS_PATH: &str = "/bin/tool-sys";
43const TOOL_SPAWN_PATH: &str = "/bin/tool-spawn";
44// Prompt must fit one IPC message (os-gap G5: Grant for large prompts later).
45const PROMPT_MAX: usize = 3500;
46// Tool call agentic loop guard — prevents infinite tool chains.
47const MAX_TOOL_ROUNDS: usize = 5;
48
49const SYSTEM_PREAMBLE: &str = "\
50system: You are Hypha, a helpful AI agent inside Cellos OS. \
51Writable dirs: /data (persistent), /tmp (scratch). Read-only: /bin (binaries). \
52When you need a tool, reply with ONLY this line: \
53TOOL_CALL: {\"name\":\"TOOL\",\"args\":{ARGS_JSON}} \
54File tools: read_file({\"path\":\"...\"}), write_file({\"path\":\"...\",\"content\":\"...\"}), list_dir({\"path\":\"...\"}). \
55Sys tools: list_cells({}), sys_info({}), lookup_service({\"name\":\"vfs|net|input\"}). \
56Spawn tools: spawn_cell({\"path\":\"/bin/...\"}), kill_cell({\"tid\":N}). \
57After tool_result: incorporate it into your answer.\n";
58
59/// Holds the tid for each tool cell. 0 means unavailable (failed to spawn).
60struct Tools {
61    fs: usize,
62    sys: usize,
63    spawn: usize,
64}
65
66impl Tools {
67    /// Route a tool name to the cell tid that handles it.
68    /// Returns 0 if no cell is available for this tool.
69    fn route(&self, name: &str) -> usize {
70        match name {
71            "read_file" | "write_file" | "list_dir" => self.fs,
72            "list_cells" | "sys_info" | "lookup_service" => self.sys,
73            "spawn_cell" | "kill_cell" => self.spawn,
74            _ => 0,
75        }
76    }
77}

79#[no_mangle]
80pub fn main() {
81    println("Hypha — Cellos AI agent (P3: fs+sys+spawn tools). Type 'exit' to quit.");
82
83    let gw = match sys_spawn_from_path(GATEWAY_PATH) {
84        SyscallResult::Ok(tid) => tid,
85        _ => {
86            println("[hypha] ERROR: cannot spawn llm-gateway");
87            sys_exit(1);
88        }
89    };
90
91    let tool_fs = match sys_spawn_from_path(TOOL_FS_PATH) {
92        SyscallResult::Ok(tid) => {
93            println("[hypha] tool-fs ready");
94            tid
95        }
96        _ => {
97            println("[hypha] WARN: tool-fs not found");
98            0
99        }
100    };
101
102    let tool_sys = match sys_spawn_from_path(TOOL_SYS_PATH) {
103        SyscallResult::Ok(tid) => {
104            println("[hypha] tool-sys ready");
105            tid
106        }
107        _ => {
108            println("[hypha] WARN: tool-sys not found");
109            0
110        }
111    };
112
113    let tool_spawn = match sys_spawn_from_path(TOOL_SPAWN_PATH) {
114        SyscallResult::Ok(tid) => {
115            println("[hypha] tool-spawn ready");
116            tid
117        }
118        _ => {
119            println("[hypha] WARN: tool-spawn not found");
120            0
121        }
122    };
123
124    let tools = Tools {
125        fs: tool_fs,
126        sys: tool_sys,
127        spawn: tool_spawn,
128    };
129    let mut conversation: Vec<(&'static str, String)> = Vec::new();
130    let sin = stdin();
131
132    // Register with the input service before entering the readline loop.
133    // Retry a few times in case of a boot race where input service is still starting.
134    for _ in 0..10 {
135        if request_focus() {
136            break;
137        }
138    }
139
140    loop {
141        print("\nyou> ");
142        let mut line = String::new();
143        if sin.read_line(&mut line).is_err() {
144            break;
145        }
146        let user = line.trim();
147        if user.is_empty() {
148            continue;
149        }
150        if user == "exit" || user == "quit" {
151            break;
152        }
153
154        conversation.push(("user", String::from(user)));
155        let prompt = render_prompt(&conversation);
156
157        match run_turn(gw, &tools, &prompt) {
158            Ok(reply) => {
159                print("hypha> ");
160                println(reply.as_str());
161                conversation.push(("assistant", reply));
162            }
163            Err(e) => {
164                print("[hypha] error: ");
165                println(e.as_str());
166                // Drop the failed turn so it doesn't poison later context.
167                conversation.pop();
168            }
169        }
170    }
171
172    println("[hypha] bye");
173    sys_exit(0);
174}

176/// Run one agentic turn: call the LLM, dispatch any tool requests, and return
177/// the final text reply. Loops up to `MAX_TOOL_ROUNDS` times to handle chained
178/// tool calls. Tool interactions are appended to `working_prompt` in-place
179/// (not stored in the permanent conversation — only the final reply is kept).
180fn run_turn(gw: usize, tools: &Tools, prompt: &str) -> Result<String, String> {
181    let mut working = String::from(prompt);
182    for _ in 0..MAX_TOOL_ROUNDS {
183        match ask(gw, &working)? {
184            LlmReply::Text(t) => return Ok(t),
185            LlmReply::ToolCalls(calls) => {
186                for call in &calls {
187                    print("[hypha] tool: ");
188                    println(call.name.as_str());
189                    let result = dispatch_tool(tools, call)?;
190                    working.push_str("\ntool_call: ");
191                    working.push_str(&call.name);
192                    working.push(' ');
193                    working.push_str(&call.args_json);
194                    working.push_str("\ntool_result: ");
195                    working.push_str(&result);
196                }
197                working.push_str("\nassistant: ");
198                working = trim_front(working, PROMPT_MAX);
199            }
200            LlmReply::Error(e) => return Err(e),
201        }
202    }
203    Err(String::from(
204        "[tool limit reached — too many sequential calls]",
205    ))
206}

208/// One IPC round-trip with the gateway: send `LlmRequest`, receive `LlmReply`.
209fn ask(gw: usize, prompt: &str) -> Result<LlmReply, String> {
210    let req = LlmRequest::Complete { prompt };
211    let mut out = [0u8; 4096];
212    let encoded = postcard::to_slice(&req, &mut out)
213        .map_err(|_| String::from("prompt too large for one IPC message"))?;
214
215    match sys_send(gw, encoded) {
216        SyscallResult::Ok(_) => {}
217        _ => return Err(String::from("send to gateway failed")),
218    }
219
220    let mut buf = [0u8; 4096];
221    // mask=0 (wildcard): accept from any sender so that input-service EV_ASCII events
222    // dispatched to this cell (because it holds keyboard focus) don't deadlock the
223    // input service. G18 enforces the mask in ipc_send — a masked recv blocks senders
224    // that don't match, so we must use 0 here and filter by sender in the match arm.
225    loop {
226        match sys_recv(0, &mut buf) {
227            SyscallResult::Ok(sender) if sender == gw => {
228                return match postcard::take_from_bytes::<LlmReply>(&buf) {
229                    Ok((reply, _)) => Ok(reply),
230                    Err(_) => Err(String::from("bad LlmReply encoding")),
231                };
232            }
233            SyscallResult::Ok(_) => continue, // non-gw message (e.g. input event) — discard
234            _ => return Err(String::from("no reply from gateway")),
235        }
236    }
237}

239/// Dispatch one tool call to the appropriate tool cell via `AgentToolRequest` IPC.
240/// Routes by tool name: fs / sys / spawn tools go to their respective cells.
241fn dispatch_tool(tools: &Tools, call: &ToolCall) -> Result<String, String> {
242    let cell_tid = tools.route(&call.name);
243    if cell_tid == 0 {
244        return Err(alloc::format!(
245            "tool '{}' not available (cell not spawned)",
246            call.name
247        ));
248    }
249
250    let req = AgentToolRequest::Invoke {
251        name: &call.name,
252        args_json: &call.args_json,
253    };
254    let mut out = [0u8; 4096];
255    let encoded =
256        postcard::to_slice(&req, &mut out).map_err(|_| String::from("tool request too large"))?;
257
258    match sys_send(cell_tid, encoded) {
259        SyscallResult::Ok(_) => {}
260        _ => return Err(alloc::format!("send to tool cell '{}' failed", call.name)),
261    }
262
263    let mut buf = [0u8; 4096];
264    // mask=0: same fix as ask() — accept from any sender to avoid G18 deadlock with
265    // the input service; filter by cell_tid in the match arm.
266    loop {
267        match sys_recv(0, &mut buf) {
268            SyscallResult::Ok(sender) if sender == cell_tid => {
269                return match postcard::take_from_bytes::<AgentToolResponse>(&buf) {
270                    Ok((AgentToolResponse::Ok { result_json }, _)) => Ok(result_json),
271                    Ok((AgentToolResponse::Err { message }, _)) => Err(message),
272                    Err(_) => Err(String::from("bad AgentToolResponse encoding")),
273                };
274            }
275            SyscallResult::Ok(_) => continue,
276            _ => return Err(alloc::format!("no reply from tool cell '{}'", call.name)),
277        }
278    }
279}

281/// Flatten the heap conversation into a single role-tagged transcript.
282/// The system preamble is prepended so the LLM always knows about tools.
283/// Trimmed from the front if it would exceed the one-message IPC budget.
284fn render_prompt(conv: &[(&'static str, String)]) -> String {
285    let mut s = String::from(SYSTEM_PREAMBLE);
286    for (role, text) in conv {
287        s.push_str(role);
288        s.push_str(": ");
289        s.push_str(text);
290        s.push('\n');
291    }
292    s.push_str("assistant: ");
293    trim_front(s, PROMPT_MAX)
294}
295
296/// Keep the tail of `s` within `max` bytes (drop oldest content), char-boundary safe.
297fn trim_front(s: String, max: usize) -> String {
298    if s.len() <= max {
299        return s;
300    }
301    let mut start = s.len() - max;
302    while start < s.len() && !s.is_char_boundary(start) {
303        start += 1;
304    }
305    String::from(&s[start..])
306}
