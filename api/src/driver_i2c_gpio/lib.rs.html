4//! Bit-bang I2C master over two GPIO pins.
5//!
6//! Generic over any `G: ViGpio`. Concrete use:
7//! `BitBangI2c::<Pl061Gpio>::new(gpio)` — calls `Pl061Gpio::open()` in the caller.
8//!
9//! SDA open-drain emulation with `ViGpio`:
10//! - "Drive low" → `set_direction(SDA, Output) + write_pin(SDA, false)`
11//! - "Release / float high" → `set_direction(SDA, Input)` (pull-up → reads 1 in QEMU)
12//!
13//! QEMU note: without a real I2C slave, SDA stays 1 — every address NACKs.
14//! The caller should detect `NackAddress` and fall back to synthetic data.
15
16use hal_gpio::{PinDir, ViGpio};
17use hal_i2c::{I2cError, ViI2c};
18
19const SCL: u8 = 0; // clock pin
20const SDA: u8 = 1; // data pin
21
22// Busy-wait loops per I2C half-period. QEMU TCG timing is not precise;
23// this value is chosen for protocol correctness, not exact frequency.
24const HALF_PERIOD: usize = 50;
25
26/// Bit-bang I2C master backed by a `ViGpio` implementation.
27pub struct BitBangI2c<G: ViGpio> {
28    gpio: G,
29}
30
31impl<G: ViGpio> BitBangI2c<G> {
32    /// Take ownership of `gpio` and prepare it as an I2C master.
33    ///
34    /// The caller is responsible for opening/acquiring `gpio` before passing it here.
35    pub fn new(gpio: G) -> Self {
36        Self { gpio }
37    }
38
39    /// Release the underlying GPIO resource (e.g. back to the cell owner).
40    pub fn into_gpio(self) -> G {
41        self.gpio
42    }
43
44    fn delay(&self) {
45        for _ in 0..HALF_PERIOD {
46            core::hint::spin_loop();
47        }
48    }
49
50    // ── SDA helpers (open-drain emulation) ──────────────────────────────────
51
52    fn sda_low(&mut self) -> Result<(), I2cError> {
53        self.gpio
54            .set_direction(SDA, PinDir::Output)
55            .map_err(|_| I2cError::BusError)?;
56        self.gpio
57            .write_pin(SDA, false)
58            .map_err(|_| I2cError::BusError)
59    }
60
61    fn sda_release(&mut self) -> Result<(), I2cError> {
62        // Input mode → SDA floats high via pull-up; slave can pull low for ACK.
63        self.gpio
64            .set_direction(SDA, PinDir::Input)
65            .map_err(|_| I2cError::BusError)
66    }
67
68    fn read_sda(&self) -> Result<bool, I2cError> {
69        self.gpio.read_pin(SDA).map_err(|_| I2cError::BusError)
70    }
71
72    // ── SCL helpers (master always drives clock) ─────────────────────────────
73
74    fn scl_high(&mut self) -> Result<(), I2cError> {
75        self.gpio
76            .write_pin(SCL, true)
77            .map_err(|_| I2cError::BusError)
78    }
79
80    fn scl_low(&mut self) -> Result<(), I2cError> {
81        self.gpio
82            .write_pin(SCL, false)
83            .map_err(|_| I2cError::BusError)
84    }
85
86    // ── Bus conditions ───────────────────────────────────────────────────────
87
88    fn start(&mut self) -> Result<(), I2cError> {
89        // START: SDA 1→0 while SCL is high.
90        self.sda_release()?;
91        self.scl_high()?;
92        self.delay();
93        self.sda_low()?;
94        self.delay();
95        self.scl_low()?;
96        Ok(())
97    }
98
99    fn stop(&mut self) -> Result<(), I2cError> {
100        // STOP: SDA 0→1 while SCL is high.
101        self.sda_low()?;
102        self.delay();
103        self.scl_high()?;
104        self.delay();
105        self.sda_release()?;
106        self.delay();
107        Ok(())
108    }
109
110    // ── Bit-level I/O ────────────────────────────────────────────────────────
111
112    fn send_bit(&mut self, bit: bool) -> Result<(), I2cError> {
113        if bit {
114            self.sda_release()?;
115        } else {
116            self.sda_low()?;
117        }
118        self.delay();
119        self.scl_high()?;
120        self.delay();
121        self.scl_low()?;
122        Ok(())
123    }
124
125    fn recv_bit(&mut self) -> Result<bool, I2cError> {
126        self.sda_release()?; // let slave drive SDA
127        self.delay();
128        self.scl_high()?;
129        let bit = self.read_sda()?;
130        self.delay();
131        self.scl_low()?;
132        Ok(bit)
133    }
134
135    // ── Byte-level I/O ───────────────────────────────────────────────────────
136
137    /// Send one byte MSB-first. Returns `true` if slave ACKed (SDA low in ACK slot).
138    fn send_byte(&mut self, byte: u8) -> Result<bool, I2cError> {
139        for i in (0..8).rev() {
140            self.send_bit((byte >> i) & 1 != 0)?;
141        }
142        // ACK slot: master releases SDA, slave pulls low to ACK.
143        let nack = self.recv_bit()?;
144        Ok(!nack)
145    }
146
147    /// Receive one byte MSB-first. `ack`: true sends ACK (more bytes coming), false sends NACK.
148    fn recv_byte(&mut self, ack: bool) -> Result<u8, I2cError> {
149        let mut byte = 0u8;
150        for _ in 0..8 {
151            byte = (byte << 1) | (self.recv_bit()? as u8);
152        }
153        // ACK/NACK: master drives SDA.
154        if ack {
155            self.sda_low()?;
156        } else {
157            self.sda_release()?;
158        }
159        self.delay();
160        self.scl_high()?;
161        self.delay();
162        self.scl_low()?;
163        Ok(byte)
164    }
165
166    fn send_addr(&mut self, addr: u8, read: bool) -> Result<bool, I2cError> {
167        self.send_byte((addr << 1) | read as u8)
168    }
169}
170
171impl<G: ViGpio> ViI2c for BitBangI2c<G> {
172    type Error = I2cError;
173
174    fn write(&mut self, addr: u8, bytes: &[u8]) -> Result<(), I2cError> {
175        // SCL must be Output throughout; set once before START.
176        self.gpio
177            .set_direction(SCL, PinDir::Output)
178            .map_err(|_| I2cError::BusError)?;
179        self.start()?;
180        if !self.send_addr(addr, false)? {
181            let _ = self.stop();
182            return Err(I2cError::NackAddress);
183        }
184        for &b in bytes {
185            if !self.send_byte(b)? {
186                let _ = self.stop();
187                return Err(I2cError::NackData);
188            }
189        }
190        self.stop()
191    }
192
193    fn read(&mut self, addr: u8, buf: &mut [u8]) -> Result<(), I2cError> {
194        self.gpio
195            .set_direction(SCL, PinDir::Output)
196            .map_err(|_| I2cError::BusError)?;
197        self.start()?;
198        if !self.send_addr(addr, true)? {
199            let _ = self.stop();
200            return Err(I2cError::NackAddress);
201        }
202        let last = buf.len().saturating_sub(1);
203        for (i, slot) in buf.iter_mut().enumerate() {
204            *slot = self.recv_byte(i < last)?; // ACK all but final byte
205        }
206        self.stop()
207    }
208
209    fn write_read(&mut self, addr: u8, wr: &[u8], rd: &mut [u8]) -> Result<(), I2cError> {
210        self.gpio
211            .set_direction(SCL, PinDir::Output)
212            .map_err(|_| I2cError::BusError)?;
213        // Write phase
214        self.start()?;
215        if !self.send_addr(addr, false)? {
216            let _ = self.stop();
217            return Err(I2cError::NackAddress);
218        }
219        for &b in wr {
220            if !self.send_byte(b)? {
221                let _ = self.stop();
222                return Err(I2cError::NackData);
223            }
224        }
225        // Repeated START → read phase
226        self.start()?;
227        if !self.send_addr(addr, true)? {
228            let _ = self.stop();
229            return Err(I2cError::NackAddress);
230        }
231        let last = rd.len().saturating_sub(1);
232        for (i, slot) in rd.iter_mut().enumerate() {
233            *slot = self.recv_byte(i < last)?;
234        }
235        self.stop()
236    }
237}
