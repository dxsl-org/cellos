1// 8×8 IBM VGA CP437 bitmap font, glyphs 32–126 (95 chars, 760 bytes).
2// Bit 7 of each row byte = leftmost pixel.
3
4pub const WIDTH: usize = 8;
5pub const HEIGHT: usize = 8;
6
7/// Return the 8-row bitmap for ASCII `c`.  Falls back to a solid block for
8/// characters outside 32–126.
9pub fn glyph(c: u8) -> &'static [u8] {
10    let idx = if (32..=126).contains(&c) {
11        (c - 32) as usize
12    } else {
13        94
14    };
15    &FONT_DATA[idx * HEIGHT..(idx + 1) * HEIGHT]
16}
