1//! fb-console — mirrors kernel user-log output to the HDMI display.
2//!
3//! Creates a full-screen background surface in the compositor and renders
4//! incoming log bytes as white-on-black 8×8 bitmap text. Wraps at the right
5//! edge; scrolls one line up when the bottom row is reached. No UART I/O.
6//!
7//! Requires `ReadLog` capability (allowlist bit 54).
8
9#![no_std]
10#![no_main]
11
12mod font;
13
14extern crate ostd;
15
16use api::display::PixelFormat;
17use ostd::display::{wait_for_compositor, ViSurface};
18use ostd::syscall::{sys_exit, sys_read_log, sys_yield};

20api::declare_manifest!(block_io = false, network = false, spawn = false);
21api::declare_syscalls![
22    Log,
23    GpuGetResolution,
24    GrantRegister,
25    GrantShare,
26    GrantSlice,
27    GrantUnregister,
28    Send,
29    Recv,

31    ReadLog

34// Foreground/background colours (BGRA).
35const FG: [u8; 4] = [0xCC, 0xCC, 0xCC, 0xFF]; // light grey
36const BG: [u8; 4] = [0x00, 0x00, 0x00, 0xFF]; // black
37
38#[no_mangle]
39pub fn main() {
40    let (width, height) = ostd::syscall::sys_get_resolution();
41    let width = width as usize;
42    let height = height as usize;
43
44    let comp = wait_for_compositor();
45    let mut surf = match ViSurface::create(comp, width as u32, height as u32, PixelFormat::Bgra8888)
46    {
47        Ok(s) => s,
48        Err(_) => {
49            sys_exit(1);
50        }
51    };
52
53    // Clear to background.
54    let px = surf.pixels_mut();
55    for chunk in px.chunks_exact_mut(4) {
56        chunk.copy_from_slice(&BG);
57    }
58    surf.damage_all();
59
60    let cols = width / font::WIDTH;
61    let rows = height / font::HEIGHT;
62
63    let mut col: usize = 0;
64    let mut row: usize = 0;
65    let mut buf = [0u8; 256];
66
67    loop {
68        let n = sys_read_log(&mut buf);
69        if n == 0 {
70            sys_yield();
71            continue;
72        }
73
74        let mut dirty = false;
75        for &b in &buf[..n] {
76            match b {
77                b'\n' => {
78                    col = 0;
79                    row += 1;
80                    if row >= rows {
81                        scroll_up(&mut surf, width, height, rows);
82                        row = rows - 1;
83                    }
84                    dirty = true;
85                }
86                b'\r' => {
87                    col = 0;
88                }
89                _ => {
90                    draw_char(&mut surf, b, col, row, width);
91                    col += 1;
92                    dirty = true;
93                    if col >= cols {
94                        col = 0;
95                        row += 1;
96                        if row >= rows {
97                            scroll_up(&mut surf, width, height, rows);
98                            row = rows - 1;
99                        }
100                    }
101                }
102            }
103        }
104        if dirty {
105            surf.damage_all();
106        }
107    }
108}

110/// Draw one ASCII character at grid cell (col, row).
111fn draw_char(surf: &mut ViSurface, c: u8, col: usize, row: usize, _width: usize) {
112    let glyph = font::glyph(c);
113    let stride = surf.stride(); // bytes per row = width*4
114    let px = surf.pixels_mut();
115    let base_x = col * font::WIDTH;
116    let base_y = row * font::HEIGHT;
117
118    for (gy, &row_bits) in glyph.iter().enumerate() {
119        let y = base_y + gy;
120        for gx in 0..font::WIDTH {
121            let x = base_x + gx;
122            let off = y * stride + x * 4;
123            if off + 4 > px.len() {
124                continue;
125            }
126            let lit = (row_bits >> (7 - gx)) & 1 != 0;
127            px[off..off + 4].copy_from_slice(if lit { &FG } else { &BG });
128        }
129    }
130}
131
132/// Scroll the framebuffer up by one text row.
133fn scroll_up(surf: &mut ViSurface, width: usize, height: usize, rows: usize) {
134    let stride = width * 4;
135    let row_h = font::HEIGHT;
136    let move_h = (rows - 1) * row_h; // rows to keep
137    let move_bytes = move_h * stride;
138    let px = surf.pixels_mut();
139
140    // Shift pixel data up by one text row.
141    px.copy_within(row_h * stride..row_h * stride + move_bytes, 0);
142
143    // Clear the last row.
144    let clear_start = move_h * stride;
145    let clear_end = height * stride;
146    for chunk in px[clear_start..clear_end].chunks_exact_mut(4) {
147        chunk.copy_from_slice(&BG);
148    }
149}
