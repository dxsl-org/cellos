4//! BCM2837 GPIO Driver Cell — Raspberry Pi 3 (BCM2837).
5//!
6//! Base: 0x3F200000, 54 pins, 32-bit GPFSEL/GPSET/GPCLR/GPLEV/GPREN/GPFEN.
7//! Pin function select: 3 bits per pin, 10 pins per GPFSEL register.
8//!
9//! IRQ routing: BCM2835 legacy controller bank0 (pins 0–27) / bank1 (pins 28–45).
10//! The kernel IRQ handler notifies the MMIO owner via opcode 0xA0 when a GPIO
11//! edge fires — same protocol as the PL061 driver.
12//!
13//! Pull-up/down: BCM2837 requires a 3-step GPPUD + GPPUDCLK sequence (≥150 cycles
14//! hold time). This driver omits the timing (no busy-wait in no_std), so pull
15//! configuration is caller responsibility if needed.
16
17use hal_gpio::{Edge, PinDir, ViGpio};
18use ostd::mmio::MmioRegion;
19use types::{ViError, ViResult};
20
21/// BCM2837 GPIO MMIO base on Raspberry Pi 3.
22pub const BCM_GPIO_BASE: usize = 0x3F20_0000;
23/// MMIO region size mapped for GPIO access.
24pub const BCM_GPIO_SIZE: usize = 0x0001_0000; // 64 KiB (covers GPPUDCLK1 at 0x9C)
25
26// Register offsets (u32, byte-addressed)
27const GPFSEL: [usize; 6] = [0x00, 0x04, 0x08, 0x0C, 0x10, 0x14]; // function select
28const GPSET: [usize; 2] = [0x1C, 0x20]; // set output high
29const GPCLR: [usize; 2] = [0x28, 0x2C]; // set output low
30const GPLEV: [usize; 2] = [0x34, 0x38]; // read level
31const GPEDS: [usize; 2] = [0x40, 0x44]; // event detect status (write 1 to clear)
32const GPREN: [usize; 2] = [0x4C, 0x50]; // rising edge detect enable
33const GPFEN: [usize; 2] = [0x58, 0x5C]; // falling edge detect enable
34
35// Function select encoding (3 bits per pin)
36const FSEL_INPUT: u32 = 0b000;
37const FSEL_OUTPUT: u32 = 0b001;
38
39/// BCM2837 GPIO controller for Raspberry Pi 3 (54 pins).
40pub struct BcmGpio {
41    mmio: MmioRegion,
42}
43
44impl BcmGpio {
45    /// Request exclusive MMIO ownership from the kernel resource registry.
46    ///
47    /// Returns `Err(PermissionDenied)` when not running on a `board-rpi3` target
48    /// (the kernel allowlist excludes 0x3F200000 on QEMU virt/x86_64/riscv64).
49    pub fn open() -> ViResult<Self> {
50        let mmio = ostd::mmio::request_region(BCM_GPIO_BASE, BCM_GPIO_SIZE)?;
51        Ok(Self { mmio })
52    }
53
54    /// Read GPEDS (event detect status) for a pin bank (0 = pins 0-31, 1 = pins 32-53).
55    pub fn read_eds(&self, bank: usize) -> ViResult<u32> {
56        self.mmio.read_u32(GPEDS[bank])
57    }
58
59    /// Clear GPEDS bits indicated by `mask` in the given bank.
60    pub fn clear_eds(&self, bank: usize, mask: u32) -> ViResult<()> {
61        self.mmio.write_u32(GPEDS[bank], mask)
62    }
63
64    /// Set the function for a pin (GPFSEL register).
65    fn set_fsel(&self, pin: u8, func: u32) -> ViResult<()> {
66        let reg_idx = (pin / 10) as usize;
67        let bit_pos = ((pin % 10) * 3) as u32;
68        let cur = self.mmio.read_u32(GPFSEL[reg_idx])?;
69        let new = (cur & !(0x7 << bit_pos)) | ((func & 0x7) << bit_pos);
70        self.mmio.write_u32(GPFSEL[reg_idx], new)
71    }
72}
73
74impl ViGpio for BcmGpio {
75    fn set_direction(&mut self, pin: u8, dir: PinDir) -> ViResult<()> {
76        if pin > 53 {
77            return Err(ViError::InvalidInput);
78        }
79        let func = match dir {
80            PinDir::Output => FSEL_OUTPUT,
81            PinDir::Input => FSEL_INPUT,
82        };
83        self.set_fsel(pin, func)
84    }
85
86    fn read_pin(&self, pin: u8) -> ViResult<bool> {
87        if pin > 53 {
88            return Err(ViError::InvalidInput);
89        }
90        let (bank, bit) = ((pin / 32) as usize, pin % 32);
91        let val = self.mmio.read_u32(GPLEV[bank])?;
92        Ok(val & (1 << bit) != 0)
93    }
94
95    fn write_pin(&mut self, pin: u8, high: bool) -> ViResult<()> {
96        if pin > 53 {
97            return Err(ViError::InvalidInput);
98        }
99        let (bank, bit) = ((pin / 32) as usize, pin % 32);
100        if high {
101            self.mmio.write_u32(GPSET[bank], 1 << bit)
102        } else {
103            self.mmio.write_u32(GPCLR[bank], 1 << bit)
104        }
105    }
106
107    fn enable_edge_irq(&mut self, pin: u8, edge: Edge) -> ViResult<()> {
108        if pin > 53 {
109            return Err(ViError::InvalidInput);
110        }
111        let (bank, bit) = ((pin / 32) as usize, pin % 32);
112        let mask = 1u32 << bit;
113        if matches!(edge, Edge::Rising | Edge::Both) {
114            let cur = self.mmio.read_u32(GPREN[bank])?;
115            self.mmio.write_u32(GPREN[bank], cur | mask)?;
116        }
117        if matches!(edge, Edge::Falling | Edge::Both) {
118            let cur = self.mmio.read_u32(GPFEN[bank])?;
119            self.mmio.write_u32(GPFEN[bank], cur | mask)?;
120        }
121        Ok(())
122    }
123
124    fn disable_irq(&mut self, pin: u8) -> ViResult<()> {
125        if pin > 53 {
126            return Err(ViError::InvalidInput);
127        }
128        let (bank, bit) = ((pin / 32) as usize, pin % 32);
129        let mask = 1u32 << bit;
130        let cur_ren = self.mmio.read_u32(GPREN[bank])?;
131        self.mmio.write_u32(GPREN[bank], cur_ren & !mask)?;
132        let cur_fen = self.mmio.read_u32(GPFEN[bank])?;
133        self.mmio.write_u32(GPFEN[bank], cur_fen & !mask)
134    }
135}
