1//! Byte-source helpers shared by the chunked decoder: pull bytes from the
2//! seeded `leftover` region first, then from the transport.  A 0-length
3//! transport read while the body is incomplete is a genuine EOF (the connection
4//! closed mid-body), surfaced as [`Byte::Eof`] — the callers translate that into
5//! `HttpError::UnexpectedEof` (see the `body` module doc).
6//!
7//! CRITICAL distinction: [`Byte::Eof`] / `fill() == 0` are returned ONLY when a
8//! real `transport.read()` produced 0 bytes.  Leftover bytes are always served
9//! first, so "leftover drained but transport not yet polled" can never be
10//! mistaken for EOF here — the transport is polled in the same call.
11
12use super::{BodyReader, HttpError, Read};
13
14/// Outcome of pulling a single byte from the leftover/transport source.
15pub(super) enum Byte {
16    Got(u8),
17    /// Transport returned a 0-length read (EOF) with no leftover available.
18    Eof,
19}
20
21impl BodyReader {
22    /// Pull a single byte from leftover-then-transport.
23    pub(super) fn next_byte<R: Read>(&mut self, transport: &mut R) -> Result<Byte, HttpError> {
24        if !self.leftover_empty() {
25            let b = self.leftover[self.leftover_pos];
26            self.leftover_pos += 1;
27            return Ok(Byte::Got(b));
28        }
29        let mut one = [0u8; 1];
30        let n = transport
31            .read(&mut one)
32            .map_err(|e| HttpError::Other(alloc::format!("transport read: {e:?}")))?;
33        if n == 0 {
34            Ok(Byte::Eof)
35        } else {
36            Ok(Byte::Got(one[0]))
37        }
38    }
39
40    /// Fill `dst` from leftover-then-transport, returning bytes copied.  Returns
41    /// `0` only when the transport itself returned 0 (EOF) with no leftover
42    /// available — never for an empty `dst` quirk (callers guard `dst`).
43    pub(super) fn fill<R: Read>(
44        &mut self,
45        transport: &mut R,
46        dst: &mut [u8],
47    ) -> Result<usize, HttpError> {
48        if dst.is_empty() {
49            return Ok(0);
50        }
51        if !self.leftover_empty() {
52            return Ok(self.drain_leftover(dst));
53        }
54        transport
55            .read(dst)
56            .map_err(|e| HttpError::Other(alloc::format!("transport read: {e:?}")))
57    }
58}
