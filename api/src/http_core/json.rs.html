1//! JSON codec — thin re-export layer over `serde_json` (alloc feature).
2//!
3//! Re-exports provide a stable surface so callers can `use http_core::json::*`
4//! without taking a direct `serde_json` dependency.  If the underlying codec
5//! ever changes, only this module needs updating.
6
7pub use serde_json::{from_slice, json, to_string, to_vec, Error, Value};
8
9/// Walk a nested JSON object by a sequence of string keys and return the
10/// innermost value as a `&str`, or `None` if any step is absent or non-string.
11///
12/// An empty `path` slice skips all navigation and calls `as_str()` on `value`
13/// directly — returning `Some` if it is a JSON string, `None` otherwise.
14///
15/// # Duplicate-key behaviour
16///
17/// `serde_json` deserialises objects into a `Map<String, Value>` backed by an
18/// insertion-order-preserving `IndexMap`.  When a JSON object contains duplicate
19/// keys, **the last occurrence wins** (serde_json 1.x default; documented here
20/// so callers are not surprised by silently-dropped earlier values).
21///
22/// # Example
23///
24/// ```
25/// use http_core::json::{json, get_str};
26///
27/// let v = json!({"choices": [{"message": {"content": "hello"}}]});
28/// // get_str only walks object keys, not array indices
29/// assert_eq!(get_str(&v, &["choices"]), None); // value is an array, not a string
30/// assert_eq!(get_str(&v["choices"][0], &["message", "content"]), Some("hello"));
31/// ```
32pub fn get_str<'a>(value: &'a Value, path: &[&str]) -> Option<&'a str> {
33    let mut current = value;
34    for key in path {
35        current = current.as_object()?.get(*key)?;
36    }
37    current.as_str()
38}
