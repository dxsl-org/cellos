1//! Chunked transfer-encoding decoder for [`BodyReader`].
2//!
3//! Drives the classic chunked state machine via [`ChunkPhase`]:
4//!
5//! ```text
6//!   Size  ->  Data  ->  TrailingCrlf  ->  Size  ->  ...
7//!   ... a "0" size line switches to Trailer, drained to its final CRLF, done.
8//! ```
9//!
10//! Every byte is sourced through [`fill`] / [`next_byte`], which drain the
11//! seeded `leftover` (the bytes that arrived in the same packet as the headers)
12//! before touching the transport.  A 0-length transport read while the body is
13//! incomplete is a genuine EOF — the connection closed mid-body, so it is
14//! surfaced as `HttpError::UnexpectedEof` (a truncated chunked stream), never as
15//! a "retry" signal (see module doc).
16//!
17//! Three classic bugs this defends against (research finding 5):
18//!  1. not consuming the CRLF that trails each chunk's data (the `TrailingCrlf`
19//!     phase is a distinct, resumable step);
20//!  2. a chunk-size line split across two transport reads (accumulated in
21//!     `line_buf`, retried via `parse_chunk_size`'s `Partial`);
22//!  3. `u64 -> usize` overflow on a hostile chunk size (capped at
23//!     `MAX_CHUNK_SIZE`).
24
25use super::source::Byte;
26use super::{BodyReader, ChunkPhase, HttpError, Read, State, LINE_BUF_LEN, MAX_CHUNK_SIZE};

28impl BodyReader {
29    pub(super) fn read_chunked<R: Read>(
30        &mut self,
31        transport: &mut R,
32        out: &mut [u8],
33    ) -> Result<usize, HttpError> {
34        loop {
35            let phase = match &self.state {
36                State::Chunked { done: true, .. } => return Ok(0),
37                State::Chunked { phase, .. } => *phase,
38                _ => return Err(HttpError::BadChunk),
39            };
40            match phase {
41                ChunkPhase::Size => {
42                    // Parse the next size line; loop on to serve data, or finish.
43                    match self.parse_size_line(transport)? {
44                        // EOF mid-size-line: stream truncated before a complete
45                        // chunk-size header arrived.
46                        None => return Err(HttpError::UnexpectedEof),
47                        Some(0) => self.set_phase(ChunkPhase::Trailer),
48                        Some(size) => {
49                            self.begin_chunk(size);
50                            // fall through the loop into the Data phase
51                        }
52                    }
53                }
54                ChunkPhase::Data => return self.serve_chunk_data(transport, out),
55                ChunkPhase::TrailingCrlf => {
56                    // Resumable: consume the 2-byte CRLF, then back to Size.
57                    self.consume_crlf(transport)?;
58                    self.set_phase(ChunkPhase::Size);
59                }
60                ChunkPhase::Trailer => {
61                    self.drain_trailer(transport)?;
62                    if let State::Chunked { done, .. } = &mut self.state {
63                        *done = true;
64                    }
65                    return Ok(0);
66                }
67            }
68        }
69    }

71    /// Copy chunk data into `out`.  When the chunk is fully delivered, advance to
72    /// `TrailingCrlf` (the CRLF is consumed on the NEXT call, so already-returned
73    /// data is never lost to a split-CRLF error).
74    fn serve_chunk_data<R: Read>(
75        &mut self,
76        transport: &mut R,
77        out: &mut [u8],
78    ) -> Result<usize, HttpError> {
79        let remaining = match &self.state {
80            State::Chunked {
81                chunk_remaining, ..
82            } => *chunk_remaining,
83            _ => return Err(HttpError::BadChunk),
84        };
85        // `remaining` fits in usize: bounded by MAX_CHUNK_SIZE at parse time.
86        let want = (remaining as usize).min(out.len());
87        let mut written = 0;
88        while written < want {
89            match self.fill(transport, &mut out[written..want])? {
90                0 => {
91                    // Transport hit EOF mid-chunk: the chunk's `chunk_remaining`
92                    // data bytes never fully arrived → truncated stream.  (If we
93                    // already copied some bytes this call, return them first; the
94                    // EOF surfaces on the next call when `written == 0`.)
95                    if written == 0 {
96                        return Err(HttpError::UnexpectedEof);
97                    }
98                    break;
99                }
100                n => written += n,
101            }
102        }
103
104        let new_remaining = remaining - written as u64;
105        if let State::Chunked {
106            chunk_remaining,
107            phase,
108            ..
109        } = &mut self.state
110        {
111            *chunk_remaining = new_remaining;
112            if new_remaining == 0 {
113                *phase = ChunkPhase::TrailingCrlf;
114            }
115        }
116        Ok(written)
117    }
118
119    fn begin_chunk(&mut self, size: u64) {
120        if let State::Chunked {
121            chunk_remaining,
122            phase,
123            line_len,
124            ..
125        } = &mut self.state
126        {
127            *chunk_remaining = size;
128            *line_len = 0;
129            *phase = ChunkPhase::Data;
130        }
131    }
132
133    fn set_phase(&mut self, p: ChunkPhase) {
134        if let State::Chunked { phase, .. } = &mut self.state {
135            *phase = p;
136        }
137    }

139    /// Accumulate and parse a chunk-size line.  Returns `Some(size)` when the
140    /// line is complete, `None` if more transport data is needed (caller retry).
141    fn parse_size_line<R: Read>(&mut self, transport: &mut R) -> Result<Option<u64>, HttpError> {
142        loop {
143            let line_len = match &self.state {
144                State::Chunked { line_len, .. } => *line_len,
145                _ => return Err(HttpError::BadChunk),
146            };
147            if line_len > 0 {
148                let parsed = match &self.state {
149                    State::Chunked { line_buf, .. } => {
150                        httparse::parse_chunk_size(&line_buf[..line_len])
151                    }
152                    _ => return Err(HttpError::BadChunk),
153                };
154                match parsed {
155                    Ok(httparse::Status::Complete((consumed, size))) => {
156                        if size > MAX_CHUNK_SIZE {
157                            return Err(HttpError::BadChunk);
158                        }
159                        self.requeue_after_size_line(consumed, line_len);
160                        return Ok(Some(size));
161                    }
162                    Ok(httparse::Status::Partial) => { /* need another byte */ }
163                    Err(_) => return Err(HttpError::BadChunk),
164                }
165            }
166            match self.next_byte(transport)? {
167                // EOF before the size line terminated: caller maps None to
168                // UnexpectedEof (truncated mid-size-line).
169                Byte::Eof => return Ok(None),
170                Byte::Got(b) => self.push_line_byte(b)?,
171            }
172        }
173    }
174
175    /// Push one byte into the size-line accumulator, guarding its capacity.
176    fn push_line_byte(&mut self, b: u8) -> Result<(), HttpError> {
177        if let State::Chunked {
178            line_buf, line_len, ..
179        } = &mut self.state
180        {
181            if *line_len >= LINE_BUF_LEN {
182                // A size line longer than the buffer is malformed/hostile.
183                return Err(HttpError::BadChunk);
184            }
185            line_buf[*line_len] = b;
186            *line_len += 1;
187            Ok(())
188        } else {
189            Err(HttpError::BadChunk)
190        }
191    }
192
193    /// Reset the size-line accumulator after a size line is fully parsed.
194    ///
195    /// INVARIANT: `consumed == line_len`.  `parse_size_line` feeds exactly one
196    /// byte into `line_buf` and re-parses after each, so `parse_chunk_size`
197    /// signals `Complete` on the very byte that finishes the trailing CRLF —
198    /// there are never leftover data bytes in `line_buf` to splice back.  We keep
199    /// the `line_len` argument only to assert that invariant.
200    fn requeue_after_size_line(&mut self, consumed: usize, line_len: usize) {
201        debug_assert_eq!(
202            consumed, line_len,
203            "size-line accumulator parsed one byte at a time; no trailing data expected",
204        );
205        if let State::Chunked { line_len, .. } = &mut self.state {
206            *line_len = 0;
207        }
208    }

210    /// Consume an expected CRLF.  Tolerates the split where `\r` and `\n` arrive
211    /// in separate reads by sourcing each byte through `next_byte`.
212    fn consume_crlf<R: Read>(&mut self, transport: &mut R) -> Result<(), HttpError> {
213        for &expected in b"\r\n" {
214            match self.next_byte(transport)? {
215                // EOF mid-CRLF: the chunk's trailing CRLF never arrived.
216                Byte::Eof => return Err(HttpError::UnexpectedEof),
217                Byte::Got(b) if b == expected => {}
218                Byte::Got(_) => return Err(HttpError::BadChunk),
219            }
220        }
221        Ok(())
222    }

224    /// Drain the trailer after the `0` terminator up to its final empty line.
225    /// A trailer is `*(trailer-field CRLF) CRLF`; everything is discarded.  The
226    /// "0" line's own CRLF was already consumed by `parse_chunk_size`, so the
227    /// first thing seen here is either the closing CRLF (no trailer fields) or a
228    /// trailer field.
229    fn drain_trailer<R: Read>(&mut self, transport: &mut R) -> Result<(), HttpError> {
230        let mut prev_cr = false;
231        let mut line_has_content = false;
232        loop {
233            match self.next_byte(transport)? {
234                // EOF before the trailer's final blank line: truncated stream.
235                Byte::Eof => return Err(HttpError::UnexpectedEof),
236                Byte::Got(b'\r') => {
237                    // A `\r` after a `\r` is itself trailer content (a malformed
238                    // `\r\r` line), so the prior `\r` did not start a CRLF.
239                    if prev_cr {
240                        line_has_content = true;
241                    }
242                    prev_cr = true;
243                }
244                Byte::Got(b'\n') if prev_cr => {
245                    if !line_has_content {
246                        return Ok(()); // blank line -> end of trailer
247                    }
248                    prev_cr = false;
249                    line_has_content = false;
250                }
251                Byte::Got(_) => {
252                    prev_cr = false;
253                    line_has_content = true;
254                }
255            }
256        }
257    }
