1//! HTTP/1.1 request serialisation.
2//!
3//! Generalises `build_post` from `cells/apps/hypha/llm-gateway/src/http.rs`.
4//! That function emitted HTTP/1.0; this builder emits HTTP/1.1, which enables
5//! chunked responses from modern servers (needed by Phase 02 `BodyReader`).
6
7use alloc::vec::Vec;
8
9/// Builds a well-formed HTTP/1.1 request and serialises it into a `Vec<u8>`.
10///
11/// # Auto-emitted headers
12///
13/// - `Host: <host>` — always first after the request line.
14/// - `Content-Length: <n>` — only when `body` is `Some` and non-empty.
15/// - `Connection: close` — always last in the auto-headers.
16///
17/// # Caller-supplied headers
18///
19/// `extra_headers` are emitted between `Host` and `Connection: close`.
20/// Callers MUST NOT repeat `Host`, `Content-Length`, or `Connection` in
21/// `extra_headers`; doing so produces a technically valid but redundant request.
22///
23/// # Panics
24///
25/// Never — all formatting is done via `Vec<u8>` writes which cannot fail in
26/// a healthy allocator.  Malformed method/path strings are passed through as-is
27/// (no validation — callers own the correctness of those fields).
28pub struct RequestBuilder<'a> {
29    method: &'a str,
30    path: &'a str,
31    host: &'a str,
32    extra_headers: &'a [(&'a str, &'a str)],
33    body: Option<&'a [u8]>,
34}
35
36impl<'a> RequestBuilder<'a> {
37    /// Create a new builder.
38    ///
39    /// - `method` — HTTP verb, e.g. `"POST"`.
40    /// - `path` — Request-URI, e.g. `"/v1/chat/completions"`.
41    /// - `host` — value for the `Host` header, e.g. `"api.openai.com"`.
42    /// - `extra_headers` — ordered list of additional headers to emit.
43    /// - `body` — optional request body bytes; triggers `Content-Length`.
44    pub fn new(
45        method: &'a str,
46        path: &'a str,
47        host: &'a str,
48        extra_headers: &'a [(&'a str, &'a str)],
49        body: Option<&'a [u8]>,
50    ) -> Self {
51        Self {
52            method,
53            path,
54            host,
55            extra_headers,
56            body,
57        }
58    }
59
60    /// Serialise the request into a byte vector ready to write to a stream.
61    ///
62    /// Layout (RFC 9112):
63    /// ```text
64    /// <METHOD> <path> HTTP/1.1\r\n
65    /// Host: <host>\r\n
66    /// [<extra headers>\r\n]*
67    /// [Content-Length: <n>\r\n]
68    /// Connection: close\r\n
69    /// \r\n
70    /// [<body>]
71    /// ```
72    pub fn build(self) -> Vec<u8> {
73        let mut out = Vec::new();
74
75        // Request line
76        push_str(&mut out, self.method);
77        out.push(b' ');
78        push_str(&mut out, self.path);
79        out.extend_from_slice(b" HTTP/1.1\r\n");
80
81        // Mandatory Host header
82        push_header(&mut out, "Host", self.host);
83
84        // Caller-supplied headers
85        for (name, value) in self.extra_headers {
86            push_header(&mut out, name, value);
87        }
88
89        // Content-Length — only when body is present and non-empty
90        if let Some(body) = self.body {
91            if !body.is_empty() {
92                let len_str = usize_to_decimal(body.len());
93                push_header(&mut out, "Content-Length", &len_str);
94            }
95        }
96
97        // Connection: close — signals to the server that this is a one-shot req
98        push_header(&mut out, "Connection", "close");
99
100        // Header terminator
101        out.extend_from_slice(b"\r\n");
102
103        // Body
104        if let Some(body) = self.body {
105            out.extend_from_slice(body);
106        }
107
108        out
109    }
110}

112// ─── helpers ────────────────────────────────────────────────────────────────
113
114fn push_str(buf: &mut Vec<u8>, s: &str) {
115    buf.extend_from_slice(s.as_bytes());
116}
117
118fn push_header(buf: &mut Vec<u8>, name: &str, value: &str) {
119    push_str(buf, name);
120    buf.extend_from_slice(b": ");
121    push_str(buf, value);
122    buf.extend_from_slice(b"\r\n");
123}

125/// Convert a `usize` to its decimal string representation without `std::fmt`.
126///
127/// Used in no_std context where `format!` works but we want to avoid allocating
128/// an extra `String`; this writes directly to a small stack buffer.
129fn usize_to_decimal(mut n: usize) -> alloc::string::String {
130    if n == 0 {
131        return alloc::string::String::from("0");
132    }
133    // 20 digits is enough for u64::MAX
134    let mut buf = [0u8; 20];
135    let mut i = 0;
136    while n > 0 {
137        buf[i] = b'0' + (n % 10) as u8;
138        n /= 10;
139        i += 1;
140    }
141    buf[..i].reverse();
142    alloc::string::String::from(core::str::from_utf8(&buf[..i]).unwrap())
143}
