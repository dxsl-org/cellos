1//! Incremental HTTP/1.1 response body decoder (`BodyReader`).
2//!
3//! Decodes **Content-Length** and **chunked** framing over a blocking
4//! `embedded_io::Read` transport, returning body bytes a slice at a time
5//! WITHOUT ever buffering the whole (possibly unbounded, e.g. a streaming LLM
6//! response) body.  `UntilClose` framing is deliberately unsupported (plan
7//! decision 4): a TLS transport cannot distinguish "no data yet" from
8//! "connection closed", so there is no reliable EOF signal to terminate on.
9//!
10//! # Completion vs. transport EOF (the load-bearing contract)
11//!
12//! `read()` returning `Ok(0)` means the body is **complete per its framing** —
13//! Content-Length reached `remaining == 0`, or the chunked stream hit its
14//! `0\r\n` terminator.  That is the ONLY success-completion signal.
15//!
16//! When the underlying `transport.read()` returns `0` while the body is still
17//! **incomplete**, that is a genuine EOF: per `embedded_io::Read`, `Ok(0)` means
18//! "no more bytes will ever come".  Both real transports honour this —
19//! `TcpStream::read` returns `0` only on FIN (it loops internally on a transient
20//! empty read), and `TlsStream::read` never returns `0` (it converts a
21//! no-progress retry budget to `Err`).  So a 0-read mid-body is a definitive
22//! truncation, and `BodyReader` returns `Err(HttpError::UnexpectedEof)` — NOT
23//! `NeedMoreData`.  Returning `NeedMoreData` here would make a `NeedMoreData →
24//! retry` caller spin forever over a closed TCP connection (the FIN-spin bug).
25//!
26//! `HttpError::NeedMoreData` belongs exclusively to header parsing
27//! (`parse_response_headers`); the body decoder never emits it.  Keeping all
28//! transport policy (retry, timeout) out of `http-core` is what lets this crate
29//! stay pure and host-testable: no spin, no yield, no timer lives here.
30
31use crate::response::{Framing, HttpError};
32use alloc::vec::Vec;
33use embedded_io::Read;
34
35/// Size of the chunk-size line accumulator.
36///
37/// A chunk-size line is `1*HEXDIG [ chunk-ext ] CRLF`.  httparse rejects more
38/// than 16 hex digits, so 16 + CRLF = 18 bytes covers any well-formed size with
39/// no extension; 32 leaves room for a short extension token while still bounding
40/// internal state.  A size line that does not complete within 32 bytes is
41/// treated as malformed (`HttpError::BadChunk`) — defends against a hostile
42/// server streaming an unbounded "size line".
43const LINE_BUF_LEN: usize = 32;
44
45/// Upper bound on a single chunk's declared size, guarding the `u64 -> usize`
46/// cast.  16 MiB comfortably exceeds any realistic response chunk yet stays far
47/// below `usize::MAX` even on 32-bit targets (riscv32: `usize` is 4 bytes), so
48/// the cast below can never truncate.  A chunk claiming more is rejected as
49/// `HttpError::BadChunk` rather than risking a wraparound.
50const MAX_CHUNK_SIZE: u64 = 16 * 1024 * 1024;
51
52/// Which part of a chunk the decoder expects next.
53///
54/// NOTE (spec deviation): the phase spec listed the chunked state as
55/// `{ chunk_remaining, line_buf, line_len, done }`.  An explicit `phase` is
56/// added so the "data fully delivered, trailing CRLF not yet consumed" moment
57/// is representable WITHOUT eagerly consuming the CRLF in the same call that
58/// returned the data — eager consumption would lose returned bytes if the CRLF
59/// were split across reads.  This is a correctness fix, documented per the
60/// no-silent-deviation rule.
61#[derive(PartialEq, Eq, Clone, Copy)]
62enum ChunkPhase {
63    /// Expecting (the next byte of) a chunk-size line.
64    Size,
65    /// Inside a chunk's data section (`chunk_remaining` bytes still owed).
66    Data,
67    /// Data delivered; expecting the 2-byte CRLF that trails it.
68    TrailingCrlf,
69    /// `0`-size terminator seen; draining the trailer up to its final CRLF.
70    Trailer,
71}
72
73/// Decoder state, one variant per supported framing.
74enum State {
75    /// Content-Length framing: serve bytes until `remaining` hits zero.
76    ContentLength { remaining: usize },
77    /// Chunked framing: drives size-line -> data -> trailing-CRLF -> repeat.
78    Chunked {
79        /// What the decoder expects to read next.
80        phase: ChunkPhase,
81        /// Bytes still owed from the current chunk's data section.
82        chunk_remaining: u64,
83        /// Accumulator for a chunk-size line split across transport reads.
84        line_buf: [u8; LINE_BUF_LEN],
85        /// Valid prefix length within `line_buf`.
86        line_len: usize,
87        /// Set once the `0\r\n` terminator (and its trailer) is fully consumed.
88        done: bool,
89    },
90}
91
92/// Incremental body reader.  Construct with [`BodyReader::new`], then call
93/// [`BodyReader::read`] repeatedly until it returns `Ok(0)`.
94pub struct BodyReader {
95    state: State,
96    /// Bytes already received past the header terminator during header parsing.
97    /// CRITICAL: the first chunk-size line frequently arrives in the SAME packet
98    /// as the headers, so these MUST be consumed before any transport read.
99    leftover: Vec<u8>,
100    /// Read cursor into `leftover`.
101    leftover_pos: usize,
102}
103
104impl BodyReader {
105    /// Build a reader for `framing`.
106    ///
107    /// `content_length` is required (and used) only for `Framing::ContentLength`;
108    /// it is ignored for chunked.  A `ContentLength` framing with `None` length
109    /// is treated as a zero-length body (the header parser only yields
110    /// `ContentLength` when it saw a valid length, but we stay total here).
111    ///
112    /// `leftover` is the slice of post-header bytes already buffered by the
113    /// caller; it is copied in and drained before the transport is touched.
114    pub fn new(framing: Framing, content_length: Option<usize>, leftover: &[u8]) -> Self {
115        let state = match framing {
116            Framing::ContentLength => State::ContentLength {
117                remaining: content_length.unwrap_or(0),
118            },
119            Framing::Chunked => State::Chunked {
120                phase: ChunkPhase::Size,
121                chunk_remaining: 0,
122                line_buf: [0u8; LINE_BUF_LEN],
123                line_len: 0,
124                done: false,
125            },
126        };
127        BodyReader {
128            state,
129            leftover: leftover.to_vec(),
130            leftover_pos: 0,
131        }
132    }
133
134    /// Decode the next slice of body into `out`.
135    ///
136    /// Returns the number of bytes written (`<= out.len()`).  `Ok(0)` means the
137    /// body is complete per its framing.  A transport 0-read on an incomplete
138    /// body yields `Err(HttpError::UnexpectedEof)` (truncated).  See the module
139    /// doc for the precise completion-vs-EOF contract.
140    pub fn read<R: Read>(&mut self, transport: &mut R, out: &mut [u8]) -> Result<usize, HttpError> {
141        if out.is_empty() {
142            return Ok(0);
143        }
144        match &mut self.state {
145            State::ContentLength { .. } => self.read_content_length(transport, out),
146            State::Chunked { .. } => self.read_chunked(transport, out),
147        }
148    }
149
150    /// Drain pending leftover bytes into `dst`, returning how many were copied.
151    fn drain_leftover(&mut self, dst: &mut [u8]) -> usize {
152        let avail = &self.leftover[self.leftover_pos..];
153        let n = avail.len().min(dst.len());
154        dst[..n].copy_from_slice(&avail[..n]);
155        self.leftover_pos += n;
156        n
157    }
158
159    /// True once all buffered leftover bytes have been consumed.
160    fn leftover_empty(&self) -> bool {
161        self.leftover_pos >= self.leftover.len()
162    }
163}
