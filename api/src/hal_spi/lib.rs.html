3/// SPI bus error kinds.
4#[derive(Debug, PartialEq)]
5pub enum SpiError {
6    /// GPIO or bus-state fault (pin direction, MMIO error).
7    BusError,
8    /// Transfer aborted mid-frame (reserved for future hardware SPI adapters).
9    TransferError,
10}
11
12/// Synchronous SPI master trait (Mode 0: CPOL=0, CPHA=0).
13///
14/// # Contract
15/// - Implementations shift **MSB-first**.
16/// - `transfer` and `write` manage chip-select internally (CS asserted before
17///   first clock, deasserted after last clock or on error).
18/// - `cs_select` / `cs_deselect` are provided for multi-transfer transactions
19///   where the caller needs to keep CS asserted across multiple `transfer`/`write`
20///   calls (advanced use — most callers should use `transfer`/`write` directly).
21/// - `transfer`: clocks `max(tx.len(), rx.len())` SCK pulses. TX bytes beyond
22///   `tx.len()` are padded with `0x00`; RX bytes beyond `rx.len()` are discarded.
23/// - All methods are synchronous (no async boundary) so `&[u8]` / `&mut [u8]` are valid.
24pub trait ViSpi {
25    type Error: core::fmt::Debug;
26
27    /// Assert chip-select (active-low).
28    ///
29    /// For advanced multi-transfer use only; `transfer`/`write` assert CS automatically.
30    fn cs_select(&mut self) -> Result<(), Self::Error>;
31
32    /// Deassert chip-select.
33    ///
34    /// For advanced multi-transfer use only; `transfer`/`write` deassert CS automatically.
35    fn cs_deselect(&mut self) -> Result<(), Self::Error>;
36
37    /// Full-duplex transfer: shift out `tx[i]`, shift in `rx[i]` simultaneously.
38    ///
39    /// Asserts CS before the first clock and deasserts it after the last clock (or on error).
40    /// Clocks `max(tx.len(), rx.len())` SCK pulses.
41    /// TX bytes beyond `tx.len()` are driven as `0x00`.
42    /// RX bytes beyond `rx.len()` are discarded (MISO still clocked in but dropped).
43    fn transfer(&mut self, tx: &[u8], rx: &mut [u8]) -> Result<(), Self::Error>;
44
45    /// Write-only transfer: shift out every byte in `bytes`, discard MISO.
46    ///
47    /// Asserts CS before the first clock and deasserts it after the last clock (or on error).
48    fn write(&mut self, bytes: &[u8]) -> Result<(), Self::Error>;
49}
