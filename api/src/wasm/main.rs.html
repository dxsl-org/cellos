1//! WASM host cell — runs WebAssembly (.wasm) binaries as Tier 2 cells.
2//!
3//! Usage: spawn `/bin/wasm` with the `.wasm` file path as argv.
4//! The shell sets this via `sys_set_spawn_args("/data/apps/app.wasm")`.
5
6#![no_std]
7#![no_main]
8extern crate alloc;
9
10use alloc::vec::Vec;
11use driver_wasm::imports::register_vi_imports;
12use driver_wasm::{HostState, WasmConfig, WasmRuntime};
13
14api::declare_syscalls![Send, Recv, Log, Heartbeat];
15
16/// VFS service task ID — boot order: init=1, user_hello=2, vfs=3.
17const VFS_ENDPOINT: usize = 3;
18
19#[no_mangle]
20pub fn main() {
21    // 1. Read .wasm path from spawn-argv stash (set by shell before spawning).
22    let mut argv_buf = [0u8; 256];
23    let n = ostd::syscall::sys_spawn_args(&mut argv_buf);
24    let path = if n > 0 {
25        core::str::from_utf8(&argv_buf[..n]).unwrap_or("/data/apps/app.wasm")
26    } else {
27        "/data/apps/app.wasm"
28    };
29
30    // 2. Load .wasm binary from VFS.
31    let wasm_bytes = load_from_vfs(path);
32    if wasm_bytes.is_empty() {
33        ostd::io::print("[wasm] error: could not read ");
34        ostd::io::println(path);
35        ostd::syscall::sys_exit(1);
36        // sys_exit diverges; unreachable but satisfies the compiler with no warning
37    }
38
39    // 3. Initialise wasmi with fuel metering.
40    let config = WasmConfig::default();
41    let runtime = WasmRuntime::new(&config);
42
43    // 4. Parse and validate the WASM module.
44    let module = match runtime.load_module(&wasm_bytes) {
45        Ok(m) => m,
46        Err(_) => {
47            ostd::io::print("[wasm] error: invalid WASM module at ");
48            ostd::io::println(path);
49            ostd::syscall::sys_exit(1);
50        }
51    };
52
53    // 5. Create store + linker and register vi.* imports.
54    let mut store = runtime.new_store(&config, HostState { cell_task_id: 0 });
55    let mut linker = runtime.new_linker();
56    register_vi_imports(&mut linker);
57
58    // 6. Instantiate the module and run its start function (if any).
59    //    wasmi v1: use `instantiate_and_start` — NOT `.instantiate().and_then(.start())`.
60    let instance = match linker.instantiate_and_start(&mut store, &module) {
61        Ok(i) => i,
62        Err(_) => {
63            ostd::io::print("[wasm] error: instantiation failed\n");
64            ostd::syscall::sys_exit(1);
65        }
66    };
67
68    // 7. Retrieve the exported `run` function.
69    let run_fn = match instance.get_typed_func::<(), ()>(&store, "run") {
70        Ok(f) => f,
71        Err(_) => {
72            ostd::io::print("[wasm] error: module must export 'run: () -> ()'\n");
73            ostd::syscall::sys_exit(1);
74        }
75    };
76
77    // 8. Run in a fuel-cooperative loop.
78    //    vi.exit() calls sys_exit() which never returns, so we only reach
79    //    this loop body on fuel exhaustion or the module's run() returning.
80    //    NOTE: verify exact wasmi v1 OutOfFuel error variant before extending
81    //    this to distinguish trap types for untrusted WASM cells.
82    loop {
83        match run_fn.call(&mut store, ()) {
84            Ok(()) => break, // module returned normally from run()
85            Err(_) => {
86                // For Phase 28 MVP (trusted WASM only): treat all traps as fuel
87                // exhaustion since vi.exit() never returns to this point.
88                store.set_fuel(config.fuel_per_tick).ok();
89                ostd::task::yield_now();
90            }
91        }
92    }
93
94    ostd::syscall::sys_exit(0);
95}

97/// Read a `.wasm` binary from the VFS service via typed IPC.
98///
99/// Uses `VfsResponse::DataPtr` for zero-copy access to VFS memory.
100fn load_from_vfs(path: &str) -> Vec<u8> {
101    let mut send_buf = [0u8; 512];
102    let n = match api::ipc::encode(&api::ipc::VfsRequest::GetFile(path), &mut send_buf) {
103        Ok(s) => s.len(),
104        Err(_) => return Vec::new(),
105    };
106    ostd::syscall::sys_send(VFS_ENDPOINT, &send_buf[..n]);
107
108    let mut reply = [0u8; 512];
109    // sys_recv always returns SyscallResult::Ok(sender_id) — the Err arm is
110    // unreachable with the current ostd API.  If VFS isn't ready or replies
111    // with an unexpected response, postcard decode will fail → empty Vec.
112    match ostd::syscall::sys_recv(0, &mut reply) {
113        ostd::syscall::SyscallResult::Ok(sender) if sender > 0 => {
114            match api::ipc::decode::<api::ipc::VfsResponse>(&reply) {
115                Ok(api::ipc::VfsResponse::DataPtr { ptr, len }) => {
116                    // SAFETY: VFS replied with a pointer into SAS-shared VFS memory.
117                    // The decode succeeded, confirming this is a valid DataPtr response.
118                    // VFS is blocked waiting for its next sys_recv, so no concurrent
119                    // modification of the pointed-to memory can occur while we copy.
120                    unsafe { core::slice::from_raw_parts(ptr as *const u8, len as usize).to_vec() }
121                }
122                _ => Vec::new(),
123            }
124        }
125        _ => Vec::new(),
126    }
127}
