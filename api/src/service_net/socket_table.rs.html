1//! CapId-keyed socket handle table for the net service cell.
2//!
3//! Maps kernel-issued `CapId`s (u64) to smoltcp `SocketHandle`s so that
4//! any consumer cell can reference an open socket across IPC calls without
5//! exposing smoltcp-internal handles.
6
7extern crate alloc;
8
9use crate::socket_state::SocketState;
10use alloc::collections::{BTreeMap, BTreeSet};
11use smoltcp::iface::SocketHandle;
12use types::ViError;
13
14/// Maximum simultaneous sockets (including the DHCP management socket).
15pub const MAX_SOCKETS: usize = 18; // 16 user + 1 DHCP + 1 ARP
16
17/// Maps a `CapId` to a smoltcp `SocketHandle` and connection state.
18#[derive(Default)]
19pub struct SocketTable {
20    entries: BTreeMap<u64, SocketHandle>,
21    states: BTreeMap<u64, SocketState>,
22    /// Bound listen port per cap — needed so ACCEPT can renew the listener on
23    /// the same port after the original socket transitions to Established.
24    listen_ports: BTreeMap<u64, u16>,
25    /// Tracks which caps hold UDP sockets so TCP-only opcodes can reject them
26    /// before calling `sockets.get_mut::<tcp::Socket>()`, which panics on a
27    /// wrong-type handle.
28    udp_caps: BTreeSet<u64>,
29    next_cap: u64,
30}
31
32impl SocketTable {
33    pub fn new() -> Self {
34        Self {
35            entries: BTreeMap::new(),
36            states: BTreeMap::new(),
37            listen_ports: BTreeMap::new(),
38            udp_caps: BTreeSet::new(),
39            next_cap: 1,
40        }
41    }
42
43    /// Allocate a new `CapId` and associate it with `handle`.
44    ///
45    /// # Errors
46    /// Returns `ViError::OutOfMemory` if `MAX_SOCKETS` is already reached.
47    pub fn insert(&mut self, handle: SocketHandle) -> Result<u64, ViError> {
48        if self.entries.len() >= MAX_SOCKETS {
49            return Err(ViError::OutOfMemory);
50        }
51        let cap = self.next_cap;
52        self.next_cap += 1;
53        self.entries.insert(cap, handle);
54        self.states.insert(cap, SocketState::Created);
55        Ok(cap)
56    }
57
58    /// Allocate a new `CapId` for `handle` with an explicit initial state.
59    ///
60    /// Unlike `insert` (which defaults to `Created`), this sets `state` directly.
61    /// Only call from ACCEPT — the handle is already Established and must be
62    /// surfaced as `Connected` to the consumer.
63    ///
64    /// # Errors
65    /// Returns `ViError::OutOfMemory` if `MAX_SOCKETS` is already reached.
66    pub fn insert_with_state(
67        &mut self,
68        handle: SocketHandle,
69        state: SocketState,
70    ) -> Result<u64, ViError> {
71        if self.entries.len() >= MAX_SOCKETS {
72            return Err(ViError::OutOfMemory);
73        }
74        let cap = self.next_cap;
75        self.next_cap += 1;
76        self.entries.insert(cap, handle);
77        self.states.insert(cap, state);
78        Ok(cap)
79    }
80
81    /// Look up the smoltcp `SocketHandle` for a given `CapId`.
82    pub fn get(&self, cap: u64) -> Option<SocketHandle> {
83        self.entries.get(&cap).copied()
84    }
85
86    /// Read the connection state for `cap`.
87    pub fn get_state(&self, cap: u64) -> Option<SocketState> {
88        self.states.get(&cap).copied()
89    }
90
91    /// Update the connection state for `cap`.
92    pub fn set_state(&mut self, cap: u64, state: SocketState) {
93        if self.entries.contains_key(&cap) {
94            self.states.insert(cap, state);
95        }
96    }
97
98    /// Record the port a listening socket is bound to, so ACCEPT can renew it.
99    pub fn set_listen_port(&mut self, cap: u64, port: u16) {
100        if self.entries.contains_key(&cap) {
101            self.listen_ports.insert(cap, port);
102        }
103    }
104
105    /// Read the bound listen port for `cap`, if any.
106    pub fn get_listen_port(&self, cap: u64) -> Option<u16> {
107        self.listen_ports.get(&cap).copied()
108    }
109
110    /// Repoint an existing cap at a new smoltcp handle.
111    ///
112    /// Used by ACCEPT to swap the exhausted listening handle for a fresh one
113    /// without changing the cap the consumer holds.
114    pub fn update_handle(&mut self, cap: u64, new_handle: SocketHandle) {
115        if self.entries.contains_key(&cap) {
116            self.entries.insert(cap, new_handle);
117        }
118    }
119
120    /// Mark a cap as holding a UDP socket.
121    ///
122    /// TCP-only opcodes (CONNECT, SEND, RECV, etc.) check this to avoid calling
123    /// `sockets.get_mut::<tcp::Socket>` on a UDP handle, which panics.
124    pub fn mark_udp(&mut self, cap: u64) {
125        self.udp_caps.insert(cap);
126    }
127
128    /// Returns `true` if `cap` holds a UDP socket.
129    pub fn is_udp(&self, cap: u64) -> bool {
130        self.udp_caps.contains(&cap)
131    }
132
133    /// Remove a socket from the table (called on close).
134    pub fn remove(&mut self, cap: u64) -> Option<SocketHandle> {
135        self.states.remove(&cap);
136        self.listen_ports.remove(&cap);
137        self.udp_caps.remove(&cap);
138        self.entries.remove(&cap)
139    }
140}
