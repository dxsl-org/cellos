1//! SmoltcpTlsTransport — smoltcp TCP socket as an embedded-io Read+Write transport.
2//!
3//! Since `TlsConnection` does not expose `delegate_mut()`, the smoltcp context
4//! pointers (iface/device/sockets) are stored in module-level AtomicPtrs.
5//! Before any TLS I/O call, invoke `set_tls_context()` with the current mutable
6//! references.  Safe because the net cell is single-core.
7//!
8//! # Transport timeouts (M4)
9//! Both `Read` and `Write` spin-poll smoltcp.  The original iteration-count cap
10//! (`MAX_SPIN`) was replaced with a wall-clock deadline: a software ECDSA/RSA
11//! verify on QEMU TCG can take tens of milliseconds and runs *between* `read()`
12//! calls, easily exceeding a spin count while the work is still making progress.
13//! Using `sys_get_time()` (10 MHz monotonic counter) ties the timeout to real
14//! elapsed time regardless of CPU speed.
15//!
16//! `sys_heartbeat()` is called inside both spin loops so a long but progressing
17//! verify does not trip the RT watchdog.  The wall-clock deadline still fires
18//! if the transport genuinely hangs.
19

24use embedded_io::{ErrorKind, ErrorType, Read, Write};

32// ── Thread-local smoltcp context (single-core net cell) ───────────────────────
33
34static TLS_IFACE: AtomicPtr<()> = AtomicPtr::new(core::ptr::null_mut());
35static TLS_DEVICE: AtomicPtr<()> = AtomicPtr::new(core::ptr::null_mut());
36static TLS_SOCKETS: AtomicPtr<()> = AtomicPtr::new(core::ptr::null_mut());
37
38/// Update module-level smoltcp context pointers before any TLS I/O call.
39///
40/// # Safety
41/// Caller must hold valid `&mut` references to all three objects. On single-core
42/// hardware the absence of data races makes this safe for the duration of one IPC
43/// handler invocation.
44pub unsafe fn set_tls_context(
45    iface: *mut Interface,
46    device: *mut VirtioNetDevice,
47    sockets: *mut (),
48) {
49    TLS_IFACE.store(iface as *mut (), Ordering::Relaxed);
50    TLS_DEVICE.store(device as *mut (), Ordering::Relaxed);
51    TLS_SOCKETS.store(sockets, Ordering::Relaxed);
52}
53
54// ── Timeout constants ─────────────────────────────────────────────────────────
55
56/// Wall-clock TLS I/O timeout in sys_get_time ticks (10 MHz monotonic clock).
57///
58/// 30 seconds: generous enough for software RSA-PSS verify on QEMU TCG (~1-5 s
59/// measured), with headroom for a loaded system.  P03 e2e will tighten if needed.
60/// Formula: 30 s * 10_000_000 ticks/s = 300_000_000.
61const TLS_IO_TIMEOUT_TICKS: u64 = 300_000_000;
62
63/// Heartbeat interval: reset the RT watchdog every 500 ms worth of ticks while
64/// spinning.  500 ms * 10_000_000 ticks/s = 5_000_000 ticks.
65const HEARTBEAT_INTERVAL_TICKS: u64 = 5_000_000;
66
67// ── Transport implementation ──────────────────────────────────────────────────
68
69/// Wraps a smoltcp TCP socket handle as an embedded-io Read+Write transport.
70///
71/// Reads the smoltcp context from the module-level AtomicPtrs set by `set_tls_context`.
72pub struct SmoltcpTlsTransport {
73    pub handle: SocketHandle,
74}
75
76// SAFETY: Net cell is single-core; AtomicPtrs are updated under exclusive access.
77unsafe impl Send for SmoltcpTlsTransport {}
78unsafe impl Sync for SmoltcpTlsTransport {}
79
80impl SmoltcpTlsTransport {
81    pub fn new(handle: SocketHandle) -> Self {
82        Self { handle }
83    }
84
85    unsafe fn iface() -> &'static mut Interface {
86        &mut *(TLS_IFACE.load(Ordering::Relaxed) as *mut Interface)
87    }
88    unsafe fn device() -> &'static mut VirtioNetDevice {
89        &mut *(TLS_DEVICE.load(Ordering::Relaxed) as *mut VirtioNetDevice)
90    }
91    unsafe fn sockets() -> &'static mut SocketSet<'static> {
92        &mut *(TLS_SOCKETS.load(Ordering::Relaxed) as *mut SocketSet<'static>)
93    }
94
95    fn now() -> Instant {
96        Instant::from_micros((sys_get_time() / 10) as i64)
97    }
98
99    unsafe fn poll() {
100        Self::device().pump_rx();
101        Self::iface().poll(Self::now(), Self::device(), Self::sockets());
102    }
103}
104
105impl ErrorType for SmoltcpTlsTransport {
106    type Error = ErrorKind; // embedded_io::ErrorKind implements embedded_io::Error
107}

109impl Read for SmoltcpTlsTransport {
110    fn read(&mut self, buf: &mut [u8]) -> Result<usize, ErrorKind> {
111        let deadline = sys_get_time() + TLS_IO_TIMEOUT_TICKS;
112        let mut next_heartbeat = sys_get_time() + HEARTBEAT_INTERVAL_TICKS;
113
114        loop {
115            // SAFETY: set_tls_context was called before open()/read().
116            unsafe {
117                Self::poll();
118            }
119            let (n, eof) = unsafe {
120                let socket = Self::sockets().get_mut::<tcp::Socket>(self.handle);
121                let n = if socket.can_recv() {
122                    socket.recv_slice(buf).unwrap_or(0)
123                } else {
124                    0
125                };
126                // EOF: remote sent FIN and RX buffer is drained — no more data will arrive.
127                (n, !socket.may_recv() && !socket.can_recv())
128            };
129            if n > 0 {
130                return Ok(n);
131            }
132            // Signal connection close so embedded-tls returns an error immediately
133            // instead of spinning all the way to the 30-second deadline.
134            if eof {
135                return Err(ErrorKind::ConnectionReset);
136            }
137
138            let now = sys_get_time();
139            if now >= deadline {
140                return Err(ErrorKind::TimedOut);
141            }
142            // Periodically reset the RT watchdog so a slow software verify
143            // (ECDSA/RSA on QEMU TCG) does not look like a hung task.
144            // The deadline above still fires if the transport genuinely hangs.
145            if now >= next_heartbeat {
146                sys_heartbeat(500);
147                next_heartbeat = now + HEARTBEAT_INTERVAL_TICKS;
148            }
149            core::hint::spin_loop();
150        }
151    }
152}
153
154impl Write for SmoltcpTlsTransport {
155    fn write(&mut self, buf: &[u8]) -> Result<usize, ErrorKind> {
156        let deadline = sys_get_time() + TLS_IO_TIMEOUT_TICKS;
157        let mut next_heartbeat = sys_get_time() + HEARTBEAT_INTERVAL_TICKS;
158        let mut written = 0usize;
159
160        while written < buf.len() {
161            let n = unsafe {
162                let socket = Self::sockets().get_mut::<tcp::Socket>(self.handle);
163                if socket.can_send() {
164                    socket.send_slice(&buf[written..]).unwrap_or(0)
165                } else {
166                    0
167                }
168            };
169            if n > 0 {
170                written += n;
171                // Note: `deadline` is a fixed total-write budget (30s), not reset on
172                // progress. Handshake writes are small, so this is ample; revisit only
173                // if bulk TLS writes ever need a progress-reset deadline.
174            } else {
175                // Detect connection close before spinning: if the remote sent FIN/RST
176                // may_send() returns false and we'd spin to the 30-second deadline.
177                let gone = unsafe {
178                    !Self::sockets()
179                        .get_mut::<tcp::Socket>(self.handle)
180                        .may_send()
181                };
182                if gone {
183                    return Err(ErrorKind::BrokenPipe);
184                }
185                unsafe {
186                    Self::poll();
187                }
188
189                let now = sys_get_time();
190                if now >= deadline {
191                    return Err(ErrorKind::TimedOut);
192                }
193                if now >= next_heartbeat {
194                    sys_heartbeat(500);
195                    next_heartbeat = now + HEARTBEAT_INTERVAL_TICKS;
196                }
197                core::hint::spin_loop();
198            }
199        }
200        unsafe {
201            Self::poll();
202        }
203        Ok(written)
204    }
205
206    fn flush(&mut self) -> Result<(), ErrorKind> {
207        unsafe {
208            Self::poll();
209        }
210        Ok(())
211    }
