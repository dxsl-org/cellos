1//! ViTlsProvider — verifying CryptoProvider for the embedded TLS build.
2//!
3//! Wraps a `pki::CertVerifier` (single CA anchor) and `ViTlsClock` so that
4//! every TLS handshake in the default build verifies the server certificate
5//! chain, validity dates, and hostname.
6//!
7//! # Security invariant (B3)
8//! `verifier()` MUST return `Ok` unconditionally.  `connection.rs:455` in
9//! embedded-tls silently skips verification when `verifier()` returns `Err`
10//! — a single fallible path = silent MITM.  The unit test `verifier_always_ok`
11//! guards this property.
12//!
13//! This module is only compiled when `tls-roots-embedded` is active (i.e. not
14//! in `tls-insecure` builds).
15
16#![cfg(feature = "tls-roots-embedded")]
17
18use crate::tls::clock::ViTlsClock;
19use crate::tls::rng::ViRng;
20use crate::tls::roots::ca_cert;
21use embedded_tls::pki::CertVerifier;
22use embedded_tls::CryptoRngCore;
23use embedded_tls::{Aes128GcmSha256, CryptoProvider, TlsError, TlsVerifier};
24
25/// Cert buffer size for the leaf certificate held in-memory during handshake.
26/// 4096 bytes covers typical ECC certificates with SAN extensions.
27const CERT_SIZE: usize = 4096;
28
29/// A verifying TLS crypto provider for ViCell's embedded build.
30///
31/// Constructed per-handshake inside `socket::TlsSocketEntry::handshake()`.
32/// Never stored as a global — the verifier holds per-connection transcript state.
33pub struct ViTlsProvider {
34    rng: ViRng,
35    verifier: CertVerifier<'static, Aes128GcmSha256, ViTlsClock, CERT_SIZE>,
36}
37
38impl ViTlsProvider {
39    /// Build a provider anchored to the build-selected CA, with hostname verification.
40    ///
41    /// `hostname` must match the server's certificate CN or SAN.  An empty
42    /// hostname is rejected by `handshake()` before this is called, so
43    /// `set_hostname_verification` here is always called with a non-empty value.
44    ///
45    /// # Errors
46    /// Returns `TlsError::InsufficientSpace` if `hostname` exceeds 64 characters
47    /// (the `pki::CertVerifier` limit documented in spike-00-findings.md).
48    pub fn new(rng: ViRng, hostname: &str) -> Result<Self, TlsError> {
49        let mut verifier = CertVerifier::new(ca_cert());
50        verifier.set_hostname_verification(hostname)?;
51        Ok(Self { rng, verifier })
52    }
53}
54
55impl CryptoProvider for ViTlsProvider {
56    type CipherSuite = Aes128GcmSha256;
57    // [u8; 64] satisfies AsRef<[u8]>; unused in server-auth (no client cert / mTLS).
58    type Signature = [u8; 64];
59
60    fn rng(&mut self) -> impl CryptoRngCore {
61        &mut self.rng
62    }
63
64    /// Return the certificate verifier — MUST be infallible (B3 guard).
65    ///
66    /// embedded-tls connection.rs:455 silently skips verification when this
67    /// returns Err.  Any `Err` path here = silent MITM.  Never add fallible
68    /// logic to this method.
69    fn verifier(&mut self) -> Result<&mut impl TlsVerifier<Self::CipherSuite>, TlsError> {
70        Ok(&mut self.verifier)
71    }
72
73    // signer() and client_cert() keep their default (no mTLS in this build).
74}

78#[cfg(test)]
79mod tests {
80    use super::*;
81
82    /// B3 guard: verifier() must always return Ok, regardless of the hostname
83    /// passed to ViTlsProvider::new().  A single Err path = MITM bypass.
84    #[test]
85    fn verifier_always_ok() {
86        let rng = ViRng::new();
87        // Use a valid hostname — set_hostname_verification only errors on
88        // strings > 64 chars; a normal hostname always succeeds.
89        let mut provider = ViTlsProvider::new(rng, "broker.example.com")
90            .expect("ViTlsProvider::new must succeed for a valid hostname");
91
92        // The invariant: verifier() returns Ok unconditionally.
93        assert!(
94            provider.verifier().is_ok(),
95            "verifier() returned Err — B3 violation: embedded-tls would skip verification"
96        );
97    }
98
99    /// Confirm that a freshly constructed provider also satisfies the invariant.
100    /// (Redundant with above, but makes the guard explicit for reviewers.)
101    #[test]
102    fn verifier_ok_on_fresh_provider() {
103        let rng = ViRng::new();
104        let mut provider =
105            ViTlsProvider::new(rng, "test.internal").expect("new must not fail on short hostname");
106        let result = provider.verifier();
107        assert!(result.is_ok());
108    }
109}
