1//! TlsSocketEntry — persists a TlsConnection across SEND/RECV IPC calls.
2//!
3//! Uses Box::leak for 16 KiB read/write buffers so TlsConnection<'static> can be
4//! stored in the socket table.  Memory cost: 32 KiB per concurrent TLS connection.
5//! Max 4 simultaneous TLS connections in G1 robot = 128 KiB overhead total.
6//!
7//! Build flavors:
8//!   default (tls-roots-embedded)  — ViTlsProvider; server certs verified
9//!   tls-insecure                  — UnsecureProvider; no verification (dev only)
10
11extern crate alloc;
12
13use crate::interface::VirtioNetDevice;
14use crate::tls::rng::ViRng;
15use crate::tls::transport::{set_tls_context, SmoltcpTlsTransport};
16use alloc::boxed::Box;
17use embedded_tls::blocking::{TlsConfig, TlsConnection, TlsContext};
18use embedded_tls::{Aes128GcmSha256, TlsError};
19use smoltcp::iface::{Interface, SocketHandle};
20
21// Verifying path — default build.
22#[cfg(feature = "tls-roots-embedded")]
23use crate::tls::provider::ViTlsProvider;
24
25// Insecure path — dev/lab builds only.
26#[cfg(feature = "tls-insecure")]
27use embedded_tls::UnsecureProvider;
28
29// A TLS flavor that actually drives `conn.open()` MUST be selected. Without this
30// guard, a build with neither feature (e.g. the unimplemented `tls-roots-full`, or
31// `--no-default-features`) would compile, skip the handshake entirely, and return a
32// TLS entry with ZERO verification — a silent bypass. (haily-reviewer MAJOR-1.)
33#[cfg(not(any(feature = "tls-roots-embedded", feature = "tls-insecure")))]
34compile_error!(
35    "service-net: select a TLS flavor — `tls-roots-embedded` (verifying) or `tls-insecure` (dev only). \
36     `tls-roots-full` is not yet implemented (see plan phase-04)."
37);
38
39/// TLS record buffer size (covers the max 16 KB TLS record + overhead).
40const TLS_BUF: usize = 16640;
41
42/// A live TLS connection stored in the socket table.
43///
44/// After construction the TLS handshake has already completed.  Before calling
45/// `write()` or `read()` the caller must call `set_tls_context()` so the embedded
46/// transport can drive smoltcp.
47pub struct TlsSocketEntry {
48    pub conn: TlsConnection<'static, SmoltcpTlsTransport, Aes128GcmSha256>,
49    // reason: socket teardown goes through the separate SocketTable (by cap_id),
50    // not through this field — kept for API completeness / future direct-close use.
51    #[allow(dead_code)]
52    pub handle: SocketHandle,
53}
54
55impl TlsSocketEntry {
56    /// Perform the TLS 1.3 handshake over `handle` and return a live entry.
57    ///
58    /// Blocks (spin-polls smoltcp via the global transport context) until the
59    /// handshake completes or times out.
60    ///
61    /// In verifying builds (default), an empty `hostname` is rejected before
62    /// `open()` to produce a clear error rather than a silent SAN mismatch.
63    ///
64    /// # Safety
65    /// `set_tls_context()` must have been called with valid pointers before this.
66    pub unsafe fn handshake(handle: SocketHandle, hostname: &str) -> Result<Self, TlsError> {
67        // Leak 32 KiB for TLS record buffers — intentional, documented cost.
68        let read_buf: &'static mut [u8] = Box::leak(Box::new([0u8; TLS_BUF]));
69        let write_buf: &'static mut [u8] = Box::leak(Box::new([0u8; TLS_BUF]));
70
71        let transport = SmoltcpTlsTransport::new(handle);
72        let mut conn = TlsConnection::new(transport, read_buf, write_buf);
73
74        let config = if hostname.is_empty() {
75            TlsConfig::new()
76        } else {
77            TlsConfig::new().with_server_name(hostname)
78        };
79
80        // ── Provider selection ────────────────────────────────────────────────
81        //
82        // tls-roots-embedded (default): verifying path.
83        //   Empty hostname rejected here for a clear diagnostic before open().
84        //   ViTlsProvider::new() sets hostname verification on CertVerifier.
85        //
86        // tls-insecure: accept-all path, dev/lab only.
87        //   Banner printed once on first handshake.
88
89        #[cfg(feature = "tls-roots-embedded")]
90        {
91            if hostname.is_empty() {
92                return Err(TlsError::InvalidCertificate);
93            }
94            let rng = ViRng::new();
95            let provider = ViTlsProvider::new(rng, hostname)?;
96            let ctx = TlsContext::new(&config, provider);
97            conn.open(ctx)?;
98        }
99
100        #[cfg(feature = "tls-insecure")]
101        {
102            // One-time banner: printed on every handshake in insecure builds so
103            // the danger is never invisible in logs.
104            // Safety: single-core net cell; concurrent call not possible.
105            static BANNER_PRINTED: core::sync::atomic::AtomicBool =
106                core::sync::atomic::AtomicBool::new(false);
107            if !BANNER_PRINTED.swap(true, core::sync::atomic::Ordering::Relaxed) {
108                // The message is intentionally alarming — it must not be missed.
109                // sys_log is best-effort; we don't propagate errors.
110                let _ = ostd::syscall::sys_log(
111                    "[net/tls] !!! INSECURE TLS BUILD - server certs NOT verified !!!",
112                );
113            }
114
115            let rng = ViRng::new();
116            let ctx = TlsContext::new(&config, UnsecureProvider::new::<Aes128GcmSha256>(rng));
117            conn.open(ctx)?;
118        }
119
120        Ok(Self { conn, handle })
121    }
122
123    /// Convenience: set TLS context and write data.
124    ///
125    /// # Safety
126    /// Same as `set_tls_context`.
127    pub unsafe fn send(
128        &mut self,
129        data: &[u8],
130        iface: *mut Interface,
131        device: *mut VirtioNetDevice,
132        sockets: *mut (),
133    ) -> Result<usize, TlsError> {
134        set_tls_context(iface, device, sockets);
135        // embedded-tls `write()` only BUFFERS the record (blocking.rs:130) — without
136        // an explicit `flush()` the encrypted record never reaches the transport and
137        // is never transmitted. `conn.open()` flushes its own handshake records, which
138        // is why the handshake completes but the first application write (the HTTP POST)
139        // silently stalled: the peer waited forever for a request that never left.
140        let n = self.conn.write(data)?;
141        self.conn.flush()?;
142        Ok(n)
143    }
144
145    /// Convenience: set TLS context and read data.
146    ///
147    /// # Safety
148    /// Same as `set_tls_context`.
149    pub unsafe fn recv(
150        &mut self,
151        buf: &mut [u8],
152        iface: *mut Interface,
153        device: *mut VirtioNetDevice,
154        sockets: *mut (),
155    ) -> Result<usize, TlsError> {
156        set_tls_context(iface, device, sockets);
157        self.conn.read(buf)
158    }
159}
