1//! ViTlsClock — TlsClock impl backed by the hardware RTC syscall.
2//!
3//! Contract: `now()` returns `Some(epoch_seconds)` clamped to at least
4//! `VICELL_MIN_UNIX` so that a missing or unsynced RTC does not cause every
5//! certificate to appear expired.  It NEVER returns `None` — an absent RTC
6//! produces the floor value, which is a known-safe past date, not a bypass.
7//!
8//! The floor is intentionally conservative: it only protects against epoch-0
9//! (RTC unset).  Valid future dates always override the floor.
10
11use embedded_tls::TlsClock;
12use ostd::syscall::sys_get_wall_secs;
13
14/// Fixed floor: 2025-06-01T00:00:00Z.
15///
16/// Injected by build.rs via `VICELL_MIN_UNIX` (may use SOURCE_DATE_EPOCH for
17/// reproducible builds).  Fallback is the compile-time constant below.
18const VICELL_MIN_UNIX: u64 = {
19    // option_env! returns None at runtime; resolve at compile time.
20    // The value is emitted by build.rs as a string env var.
21    match option_env!("VICELL_MIN_UNIX") {
22        Some(s) => {
23            // const parse u64 from decimal string — no std needed.
24            let bytes = s.as_bytes();
25            let mut val: u64 = 0;
26            let mut i = 0;
27            while i < bytes.len() {
28                val = val * 10 + (bytes[i] - b'0') as u64;
29                i += 1;
30            }
31            val
32        }
33        None => 1_748_736_000,
34    }
35};
36
37/// A TLS clock that reads Unix epoch seconds from the hardware RTC.
38///
39/// Returned time is clamped: `max(rtc_secs, VICELL_MIN_UNIX)`.
40/// This is a zero-sized type — all state lives in the syscall layer.
41pub struct ViTlsClock;
42
43impl TlsClock for ViTlsClock {
44    /// Returns `Some(epoch_seconds)`, clamped to the build-time floor.
45    ///
46    /// Never returns `None` — a missing RTC produces the floor timestamp
47    /// rather than disabling expiry checks, which would silently accept
48    /// expired certificates.
49    fn now() -> Option<u64> {
50        let rtc_secs = sys_get_wall_secs();
51        Some(rtc_secs.max(VICELL_MIN_UNIX))
52    }
53}

55// ── Tests ─────────────────────────────────────────────────────────────────────
56
57#[cfg(test)]
58mod tests {
59    use super::*;
60
61    #[test]
62    fn floor_is_sane() {
63        // The floor must be at least 2025-06-01 so that modern certs are not
64        // immediately expired when the RTC is absent.
65        assert!(VICELL_MIN_UNIX >= 1_748_736_000, "floor too early");
66    }
67
68    #[test]
69    fn now_never_below_floor() {
70        // Simulate an epoch-0 RTC by observing that max(0, floor) == floor.
71        // We cannot call sys_get_wall_secs in a host test (no kernel), but we
72        // can verify the clamp arithmetic directly.
73        let rtc_zero: u64 = 0;
74        let clamped = rtc_zero.max(VICELL_MIN_UNIX);
75        assert_eq!(clamped, VICELL_MIN_UNIX);
76    }
77
78    #[test]
79    fn now_passes_through_valid_time() {
80        // A time well past the floor should pass through unchanged.
81        let future_time: u64 = 1_800_000_000; // ~2027
82        let clamped = future_time.max(VICELL_MIN_UNIX);
83        assert_eq!(clamped, future_time);
84    }
85}
