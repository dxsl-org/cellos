1//! DHCPv4 boot client for the net service cell.
2//!
3//! Adds a smoltcp `Dhcpv4Socket` to the socket set and drives it through the
4//! DISCOVER → OFFER → REQUEST → ACK sequence.  On success it calls
5//! `interface.update_ip_addrs()` and `interface.routes_mut()` to apply the
6//! leased address and default gateway.
7
8use crate::interface::VirtioNetDevice;
9use ostd::io::println;
10use smoltcp::{
11    iface::{Interface, SocketSet},
12    socket::dhcpv4,
13    time::Instant,
14    wire::IpCidr,
15};
16
17/// State of the DHCP boot sequence.
18#[derive(Debug, PartialEq, Eq)]
19pub enum DhcpState {
20    /// Awaiting DHCP ACK.
21    Pending,
22    /// IP address acquired successfully.
23    Acquired,
24    #[allow(dead_code)] // reason: will be used when retry limits are wired in
25    Failed,
26}
27
28/// Add a `Dhcpv4Socket` to `sockets` and return its handle.
29pub fn add_dhcp_socket(sockets: &mut SocketSet<'_>) -> smoltcp::iface::SocketHandle {
30    sockets.add(dhcpv4::Socket::new())
31}
32
33/// Poll the DHCP socket and apply a lease when one arrives.
34///
35/// Returns `DhcpState::Acquired` once an IP address is configured on `iface`.
36pub fn poll_dhcp(
37    handle: smoltcp::iface::SocketHandle,
38    iface: &mut Interface,
39    sockets: &mut SocketSet<'_>,
40    device: &mut VirtioNetDevice,
41    now: Instant,
42) -> DhcpState {
43    iface.poll(now, device, sockets);
44
45    let socket = sockets.get_mut::<dhcpv4::Socket>(handle);
46    if let Some(event) = socket.poll() {
47        match event {
48            dhcpv4::Event::Configured(config) => {
49                iface.update_ip_addrs(|addrs| {
50                    if let Some(slot) = addrs.iter_mut().find(|a| matches!(a, IpCidr::Ipv4(_))) {
51                        *slot = IpCidr::Ipv4(config.address);
52                    } else {
53                        let _ = addrs.push(IpCidr::Ipv4(config.address));
54                    }
55                });
56                if let Some(gw) = config.router {
57                    iface.routes_mut().add_default_ipv4_route(gw).ok();
58                }
59                println("[net] DHCP acquired — IP configured");
60                return DhcpState::Acquired;
61            }
62            dhcpv4::Event::Deconfigured => {
63                println("[net] DHCP: deconfigured");
64            }
65        }
66    }
67    DhcpState::Pending
68}
