3/// PWM channel error kinds.
4#[derive(Debug, Clone, Copy, PartialEq, Eq)]
5pub enum PwmError {
6    /// Channel index out of range.
7    InvalidChannel,
8    /// Frequency is zero or exceeds implementation maximum.
9    InvalidFrequency,
10    /// Duty cycle exceeds 1000 per mille.
11    InvalidDuty,
12    /// Operation on a channel that has not been enabled.
13    NotEnabled,
14}

16/// Pulse-width modulation output trait.
17///
18/// # Contract
19/// - `channel` is a zero-based index. Implementations define the maximum number of channels.
20/// - `duty_per_mille` is the on-time fraction × 1000: 0 = always LOW, 1000 = always HIGH,
21///   500 = 50% duty cycle.
22/// - `tick()` MUST be called continuously in a tight loop to advance the waveform state machine.
23///   Callers that stop calling `tick()` will observe a stuck output level.
24/// - After `disable(channel)`, the pin is driven LOW.
25pub trait ViPwm {
26    type Error: core::fmt::Debug;
27
28    /// Configure waveform frequency (Hz) for the given channel.
29    /// Valid range: 1–50 000 Hz (implementation may restrict further).
30    fn set_frequency(&mut self, channel: u8, hz: u32) -> Result<(), Self::Error>;
31
32    /// Configure duty cycle in per mille (0–1000) for the given channel.
33    fn set_duty(&mut self, channel: u8, duty_per_mille: u16) -> Result<(), Self::Error>;
34
35    /// Start waveform generation on `channel` at the configured frequency and duty.
36    fn enable(&mut self, channel: u8) -> Result<(), Self::Error>;
37
38    /// Stop waveform generation; the pin is driven LOW.
39    fn disable(&mut self, channel: u8) -> Result<(), Self::Error>;
40
41    /// Advance the PWM state machine by one step.
42    ///
43    /// Call this in a tight pinned-poll loop (see spec 13-peripherals.md §6).
44    /// Each call may toggle output pins for any enabled channel whose deadline has elapsed.
45    fn tick(&mut self);
46}
