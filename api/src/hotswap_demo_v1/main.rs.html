3//! hotswap-demo-v1 — Phase 04 hot-migration demo (old version).
4//!
5//! Maintains a u32 counter, replies to "inc"/"get" messages, and implements
6//! the full ViStateTransfer protocol so a live upgrade to hotswap-demo-v2
7//! preserves the counter value across the cell boundary.
8//!
9//! # Sequence
10//! 1. Run v1, send "inc" N times, send "get" → receives "v1:<counter_le4>".
11//! 2. Call `sys_hotswap(cell_id, "/bin/hotswap-demo-v2")` from the shell.
12//! 3. v1 receives AppEvent::Snapshot → serializes counter → stashes.
13//! 4. v2 spawns, receives AppEvent::Restore → deserializes counter.
14//! 5. Send "get" to v2 → receives "v2:<counter_le4>" with the same counter.
15//!
16//! # Law 4 compliance
17//! `#![forbid(unsafe_code)]` — global state is guarded by `spin::Mutex`.
18
19#![no_std]
20#![no_main]
21#![forbid(unsafe_code)]
22

33// spawn = true grants StateStash / StateRestore / HotSwapReady allowlist bits
34// (bits 33, 34, 32) that the hotswap state-transfer protocol requires.
35// The cell does NOT use SpawnCap to spawn other cells, but the bits live in the
36// same allowlist group in runtime.rs:app_syscall_set.
37ostd::app_entry!(spawn = true, handler = main_handler);
38
39// ── State ─────────────────────────────────────────────────────────────────────
40
41/// Persistent counter — the state that survives a hot-swap to v2.
42struct DemoState {
43    counter: u32,
44}
45
46impl DemoState {
47    const fn new() -> Self {
48        Self { counter: 0 }
49    }
50}
51
52impl ViStateTransfer for DemoState {
53    /// Schema version 1: 4-byte little-endian counter.
54    const SCHEMA_VERSION: u32 = 1;
55
56    fn serialize(&self) -> Result<Box<[u8]>, ViError> {
57        Ok(Box::from(self.counter.to_le_bytes().as_slice()))
58    }
59
60    fn deserialize(version: u32, bytes: &[u8]) -> Result<Self, ViError> {
61        if version != 1 || bytes.len() < 4 {
62            return Err(ViError::InvalidInput);
63        }
64        let counter = u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
65        Ok(Self { counter })
66    }
67}
68
69// Single-threaded cell — Mutex<DemoState> ensures Sync without unsafe.
70static STATE: Mutex<DemoState> = Mutex::new(DemoState::new());
71
72// ── Event handler ─────────────────────────────────────────────────────────────
73
74fn main_handler(ctx: &mut AppContext, event: AppEvent) {
75    match event {
76        AppEvent::Init => {
77            println("[hotswap-demo-v1] ready — send 'inc' or 'get'");
78        }
79
80        AppEvent::Snapshot { swap_id } => {
81            // Old cell: serialize counter and stash it so v2 can restore.
82            // hotswap_key(swap_id) must match what hotswap.rs writes — both
83            // derive the same numeric key from the monotonic swap_id.
84            let key = hotswap_key(swap_id);
85            let state = STATE.lock();
86            match stash_transfer(key, &*state) {
87                Ok(()) => {
88                    println("[hotswap-demo-v1] state stashed");
89                }
90                Err(e) => {
91                    // Log and continue — kernel will proceed with an empty stash
92                    // (v2 starts cold) rather than aborting the swap.
93                    let _ = e;
94                    println("[hotswap-demo-v1] WARN: stash failed — v2 starts cold");
95                }
96            }
97        }
98
99        AppEvent::Restore { key } => {
100            // New instance of this cell (upgraded-to path); restore from stash.
101            // key is a null-terminated decimal string of swap_id.
102            let swap_id = parse_key_to_swap_id(&key);
103            let stash_key = hotswap_key(swap_id);
104            match restore_transfer::<DemoState>(stash_key) {
105                Ok(restored) => {
106                    let mut state = STATE.lock();
107                    state.counter = restored.counter;
108                    println("[hotswap-demo-v1] state restored");
109                }
110                Err(_) => {
111                    // Cold start (no prior stash) — counter stays at 0.
112                    println("[hotswap-demo-v1] WARN: no stash found — starting cold");
113                }
114            }
115            // MUST be the last call in the Restore handler.
116            // Unblocks the hotswap orchestrator's Step 4 wait.
117            sys_hotswap_ready();
118        }
119
120        AppEvent::Message { sender_tid, data } => {
121            match data.as_slice() {
122                b"inc" => {
123                    let mut state = STATE.lock();
124                    state.counter = state.counter.saturating_add(1);
125                    // Reply with raw "ok" bytes so tests can assert on a simple string.
126                    ctx.send(sender_tid, b"ok").ok();
127                }
128                b"get" => {
129                    // Reply with "v1:" + 4 LE bytes of counter so the test can assert
130                    // the version prefix AND the counter value in one message.
131                    let state = STATE.lock();
132                    let mut resp = [0u8; 7];
133                    resp[0] = b'v';
134                    resp[1] = b'1';
135                    resp[2] = b':';
136                    resp[3..7].copy_from_slice(&state.counter.to_le_bytes());
137                    ctx.send(sender_tid, &resp).ok();
138                }
139                _ => {
140                    ctx.send(sender_tid, b"unknown").ok();
141                }
142            }
143        }
144
145        AppEvent::Shutdown | AppEvent::ShutdownWith { .. } => {
146            ostd::syscall::sys_exit(0);
147        }
148
149        _ => {}
150    }
151}

153// ── Helpers ───────────────────────────────────────────────────────────────────
154
155/// Parse a null-terminated decimal ASCII key (from AppEvent::Restore) to u64.
156///
157/// The hotswap orchestrator writes the decimal swap_id into a 64-byte field
158/// with a null terminator. We stop at the first 0x00 byte or non-ASCII-digit.
159fn parse_key_to_swap_id(key: &[u8; 64]) -> u64 {
160    let mut val = 0u64;
161    for &b in key.iter() {
162        if b == 0 || !b.is_ascii_digit() {
163            break;
164        }
165        val = val.saturating_mul(10).saturating_add((b - b'0') as u64);
166    }
167    val
168}
