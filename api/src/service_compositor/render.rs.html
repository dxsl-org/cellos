1//! Software rasterizer — blends damaged surfaces into the screen framebuffer
2//! and flushes the dirty region to the VirtIO GPU via the `GpuFlush` syscall.
3

7use crate::surface_table::SurfaceTable;

11use ostd::syscall::{sys_get_resolution, sys_gpu_flush};

13/// Screen framebuffer owned by the compositor (BGRA8888).
14pub struct ScreenFb {
15    pixels: alloc::vec::Vec<u8>,
16    /// Reusable staging buffer for GPU flush — pre-allocated to avoid per-frame heap churn.
17    staging: alloc::vec::Vec<u8>,
18    pub width: u32,
19    pub height: u32,
20}
21
22impl ScreenFb {
23    /// Allocate a zeroed framebuffer of the given dimensions.
24    pub fn new(width: u32, height: u32) -> Self {
25        assert!(
26            width > 0 && height > 0 && width <= 4096 && height <= 4096,
27            "ScreenFb dimensions out of range: {}x{}",
28            width,
29            height
30        );
31        let full = (width * height * 4) as usize;
32        Self {
33            pixels: vec![0u8; full],
34            staging: vec![0u8; full],
35            width,
36            height,
37        }
38    }
39
40    /// Blit one surface's pixels into the screen FB at its screen position.
41    ///
42    /// Clips to the screen boundary; surfaces that are partially off-screen
43    /// are rendered up to the edge.
44    fn blit_surface(&mut self, s: &crate::surface_table::SurfaceState) {
45        let sx = s.x.max(0) as u32;
46        let sy = s.y.max(0) as u32;
47        let clip_x = (-s.x).max(0) as u32; // surface offset if partially off-screen
48        let clip_y = (-s.y).max(0) as u32;
49        let w = (s.w.saturating_sub(clip_x)).min(self.width.saturating_sub(sx));
50        let h = (s.h.saturating_sub(clip_y)).min(self.height.saturating_sub(sy));
51        if w == 0 || h == 0 {
52            return;
53        }
54
55        let screen_stride = self.width as usize * 4;
56        let surf_stride = s.w as usize * 4;
57        let surf_pixels = s.pixels();
58
59        for row in 0..h as usize {
60            let dst_off = (sy as usize + row) * screen_stride + sx as usize * 4;
61            let src_off = (clip_y as usize + row) * surf_stride + clip_x as usize * 4;
62            let n = w as usize * 4;
63            if dst_off + n <= self.pixels.len() && src_off + n <= surf_pixels.len() {
64                self.pixels[dst_off..dst_off + n]
65                    .copy_from_slice(&surf_pixels[src_off..src_off + n]);
66            }
67        }
68    }
69
70    /// Alpha-blend the 16×16 cursor sprite at `(cx, cy)` into the FB.
71    ///
72    /// Only pixels that fall inside `dirty` are written — avoids painting outside
73    /// the region that will be flushed to the GPU this frame.
74    /// Clips the sprite to screen bounds so a cursor near an edge is safe.
75    ///
76    /// Pixel format is BGRA8888 (native VirtIO GPU order). Alpha blend formula:
77    ///   out = src * alpha/255 + dst * (255 - alpha)/255
78    fn composite_cursor(&mut self, cx: i32, cy: i32, dirty: Rect) {
79        let stride = self.width as usize * 4;
80        for row in 0..CURSOR_H {
81            let sy = cy + row as i32;
82            if sy < 0 || sy >= self.height as i32 {
83                continue;
84            }
85            for col in 0..CURSOR_W {
86                let sx = cx + col as i32;
87                if sx < 0 || sx >= self.width as i32 {
88                    continue;
89                }
90                // Skip pixels outside the dirty region (won't be flushed).
91                if sx < dirty.x
92                    || sx >= dirty.x + dirty.w as i32
93                    || sy < dirty.y
94                    || sy >= dirty.y + dirty.h as i32
95                {
96                    continue;
97                }
98                let Some(src) = cursor_pixel(row, col) else {
99                    continue;
100                };
101                let alpha = src[3] as u32;
102                if alpha == 0 {
103                    continue;
104                }
105                let dst_off = sy as usize * stride + sx as usize * 4;
106                if dst_off + 4 > self.pixels.len() {
107                    continue;
108                }
109                // Straight-alpha blend over destination (BGRA channel order).
110                for (ch, &s_byte) in src.iter().enumerate().take(3) {
111                    let d = self.pixels[dst_off + ch] as u32;
112                    let s = s_byte as u32;
113                    self.pixels[dst_off + ch] = ((s * alpha + d * (255 - alpha)) / 255) as u8;
114                }
115                self.pixels[dst_off + 3] = 255; // opaque result
116            }
117        }
118    }
119
120    /// Flush `dirty_rect` from the screen FB to the GPU.
121    ///
122    /// Copies the dirty region into the pre-allocated staging buffer (no heap allocation)
123    /// then hands the sub-slice to the kernel. Clamps to screen boundary.
124    fn flush_rect(&mut self, dirty: Rect) {
125        let x = dirty.x.max(0) as u32;
126        let y = dirty.y.max(0) as u32;
127        let w = dirty.w.min(self.width.saturating_sub(x));
128        let h = dirty.h.min(self.height.saturating_sub(y));
129        if w == 0 || h == 0 {
130            return;
131        }
132
133        let stride = self.width as usize * 4;
134        let sub_len = (w * h * 4) as usize;
135        for row in 0..h as usize {
136            let src = (y as usize + row) * stride + x as usize * 4;
137            let dst = row * w as usize * 4;
138            let n = w as usize * 4;
139            if src + n <= self.pixels.len() && dst + n <= self.staging.len() {
140                self.staging[dst..dst + n].copy_from_slice(&self.pixels[src..src + n]);
141            }
142        }
143        let _ = sys_gpu_flush(&self.staging[..sub_len], x, y, w, h);
144    }
145}

147/// Render one frame: blit all damaged surfaces, composite the cursor, then
148/// flush the combined dirty rect to the VirtIO GPU.
149///
150/// `extra_dirty` is a compositor-initiated repaint region (cursor move, surface
151/// destroyed/raised) that is unioned with per-surface damage before blitting.
152/// `cursor_x/cursor_y` is the current logical mouse position used to draw the
153/// 16×16 software cursor sprite on top of all surfaces.
154///
155/// `extra_dirty` is a compositor-initiated repaint region (e.g. surface just
156/// destroyed or raised) that is unioned with per-surface damage before blitting.
157///
158/// Returns the dirty rect that was flushed, or `None` if nothing was dirty.
159/// `fb` requires `&mut` because `flush_rect` writes into the staging buffer.
160pub fn render_frame(
161    fb: &mut ScreenFb,
162    table: &mut SurfaceTable,
163    z_order: &ZOrder,
164    extra_dirty: Option<Rect>,
165    cursor_x: i32,
166    cursor_y: i32,
167) -> Option<Rect> {
168    // Seed with compositor-initiated dirty region (cursor move, surface destroyed/raised).
169    let mut dirty: Option<Rect> = extra_dirty;
170    for cap in z_order.iter_bottom_to_top() {
171        if let Some(s) = table.get(cap) {
172            if let Some(dmg) = s.damage {
173                // Translate surface-local damage to screen coordinates.
174                let screen_dmg = Rect {
175                    x: s.x + dmg.x,
176                    y: s.y + dmg.y,
177                    w: dmg.w,
178                    h: dmg.h,
179                };
180                dirty = Some(match dirty {
181                    Some(acc) => acc.union(&screen_dmg),
182                    None => screen_dmg,
183                });
184            }
185        }
186    }
187
188    let dirty = dirty?;
189
190    // Re-blit all surfaces that overlap the dirty rect (bottom to top).
191    for cap in z_order.iter_bottom_to_top() {
192        if let Some(s) = table.get(cap) {
193            if s.screen_rect().intersects(&dirty) {
194                fb.blit_surface(s);
195            }
196        }
197    }
198
199    // Clear damage on all surfaces.
200    for cap in z_order.iter_bottom_to_top() {
201        if let Some(s) = table.get_mut(cap) {
202            s.clear_damage();
203        }
204    }
205
206    // Composite the cursor sprite on top of all surfaces within the dirty rect.
207    // The cursor is always drawn last so it occludes surface pixels.
208    fb.composite_cursor(cursor_x, cursor_y, dirty);
209
210    // Flush the dirty rect to the GPU.
211    fb.flush_rect(dirty);
212    Some(dirty)
213}

215/// Return the screen dimensions from the GPU's actual scanout resolution.
216///
217/// Calls `GpuGetResolution` syscall so all callers (ScreenFb alloc, GET_SCREEN_SIZE
218/// replies, etc.) always agree with the hardware — no hardcoded fallback constants.
219pub fn default_screen_size() -> (u32, u32) {
220    sys_get_resolution()
