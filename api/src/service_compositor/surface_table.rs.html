1//! CapId-keyed surface state table.
2//!
3//! Each surface references pixel data via one of two sources:
4//!   - `PixelSource::Grant` — a read-only pointer into the app cell's own Grant buffer
5//!     (zero-copy; app writes directly, compositor reads directly).
6//!   - `PixelSource::Owned` — a compositor-owned buffer (legacy `WRITE_PIXELS` path).
7//!
8//! New code should always use the Grant path via `attach_grant`.
9
10extern crate alloc;
11
12use alloc::{boxed::Box, collections::BTreeMap};
13use api::display::{PixelFormat, Rect};
14use types::ViError;
15
16/// Maximum number of simultaneous surfaces.
17///
18/// 32 covers typical desktop use; reduce to 2 for kiosk/embedded profiles.
19pub const MAX_SURFACES: usize = 32;
20
21/// Pixel data source for a surface.
22#[allow(dead_code)] // reg_id reserved for future cleanup on cell exit
23enum PixelSource {
24    /// App Cell's Grant buffer — compositor reads directly via a read-only pointer.
25    ///
26    /// SAFETY invariant: `ptr` is valid for reads as long as the owning cell holds its
27    /// `sys_grant_register` buffer.  The protocol requires the app to send `DETACH_GRANT`
28    /// before calling `sys_grant_unregister`, so the pointer is never dangling during a
29    /// render tick.  The compositor never writes through this pointer.
30    Grant { ptr: *const u8, reg_id: usize },
31    /// Compositor-owned fallback buffer (legacy `WRITE_PIXELS` path).
32    Owned(Box<[u8]>),
33}
34
35// SAFETY: compositor runs as a single cooperative task; no other task touches
36// PixelSource concurrently.  The Grant pointer is a stable physical page whose
37// lifetime is enforced by the DETACH_GRANT protocol above.
38unsafe impl Send for PixelSource {}
39
40/// State for one live surface.
41pub struct SurfaceState {
42    /// Screen position.
43    pub x: i32,
44    pub y: i32,
45    /// Dimensions in pixels.
46    pub w: u32,
47    pub h: u32,
48    /// Pixel format (default: Bgra8888).
49    pub fmt: PixelFormat,
50    /// Pixel data source.
51    source: PixelSource,
52    /// Accumulated damage since last flush.  `None` = no damage.
53    pub damage: Option<Rect>,
54    /// TID of the cell that created this surface (input routing + ownership checks).
55    pub owner: usize,
56}
57
58impl SurfaceState {
59    /// Allocate a new surface with a compositor-owned pixel buffer.
60    ///
61    /// Used by `CREATE_SURFACE` before the app attaches a Grant.  Preserves
62    /// compatibility with the legacy `WRITE_PIXELS` path.
63    pub fn new(x: i32, y: i32, w: u32, h: u32, owner: usize) -> Self {
64        // No eager pixel buffer. The Grant path (attach_grant) is the standard flow
65        // and replaces `source` immediately after CREATE_SURFACE, so pre-allocating
66        // w*h*4 here is wasted — and for a full-screen surface (3 MiB at 1024×768) it
67        // would briefly sit on top of the compositor's own framebuffers and OOM its
68        // 8 MiB cell heap. The legacy WRITE_PIXELS path grows the buffer lazily.
69        Self {
70            x,
71            y,
72            w,
73            h,
74            fmt: PixelFormat::Bgra8888,
75            source: PixelSource::Owned(alloc::vec::Vec::new().into_boxed_slice()),
76            damage: None,
77            owner,
78        }
79    }
80
81    /// Attach a Grant buffer from the app cell.
82    ///
83    /// `ptr` comes from `sys_grant_slice` after the app shared the grant read-only.
84    /// Updates dimensions and format from the `AttachGrant` message.
85    pub fn attach_grant(
86        &mut self,
87        ptr: *const u8,
88        reg_id: usize,
89        w: u32,
90        h: u32,
91        fmt: PixelFormat,
92    ) {
93        self.w = w;
94        self.h = h;
95        self.fmt = fmt;
96        self.source = PixelSource::Grant { ptr, reg_id };
97    }
98
99    /// Detach the Grant and fall back to an empty Owned buffer.
100    ///
101    /// Must be called when the app sends `DETACH_GRANT`, before it frees the Grant.
102    /// We do NOT eagerly allocate a full-screen replacement — a large surface would
103    /// OOM the compositor heap, and a surface is almost always destroyed right after
104    /// detach. The legacy WRITE_PIXELS path regrows the buffer lazily if reused.
105    pub fn detach_grant(&mut self) {
106        self.source = PixelSource::Owned(alloc::vec::Vec::new().into_boxed_slice());
107    }
108
109    /// Read access to pixel data — either from the Grant or the Owned buffer.
110    pub fn pixels(&self) -> &[u8] {
111        match &self.source {
112            PixelSource::Grant { ptr, .. } => {
113                let len = (self.w * self.h * self.fmt.bpp()) as usize;
114                // SAFETY: ptr comes from sys_grant_slice after the app called
115                // sys_grant_share(perm=0, ReadOnly).  The buffer is registered
116                // (sys_grant_register) for the surface's lifetime; the app sends
117                // DETACH_GRANT before sys_grant_unregister, ensuring the pointer
118                // is valid for any render tick that follows ATTACH_GRANT.
119                // The compositor only reads — no write through this pointer.
120                unsafe { core::slice::from_raw_parts(*ptr, len) }
121            }
122            PixelSource::Owned(buf) => buf,
123        }
124    }
125
126    /// Write pixel data into an Owned surface (legacy `WRITE_PIXELS` path).
127    ///
128    /// Silently ignores writes on Grant surfaces — those are written directly by
129    /// the app cell.
130    pub fn write_pixels(&mut self, px: i32, py: i32, pw: u32, ph: u32, data: &[u8]) {
131        // Lazily allocate the legacy owned buffer to full surface size on first use
132        // (new()/detach_grant() leave it empty to avoid eager full-screen allocs).
133        let needed = (self.w * self.h * 4) as usize;
134        if let PixelSource::Owned(b) = &self.source {
135            if b.len() < needed {
136                self.source = PixelSource::Owned(alloc::vec![0u8; needed].into_boxed_slice());
137            }
138        }
139        let buf = match &mut self.source {
140            PixelSource::Owned(b) => b,
141            PixelSource::Grant { .. } => return,
142        };
143        let expected = (pw * ph * 4) as usize;
144        if data.len() < expected {
145            return;
146        }
147        let stride = self.w as usize * 4;
148        for row in 0..ph as usize {
149            let dst_off = (py as usize + row) * stride + px as usize * 4;
150            let src_off = row * pw as usize * 4;
151            let row_bytes = pw as usize * 4;
152            if dst_off + row_bytes <= buf.len() {
153                buf[dst_off..dst_off + row_bytes]
154                    .copy_from_slice(&data[src_off..src_off + row_bytes]);
155            }
156        }
157        let new_dmg = Rect {
158            x: px,
159            y: py,
160            w: pw,
161            h: ph,
162        };
163        self.damage = Some(match self.damage {
164            Some(existing) => existing.union(&new_dmg),
165            None => new_dmg,
166        });
167    }
168
169    /// Clear the damage accumulator after a flush.
170    pub fn clear_damage(&mut self) {
171        self.damage = None;
172    }
173
174    /// Bounding rect of this surface on screen.
175    pub fn screen_rect(&self) -> Rect {
176        Rect {
177            x: self.x,
178            y: self.y,
179            w: self.w,
180            h: self.h,
181        }
182    }
183}

185/// CapId-keyed surface registry.
186#[derive(Default)]
187pub struct SurfaceTable {
188    entries: BTreeMap<u64, SurfaceState>,
189    next_cap: u64,
190}
191
192impl SurfaceTable {
193    pub fn new() -> Self {
194        Self {
195            entries: BTreeMap::new(),
196            next_cap: 1,
197        }
198    }
199
200    /// Allocate a new surface slot and return its CapId.
201    ///
202    /// `owner` is the TID of the creating cell (ownership checks + input routing).
203    ///
204    /// # Errors
205    /// Returns `OutOfMemory` if `MAX_SURFACES` is already reached.
206    pub fn create(&mut self, x: i32, y: i32, w: u32, h: u32, owner: usize) -> Result<u64, ViError> {
207        if self.entries.len() >= MAX_SURFACES {
208            return Err(ViError::OutOfMemory);
209        }
210        let cap = self.next_cap;
211        self.next_cap += 1;
212        self.entries
213            .insert(cap, SurfaceState::new(x, y, w, h, owner));
214        Ok(cap)
215    }
216
217    /// Look up a surface mutably.
218    pub fn get_mut(&mut self, cap: u64) -> Option<&mut SurfaceState> {
219        self.entries.get_mut(&cap)
220    }
221
222    /// Look up a surface immutably.
223    pub fn get(&self, cap: u64) -> Option<&SurfaceState> {
224        self.entries.get(&cap)
225    }
226
227    /// Remove a surface.
228    pub fn remove(&mut self, cap: u64) -> Option<SurfaceState> {
229        self.entries.remove(&cap)
230    }
231
232    /// Returns true if any live surface has accumulated damage.
233    ///
234    /// Used by the main loop to decide whether to call render_frame.
235    pub fn has_damage(&self) -> bool {
236        self.entries.values().any(|s| s.damage.is_some())
237    }
238
239    /// Find all surfaces owned by `tid` and return their caps.
240    #[allow(dead_code)] // used by future NotifyOnExit cleanup path
241    pub fn caps_owned_by(&self, tid: usize) -> alloc::vec::Vec<u64> {
242        self.entries
243            .iter()
244            .filter(|(_, s)| s.owner == tid)
245            .map(|(&cap, _)| cap)
246            .collect()
247    }
248}
